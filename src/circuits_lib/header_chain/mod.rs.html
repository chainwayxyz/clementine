25/// The main entry point of the header chain circuit.
26pub fn header_chain_circuit(guest: &impl ZkvmGuest) {
27    // let start = risc0_zkvm::guest::env::cycle_count();
28
29    let input: HeaderChainCircuitInput = guest.read_from_host();
30    // println!("Detected network: {:?}", NETWORK_TYPE);
31    // println!("NETWORK_CONSTANTS: {:?}", NETWORK_CONSTANTS);
32    let mut chain_state = match input.prev_proof {
33        HeaderChainPrevProofType::GenesisBlock => ChainState::new(),
34        HeaderChainPrevProofType::PrevProof(prev_proof) => {
35            assert_eq!(prev_proof.method_id, input.method_id);
36            guest.verify(input.method_id, &prev_proof);
37            prev_proof.chain_state
38        }
39    };
40
41    chain_state.apply_blocks(input.block_headers);
42
43    guest.commit(&BlockHeaderCircuitOutput {
44        method_id: input.method_id,
45        chain_state,
46    });
47    // let end = risc0_zkvm::guest::env::cycle_count();
48    // println!("Header chain circuit took {:?} cycles", end - start);
49}

131/// Bitcoin block header.
132#[derive(Serialize, Deserialize, Eq, PartialEq, Clone, Debug, BorshDeserialize, BorshSerialize)]
133pub struct CircuitBlockHeader {
134    pub version: i32,
135    pub prev_block_hash: [u8; 32],
136    pub merkle_root: [u8; 32],
137    pub time: u32,
138    pub bits: u32,
139    pub nonce: u32,
140}
141
142impl CircuitBlockHeader {
143    pub fn compute_block_hash(&self) -> [u8; 32] {
144        let mut hasher = Sha256::new();
145        hasher.update(self.version.to_le_bytes());
146        hasher.update(self.prev_block_hash);
147        hasher.update(self.merkle_root);
148        hasher.update(self.time.to_le_bytes());
149        hasher.update(self.bits.to_le_bytes());
150        hasher.update(self.nonce.to_le_bytes());
151        let first_hash_result = hasher.finalize_reset();
152
153        hasher.update(first_hash_result);
154        let result: [u8; 32] = hasher.finalize().into();
155        result
156    }
157}
158
159impl From<Header> for CircuitBlockHeader {
160    fn from(header: Header) -> Self {
161        CircuitBlockHeader {
162            version: header.version.to_consensus(),
163            prev_block_hash: header.prev_blockhash.to_byte_array(),
164            merkle_root: header.merkle_root.as_raw_hash().to_byte_array(),
165            time: header.time,
166            bits: header.bits.to_consensus(),
167            nonce: header.nonce,
168        }
169    }
170}
171
172impl From<CircuitBlockHeader> for Header {
173    fn from(val: CircuitBlockHeader) -> Self {
174        Header {
175            version: Version::from_consensus(val.version),
176            prev_blockhash: BlockHash::from_slice(&val.prev_block_hash)
177                .expect("Previous block hash is 32 bytes"),
178            merkle_root: TxMerkleNode::from_slice(&val.merkle_root)
179                .expect("Merkle root is 32 bytes"),
180            time: val.time,
181            bits: CompactTarget::from_consensus(val.bits),
182            nonce: val.nonce,
183        }
184    }
185}

187#[derive(Serialize, Deserialize, Eq, PartialEq, Clone, Debug, BorshDeserialize, BorshSerialize)]
188pub struct ChainState {
189    pub block_height: u32,
190    pub total_work: [u8; 32],
191    pub best_block_hash: [u8; 32],
192    pub current_target_bits: u32,
193    pub epoch_start_time: u32,
194    pub prev_11_timestamps: [u32; 11],
195    pub block_hashes_mmr: MMRGuest,
196}
197
198impl Default for ChainState {
199    fn default() -> Self {
200        ChainState::new()
201    }
202}
203
204impl ChainState {
205    pub fn new() -> Self {
206        ChainState {
207            block_height: u32::MAX,
208            total_work: [0u8; 32],
209            best_block_hash: [0u8; 32],
210            current_target_bits: NETWORK_CONSTANTS.max_bits,
211            epoch_start_time: 0,
212            prev_11_timestamps: [0u32; 11],
213            block_hashes_mmr: MMRGuest::new(),
214        }
215    }
216
217    pub fn apply_blocks(&mut self, block_headers: Vec<CircuitBlockHeader>) {
218        println!("Network type: {:?}", NETWORK_TYPE);
219        let mut current_target_bytes = if IS_REGTEST {
220            NETWORK_CONSTANTS.max_target.to_be_bytes()
221        } else {
222            bits_to_target(self.current_target_bits)
223        };
224        let mut current_work: U256 = U256::from_be_bytes(self.total_work);
225
226        let mut last_block_time = if IS_TESTNET4 {
227            if self.block_height == u32::MAX {
228                0
229            } else {
230                self.prev_11_timestamps[self.block_height as usize % 11]
231            }
232        } else {
233            0
234        };
235
236        for block_header in block_headers {
237            self.block_height = self.block_height.wrapping_add(1);
238
239            let (target_to_use, expected_bits, work_to_add) = if IS_TESTNET4 {
240                if block_header.time > last_block_time + 1200 {
241                    // If the block is an epoch block, then it still has to have the real target.
242                    if self.block_height % BLOCKS_PER_EPOCH == 0 {
243                        (
244                            current_target_bytes,
245                            self.current_target_bits,
246                            calculate_work(&current_target_bytes),
247                        )
248                    } else {
249                        (
250                            NETWORK_CONSTANTS.max_target_bytes,
251                            NETWORK_CONSTANTS.max_bits,
252                            MINIMUM_WORK_TESTNET,
253                        )
254                    }
255                } else {
256                    (
257                        current_target_bytes,
258                        self.current_target_bits,
259                        calculate_work(&current_target_bytes),
260                    )
261                }
262            } else {
263                (
264                    current_target_bytes,
265                    self.current_target_bits,
266                    calculate_work(&current_target_bytes),
267                )
268            };
269
270            let new_block_hash = block_header.compute_block_hash();
271
272            assert_eq!(block_header.prev_block_hash, self.best_block_hash);
273
274            if IS_REGTEST {
275                assert_eq!(block_header.bits, NETWORK_CONSTANTS.max_bits);
276            } else {
277                assert_eq!(block_header.bits, expected_bits);
278            }
279
280            check_hash_valid(&new_block_hash, &target_to_use);
281
282            if !validate_timestamp(block_header.time, self.prev_11_timestamps) {
283                panic!("Timestamp is not valid");
284            }
285
286            self.block_hashes_mmr.append(new_block_hash);
287            self.best_block_hash = new_block_hash;
288            current_work = current_work.wrapping_add(&work_to_add);
289
290            if !IS_REGTEST && self.block_height % BLOCKS_PER_EPOCH == 0 {
291                self.epoch_start_time = block_header.time;
292            }
293
294            self.prev_11_timestamps[self.block_height as usize % 11] = block_header.time;
295
296            if IS_TESTNET4 {
297                last_block_time = block_header.time;
298            }
299
300            if !IS_REGTEST && self.block_height % BLOCKS_PER_EPOCH == BLOCKS_PER_EPOCH - 1 {
301                current_target_bytes = calculate_new_difficulty(
302                    self.epoch_start_time,
303                    block_header.time,
304                    self.current_target_bits,
305                );
306                self.current_target_bits = target_to_bits(&current_target_bytes);
307            }
308        }
309
310        self.total_work = current_work.to_be_bytes();
311    }
312}

314fn median(arr: [u32; 11]) -> u32 {
315    let mut sorted_arr = arr;
316    sorted_arr.sort_unstable();
317    sorted_arr[5]
318}
319
320fn validate_timestamp(block_time: u32, prev_11_timestamps: [u32; 11]) -> bool {
321    let median_time = median(prev_11_timestamps);
322    block_time > median_time
323}

325pub fn bits_to_target(bits: u32) -> [u8; 32] {
326    let size = (bits >> 24) as usize;
327    let mantissa = bits & 0x00ffffff;
328
329    let target = if size <= 3 {
330        U256::from(mantissa >> (8 * (3 - size)))
331    } else {
332        U256::from(mantissa) << (8 * (size - 3))
333    };
334    // println!("Output: {:?}", target.to_be_bytes());
335    target.to_be_bytes()
336}
337
338fn target_to_bits(target: &[u8; 32]) -> u32 {
339    let target_u256 = U256::from_be_slice(target);
340    let target_bits = target_u256.bits();
341    let size = (263 - target_bits) / 8;
342    let mut compact_target = [0u8; 4];
343    compact_target[0] = 33 - size as u8;
344    compact_target[1] = target[size - 1_usize];
345    compact_target[2] = target[size];
346    compact_target[3] = target[size + 1_usize];
347    u32::from_be_bytes(compact_target)
348}
349
350fn calculate_new_difficulty(
351    epoch_start_time: u32,
352    last_timestamp: u32,
353    current_target: u32,
354) -> [u8; 32] {
355    let mut actual_timespan = last_timestamp - epoch_start_time;
356    if actual_timespan < EXPECTED_EPOCH_TIMESPAN / 4 {
357        actual_timespan = EXPECTED_EPOCH_TIMESPAN / 4;
358    } else if actual_timespan > EXPECTED_EPOCH_TIMESPAN * 4 {
359        actual_timespan = EXPECTED_EPOCH_TIMESPAN * 4;
360    }
361
362    let new_target_bytes = bits_to_target(current_target);
363    let mut new_target = U256::from_be_bytes(new_target_bytes)
364        .wrapping_mul(&U256::from(actual_timespan))
365        .wrapping_div(&U256::from(EXPECTED_EPOCH_TIMESPAN));
366
367    if new_target > NETWORK_CONSTANTS.max_target {
368        new_target = NETWORK_CONSTANTS.max_target;
369    }
370    new_target.to_be_bytes()
371}
372
373fn check_hash_valid(hash: &[u8; 32], target_bytes: &[u8; 32]) {
374    for i in 0..32 {
375        match hash[31 - i].cmp(&target_bytes[i]) {
376            Ordering::Less => return,
377            Ordering::Greater => panic!("Hash is not valid"),
378            Ordering::Equal => continue,
379        }
380    }
381}
382
383fn calculate_work(target: &[u8; 32]) -> U256 {
384    let target = U256::from_be_slice(target);
385    let target_plus_one = target.saturating_add(&U256::ONE);
386    U256::MAX.wrapping_div(&target_plus_one)
387}
