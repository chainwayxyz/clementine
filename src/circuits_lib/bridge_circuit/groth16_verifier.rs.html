1//! # Bridge Circuit Groth16 Verifier
2//! This module implements the Groth16 verifier for the bridge circuit.
3//! It includes functions to create digests for the work-only circuit output,
4//! verify Groth16 proofs, and handle the conversion of hexadecimal strings to decimal.
5//! The verifier uses the prepared verification keys and constants defined in the `constants.rs` module.
6//!
7//! ## Key Functions
8//! - `create_journal_digest`: Creates a digest for the journal of the work-only circuit output.
9//! - `create_output_digest`: Creates an output digest for the work-only circuit output.
10//! - `create_claim_digest`: Creates a claim digest for the work-only circuit output.
11//! - `CircuitGroth16WithTotalWork`: A struct that encapsulates the Groth16 proof with total work and genesis state hash.
12//! - `verify`: Verifies the Groth16 proof against the prepared verification key and public inputs.
13//! - `to_decimal`: Converts a hexadecimal string to a decimal string representation.
14

32/// Creates a digest for the journal of the work-only circuit output.
33pub fn create_journal_digest(work_only_circuit_output: &WorkOnlyCircuitOutput) -> [u8; 32] {
34    let pre_digest = borsh::to_vec(work_only_circuit_output).unwrap();
35    Sha256::digest(pre_digest).into()
36}
37
38/// Creates an output digest for the work-only circuit output.
39pub fn create_output_digest(work_only_circuit_output: &WorkOnlyCircuitOutput) -> [u8; 32] {
40    let journal_digest: [u8; 32] = create_journal_digest(work_only_circuit_output);
41    let len_output: u16 = 2;
42
43    let output_pre_digest: [u8; 98] = [
44        &OUTPUT_TAG,
45        &journal_digest[..],
46        &ASSUMPTIONS[..],
47        &len_output.to_le_bytes(),
48    ]
49    .concat()
50    .try_into()
51    .expect("Slice has correct length");
52
53    Sha256::digest(output_pre_digest).into()
54}
55
56/// Creates a claim digest for the work-only circuit output.
57pub fn create_claim_digest(output_digest: &[u8; 32], pre_state: &[u8; 32]) -> [u8; 32] {
58    let data: [u8; 8] = [0; 8];
59
60    let claim_len: u16 = 4;
61
62    let concatenated = [
63        &CLAIM_TAG,
64        &INPUT,
65        pre_state,
66        &POST_STATE,
67        output_digest,
68        &data[..],
69        &claim_len.to_le_bytes(),
70    ]
71    .concat();
72
73    let mut claim_digest = Sha256::digest(concatenated);
74    claim_digest.reverse();
75
76    claim_digest.into()
77}

79/// Groth16 proof with total work and genesis state hash. In Clementine, this is provided by
80/// the watchtowers who challenge the operator whom they suspect of malicious behavior. Just
81/// by knowing the Groth16 proof and the total work, we can reconstruct the public outputs of
82/// the proof and verify it against the Verifying Key (VK) of the Groth16 proof.
83pub struct CircuitGroth16WithTotalWork {
84    groth16_seal: CircuitGroth16Proof,
85    total_work: [u8; 16],
86    genesis_state_hash: [u8; 32],
87}
88
89impl CircuitGroth16WithTotalWork {
90    /// Creates a new instance of `CircuitGroth16WithTotalWork`.
91    pub fn new(
92        groth16_seal: CircuitGroth16Proof,
93        total_work: [u8; 16],
94        genesis_state_hash: [u8; 32],
95    ) -> CircuitGroth16WithTotalWork {
96        CircuitGroth16WithTotalWork {
97            groth16_seal,
98            total_work,
99            genesis_state_hash,
100        }
101    }
102
103    /// Given the `pre_state` (which is actually the `method ID` of the work-only circuit),
104    /// verifies the Groth16 proof against the prepared Verifying Key (VK) and the public inputs.
105    pub fn verify(&self, pre_state: &[u8; 32]) -> bool {
106        let ark_proof: Proof<Bn254> = self.groth16_seal.into();
107
108        let prepared_vk: &[u8] = get_prepared_vk();
109
110        let prepared_vk: PreparedVerifyingKey<Bn254> =
111            CanonicalDeserialize::deserialize_uncompressed(prepared_vk).unwrap();
112
113        let output_digest = create_output_digest(&WorkOnlyCircuitOutput {
114            work_u128: self.total_work,
115            genesis_state_hash: self.genesis_state_hash,
116        });
117
118        let claim_digest: [u8; 32] = create_claim_digest(&output_digest, pre_state);
119
120        let claim_digest_hex: String = claim_digest.encode_hex();
121        let c0_str = &claim_digest_hex[32..64];
122        let c1_str = &claim_digest_hex[0..32];
123
124        let c0_dec = to_decimal(c0_str).unwrap();
125        let c1_dec = to_decimal(c1_str).unwrap();
126
127        let c0 = Fr::from_str(&c0_dec).unwrap();
128        let c1 = Fr::from_str(&c1_dec).unwrap();
129
130        let public_inputs = vec![A0_ARK, A1_ARK, c0, c1, BN_254_CONTROL_ID_ARK];
131
132        ark_groth16::Groth16::<Bn254>::verify_proof(&prepared_vk, &ark_proof, &public_inputs)
133            .unwrap_or(false)
134    }
135}

137/// Converts a hexadecimal string to a decimal string representation.
138pub fn to_decimal(s: &str) -> Option<String> {
139    let int = BigUint::from_str_radix(s, 16).ok();
140    int.map(|n| n.to_str_radix(10))
141}
