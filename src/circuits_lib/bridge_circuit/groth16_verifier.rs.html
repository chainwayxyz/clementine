17pub fn create_output_digest(total_work: &[u8; 16]) -> [u8; 32] {
18    let total_work_digest: [u8; 32] = Sha256::digest(total_work).into();
19    let len_output: u16 = 2;
20
21    let output_pre_digest: [u8; 98] = [
22        &OUTPUT_TAG,
23        &total_work_digest[..],
24        &ASSUMPTIONS[..],
25        &len_output.to_le_bytes(),
26    ]
27    .concat()
28    .try_into()
29    .expect("slice has correct length");
30
31    Sha256::digest(output_pre_digest).into()
32}

34pub fn create_claim_digest(output_digest: &[u8; 32], pre_state: &[u8; 32]) -> [u8; 32] {
35    let data: [u8; 8] = [0; 8];
36
37    let claim_len: u16 = 4;
38
39    let concatenated = [
40        &CLAIM_TAG,
41        &INPUT,
42        pre_state,
43        &POST_STATE,
44        output_digest,
45        &data[..],
46        &claim_len.to_le_bytes(),
47    ]
48    .concat();
49
50    let mut claim_digest = Sha256::digest(concatenated);
51    claim_digest.reverse();
52
53    claim_digest.into()
54}
55pub struct CircuitGroth16WithTotalWork {
56    groth16_seal: CircuitGroth16Proof,
57    total_work: [u8; 16],
58}
59
60impl CircuitGroth16WithTotalWork {
61    pub fn new(
62        groth16_seal: CircuitGroth16Proof,
63        total_work: [u8; 16],
64    ) -> CircuitGroth16WithTotalWork {
65        CircuitGroth16WithTotalWork {
66            groth16_seal,
67            total_work,
68        }
69    }
70
71    pub fn verify(&self, pre_state: &[u8; 32]) -> bool {
72        let ark_proof: Proof<Bn254> = self.groth16_seal.into();
73        let prepared_vk: PreparedVerifyingKey<ark_ec::bn::Bn<ark_bn254::Config>> =
74            CanonicalDeserialize::deserialize_uncompressed(PREPARED_VK).unwrap();
75
76        let output_digest = create_output_digest(&self.total_work);
77
78        let claim_digest: [u8; 32] = create_claim_digest(&output_digest, pre_state);
79
80        let claim_digest_hex: String = claim_digest.encode_hex();
81        let c0_str = &claim_digest_hex[32..64];
82        let c1_str = &claim_digest_hex[0..32];
83
84        let c0_dec = to_decimal(c0_str).unwrap();
85        let c1_dec = to_decimal(c1_str).unwrap();
86
87        let c0 = Fr::from_str(&c0_dec).unwrap();
88        let c1 = Fr::from_str(&c1_dec).unwrap();
89
90        let public_inputs = vec![A0_ARK, A1_ARK, c0, c1, BN_254_CONTROL_ID_ARK];
91
92        ark_groth16::Groth16::<Bn254>::verify_proof(&prepared_vk, &ark_proof, &public_inputs)
93            .unwrap()
94    }
