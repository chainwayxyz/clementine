58/// Executes the bridge circuit in a zkVM environment, verifying multiple cryptographic proofs
59/// related to watchtower work, SPV, and storage proofs.
60///
61/// # Parameters
62///
63/// - `guest`: A reference to a zkVM guest implementing `ZkvmGuest`.
64/// - `work_only_image_id`: A 32-byte array representing the work-only image ID used in verification.
65///
66/// # Functionality
67///
68/// 1. Reads the `BridgeCircuitInput` from the host.
69/// 2. Ensures the method ID in `hcp` (header chain proof) matches `HEADER_CHAIN_METHOD_ID`.
70/// 3. Verifies the header chain proof (`hcp`).
71/// 4. Computes total work and watchtower challenge flags using `total_work_and_watchtower_flags`.
72/// 5. Validates that the computed `total_work` does not exceed the total work in `hcp.chain_state`.
73/// 6. Fetches the MMR (Merkle Mountain Range) for block hashes from `hcp.chain_state`.
74/// 7. Verifies the SPV proof (`payout_spv`) using the fetched MMR.
75/// 8. Verifies the light client proof using `lc_proof_verifier`.
76/// 9. Checks storage proofs for deposit and withdrawal transaction indices using `verify_storage_proofs`.
77/// 10. Converts the verified withdrawal outpoint into a Bitcoin transaction ID.
78/// 11. Ensures the withdrawal transaction ID matches the input reference in `payout_spv.transaction`.
79/// 12. Computes the `deposit_constant` using the last output of the payout transaction.
80/// 13. Extracts and truncates the latest block hash and the payout transactionâ€™s block hash.
81/// 14. Computes a Blake3 hash over concatenated block hash and watchtower flags.
82/// 15. Generates a final journal hash using Blake3 over concatenated data and commits it.
83///
84/// # Panics
85///
86/// - If the method ID in `hcp` does not match `HEADER_CHAIN_METHOD_ID`.
87/// - If `max_total_work` given by watchtowers is greater than `hcp.chain_state.total_work`.
88/// - If the SPV proof is invalid.
89/// - If the storage proof verification fails.
90/// - If the withdrawal transaction ID does not match the referenced input in `payout_spv`.
91pub fn bridge_circuit(guest: &impl ZkvmGuest, work_only_image_id: [u8; 32]) {
92    let input: BridgeCircuitInput = guest.read_from_host();
93    assert_eq!(HEADER_CHAIN_METHOD_ID, input.hcp.method_id);
94
95    // Verify the HCP
96    guest.verify(input.hcp.method_id, &input.hcp);
97
98    let kickoff_txid = input.kickoff_tx_id; // TODO: Maybe use `txid` instead of `compute_txid`
99
100    let (max_total_work, challenge_sending_watchtowers) =
101        total_work_and_watchtower_flags(&input, &work_only_image_id);
102
103    // Why is that 32 bytes in the first place?
104    let total_work: TotalWork = input.hcp.chain_state.total_work[16..32]
105        .try_into()
106        .expect("Cannot fail");
107
108    // If total work is less than the max total work of watchtowers, panic
109    if total_work < max_total_work {
110        panic!(
111            "Invalid total work: Total Work {:?} - Max Total Work: {:?}",
112            input.hcp.chain_state.total_work, max_total_work
113        );
114    }
115
116    // MMR WILL BE FETCHED FROM LC PROOF WHEN IT IS READY - THIS IS JUST FOR PROOF OF CONCEPT
117    let mmr = input.hcp.chain_state.block_hashes_mmr.clone();
118
119    if !input.payout_spv.verify(mmr) {
120        panic!("Invalid SPV proof");
121    }
122
123    // Light client proof verification
124    let light_client_circuit_output = lc_proof_verifier(input.lcp.clone());
125
126    // Storage proof verification for deposit tx index and withdrawal outpoint
127    let (user_wd_outpoint, vout, move_txid) =
128        verify_storage_proofs(&input.sp, light_client_circuit_output.l2_state_root);
129
130    let user_wd_txid = bitcoin::Txid::from_byte_array(*user_wd_outpoint);
131
132    let payout_input_index: usize = input.payout_input_index as usize;
133
134    assert_eq!(
135        user_wd_txid,
136        input.payout_spv.transaction.input[payout_input_index]
137            .previous_output
138            .txid,
139        "Invalid withdrawal transaction ID"
140    );
141
142    assert_eq!(
143        vout,
144        input.payout_spv.transaction.input[payout_input_index]
145            .previous_output
146            .vout,
147        "Invalid withdrawal transaction output index"
148    );
149
150    let last_output = input.payout_spv.transaction.output.last().unwrap();
151
152    let deposit_constant = deposit_constant(
153        last_output,
154        &kickoff_txid,
155        input.watchtower_challenge_connector_start_idx,
156        &input.all_tweaked_watchtower_pubkeys,
157        *move_txid,
158    );
159
160    let latest_blockhash: LatestBlockhash = input.hcp.chain_state.best_block_hash[12..32]
161        .try_into()
162        .unwrap();
163    let payout_tx_blockhash: PayoutTxBlockhash = input.payout_spv.block_header.compute_block_hash()
164        [12..32]
165        .try_into()
166        .unwrap();
167
168    let journal_hash = journal_hash(
169        payout_tx_blockhash,
170        latest_blockhash,
171        challenge_sending_watchtowers,
172        deposit_constant,
173    );
174
175    guest.commit(journal_hash.as_bytes());
176}

178/// Converts a compressed Groth16 proof into a proof structure and verifies it against a given image ID.
179///
180/// # Parameters
181///
182/// - `compressed_proof`: A reference to a 128-byte array containing the compressed Groth16 proof.
183/// - `total_work`: A 16-byte array representing the total accumulated work associated with the proof.
184/// - `image_id`: A reference to a 32-byte array representing the image ID used for verification.
185///
186/// # Returns
187///
188/// - `true` if the Groth16 proof is successfully deserialized and verified.
189/// - `false` if any step in the process fails (e.g., failed deserialization or proof verification).
190///
191/// # Failure Cases
192///
193/// - If deserialization of the compressed proof fails, it returns `false`.
194/// - If Groth16 proof verification fails, it returns `false`.
195fn convert_to_groth16_and_verify(
196    compressed_proof: &[u8; 128],
197    total_work: [u8; 16],
198    image_id: &[u8; 32],
199) -> bool {
200    let seal = match CircuitGroth16Proof::from_compressed(compressed_proof) {
201        Ok(seal) => seal,
202        Err(_) => return false,
203    };
204
205    let groth16_proof = CircuitGroth16WithTotalWork::new(seal, total_work);
206    groth16_proof.verify(image_id)
207}

209/// Verifies watchtower challenge transactions and collects their outputs.
210///
211/// This function performs validation on a set of watchtower challenge transactions
212/// and their associated inputs, witnesses, and public keys. It checks that:
213/// - Each challenge input corresponds to the correct `kickoff_tx` output (P2TR),
214/// - The signature is valid under the Taproot sighash rules,
215/// - The public key matches the one registered for the watchtower,
216/// - And, if all checks pass, it marks the corresponding bit in a 20-byte bitmap
217///   (`challenge_sending_watchtowers`) and collects the first 3 outputs of the
218///   watchtower transaction into `watchtower_challenges_outputs`.
219///
220///   Note: This function only verifies keypath spends.
221///
222/// # Parameters
223/// - `circuit_input`: Data structure holding serialized watchtower transactions, UTXOs, input indices, and pubkeys.
224/// - `kickoff_txid`: The transaction ID of the `kickoff_tx`.
225///
226/// # Returns
227/// A tuple containing:
228/// - A 20-byte bitmap indicating which watchtower challenges were valid,
229/// - A vector of the first 3 outputs from each valid watchtower transaction.
230///
231/// # Notes
232/// Invalid or malformed challenge data (e.g., decoding errors, invalid signatures)
233/// will be skipped gracefully without causing the function to panic.
234pub fn verify_watchtower_challenges(circuit_input: &BridgeCircuitInput) -> WatchtowerChallengeSet {
235    let mut challenge_sending_watchtowers: [u8; 20] = [0u8; 20];
236    let mut watchtower_challenges_outputs: Vec<[TxOut; 3]> = vec![];
237
238    if circuit_input.watchtower_inputs.len() > NUMBER_OF_WATCHTOWERS {
239        panic!("Invalid number of watchtower challenge transactions");
240    }
241
242    for watchtower_input in circuit_input.watchtower_inputs.iter() {
243        let inner_txouts: Vec<TxOut> = watchtower_input
244            .watchtower_challenge_utxos
245            .iter()
246            .map(|utxo| utxo.0.clone()) // TODO: Get rid of this clone if possible
247            .collect::<Vec<TxOut>>();
248
249        let prevouts = Prevouts::All(&inner_txouts);
250
251        let watchtower_input_idx = watchtower_input.watchtower_challenge_input_idx as usize;
252
253        if watchtower_input_idx >= watchtower_input.watchtower_challenge_tx.input.len() {
254            panic!(
255                "Invalid watchtower challenge input index, watchtower index: {}",
256                watchtower_input.watchtower_idx
257            );
258        }
259
260        let input = watchtower_input.watchtower_challenge_tx.input[watchtower_input_idx].clone();
261
262        let (sighash_type, sig_bytes): (TapSighashType, [u8; 64]) = {
263            let signature = watchtower_input.watchtower_challenge_witness.0.to_vec()[0].clone();
264
265            if signature.len() == 64 {
266                (
267                    TapSighashType::Default,
268                    signature[0..64].try_into().expect("Cannot fail"),
269                )
270            } else if signature.len() == 65 {
271                match TapSighashType::from_consensus_u8(signature[64]) {
272                    Ok(sighash_type) => (
273                        sighash_type,
274                        signature[0..64].try_into().expect("Cannot fail"),
275                    ),
276                    Err(_) => panic!(
277                        "Invalid sighash type, watchtower index: {}",
278                        watchtower_input.watchtower_idx
279                    ),
280                }
281            } else {
282                panic!(
283                    "Invalid witness length, expected 64 or 65 bytes, watchtower index: {}",
284                    watchtower_input.watchtower_idx
285                );
286            }
287        };
288
289        let Ok(sighash) = sighash(
290            &watchtower_input.watchtower_challenge_tx,
291            &prevouts,
292            watchtower_input_idx,
293            sighash_type,
294        ) else {
295            panic!(
296                "Sighash could not be computed, watchtower index: {}",
297                watchtower_input.watchtower_idx
298            );
299        };
300
301        if input.previous_output.txid != *circuit_input.kickoff_tx_id {
302            panic!(
303                "Invalid input: expected input to reference an output from the kickoff transaction (txid: {}), but got txid: {}, vout: {}, watchtower index: {}",
304                *circuit_input.kickoff_tx_id,
305                input.previous_output.txid,
306                input.previous_output.vout,
307                watchtower_input.watchtower_idx
308            );
309        };
310
311        if watchtower_input_idx >= inner_txouts.len() {
312            panic!(
313                "Invalid watchtower challenge input index, watchtower index: {}",
314                watchtower_input.watchtower_idx
315            );
316        }
317
318        let output = inner_txouts[watchtower_input_idx].clone();
319
320        let script_pubkey = output.script_pubkey.clone();
321
322        if !script_pubkey.is_p2tr() {
323            panic!(
324                "Invalid output script type - kickoff, watchtower index: {}",
325                watchtower_input.watchtower_idx
326            );
327        };
328
329        if watchtower_input.watchtower_idx as usize
330            >= circuit_input.all_tweaked_watchtower_pubkeys.len()
331        {
332            panic!(
333                "Invalid watchtower index, watchtower index: {}, number of watchtowers: {}",
334                watchtower_input.watchtower_idx,
335                circuit_input.all_tweaked_watchtower_pubkeys.len()
336            );
337        }
338
339        let pubkey: [u8; 32] = script_pubkey.as_bytes()[2..34]
340            .try_into()
341            .expect("Cannot fail");
342
343        if circuit_input.all_tweaked_watchtower_pubkeys[watchtower_input.watchtower_idx as usize]
344            != pubkey
345        {
346            panic!(
347                "Invalid watchtower public key, watchtower index: {}",
348                watchtower_input.watchtower_idx
349            );
350        }
351
352        let vout = watchtower_input
353            .watchtower_idx
354            .checked_mul(2)
355            .and_then(|x| x.checked_add(circuit_input.watchtower_challenge_connector_start_idx))
356            .map(u32::from)
357            .expect("Overflow occurred while calculating vout");
358
359        if vout != input.previous_output.vout {
360            panic!(
361                "Invalid output index, watchtower index: {}",
362                watchtower_input.watchtower_idx
363            );
364        }
365
366        let Ok(verifying_key) = VerifyingKey::from_bytes(&pubkey) else {
367            panic!(
368                "Invalid verifying key, watchtower index: {}",
369                watchtower_input.watchtower_idx
370            );
371        };
372
373        let Ok(signature) = Signature::try_from(sig_bytes.as_slice()) else {
374            panic!(
375                "Invalid signature, watchtower index: {}",
376                watchtower_input.watchtower_idx
377            );
378        };
379
380        if verifying_key
381            .verify_prehash(sighash.as_byte_array(), &signature)
382            .is_ok()
383        {
384            // TODO: CHECK IF THIS IS CORRECT
385            challenge_sending_watchtowers[(watchtower_input.watchtower_idx as usize) / 8] |=
386                1 << (watchtower_input.watchtower_idx % 8);
387            if watchtower_input.watchtower_challenge_tx.output.len() >= 3 {
388                watchtower_challenges_outputs.push([
389                    watchtower_input.watchtower_challenge_tx.output[0].clone(),
390                    watchtower_input.watchtower_challenge_tx.output[1].clone(),
391                    watchtower_input.watchtower_challenge_tx.output[2].clone(),
392                ]);
393            }
394        }
395    }
396
397    WatchtowerChallengeSet {
398        challenge_senders: challenge_sending_watchtowers,
399        challenge_outputs: watchtower_challenges_outputs,
400    }
401}

403/// Computes the maximum verified total work and watchtower challenge flags from challenge transactions.
404///
405/// # Parameters
406///
407/// - `kickoff_txid`: The transaction ID of the kickoff transaction.
408/// - `circuit_input`: The `BridgeCircuitInput` containing all watchtower inputs and related data.
409/// - `work_only_image_id`: A 32-byte identifier used for Groth16 verification against the work-only circuit.
410///
411/// # Returns
412///
413/// A tuple containing:
414/// - `[u8; 16]`: The total work from the highest valid watchtower challenge (after successful Groth16 verification).
415/// - `[u8; 20]`: Bitflags representing which watchtowers sent valid challenges (1 bit per watchtower).
416///
417/// # Notes
418///
419/// - Skips over any challenge with invalid encoding, invalid signature, or improper structure.
420/// - Each watchtower challenge is expected to contain exactly 3 outputs:
421///     - First two should be P2TR outputs containing the compressed Groth16 proof parts.
422///     - Third must be an OP_RETURN containing the rest of the proof and the total work value.
423/// - The function sorts valid commitments by total work and verifies the highest one using a Groth16 verifier.
424pub fn total_work_and_watchtower_flags(
425    circuit_input: &BridgeCircuitInput,
426    work_only_image_id: &[u8; 32],
427) -> (TotalWork, ChallengeSendingWatchtowers) {
428    let watchtower_challenge_set = verify_watchtower_challenges(circuit_input);
429
430    let mut valid_watchtower_challenge_commitments: Vec<WatchTowerChallengeTxCommitment> = vec![];
431
432    for outputs in watchtower_challenge_set.challenge_outputs {
433        if !outputs[0].script_pubkey.is_p2tr()
434            || !outputs[1].script_pubkey.is_p2tr()
435            || !outputs[2].script_pubkey.is_op_return()
436        {
437            continue;
438        }
439
440        let first_output: [u8; 32] = outputs[0].script_pubkey.to_bytes()[2..]
441            .try_into()
442            .expect("Cannot fail");
443        let second_output: [u8; 32] = outputs[1].script_pubkey.to_bytes()[2..]
444            .try_into()
445            .expect("Cannot fail");
446
447        let Some(Ok(third_output)) =
448            parse_op_return_data(&outputs[2].script_pubkey).map(TryInto::<[u8; 80]>::try_into)
449        else {
450            continue;
451        };
452
453        let compressed_g16_proof: [u8; 128] = [&first_output, &second_output, &third_output[0..64]]
454            .concat()
455            .try_into()
456            .expect("Cannot fail");
457
458        // Borsh deserialization of the final 16 bytes is functionally redundant in this context,
459        // as it does not alter the byte content. It is retained here for consistency and defensive safety.
460        let total_work: [u8; 16] = borsh::from_slice(&third_output[64..]).expect("Cannot fail");
461
462        let commitment = WatchTowerChallengeTxCommitment {
463            compressed_g16_proof,
464            total_work,
465        };
466
467        valid_watchtower_challenge_commitments.push(commitment);
468    }
469
470    // TODO: UPDATE THIS PART ACCORDING TO ENDIANNESS
471    valid_watchtower_challenge_commitments.sort_by(|a, b| b.total_work.cmp(&a.total_work));
472
473    let mut total_work = [0u8; 16];
474
475    for commitment in valid_watchtower_challenge_commitments {
476        // Grooth16 verification of work only circuit
477        if convert_to_groth16_and_verify(
478            &commitment.compressed_g16_proof,
479            commitment.total_work,
480            work_only_image_id,
481        ) {
482            total_work = commitment.total_work;
483            break;
484        }
485    }
486
487    (
488        TotalWork(total_work),
489        ChallengeSendingWatchtowers(watchtower_challenge_set.challenge_senders),
490    )
491}

493fn parse_op_return_data(script: &Script) -> Option<Vec<u8>> {
494    let mut instructions = script.instructions();
495    if let Some(Ok(Instruction::Op(opcodes::all::OP_RETURN))) = instructions.next() {
496        if let Some(Ok(Instruction::PushBytes(data))) = instructions.next() {
497            return Some(data.as_bytes().to_vec());
498        }
499    }
500    None
501}

503/// Computes a deposit constant hash using transaction output data, kickoff transaction ID,
504/// tweaked watchtower public keys, and a move transaction ID.
505///
506/// # Parameters
507///
508/// - `last_output`: A reference to the last transaction output (`TxOut`).
509/// - `kickoff_txid`: A reference to the kickoff transaction ID (`Txid`).
510/// - `watchtower_pubkeys`: A slice of 32-byte arrays representing tweaked watchtower public keys.
511/// - `move_txid_hex`: A 32-byte array representing the move transaction ID.
512///
513/// # Returns
514///
515/// A 32-byte array (`[u8; 32]`) representing the SHA-256 hash of the concatenated input components.
516///
517/// # Panics
518///
519/// - If the `script_pubkey` of `last_output` does not start with `OP_RETURN` (`0x6a`).
520/// - If the length of the operator ID (extracted from `script_pubkey`) exceeds 32 bytes.
521pub fn deposit_constant(
522    last_output: &TxOut,
523    kickoff_txid: &Txid,
524    watchtower_challenge_connector_start_idx: u16,
525    watchtower_pubkeys: &[[u8; 32]],
526    move_txid_hex: [u8; 32],
527) -> DepositConstant {
528    let last_output_script = last_output.script_pubkey.to_bytes();
529
530    // OP_RETURN check
531    assert!(last_output_script[0] == 0x6a);
532
533    if last_output_script.len() < 3 {
534        panic!("OP_RETURN script too short");
535    }
536
537    let len: usize = last_output_script[1] as usize;
538
539    if len > 32 {
540        panic!("Invalid operator id length");
541    }
542
543    let mut operator_id = [0u8; 32];
544    operator_id[..len].copy_from_slice(&last_output_script[2..2 + len]);
545
546    // pubkeys are 32 bytes long
547    let pubkey_concat = watchtower_pubkeys
548        .iter()
549        .flat_map(|pubkey| pubkey.to_vec())
550        .collect::<Vec<u8>>();
551
552    let watchtower_pubkeys_digest: [u8; 32] = Sha256::digest(&pubkey_concat).into();
553
554    let pre_deposit_constant = [
555        &kickoff_txid.as_byte_array()[..],
556        &move_txid_hex,
557        &watchtower_pubkeys_digest,
558        &operator_id,
559        &watchtower_challenge_connector_start_idx.to_be_bytes()[..],
560    ]
561    .concat();
562
563    DepositConstant(Sha256::digest(&pre_deposit_constant).into())
564}

566pub fn journal_hash(
567    payout_tx_blockhash: PayoutTxBlockhash,
568    latest_blockhash: LatestBlockhash,
569    challenge_sending_watchtowers: ChallengeSendingWatchtowers,
570    deposit_constant: DepositConstant,
571) -> blake3::Hash {
572    let concatenated_data = [
573        payout_tx_blockhash.0,
574        latest_blockhash.0,
575        challenge_sending_watchtowers.0,
576    ]
577    .concat();
578
579    let binding = blake3::hash(&concatenated_data);
580    let hash_bytes = binding.as_bytes();
581
582    let concat_journal = [deposit_constant.0, *hash_bytes].concat();
583
584    blake3::hash(&concat_journal)
585}

633    fn total_work_and_watchtower_flags_setup() -> (BridgeCircuitInput, Txid) {
634        let wt_tx_bytes = include_bytes!("../../test_data/wt_raw_tx.bin");
635        let kickoff_raw_tx_bytes = include_bytes!("../../test_data/kickoff_raw_tx.bin");
636        let pubkey_hex = "412c00124e48ab8b082a5fa3ee742eb763387ef67adb9f0d5405656ff12ffd50";
637
638        let mut wt_tx: Transaction =
639            Decodable::consensus_decode(&mut Cursor::new(&wt_tx_bytes)).unwrap();
640
641        let witness = wt_tx.input[0].witness.clone();
642
643        wt_tx.input[0].witness.clear();
644
645        let kickoff_tx: Transaction =
646            Decodable::consensus_decode(&mut Cursor::new(&kickoff_raw_tx_bytes))
647                .expect("Failed to decode kickoff tx");
648
649        let kickoff_txid = kickoff_tx.compute_txid();
650
651        let output = kickoff_tx.output[wt_tx.input[0].previous_output.vout as usize].clone();
652
653        // READ FROM THE FILE TO PREVENT THE ISSUE WITH ELF - IMAGE ID UPDATE CYCLE
654        let mut encoded_tx_out = vec![];
655        let _ = Encodable::consensus_encode(&output, &mut encoded_tx_out);
656
657        let tx_out = Decodable::consensus_decode(&mut Cursor::new(&encoded_tx_out))
658            .expect("Failed to decode kickoff tx");
659
660        let mut watchtower_pubkeys = vec![[0u8; 32]; 160];
661
662        let operator_idx: u16 = 6;
663
664        let pubkey = hex::decode(pubkey_hex).unwrap();
665
666        watchtower_pubkeys[operator_idx as usize] =
667            pubkey.try_into().expect("Pubkey must be 32 bytes");
668
669        let watchtower_challenge_connector_start_idx: u16 =
670            (FIRST_FIVE_OUTPUTS + NUMBER_OF_ASSERT_TXS) as u16;
671
672        let input = BridgeCircuitInput {
673            kickoff_tx_id: CircuitTxid::from(kickoff_tx.compute_txid()),
674            watchtower_inputs: vec![WatchtowerInput {
675                watchtower_idx: operator_idx,
676                watchtower_challenge_witness: CircuitWitness(witness),
677                watchtower_challenge_input_idx: 0,
678                watchtower_challenge_utxos: vec![CircuitTxOut(tx_out)],
679                watchtower_challenge_tx: CircuitTransaction(wt_tx.clone()),
680            }],
681            hcp: BlockHeaderCircuitOutput {
682                method_id: [0; 8],
683                chain_state: ChainState::new(),
684            },
685            payout_spv: SPV {
686                transaction: CircuitTransaction(wt_tx),
687                block_inclusion_proof: BlockInclusionProof::new(0, vec![]),
688                block_header: CircuitBlockHeader {
689                    version: 0,
690                    prev_block_hash: [0u8; 32],
691                    merkle_root: [0u8; 32],
692                    time: 0,
693                    bits: 0,
694                    nonce: 0,
695                },
696                mmr_inclusion_proof: MMRInclusionProof {
697                    subroot_idx: 0,
698                    internal_idx: 0,
699                    inclusion_proof: vec![],
700                },
701            },
702            lcp: LightClientProof::default(),
703            sp: StorageProof::default(),
704            all_tweaked_watchtower_pubkeys: watchtower_pubkeys,
705            watchtower_challenge_connector_start_idx,
706            payout_input_index: 0,
707        };
708
709        (input, kickoff_txid)
710    }
