1use ark_bn254::Bn254;

5type G1 = ark_bn254::G1Affine;
6type G2 = ark_bn254::G2Affine;

8#[derive(Copy, Clone, Debug)]
9pub struct CircuitGroth16Proof {
10    a: G1,
11    b: G2,
12    c: G1,
13}
14
15impl CircuitGroth16Proof {
16    pub fn new(a: G1, b: G2, c: G1) -> CircuitGroth16Proof {
17        CircuitGroth16Proof { a, b, c }
18    }
19
20    pub fn from_seal(seal: &[u8; 256]) -> CircuitGroth16Proof {
21        let a = G1::new(
22            ark_bn254::Fq::from_be_bytes_mod_order(&seal[0..32]),
23            ark_bn254::Fq::from_be_bytes_mod_order(&seal[32..64]),
24        );
25
26        let b = G2::new(
27            ark_bn254::Fq2::from_base_prime_field_elems([
28                ark_bn254::Fq::from_be_bytes_mod_order(&seal[96..128]),
29                ark_bn254::Fq::from_be_bytes_mod_order(&seal[64..96]),
30            ])
31            .unwrap(),
32            ark_bn254::Fq2::from_base_prime_field_elems([
33                ark_bn254::Fq::from_be_bytes_mod_order(&seal[160..192]),
34                ark_bn254::Fq::from_be_bytes_mod_order(&seal[128..160]),
35            ])
36            .unwrap(),
37        );
38
39        let c = G1::new(
40            ark_bn254::Fq::from_be_bytes_mod_order(&seal[192..224]),
41            ark_bn254::Fq::from_be_bytes_mod_order(&seal[224..256]),
42        );
43
44        CircuitGroth16Proof::new(a, b, c)
45    }
46
47    pub fn from_compressed(
48        compressed: &[u8; 128],
49    ) -> Result<CircuitGroth16Proof, SerializationError> {
50        let a_compressed = &compressed[0..32];
51        let b_compressed = &compressed[32..96];
52        let c_compressed = &compressed[96..128];
53        let a = ark_bn254::G1Affine::deserialize_compressed(a_compressed)?;
54        let b = ark_bn254::G2Affine::deserialize_compressed(b_compressed)?;
55        let c = ark_bn254::G1Affine::deserialize_compressed(c_compressed)?;
56
57        Ok(CircuitGroth16Proof::new(a, b, c))
58    }
59
60    pub fn to_compressed(&self) -> Result<[u8; 128], SerializationError> {
61        let mut a_compressed = [0u8; 32];
62        let mut b_compressed = [0u8; 64];
63        let mut c_compressed = [0u8; 32];
64
65        ark_bn254::G1Affine::serialize_with_mode(&self.a, &mut a_compressed[..], Compress::Yes)
66            .expect("Serialization should not fail for valid curve points");
67        ark_bn254::G2Affine::serialize_with_mode(&self.b, &mut b_compressed[..], Compress::Yes)
68            .expect("Serialization should not fail for valid curve points");
69        ark_bn254::G1Affine::serialize_with_mode(&self.c, &mut c_compressed[..], Compress::Yes)
70            .expect("Serialization should not fail for valid curve points");
71
72        let mut compressed = [0u8; 128];
73        compressed[0..32].copy_from_slice(&a_compressed);
74        compressed[32..96].copy_from_slice(&b_compressed);
75        compressed[96..128].copy_from_slice(&c_compressed);
76
77        Ok(compressed)
78    }
79
80    pub fn a(&self) -> &G1 {
81        &self.a
82    }
83
84    pub fn b(&self) -> &G2 {
85        &self.b
86    }
87
88    pub fn c(&self) -> &G1 {
89        &self.c
90    }
91}

128        assert_eq!(proof.c(), &c);

131    #[test]
132    fn test_to_compressed_and_from_compressed() {
133        for _ in 0..16 {
134            let proof = CircuitGroth16Proof::new(random_g1(), random_g2(), random_g1());
135
136            let compressed = proof.to_compressed().expect("Compression failed");
137            let decompressed_proof =
138                CircuitGroth16Proof::from_compressed(&compressed).expect("Decompression failed");
139
140            assert_eq!(proof.a(), decompressed_proof.a());
141            assert_eq!(proof.b(), decompressed_proof.b());
142            assert_eq!(proof.c(), decompressed_proof.c());
143        }
144    }
