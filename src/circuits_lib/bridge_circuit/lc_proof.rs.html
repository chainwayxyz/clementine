1//! # Light Client Proof Verifier
2//! This module implements the light client proof verifier for the bridge circuit.
3//! It includes functions to verify light client proofs and extracting the light client circuit output.
4
5use super::{
6    constants::{
7        DEVNET_LC_IMAGE_ID, MAINNET_LC_IMAGE_ID, REGTEST_LC_IMAGE_ID, TESTNET4_LC_IMAGE_ID,
8    },
9    structs::LightClientProof,
10};
11use citrea_sov_rollup_interface::zk::light_client_proof::output::LightClientCircuitOutput;
12use risc0_zkvm::guest::env;
13
14pub const LC_IMAGE_ID: [u8; 32] = {
15    match option_env!("BITCOIN_NETWORK") {
16        Some(network) if matches!(network.as_bytes(), b"regtest") => REGTEST_LC_IMAGE_ID,
17        Some(network) if matches!(network.as_bytes(), b"signet") => DEVNET_LC_IMAGE_ID,
18        Some(network) if matches!(network.as_bytes(), b"testnet4") => TESTNET4_LC_IMAGE_ID,
19        Some(network) if matches!(network.as_bytes(), b"mainnet") => MAINNET_LC_IMAGE_ID,
20        None => MAINNET_LC_IMAGE_ID,
21        _ => panic!("Unsupported BITCOIN_NETWORK environment variable"),
22    }
23};
24
25#[inline]
26fn is_regtest() -> bool {
27    option_env!("BITCOIN_NETWORK") == Some("regtest")
28}
29
30/// Deserializes the light client circuit output from journal bytes.
31fn deserialize_circuit_output(journal: &[u8]) -> LightClientCircuitOutput {
32    borsh::from_slice(journal).expect("Failed to deserialize light client circuit output")
33}
34
35/// Verifies the light client proof and returns the light client circuit output.
36pub fn lc_proof_verifier(light_client_proof: LightClientProof) -> LightClientCircuitOutput {
37    if !is_regtest() {
38        env::verify(LC_IMAGE_ID, &light_client_proof.lc_journal)
39            .expect("Failed to verify light client proof");
40    }
41
42    let light_client_circuit_output = deserialize_circuit_output(&light_client_proof.lc_journal);
43
44    if !is_regtest() {
45        assert!(
46            check_method_id(&light_client_circuit_output, LC_IMAGE_ID),
47            "Light client proof method ID does not match the expected LC image ID"
48        );
49    }
50
51    light_client_circuit_output
52}
53
54pub fn check_method_id(
55    light_client_circuit_output: &LightClientCircuitOutput,
56    lc_image_id_circuit: [u8; 32],
57) -> bool {
58    let light_client_method_id_bytes: [u8; 32] = light_client_circuit_output
59        .light_client_proof_method_id
60        .iter()
61        .flat_map(|&x| x.to_le_bytes())
62        .collect::<Vec<u8>>()
63        .try_into()
64        .expect("Conversion from [u32; 8] to [u8; 32] cannot fail");
65
66    light_client_method_id_bytes == lc_image_id_circuit
67}

86                .expect("Failed to deserialize light client circuit output");

95        lcp_receipt.verify(REGTEST_LC_IMAGE_ID).unwrap();

97        let light_client_proof: LightClientProof = LightClientProof {
98            lc_journal: lcp_receipt.journal.bytes.to_vec(),
99        };
100
101        let light_client_circuit_output: LightClientCircuitOutput =
102            borsh::from_slice(light_client_proof.lc_journal.as_slice())
103                .expect("Failed to deserialize light client circuit output");

105        assert!(
106            check_method_id(&light_client_circuit_output, REGTEST_LC_IMAGE_ID),
