9pub fn lc_proof_verifier(light_client_proof: LightClientProof) -> LightClientCircuitOutput {
10    // env::verify(LC_IMAGE_ID, &light_client_proof.lc_journal).unwrap();
11
12    if light_client_proof.lc_journal.len() < 32 {
13        panic!("Invalid light client journal");
14    }
15
16    let light_client_circuit_output: LightClientCircuitOutput =
17        borsh::from_slice(light_client_proof.lc_journal.as_slice())
18            .expect("Failed to deserialize light client circuit output");
19
20    light_client_circuit_output
21}
