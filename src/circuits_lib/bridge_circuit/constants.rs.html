1//! # Bridge Circuit Constants
2//!
3//! This module contains constants used in the bridge circuit, including method IDs for different networks,
4//! Groth16 related constants, and prepared verification keys. These constants are essential for the operation
5//! of the bridge circuit and are used in various cryptographic operations.
6//! ## Work-Only Circuit Method IDs
7//! The method IDs for different networks are used to identify the specific work-only circuits.
8//! They are used for verifying the total work done on a Bitcoin blockchain for a given Watchtower challenge.
9//! ## Groth16 Related Constants
10//! These constants are used in the Groth16 proof verification process.
11//! They include the post state, input, assumptions, claim tag, and output tag.
12//! They are used to recover all five public outputs of the Groth16 proof when Risc0 pipeline is used
13//! for generating the proof.
14//! ## Verification Keys
15//! The prepared verification keys are used to verify the Groth16 proofs. They are included in
16//! the binary format. The `get_prepared_vk` function can be used to retrieve the appropriate
17//! verification key according to the feature flags.

92mod tests {
93    use ark_bn254::Fr;
94    use ark_ff::PrimeField;
95    use risc0_zkvm::Digest;

97    use crate::bridge_circuit::constants::{A0_BIGINT, A1_BIGINT};
98
99    // This test checks that the A0 and A1 constants match the expected values derived from the control root
100    // of the Groth16 verifier parameters. If they do not match, it indicates that the constants need to be updated
101    // in the `constants.rs` file. This is important because the A0 and A1 constants are used in the bridge circuit to verify the Groth16
102    // proof, and any mismatch could lead to incorrect verification results.
103    #[test]
104    fn test_a0_and_a1() {
105        let verifier_context = risc0_zkvm::VerifierContext::default();
106        let params = verifier_context
107            .groth16_verifier_parameters
108            .as_ref()
109            .unwrap();
110        let (a0, a1) = split_digest(params.control_root);
111
112        let a0_bigint = a0.into_bigint();
113        let a1_bigint = a1.into_bigint();
114
115        assert_eq!((a0_bigint, a1_bigint), (A0_BIGINT, A1_BIGINT),
116            "A0 and A1 do not match the expected values, please update the a0 and a1 constants in constants.rs. a0: {:?}, a1: {:?}",
117            a0_bigint.0, a1_bigint.0);
118    }
119
120    // This is the exact same implementation as in risc0_groth16, but we need to re-implement it here to change
121    // the return type. Please check the original implementation each time risc0 version is updated.
122    fn split_digest(d: Digest) -> (Fr, Fr) {
123        let big_endian: Vec<u8> = d.as_bytes().to_vec().iter().rev().cloned().collect();
124        let middle = big_endian.len() / 2;
125        let (b, a) = big_endian.split_at(middle);
126        (
127            Fr::from_be_bytes_mod_order(&from_u256_hex(&hex::encode(a))),
128            Fr::from_be_bytes_mod_order(&from_u256_hex(&hex::encode(b))),
129        )
130    }
131
132    fn from_u256_hex(value: &str) -> Vec<u8> {
133        to_fixed_array(hex::decode(value).unwrap()).to_vec()
134    }
135
136    fn to_fixed_array(input: Vec<u8>) -> [u8; 32] {
137        let mut fixed_array = [0u8; 32];
138        let start = core::cmp::max(32, input.len()) - core::cmp::min(32, input.len());
139        fixed_array[start..].copy_from_slice(&input[input.len().saturating_sub(32)..]);
140        fixed_array
141    }
