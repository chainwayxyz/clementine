1//! # Common Module
2//! This module contains common constants and utility functions used across the Risc0 circuits in the Clementine protocol.
3//! It includes definitions for constants related to the bridge amount, number of outputs, transaction assertions, and watchtowers.
4//! It also provides hashing functions for cryptographic operations, such as double SHA256 and single SHA256 hashing,
5//! as well as a utility function to hash two nodes together.
6//! The `ZkvmGuest` and `ZkvmHost` traits define the interface for zkVM guest and host interactions,
7//! allowing for reading from and writing to the host, committing data, and verifying proofs.
8
9pub mod constants;
10pub mod hashes;
11pub mod zkvm;
12
13pub const NETWORK_TYPE: &str = {
14    #[cfg(test)]
15    {
16        "testnet4"
17    }
18    #[cfg(not(test))]
19    {
20        match option_env!("BITCOIN_NETWORK") {
21            Some(network) if matches!(network.as_bytes(), b"mainnet") => "mainnet",
22            Some(network) if matches!(network.as_bytes(), b"testnet4") => "testnet4",
23            Some(network) if matches!(network.as_bytes(), b"signet") => "signet",
24            Some(network) if matches!(network.as_bytes(), b"regtest") => "regtest",
25            None => "mainnet",
26            _ => panic!("Invalid network type"),
27        }
28    }
29};
30
31pub const fn get_network() -> &'static str {
32    NETWORK_TYPE
