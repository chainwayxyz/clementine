28#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
29#[serde(rename_all = "lowercase")]
30/// A pre-defined paramset name that can be converted into a
31/// [`ProtocolParamset`] reference.
32///
33/// See: [`REGTEST_PARAMSET`]
34pub enum ProtocolParamsetName {
35    // Pre-defined paramsets
36    Regtest,
37}
38
39impl FromStr for ProtocolParamsetName {
40    type Err = BridgeError;
41
42    fn from_str(s: &str) -> Result<Self, Self::Err> {
43        match s {
44            "regtest" => Ok(ProtocolParamsetName::Regtest),
45            _ => Err(BridgeError::ConfigError(format!(
46                "Unknown paramset name: {s}"
47            ))),
48        }
49    }
50}
51
52impl Display for ProtocolParamsetName {
53    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
54        match self {
55            ProtocolParamsetName::Regtest => write!(f, "regtest"),
56        }
57    }
58}
59
60impl From<ProtocolParamsetName> for &'static ProtocolParamset {
61    fn from(name: ProtocolParamsetName) -> Self {
62        match name {
63            ProtocolParamsetName::Regtest => &REGTEST_PARAMSET,
64        }
65    }
66}

68#[derive(Debug, Clone, PartialEq, Eq, Hash, Deserialize, Serialize)]
69/// Protocol parameters that affect the transactions in the contract (which also
70/// change the pre-calculated txids and sighashes).
71///
72/// These parameters are used when generating the transactions and changing them
73/// will break compatibility between actors, making deposits impossible.  A
74/// paramset is chosen by the actor by choosing a ParamsetName inside the
75/// [`BridgeConfig`].
76pub struct ProtocolParamset {
77    /// Bitcoin network to work on (mainnet, testnet, regtest).
78    pub network: Network,
79    /// Number of round transactions that the operator will create.
80    pub num_round_txs: usize,
81    /// Number of kickoff UTXOs per round transaction.
82    pub num_kickoffs_per_round: usize,
83    /// Number of kickoffs that are signed per round and deposit.
84    /// There are num_kickoffs_per_round utxo's, but only num_signed_kickoffs are signed.
85    pub num_signed_kickoffs: usize,
86    /// Bridge deposit amount that users can deposit.
87    pub bridge_amount: Amount,
88    /// Amount allocated for each kickoff UTXO.
89    pub kickoff_amount: Amount,
90    /// Amount allocated for operator challenge transactions.
91    pub operator_challenge_amount: Amount,
92    /// Collateral funding amount for operators used to fund the round transaction chain.
93    pub collateral_funding_amount: Amount,
94    /// Length of the blockhash commitment in kickoff transactions.
95    pub kickoff_blockhash_commit_length: u32,
96    /// Total number of bytes of a watchtower challenge.
97    pub watchtower_challenge_bytes: usize,
98    /// Winternitz derivation log_d (shared for all WOTS commitments)
99    /// Currently used in statics and thus cannot be different from [`WINTERNITZ_LOG_D`].
100    pub winternitz_log_d: u32,
101    /// Number of blocks after which user can take deposit back if deposit request fails.
102    pub user_takes_after: u16,
103    /// Number of blocks for operator challenge timeout timelock (currently BLOCKS_PER_WEEK)
104    pub operator_challenge_timeout_timelock: u16,
105    /// Number of blocks for operator challenge NACK timelock (currently BLOCKS_PER_WEEK * 3)
106    pub operator_challenge_nack_timelock: u16,
107    /// Number of blocks for disprove timeout timelock (currently BLOCKS_PER_WEEK * 5)
108    pub disprove_timeout_timelock: u16,
109    /// Number of blocks for assert timeout timelock (currently BLOCKS_PER_WEEK * 4)
110    pub assert_timeout_timelock: u16,
111    /// Number of blocks for latest blockhash timeout timelock (currently BLOCKS_PER_WEEK * 2.5)
112    pub latest_blockhash_timeout_timelock: u16,
113    /// Number of blocks for operator reimburse timelock (currently BLOCKS_PER_DAY * 2)
114    /// Timelocks operator from sending the next Round Tx after the Ready to Reimburse Tx.
115    pub operator_reimburse_timelock: u16,
116    /// Number of blocks for watchtower challenge timeout timelock (currently BLOCKS_PER_WEEK * 2)
117    pub watchtower_challenge_timeout_timelock: u16,
118    /// Amount of depth a block should have from the current head to be considered finalized
119    /// Also means finality_confirmations, how many confirmations are needed for a block to be considered finalized
120    /// The chain tip has 1 confirmation. Minimum value should be 1.
121    pub finality_depth: u32,
122    /// start height to sync the chain from, i.e. the height bridge was deployed
123    pub start_height: u32,
124    /// Genesis height to sync the header chain proofs from
125    pub genesis_height: u32,
126    /// Genesis chain state hash
127    pub genesis_chain_state_hash: [u8; 32],
128    /// Denotes if the bridge is non-standard, i.e. uses 0 sat outputs for round tx (except collateral) and kickoff outputs
129    pub bridge_nonstandard: bool,
130}
131
132impl ProtocolParamset {
133    /// Parse a `ProtocolParamset` from a TOML file.
134    pub fn from_toml_file(path: &Path) -> Result<Self, BridgeError> {
135        let contents = fs::read_to_string(path).wrap_err("Failed to read config file")?;
136
137        let paramset: Self = toml::from_str(&contents).wrap_err("Failed to parse TOML")?;
138        if paramset.finality_depth < 1 {
139            return Err(BridgeError::ConfigError(
140                "Finality depth must be at least 1".to_string(),
141            ));
142        }
143
144        Ok(paramset)
145    }
146
147    /// Returns the default UTXO amount based on whether the bridge is non-standard.
148    pub fn default_utxo_amount(&self) -> Amount {
149        if self.bridge_nonstandard {
150            Amount::from_sat(0)
151        } else {
152            MIN_TAPROOT_AMOUNT
153        }
154    }
155
156    /// Returns the anchor amount based on whether the bridge is non-standard.
157    pub fn anchor_amount(&self) -> Amount {
158        if self.bridge_nonstandard {
159            Amount::from_sat(0)
160        } else {
161            NON_EPHEMERAL_ANCHOR_AMOUNT
162        }
163    }
164
165    /// Checks if a block is finalized. In clementine and citrea, finality depth means the amount of confirmations needed for a block to be considered finalized.
166    /// The chain tip has 1 confirmation.
167    pub fn is_block_finalized(&self, block_height: u32, chain_tip_height: u32) -> bool {
168        if block_height > chain_tip_height {
169            return false;
170        }
171
172        chain_tip_height - block_height + 1 >= self.finality_depth
173    }
174
175    /// Returns true if this is a regtest network.
176    pub fn is_regtest(&self) -> bool {
177        self.network == Network::Regtest
178    }
179}
180
181impl Default for ProtocolParamset {
182    fn default() -> Self {
183        REGTEST_PARAMSET
184    }
185}
186
187impl Default for &'static ProtocolParamset {
188    fn default() -> Self {
189        &REGTEST_PARAMSET
190    }
191}
