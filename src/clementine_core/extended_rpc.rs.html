22/// Extended RPC client that provides additional functionality beyond the standard Bitcoin Core RPC.
23#[derive(Debug, Clone)]
24pub struct ExtendedRpc {
25    pub url: String,
26    auth: Auth,
27    pub client: Arc<Client>,
28}
29
30/// Errors that can occur during Bitcoin RPC operations.
31#[derive(Debug, thiserror::Error)]
32pub enum BitcoinRPCError {
33    #[error("Failed to bump fee for Txid of {0} and feerate of {1}")]
34    BumpFeeError(Txid, FeeRate),
35    #[error("Failed to bump fee: UTXO is already spent")]
36    BumpFeeUTXOSpent(OutPoint),
37    #[error("Transaction is already in block: {0}")]
38    TransactionAlreadyInBlock(BlockHash),
39
40    #[error(transparent)]
41    Other(#[from] eyre::Report),
42}
43
44type Result<T> = std::result::Result<T, BitcoinRPCError>;
45
46impl ExtendedRpc {
47    /// Connects to Bitcoin RPC and returns a new `ExtendedRpc`.
48    pub async fn connect(url: String, user: String, password: String) -> Result<Self> {
49        let auth = Auth::UserPass(user, password);
50
51        let rpc = Client::new(&url, auth.clone())
52            .await
53            .wrap_err("Failed to connect to Bitcoin RPC")?;
54
55        Ok(Self {
56            url,
57            auth,
58            client: Arc::new(rpc),
59        })
60    }

63    ///
64    /// # Arguments
65    /// * `txid` - The transaction ID to check

67    /// # Returns
68    /// The number of confirmations for the transaction, or an error if the transaction cannot be found.
69    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
70    pub async fn confirmation_blocks(&self, txid: &bitcoin::Txid) -> Result<u32> {
71        let raw_transaction_results = self
72            .client
73            .get_raw_transaction_info(txid, None)
74            .await
75            .wrap_err("Failed to get transaction info")?;
76
77        raw_transaction_results
78            .confirmations
79            .ok_or_eyre("No confirmation data")
80            .map_err(Into::into)
81    }

97    pub async fn get_blockhash_of_tx(&self, txid: &bitcoin::Txid) -> Result<bitcoin::BlockHash> {
98        let raw_transaction_results = self
99            .client

109    /// Retrieves the block header for a given block height.
110    ///
111    /// # Arguments
112    /// * `height` - The block height for which to retrieve the header.
113    ///
114    /// # Returns
115    /// A tuple containing the `bitcoin::BlockHash` and `bitcoin::block::Header`,
116    /// or an error if the block or header cannot be retrieved.
117    pub async fn get_block_header_by_height(
118        &self,
119        height: u64,
120    ) -> Result<(bitcoin::BlockHash, bitcoin::block::Header)> {
121        let block_hash = self.client.get_block_hash(height).await.wrap_err(format!(
122            "Couldn't retrieve block hash from height {} from rpc",
123            height
124        ))?;
125        let block_header = self
126            .client
127            .get_block_header(&block_hash)
128            .await
129            .wrap_err(format!(
130                "Couldn't retrieve block header with block hash {} from rpc",
131                block_hash
132            ))?;
133        Ok((block_hash, block_header))
134    }

136    /// Gets the transactions that created the inputs of a given transaction.
137    ///
138    /// # Arguments
139    /// * `tx` - The transaction to get the previous transactions for
140    ///
141    /// # Returns
142    /// A vector of transactions that created the inputs of the given transaction.
143    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
144    pub async fn get_prevout_txs(
145        &self,
146        tx: &bitcoin::Transaction,
147    ) -> Result<Vec<bitcoin::Transaction>> {
148        let mut prevout_txs = Vec::new();
149        for input in &tx.input {
150            let txid = input.previous_output.txid;
151            prevout_txs.push(self.get_tx_of_txid(&txid).await?);
152        }
153        Ok(prevout_txs)
154    }

183    /// Checks if a UTXO has the expected address and amount.
184    ///
185    /// # Arguments
186    /// * `outpoint` - The outpoint to check
187    /// * `address` - The expected script pubkey
188    /// * `amount_sats` - The expected amount in satoshis
189    ///
190    /// # Returns
191    /// `true` if the UTXO has the expected address and amount, `false` otherwise.
192    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
193    pub async fn check_utxo_address_and_amount(
194        &self,
195        outpoint: &OutPoint,
196        address: &ScriptBuf,
197        amount_sats: Amount,
198    ) -> Result<bool> {
199        let tx = self
200            .client
201            .get_raw_transaction(&outpoint.txid, None)
202            .await
203            .wrap_err("Failed to get transaction")?;
204
205        let current_output = tx.output[outpoint.vout as usize].clone();
206
207        let expected_output = TxOut {
208            script_pubkey: address.clone(),
209            value: amount_sats,
210        };
211
212        Ok(expected_output == current_output)
213    }

301    /// Bumps the fee of a transaction to meet or exceed a target fee rate. Does
302    /// nothing if the transaction is already confirmed. Returns the original
303    /// txid if no bump was needed.
304    ///
305    /// This function implements Replace-By-Fee (RBF) to increase the fee of an unconfirmed transaction.
306    /// It works as follows:
307    /// 1. If the transaction is already confirmed, returns Err(TransactionAlreadyInBlock)
308    /// 2. If the current fee rate is already >= the requested fee rate, returns the original txid
309    /// 3. Otherwise, increases the fee rate by adding the node's incremental fee to the current fee rate, then `bump_fee`s the transaction
310    ///
311    /// Note: This function currently only supports fee payer TXs.
312    ///
313    /// # Arguments
314    /// * `txid` - The transaction ID to bump
315    /// * `fee_rate` - The target fee rate to achieve
316    ///
317    /// # Returns
318    /// The txid of the bumped transaction (which may be the same as the input txid if no bump was needed).
319    ///
320    /// # Errors
321    /// * `TransactionAlreadyInBlock` - If the transaction is already confirmed
322    /// * `BumpFeeUTXOSpent` - If the UTXO being spent by the transaction is already spent
323    /// * `BumpFeeError` - For other errors with fee bumping
324    pub async fn bump_fee_with_fee_rate(&self, txid: Txid, fee_rate: FeeRate) -> Result<Txid> {
325        // Check if transaction is already confirmed
326        let transaction_info = self
327            .client
328            .get_transaction(&txid, None)
329            .await
330            .wrap_err("Failed to get transaction")?;
331        if transaction_info.info.blockhash.is_some() {
332            return Err(BitcoinRPCError::TransactionAlreadyInBlock(
333                transaction_info
334                    .info
335                    .blockhash
336                    .expect("Blockhash should be present"),
337            ));
338        }
339
340        // Calculate current fee rate
341        let tx = transaction_info
342            .transaction()
343            .wrap_err("Failed to get transaction")?;
344        let tx_size = tx.weight().to_vbytes_ceil();
345        let current_fee_sat = u64::try_from(
346            transaction_info
347                .fee
348                .expect("Fee should be present")
349                .to_sat()
350                .abs(),
351        )
352        .wrap_err("Failed to convert fee to sat")?;
353
354        let current_fee_rate = FeeRate::from_sat_per_kwu(1000 * current_fee_sat / tx_size);
355
356        // If current fee rate is already sufficient, return original txid
357        if current_fee_rate >= fee_rate {
358            return Ok(txid);
359        }
360
361        // Get node's incremental fee to determine how much to increase
362        let network_info = self
363            .client
364            .get_network_info()
365            .await
366            .wrap_err("Failed to get network info")?;
367        let incremental_fee = network_info.incremental_fee;
368        let incremental_fee_rate: FeeRate = FeeRate::from_sat_per_kwu(incremental_fee.to_sat());
369
370        // Calculate new fee rate by adding incremental fee to current fee rate
371        let new_fee_rate = FeeRate::from_sat_per_kwu(
372            current_fee_rate.to_sat_per_kwu() + incremental_fee_rate.to_sat_per_kwu(),
373        );
374
375        tracing::debug!(
376            "Bumping fee for txid: {txid} from {current_fee_rate} to {new_fee_rate} with incremental fee {incremental_fee_rate} - Final fee rate: {new_fee_rate}"
377        );
378
379        // Call Bitcoin Core's bumpfee RPC
380        let bump_fee_result = match self
381            .client
382            .bump_fee(
383                &txid,
384                Some(&bitcoincore_rpc::json::BumpFeeOptions {
385                    fee_rate: Some(bitcoincore_rpc::json::FeeRate::per_vbyte(Amount::from_sat(
386                        new_fee_rate.to_sat_per_vb_ceil(),
387                    ))),
388                    replaceable: Some(true),
389                    ..Default::default()
390                }),
391            )
392            .await
393        {
394            Ok(bump_fee_result) => bump_fee_result,
395            // Attempt to parse the error message to get the outpoint if the UTXO is already spent
396            Err(e) => match e {
397                bitcoincore_rpc::Error::JsonRpc(json_rpc_error) => match json_rpc_error {
398                    bitcoincore_rpc::RpcError::Rpc(rpc_error) => {
399                        if let Some((outpoint_str, _)) =
400                            rpc_error.message.split_once(" is already spent")
401                        {
402                            let outpoint = OutPoint::from_str(outpoint_str)
403                                .wrap_err(BitcoinRPCError::BumpFeeError(txid, fee_rate))?;
404
405                            return Err(BitcoinRPCError::BumpFeeUTXOSpent(outpoint));
406                        }
407
408                        return Err(eyre::eyre!(format!("{:?}", rpc_error))
409                            .wrap_err(BitcoinRPCError::BumpFeeError(txid, fee_rate))
410                            .into());
411                    }
412                    _ => {
413                        return Err(eyre::eyre!(json_rpc_error)
414                            .wrap_err(BitcoinRPCError::BumpFeeError(txid, fee_rate))
415                            .into());
416                    }
417                },
418                _ => {
419                    return Err(eyre::eyre!(e)
420                        .wrap_err(BitcoinRPCError::BumpFeeError(txid, fee_rate))
421                        .into())
422                }
423            },
424        };
425
426        // Return the new txid
427        Ok(bump_fee_result
428            .txid
429            .ok_or_eyre("Failed to get Txid from bump_fee_result")?)
430    }

432    /// Creates a new instance of the ExtendedRpc with a new client connection.
433    ///
434    /// This is useful when you need a separate connection to the Bitcoin RPC server.
435    ///
436    /// # Returns
437    /// A new `ExtendedRpc` instance on success, or an error if the connection fails.
438    pub async fn clone_inner(&self) -> std::result::Result<Self, bitcoincore_rpc::Error> {
439        let new_client = Client::new(&self.url, self.auth.clone()).await?;
440
441        Ok(Self {
442            url: self.url.clone(),
443            auth: self.auth.clone(),
444            client: Arc::new(new_client),
445        })
446    }

449        &self.url
