22/// Extended RPC client that provides additional functionality beyond the standard Bitcoin Core RPC.
23#[derive(Debug, Clone)]
24pub struct ExtendedRpc {
25    pub url: String,
26    auth: Auth,
27    pub client: Arc<Client>,
28}
29
30/// Errors that can occur during Bitcoin RPC operations.
31#[derive(Debug, thiserror::Error)]
32pub enum BitcoinRPCError {
33    #[error("Failed to bump fee for Txid of {0} and feerate of {1}")]
34    BumpFeeError(Txid, FeeRate),
35    #[error("Failed to bump fee: UTXO is already spent")]
36    BumpFeeUTXOSpent(OutPoint),
37    #[error("Transaction is already in block: {0}")]
38    TransactionAlreadyInBlock(BlockHash),
39
40    #[error(transparent)]
41    Other(#[from] eyre::Report),
42}
43
44type Result<T> = std::result::Result<T, BitcoinRPCError>;
45
46impl ExtendedRpc {
47    /// Connects to Bitcoin RPC and returns a new `ExtendedRpc`.
48    pub async fn connect(url: String, user: String, password: String) -> Result<Self> {
49        let auth = Auth::UserPass(user, password);
50
51        let rpc = Client::new(&url, auth.clone())
52            .await
53            .wrap_err("Failed to connect to Bitcoin RPC")?;
54
55        Ok(Self {
56            url,
57            auth,
58            client: Arc::new(rpc),
59        })
60    }

62    /// Returns the number of confirmations for a transaction.
63    ///
64    /// # Arguments
65    /// * `txid` - The transaction ID to check
66    ///
67    /// # Returns
68    /// The number of confirmations for the transaction, or an error if the transaction cannot be found.
69    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
70    pub async fn confirmation_blocks(&self, txid: &bitcoin::Txid) -> Result<u32> {
71        let raw_transaction_results = self
72            .client
73            .get_raw_transaction_info(txid, None)
74            .await
75            .wrap_err("Failed to get transaction info")?;
76
77        raw_transaction_results
78            .confirmations
79            .ok_or_eyre("No confirmation data")
80            .map_err(Into::into)
81    }

84    pub async fn get_blockhash_of_tx(&self, txid: &bitcoin::Txid) -> Result<bitcoin::BlockHash> {
85        let raw_transaction_results = self
86            .client
87            .get_raw_transaction_info(txid, None)
88            .await
89            .wrap_err("Failed to get transaction info")?;
90        let Some(blockhash) = raw_transaction_results.blockhash else {
91            return Err(eyre::eyre!("Transaction not confirmed: {0}", txid).into());
92        };
93        Ok(blockhash)

123    /// Checks if a UTXO has the expected address and amount.
124    ///
125    /// # Arguments
126    /// * `outpoint` - The outpoint to check
127    /// * `address` - The expected script pubkey
128    /// * `amount_sats` - The expected amount in satoshis
129    ///
130    /// # Returns
131    /// `true` if the UTXO has the expected address and amount, `false` otherwise.
132    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
133    pub async fn check_utxo_address_and_amount(
134        &self,
135        outpoint: &OutPoint,
136        address: &ScriptBuf,
137        amount_sats: Amount,
138    ) -> Result<bool> {
139        let tx = self
140            .client
141            .get_raw_transaction(&outpoint.txid, None)
142            .await
143            .wrap_err("Failed to get transaction")?;
144
145        let current_output = tx.output[outpoint.vout as usize].clone();
146
147        let expected_output = TxOut {
148            script_pubkey: address.clone(),
149            value: amount_sats,
150        };
151
152        Ok(expected_output == current_output)
153    }

216            .wrap_err("Failed to get transaction")?;

241    /// Bumps the fee of a transaction to meet or exceed a target fee rate. Does
242    /// nothing if the transaction is already confirmed. Returns the original
243    /// txid if no bump was needed.
244    ///
245    /// This function implements Replace-By-Fee (RBF) to increase the fee of an unconfirmed transaction.
246    /// It works as follows:
247    /// 1. If the transaction is already confirmed, returns Err(TransactionAlreadyInBlock)
248    /// 2. If the current fee rate is already >= the requested fee rate, returns the original txid
249    /// 3. Otherwise, increases the fee rate by adding the node's incremental fee to the current fee rate, then `bump_fee`s the transaction
250    ///
251    /// Note: This function currently only supports fee payer TXs.
252    ///
253    /// # Arguments
254    /// * `txid` - The transaction ID to bump
255    /// * `fee_rate` - The target fee rate to achieve
256    ///
257    /// # Returns
258    /// The txid of the bumped transaction (which may be the same as the input txid if no bump was needed).
259    ///
260    /// # Errors
261    /// * `TransactionAlreadyInBlock` - If the transaction is already confirmed
262    /// * `BumpFeeUTXOSpent` - If the UTXO being spent by the transaction is already spent
263    /// * `BumpFeeError` - For other errors with fee bumping
264    pub async fn bump_fee_with_fee_rate(&self, txid: Txid, fee_rate: FeeRate) -> Result<Txid> {
265        // Check if transaction is already confirmed
266        let transaction_info = self
267            .client
268            .get_transaction(&txid, None)
269            .await
270            .wrap_err("Failed to get transaction")?;
271        if transaction_info.info.blockhash.is_some() {
272            return Err(BitcoinRPCError::TransactionAlreadyInBlock(
273                transaction_info
274                    .info
275                    .blockhash
276                    .expect("Blockhash should be present"),
277            ));
278        }
279
280        // Calculate current fee rate
281        let tx = transaction_info
282            .transaction()
283            .wrap_err("Failed to get transaction")?;
284        let tx_size = tx.weight().to_vbytes_ceil();
285        let current_fee_sat = u64::try_from(
286            transaction_info
287                .fee
288                .expect("Fee should be present")
289                .to_sat()
290                .abs(),
291        )
292        .wrap_err("Failed to convert fee to sat")?;
293
294        let current_fee_rate = FeeRate::from_sat_per_kwu(1000 * current_fee_sat / tx_size);
295
296        // If current fee rate is already sufficient, return original txid
297        if current_fee_rate >= fee_rate {
298            return Ok(txid);
299        }
300
301        // Get node's incremental fee to determine how much to increase
302        let network_info = self
303            .client
304            .get_network_info()
305            .await
306            .wrap_err("Failed to get network info")?;
307        let incremental_fee = network_info.incremental_fee;
308        let incremental_fee_rate: FeeRate = FeeRate::from_sat_per_kwu(incremental_fee.to_sat());
309
310        // Calculate new fee rate by adding incremental fee to current fee rate
311        let new_fee_rate = FeeRate::from_sat_per_kwu(
312            current_fee_rate.to_sat_per_kwu() + incremental_fee_rate.to_sat_per_kwu(),
313        );
314
315        tracing::debug!(
316            "Bumping fee for txid: {txid} from {current_fee_rate} to {new_fee_rate} with incremental fee {incremental_fee_rate} - Final fee rate: {new_fee_rate}"
317        );
318
319        // Call Bitcoin Core's bumpfee RPC
320        let bump_fee_result = match self
321            .client
322            .bump_fee(
323                &txid,
324                Some(&bitcoincore_rpc::json::BumpFeeOptions {
325                    fee_rate: Some(bitcoincore_rpc::json::FeeRate::per_vbyte(Amount::from_sat(
326                        new_fee_rate.to_sat_per_vb_ceil(),
327                    ))),
328                    replaceable: Some(true),
329                    ..Default::default()
330                }),
331            )
332            .await
333        {
334            Ok(bump_fee_result) => bump_fee_result,
335            // Attempt to parse the error message to get the outpoint if the UTXO is already spent
336            Err(e) => match e {
337                bitcoincore_rpc::Error::JsonRpc(json_rpc_error) => match json_rpc_error {
338                    bitcoincore_rpc::RpcError::Rpc(rpc_error) => {
339                        if let Some((outpoint_str, _)) =
340                            rpc_error.message.split_once(" is already spent")
341                        {
342                            let outpoint = OutPoint::from_str(outpoint_str)
343                                .wrap_err(BitcoinRPCError::BumpFeeError(txid, fee_rate))?;
344
345                            return Err(BitcoinRPCError::BumpFeeUTXOSpent(outpoint));
346                        }
347
348                        return Err(eyre::eyre!(format!("{:?}", rpc_error))
349                            .wrap_err(BitcoinRPCError::BumpFeeError(txid, fee_rate))
350                            .into());
351                    }
352                    _ => {
353                        return Err(eyre::eyre!(json_rpc_error)
354                            .wrap_err(BitcoinRPCError::BumpFeeError(txid, fee_rate))
355                            .into());
356                    }
357                },
358                _ => {
359                    return Err(eyre::eyre!(e)
360                        .wrap_err(BitcoinRPCError::BumpFeeError(txid, fee_rate))
361                        .into())
362                }
363            },
364        };
365
366        // Return the new txid
367        Ok(bump_fee_result
368            .txid
369            .ok_or_eyre("Failed to get Txid from bump_fee_result")?)
370    }

372    /// Creates a new instance of the ExtendedRpc with a new client connection.
373    ///
374    /// This is useful when you need a separate connection to the Bitcoin RPC server.
375    ///
376    /// # Returns
377    /// A new `ExtendedRpc` instance on success, or an error if the connection fails.
378    pub async fn clone_inner(&self) -> std::result::Result<Self, bitcoincore_rpc::Error> {
379        let new_client = Client::new(&self.url, self.auth.clone()).await?;
380
381        Ok(Self {
382            url: self.url.clone(),
383            auth: self.auth.clone(),
384            client: Arc::new(new_client),
385        })
386    }
387
388    pub fn url(&self) -> &str {
389        &self.url
390    }
