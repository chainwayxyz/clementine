12/// Initializes `tracing` as the logger.
13///
14/// # Parameters
15///
16/// - `level`: Level ranges from 0 to 5. 0 defaults to no logs but can be
17///   overwritten with `RUST_LOG` env var. While other numbers sets log level from
18///   lowest level (1) to highest level (5). Is is advised to use 0 on tests and
19///   other values for binaries (get value from user).
20///
21/// # Returns
22///
23/// Returns `Err` if `tracing` can't be initialized. Multiple subscription error
24/// is emmitted and will return `Ok(())`.
25pub fn initialize_logger(level: Option<LevelFilter>) -> Result<(), BridgeError> {
26    // Configure JSON formatting with additional fields
27    let json_layer = fmt::layer::<Registry>()
28        .with_test_writer()
29        // .with_timer(time::UtcTime::rfc_3339())
30        .with_file(true)
31        .with_line_number(true)
32        .with_thread_ids(true)
33        .with_thread_names(true)
34        .with_target(true)
35        // .with_current_span(true)
36        // .with_span_list(true)
37        .json();
38
39    // Standard human-readable layer for non-JSON output
40    let standard_layer = fmt::layer()
41        .with_test_writer()
42        // .with_timer(time::UtcTime::rfc_3339())
43        .with_file(true)
44        .with_line_number(true)
45        .with_target(true);
46
47    let filter = match level {
48        Some(level) => EnvFilter::builder()
49            .with_default_directive(level.into())
50            .from_env_lossy(),
51        None => EnvFilter::from_default_env(),
52    };
53
54    // Try to initialize tracing, depending on the `JSON_LOGS` env var
55    let res = if std::env::var("JSON_LOGS").is_ok() {
56        tracing_subscriber::util::SubscriberInitExt::try_init(
57            tracing_subscriber::registry().with(json_layer).with(filter),
58        )
59    } else {
60        tracing_subscriber::util::SubscriberInitExt::try_init(
61            tracing_subscriber::registry()
62                .with(standard_layer)
63                .with(filter),
64        )
65    };
66
67    if let Err(e) = res {
68        // If it failed because of a re-initialization, do not care about
69        // the error.
70        if e.to_string() != "a global default trace dispatcher has already been set" {
71            return Err(BridgeError::ConfigError(e.to_string()));
72        }
73
74        tracing::trace!("Tracing is already initialized, skipping without errors...");
75    };
76
77    Ok(())
78}

194/// Monitors a JoinHandle and aborts the process if the task completes with an error.
195/// Returns a handle to the monitoring task that can be used to cancel it.
196pub fn monitor_task_with_panic<T: Send + 'static, E: Debug + Send + 'static>(
197    task_handle: tokio::task::JoinHandle<Result<T, E>>,
198    task_name: &str,
199) {
200    let task_name = task_name.to_string();
201
202    // Move task_handle into the spawned task to make it Send
203    tokio::spawn(async move {
204        match task_handle.await {
205            Ok(Ok(_)) => {
206                // Task completed successfully
207                tracing::debug!("Task {} completed successfully", task_name);
208            }
209            Ok(Err(e)) => {
210                // Task returned an error
211                tracing::error!("Task {} failed with error: {:?}", task_name, e);
212                panic!();
213            }
214            Err(e) => {
215                if e.is_cancelled() {
216                    // Task was cancelled, which is expected during cleanup
217                    tracing::debug!("Task {} was cancelled", task_name);
218                    return;
219                }
220                // Task panicked or was aborted
221                tracing::error!("Task {} panicked: {:?}", task_name, e);
222                panic!();
223            }
224        }
225    });
226}

228/// Delays the exit of the program for 15 seconds, to allow for logs to be flushed.
229/// Then panics with the given arguments.
230///
231/// # Parameters
232///
233/// - `($($arg:tt)*)`: Arguments to pass to `panic!`, in the same manner as format! and println!
234macro_rules! delayed_panic {
235    ($($arg:tt)*) => {
236        {
237            eprintln!($($arg)*);
238            eprintln!("Delaying exit for 15 seconds, to allow for logs to be flushed");
239            std::thread::sleep(std::time::Duration::from_secs(15));
240            panic!($($arg)*);
241        }
242    };
243}

258#[derive(Debug, Clone)]
259pub struct AddMethodMiddleware<S> {
260    inner: S,
261}

263type BoxFuture<'a, T> = Pin<Box<dyn std::future::Future<Output = T> + Send + 'a>>;
264
265impl<S, ReqBody, ResBody> Service<http::Request<ReqBody>> for AddMethodMiddleware<S>
266where
267    S: Service<http::Request<ReqBody>, Response = http::Response<ResBody>> + Clone + Send + 'static,
268    S::Future: Send + 'static,
269    ReqBody: Send + 'static,
270{
271    type Response = S::Response;
272    type Error = S::Error;
273    type Future = BoxFuture<'static, Result<Self::Response, Self::Error>>;
274
275    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
276        self.inner.poll_ready(cx)
277    }
278
279    fn call(&mut self, mut req: http::Request<ReqBody>) -> Self::Future {
280        // See: https://docs.rs/tower/latest/tower/trait.Service.html#be-careful-when-cloning-inner-services
281        let clone = self.inner.clone();
282        let mut inner = std::mem::replace(&mut self.inner, clone);
283
284        Box::pin(async move {
285            let path = req.uri().path();
286
287            let grpc_method =
288                if let &[_, _, method] = &path.split("/").collect::<Vec<&str>>().as_slice() {
289                    Some(method.to_string())
290                } else {
291                    None
292                };
293
294            if let Some(grpc_method) = grpc_method {
295                if let Ok(grpc_method) = HeaderValue::from_str(&grpc_method) {
296                    req.headers_mut().insert("grpc-method", grpc_method);
297                }
298            }
299
300            // Do extra async work here...
301            let response = inner.call(req).await?;
302
303            Ok(response)
304        })
305    }
