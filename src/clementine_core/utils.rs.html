29/// Initializes a [`tracing`] subscriber depending on the environment.
30/// [`EnvFilter`] is used with an optional default level. Sets up the
31/// [`color_eyre`] handler.
32///
33/// # Log Formats
34///
35/// - `json` **JSON** is used when `LOG_FORMAT=json`
36/// - `human` **Human-readable** direct logs are used when `LOG_FORMAT` is not
37///   set to `json`.
38///
39/// ## CI
40///
41/// In CI, logging is always in the human-readable format with output to the
42/// console. The `INFO_LOG_FILE` env var can be used to set an optional log file
43/// output. If not set, only console logging is used.
44///
45/// # Backtraces
46///
47/// Backtraces are enabled by default for tests. Error backtraces otherwise
48/// depend on the `RUST_LIB_BACKTRACE` env var. Please read [`color_eyre`]
49/// documentation for more details.
50///
51/// # Parameters
52///
53/// - `default_level`: Default level ranges from 0 to 5. This is overwritten through the
54///   `RUST_LOG` env var.
55///
56/// # Returns
57///
58/// Returns `Err` in CI if the file logging cannot be initialized.  Already
59/// initialized errors are ignored, so this function can be called multiple
60/// times safely.
61pub fn initialize_logger(default_level: Option<LevelFilter>) -> Result<(), BridgeError> {
62    let is_ci = std::env::var("CI")
63        .map(|v| v == "true" || v == "1")
64        .unwrap_or(false);
65
66    // UNCOMMENT TO DEBUG TOKIO TASKS
67    // console_subscriber::init();
68
69    if cfg!(test) {
70        // Enable full backtraces for tests
71        std::env::set_var("RUST_LIB_BACKTRACE", "full");
72        std::env::set_var("RUST_BACKTRACE", "full");
73    }
74
75    // Initialize color-eyre for better error handling and backtraces
76    let _ = color_eyre::config::HookBuilder::default()
77        .add_frame_filter(Box::new(|frames| {
78            // Frames with names starting with any of the str's below will be filtered out
79            let filters = &[
80                "std::",
81                "test::",
82                "tokio::",
83                "core::",
84                "<core::",
85                "<alloc::",
86                "start_thread",
87                "clone",
88            ];
89
90            frames.retain(|frame| {
91                !filters.iter().any(|f| {
92                    let name = if let Some(name) = frame.name.as_ref() {
93                        name.as_str()
94                    } else {
95                        return true;
96                    };
97
98                    name.starts_with(f)
99                })
100            });
101        }))
102        .install();
103
104    if is_ci {
105        let info_log_file = std::env::var("INFO_LOG_FILE").ok();
106        if let Some(file_path) = info_log_file {
107            try_set_global_subscriber(env_subscriber_with_file(&file_path)?);
108            tracing::trace!("Using file logging in CI, outputting to {}", file_path);
109        } else {
110            try_set_global_subscriber(env_subscriber_to_human(default_level));
111            tracing::trace!("Using console logging in CI");
112            tracing::warn!(
113                "CI is set but INFO_LOG_FILE is missing, only console logs will be used."
114            );
115        }
116    } else if is_json_logs() {
117        try_set_global_subscriber(env_subscriber_to_json(default_level));
118        tracing::trace!("Using JSON logging");
119    } else {
120        try_set_global_subscriber(env_subscriber_to_human(default_level));
121        tracing::trace!("Using human-readable logging");
122    }
123
124    tracing::info!("Tracing initialized successfully.");
125    Ok(())
126}

128pub fn initialize_telemetry(config: &TelemetryConfig) -> Result<(), BridgeError> {
129    let telemetry_addr: SocketAddr = format!("{}:{}", config.host, config.port)
130        .parse()
131        .unwrap_or_else(|_| {
132            tracing::warn!(
133                "Invalid telemetry address: {}:{}, using default address: 127.0.0.1:8081",
134                config.host,
135                config.port
136            );
137            SocketAddr::from((Ipv4Addr::new(127, 0, 0, 1), 8081))
138        });
139
140    tracing::debug!("Initializing telemetry at {}", telemetry_addr);
141
142    let builder = PrometheusBuilder::new().with_http_listener(telemetry_addr);
143
144    builder
145        .install()
146        .map_err(|e| eyre::eyre!("Failed to initialize telemetry: {}", e))?;
147
148    Ok(())
149}

151fn try_set_global_subscriber<S>(subscriber: S)
152where
153    S: Subscriber + Send + Sync + 'static,
154{
155    match tracing::subscriber::set_global_default(subscriber) {
156        Ok(_) => {}
157        // Statically, the only error possible is "already initialized"
158        Err(_) => {
159            #[cfg(test)]
160            tracing::trace!("Tracing is already initialized, skipping without errors...");
161            #[cfg(not(test))]
162            tracing::info!(
163                "Unexpected double initialization of tracing, skipping without errors..."
164            );
165        }
166    }
167}

169fn env_subscriber_with_file(path: &str) -> Result<Box<dyn Subscriber + Send + Sync>, BridgeError> {
170    if let Some(parent_dir) = std::path::Path::new(path).parent() {
171        std::fs::create_dir_all(parent_dir).map_err(|e| {
172            BridgeError::ConfigError(format!(
173                "Failed to create log directory '{}': {}",
174                parent_dir.display(),
175                e
176            ))
177        })?;
178    }
179
180    let file = File::create(path).map_err(|e| BridgeError::ConfigError(e.to_string()))?;
181
182    let file_filter = EnvFilter::from_default_env()
183        .add_directive("info".parse().expect("It should parse info level"))
184        .add_directive("ci=debug".parse().expect("It should parse ci debug level"));
185
186    let console_filter = EnvFilter::builder()
187        .with_default_directive(LevelFilter::WARN.into())
188        .from_env_lossy();
189
190    let file_layer = fmt::layer()
191        .with_writer(file)
192        .with_ansi(false)
193        .with_file(true)
194        .with_line_number(true)
195        .with_target(true)
196        .with_thread_ids(true)
197        .with_thread_names(true)
198        .with_filter(file_filter)
199        .boxed();
200
201    let console_layer = fmt::layer()
202        .with_test_writer()
203        .with_file(true)
204        .with_line_number(true)
205        .with_target(true)
206        .with_filter(console_filter)
207        .boxed();
208
209    Ok(Box::new(
210        Registry::default().with(file_layer).with(console_layer),
211    ))
212}

383/// Monitors a [`tokio::task::JoinHandle`] in the background and logs it's end
384/// result.
385pub fn monitor_standalone_task<T: Send + 'static, E: Debug + Send + 'static>(
386    task_handle: tokio::task::JoinHandle<Result<T, E>>,
387    task_name: &str,
388) {
389    let task_name = task_name.to_string();
390
391    // Move task_handle into the spawned task to make it Send
392    tokio::spawn(async move {
393        match task_handle.await {
394            Ok(Ok(_)) => {
395                tracing::debug!("Task {} completed successfully", task_name);
396            }
397            Ok(Err(e)) => {
398                tracing::error!("Task {} throw an error: {:?}", task_name, e);
399            }
400            Err(e) => {
401                if e.is_cancelled() {
402                    // Task was cancelled, which is expected during cleanup
403                    tracing::debug!("Task {} has cancelled", task_name);
404                    return;
405                }
406                tracing::error!("Task {} has panicked: {:?}", task_name, e);
407            }
408        }
409    });
410}

412/// Delays the exit of the program for 15 seconds, to allow for logs to be flushed.
413/// Then panics with the given arguments.
414///
415/// # Parameters
416///
417/// - `($($arg:tt)*)`: Arguments to pass to `panic!`, in the same manner as format! and println!
418macro_rules! delayed_panic {
419    ($($arg:tt)*) => {
420        {
421            eprintln!($($arg)*);
422            eprintln!("Delaying exit for 15 seconds, to allow for logs to be flushed");
423            std::thread::sleep(std::time::Duration::from_secs(15));
424            panic!($($arg)*);
425        }
426    };
427}

447type BoxFuture<'a, T> = Pin<Box<dyn std::future::Future<Output = T> + Send + 'a>>;
448
449impl<S, ReqBody, ResBody> Service<http::Request<ReqBody>> for AddMethodMiddleware<S>
450where
451    S: Service<http::Request<ReqBody>, Response = http::Response<ResBody>> + Clone + Send + 'static,
452    S::Future: Send + 'static,
453    ReqBody: Send + 'static,
454{
455    type Response = S::Response;
456    type Error = S::Error;
457    type Future = BoxFuture<'static, Result<Self::Response, Self::Error>>;
458
459    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
460        self.inner.poll_ready(cx)
461    }
462
463    fn call(&mut self, mut req: http::Request<ReqBody>) -> Self::Future {
464        // See: https://docs.rs/tower/latest/tower/trait.Service.html#be-careful-when-cloning-inner-services
465        let clone = self.inner.clone();
466        let mut inner = std::mem::replace(&mut self.inner, clone);
467
468        Box::pin(async move {
469            let path = req.uri().path();
470
471            let grpc_method =
472                if let &[_, _, method] = &path.split("/").collect::<Vec<&str>>().as_slice() {
473                    Some(method.to_string())
474                } else {
475                    None
476                };
477
478            if let Some(grpc_method) = grpc_method {
479                if let Ok(grpc_method) = HeaderValue::from_str(&grpc_method) {
480                    req.headers_mut().insert("grpc-method", grpc_method);
481                }
482            }
483
484            // Do extra async work here...
485            let response = inner.call(req).await?;
486
487            Ok(response)
488        })
489    }
490}

547    /// from the original "parent" transaction. The child pays a high fee, sufficient

549    /// to confirm both together. Specifically, we utilize "fee payer" UTXOs.

554    RBF,
555    /// The transaction has already been funded and no fee is needed.
556    /// Currently used for disprove tx as it has operator's collateral as input.
557    NoFunding,
558}

594impl ScriptBufExt for ScriptBuf {
595    fn try_get_taproot_pk(&self) -> Result<XOnlyPublicKey, BridgeError> {
596        if !self.is_p2tr() {
597            return Err(eyre::eyre!("Script is not a valid P2TR script (not 34 bytes)").into());
598        }
599
600        Ok(XOnlyPublicKey::from_slice(&self.as_bytes()[2..34])
601            .wrap_err("Failed to parse XOnlyPublicKey from script")?)
602    }

605impl TryLast20Bytes for &[u8] {
606    fn try_last_20_bytes(self) -> Result<[u8; 20], BridgeError> {
607        if self.len() < 20 {
608            return Err(eyre::eyre!("Input is too short to contain 20 bytes").into());
609        }
610        let mut result = [0u8; 20];
611
612        result.copy_from_slice(&self[self.len() - 20..]);
613        Ok(result)
614    }

617/// Wraps a future with a timeout, returning a `Status::deadline_exceeded` gRPC error
618/// if the future does not complete within the specified duration.
619///
620/// This is useful for enforcing timeouts on individual asynchronous operations,
621/// especially those involving network requests, to prevent them from hanging indefinitely.
622///
623/// # Arguments
624///
625/// * `duration`: The maximum `Duration` to wait for the future to complete.
626/// * `description`: A string slice describing the operation, used in the timeout error message.
627/// * `future`: The `Future` to execute. The future should return a `Result<T, BridgeError>`.
628///
629/// # Returns
630///
631/// Returns `Ok(T)` if the future completes successfully within the time limit.
632/// Returns `Err(BridgeError)` if the future returns an error or if it times out.
633/// A timeout results in a `BridgeError` that wraps a `tonic::Status::deadline_exceeded`.
634pub async fn timed_request<F, T>(
635    duration: Duration,
636    description: &str,
637    future: F,
638) -> Result<T, BridgeError>
639where
640    F: Future<Output = Result<T, BridgeError>>,
641{
642    timed_request_base(duration, description, future)
643        .await
644        .map_err(|_| {
645            Box::new(Status::deadline_exceeded(format!(
646                "{description} timed out"
647            )))
648        })?
649}
650
651/// Wraps a future with a timeout and adds a debug span with the description.
652///
653/// # Arguments
654///
655/// * `duration`: The maximum `Duration` to wait for the future to complete.
656/// * `description`: A string slice describing the operation, used in the timeout error message.
657/// * `future`: The `Future` to execute. The future should return a `Result<T, BridgeError>`.
658///
659/// # Returns
660///
661/// Returns `Ok(Ok(T))` if the future completes successfully within the time limit, returns `Ok(Err(e))`
662/// if the future returns an error, returns `Err(Elapsed)` if the request times out.
663pub async fn timed_request_base<F, T>(
664    duration: Duration,
665    description: &str,
666    future: F,
667) -> Result<Result<T, BridgeError>, Elapsed>
668where
669    F: Future<Output = Result<T, BridgeError>>,
670{
671    timeout(duration, future)
672        .instrument(debug_span!("timed_request", description = description))
673        .await
674}

676/// Concurrently executes a collection of futures, applying a timeout to each one individually.
677/// If any future fails or times out, the entire operation is aborted and an error is returned.
678///
679/// This utility is an extension of `futures::future::try_join_all` with added per-future
680/// timeout logic and improved error reporting using optional IDs.
681///
682/// # Type Parameters
683///
684/// * `I`: An iterator that yields futures.
685/// * `T`: The success type of the futures.
686/// * `D`: A type that can be displayed, used for identifying futures in error messages.
687///
688/// # Arguments
689///
690/// * `duration`: The timeout `Duration` applied to each individual future in the iterator.
691/// * `description`: A string slice describing the collective operation, used in timeout error messages.
692/// * `ids`: An optional `Vec<D>` of identifiers corresponding to each future. If provided,
693///   these IDs are used in error messages to specify which future failed or timed out.
694/// * `iter`: An iterator producing the futures to be executed.
695///
696/// # Returns
697///
698/// Returns `Ok(Vec<T>)` containing the results of all futures if they all complete successfully.
699/// Returns `Err(BridgeError)` if any future returns an error or times out.
700/// The error will be contextualized with the operation description and the specific future's ID if available.
701pub async fn timed_try_join_all<I, T, D>(
702    duration: Duration,
703    description: &str,
704    ids: Option<Vec<D>>,
705    iter: I,
706) -> Result<Vec<T>, BridgeError>
707where
708    D: Display,
709    I: IntoIterator,
710    I::Item: Future<Output = Result<T, BridgeError>>,
711{
712    let ids = Arc::new(ids);
713    try_join_all(iter.into_iter().enumerate().map(|item| {
714        let ids = ids.clone();
715        async move {
716            let id = Option::as_ref(&ids).and_then(|ids| ids.get(item.0));
717
718            timeout(duration, item.1)
719                .await
720                .map_err(|_| {
721                    Box::new(Status::deadline_exceeded(format!(
722                        "{} (id: {}) timed out",
723                        description,
724                        id.map(|id| id.to_string())
725                            .unwrap_or_else(|| "n/a".to_string())
726                    )))
727                })?
728                // Add the id to the error chain for easier debugging for other errors.
729                .wrap_err_with(|| {
730                    format!(
731                        "Failed to join {}",
732                        id.map(ToString::to_string).unwrap_or_else(|| "n/a".into())
733                    )
734                })
735                .map_err(Into::into)
736        }
737    }))
738    .instrument(debug_span!("timed_try_join_all", description = description))
739    .await
740}
