72/// Round index is used to represent the round index safely.
73/// Collateral represents the collateral utxo.
74/// Round(index) represents the rounds of the bridge operators, index is 0-indexed.
75/// As a single u32, collateral is represented as 0 and rounds are represented starting from 1.
76#[derive(
77    Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize, Ord, PartialOrd,
78)]
79pub enum RoundIndex {
80    Collateral,
81    Round(usize), // 0-indexed
82}
83
84impl RoundIndex {
85    /// Converts the round to a 0-indexed index.
86    pub fn to_index(&self) -> usize {
87        match self {
88            RoundIndex::Collateral => 0,
89            RoundIndex::Round(index) => *index + 1,
90        }
91    }
92
93    /// Converts a 0-indexed index to a RoundIndex.
94    /// Use this only when dealing with 0-indexed data. Currently these are data coming from the database and rpc.
95    pub fn from_index(index: usize) -> Self {
96        if index == 0 {
97            RoundIndex::Collateral
98        } else {
99            RoundIndex::Round(index - 1)
100        }
101    }
102
103    /// Returns the next RoundIndex.
104    pub fn next_round(&self) -> Self {
105        match self {
106            RoundIndex::Collateral => RoundIndex::Round(0),
107            RoundIndex::Round(index) => RoundIndex::Round(*index + 1),
108        }
109    }
110
111    /// Creates an iterator over rounds from 0 to num_rounds (exclusive)
112    /// Only iterates actual rounds, collateral is not included.
113    pub fn iter_rounds(num_rounds: usize) -> impl Iterator<Item = RoundIndex> {
114        Self::iter_rounds_range(0, num_rounds)
115    }
116
117    /// Creates an iterator over rounds from start to end (exclusive)
118    /// Only iterates actual rounds, collateral is not included.
119    pub fn iter_rounds_range(start: usize, end: usize) -> impl Iterator<Item = RoundIndex> {
120        (start..end).map(RoundIndex::Round)
121    }
122}

571    /// Checks if the withdrawal amount is within the acceptable range.
572    fn is_profitable(
573        input_amount: Amount,
574        withdrawal_amount: Amount,
575        bridge_amount_sats: Amount,
576        operator_withdrawal_fee_sats: Amount,
577    ) -> bool {
578        // Use checked_sub to safely handle potential underflow
579        let withdrawal_diff = match withdrawal_amount
580            .to_sat()
581            .checked_sub(input_amount.to_sat())
582        {
583            Some(diff) => Amount::from_sat(diff),
584            None => {
585                // input amount is greater than withdrawal amount, so it's profitable but doesn't make sense
586                tracing::warn!(
587                    "Some user gave more amount than the withdrawal amount as input for withdrawal"
588                );
589                return true;
590            }
591        };
592
593        if withdrawal_diff > bridge_amount_sats {
594            return false;
595        }
596
597        // Calculate net profit after the withdrawal using checked_sub to prevent panic
598        let net_profit = match bridge_amount_sats.checked_sub(withdrawal_diff) {
599            Some(profit) => profit,
600            None => return false, // If underflow occurs, it's not profitable
601        };
602
603        // Net profit must be bigger than withdrawal fee.
604        // net profit doesn't take into account the fees, but operator_withdrawal_fee_sats should
605        net_profit >= operator_withdrawal_fee_sats
606    }

609    ///
610    /// 1. Checking if the withdrawal has been made on Citrea
611    /// 2. Verifying the given signature
612    /// 3. Checking if the withdrawal is profitable or not
613    /// 4. Funding the withdrawal transaction using TxSender RBF option
614    ///
615    /// # Parameters
616    ///
617    /// - `withdrawal_idx`: Citrea withdrawal UTXO index
618    /// - `in_signature`: User's signature that is going to be used for signing
619    ///   withdrawal transaction input
620    /// - `in_outpoint`: User's input for the payout transaction
621    /// - `out_script_pubkey`: User's script pubkey which will be used
622    ///   in the payout transaction's output
623    /// - `out_amount`: Payout transaction output's value
624    ///
625    /// # Returns
626    ///
627    /// - Ok(()) if the withdrawal checks are successful and a payout transaction is added to the TxSender
628    /// - Err(BridgeError) if the withdrawal checks fail
629    pub async fn withdraw(
630        &self,
631        withdrawal_index: u32,
632        in_signature: taproot::Signature,
633        in_outpoint: OutPoint,
634        out_script_pubkey: ScriptBuf,
635        out_amount: Amount,
636    ) -> Result<Transaction, BridgeError> {
637        tracing::info!(
638            "Withdrawing with index: {}, in_signature: {:?}, in_outpoint: {:?}, out_script_pubkey: {}, out_amount: {}",
639            withdrawal_index,
640            in_signature,
641            in_outpoint,
642            out_script_pubkey,
643            out_amount
644        );
645
646        // Prepare input and output of the payout transaction.
647        let input_prevout = self.rpc.get_txout_from_outpoint(&in_outpoint).await?;
648        let input_utxo = UTXO {
649            outpoint: in_outpoint,
650            txout: input_prevout,
651        };
652        let output_txout = TxOut {
653            value: out_amount,
654            script_pubkey: out_script_pubkey,
655        };
656
657        // Check Citrea for the withdrawal state.
658        let withdrawal_utxo = self
659            .db
660            .get_withdrawal_utxo_from_citrea_withdrawal(None, withdrawal_index)
661            .await?;
662
663        if withdrawal_utxo != input_utxo.outpoint {
664            return Err(eyre::eyre!("Input UTXO does not match withdrawal UTXO from Citrea: Input Outpoint: {0}, Withdrawal Outpoint (from Citrea): {1}", input_utxo.outpoint, withdrawal_utxo).into());
665        }
666
667        let operator_withdrawal_fee_sats =
668            self.config
669                .operator_withdrawal_fee_sats
670                .ok_or(BridgeError::ConfigError(
671                    "Operator withdrawal fee sats is not specified in configuration file"
672                        .to_string(),
673                ))?;
674        if !Self::is_profitable(
675            input_utxo.txout.value,
676            output_txout.value,
677            self.config.protocol_paramset().bridge_amount,
678            operator_withdrawal_fee_sats,
679        ) {
680            return Err(eyre::eyre!("Not enough fee for operator").into());
681        }
682
683        let user_xonly_pk = &input_utxo
684            .txout
685            .script_pubkey
686            .try_get_taproot_pk()
687            .wrap_err("Input utxo script pubkey is not a valid taproot script")?;
688
689        let payout_txhandler = builder::transaction::create_payout_txhandler(
690            input_utxo,
691            output_txout,
692            self.signer.xonly_public_key,
693            in_signature,
694            self.config.protocol_paramset().network,
695        )?;
696
697        // tracing::info!("Payout txhandler: {:?}", hex::encode(bitcoin::consensus::serialize(&payout_txhandler.get_cached_tx())));
698
699        let sighash = payout_txhandler.calculate_sighash_txin(0, in_signature.sighash_type)?;
700
701        SECP.verify_schnorr(
702            &in_signature.signature,
703            &Message::from_digest(*sighash.as_byte_array()),
704            user_xonly_pk,
705        )
706        .wrap_err("Failed to verify signature received from user for payout txin. Ensure the signature uses SinglePlusAnyoneCanPay sighash type.")?;
707
708        let fee_rate_result = self
709            .rpc
710            .get_fee_rate(
711                self.config.protocol_paramset.network,
712                &self.config.mempool_api_host,
713                &self.config.mempool_api_endpoint,
714                self.config.tx_sender_limits.mempool_fee_rate_multiplier,
715                self.config.tx_sender_limits.mempool_fee_rate_offset_sat_kvb,
716                self.config.tx_sender_limits.fee_rate_hard_cap,
717            )
718            .await;
719
720        let fee_rate_option = match fee_rate_result {
721            Ok(fee_rate) => Some(Amount::from_sat(fee_rate.to_sat_per_vb_ceil() * 1000)),
722            Err(e) => {
723                tracing::warn!("Failed to get fee rate from mempool API; funding tx with automatic fee rate. Error: {e:?}");
724                None
725            }
726        };
727
728        // send payout tx using RBF
729        let funded_tx = self
730            .rpc
731            .fund_raw_transaction(
732                payout_txhandler.get_cached_tx(),
733                Some(&bitcoincore_rpc::json::FundRawTransactionOptions {
734                    add_inputs: Some(true),
735                    change_address: None,
736                    change_position: Some(1),
737                    change_type: None,
738                    include_watching: None,
739                    lock_unspents: Some(false),
740                    fee_rate: fee_rate_option,
741                    subtract_fee_from_outputs: None,
742                    replaceable: None,
743                    conf_target: None,
744                    estimate_mode: None,
745                }),
746                None,
747            )
748            .await
749            .wrap_err("Failed to fund raw transaction")?
750            .hex;
751
752        let signed_tx = self
753            .rpc
754            .sign_raw_transaction_with_wallet(&funded_tx, None, None)
755            .await
756            .wrap_err("Failed to sign withdrawal transaction")?
757            .hex;
758
759        let signed_tx: Transaction = bitcoin::consensus::deserialize(&signed_tx)
760            .wrap_err("Failed to deserialize signed withdrawal transaction")?;
761
762        self.rpc
763            .send_raw_transaction(&signed_tx)
764            .await
765            .wrap_err("Failed to send withdrawal transaction")?;
766
767        Ok(signed_tx)
768    }

827    pub fn generate_unspent_kickoff_sigs(
828        &self,
829        kickoff_wpks: &KickoffWinternitzKeys,
830    ) -> Result<Vec<Signature>, BridgeError> {
831        let mut tweak_cache = TweakCache::default();
832        let mut sigs: Vec<Signature> =
833            Vec::with_capacity(self.config.get_num_unspent_kickoff_sigs());
834        let mut prev_ready_to_reimburse: Option<TxHandler> = None;
835        let operator_data = OperatorData {
836            xonly_pk: self.signer.xonly_public_key,
837            collateral_funding_outpoint: self.collateral_funding_outpoint,
838            reimburse_addr: self.reimburse_addr.clone(),
839        };
840        for round_idx in RoundIndex::iter_rounds(self.config.protocol_paramset().num_round_txs) {
841            let txhandlers = create_round_txhandlers(
842                self.config.protocol_paramset(),
843                round_idx,
844                &operator_data,
845                kickoff_wpks,
846                prev_ready_to_reimburse.as_ref(),
847            )?;
848            for txhandler in txhandlers {
849                if let TransactionType::UnspentKickoff(kickoff_idx) =
850                    txhandler.get_transaction_type()
851                {
852                    let partial = PartialSignatureInfo {
853                        operator_idx: 0, // dummy value
854                        round_idx,
855                        kickoff_utxo_idx: kickoff_idx,
856                    };
857                    let sighashes = txhandler
858                        .calculate_shared_txins_sighash(EntityType::OperatorSetup, partial)?;
859                    let signed_sigs: Result<Vec<_>, _> = sighashes
860                        .into_iter()
861                        .map(|(sighash, sig_info)| {
862                            self.signer.sign_with_tweak_data(
863                                sighash,
864                                sig_info.tweak_data,
865                                Some(&mut tweak_cache),
866                            )
867                        })
868                        .collect();
869                    sigs.extend(signed_sigs?);
870                }
871                if let TransactionType::ReadyToReimburse = txhandler.get_transaction_type() {
872                    prev_ready_to_reimburse = Some(txhandler);
873                }
874            }
875        }
876        if sigs.len() != self.config.get_num_unspent_kickoff_sigs() {
877            return Err(eyre::eyre!(
878                "Expected {} number of unspent kickoff sigs, but got {}",
879                self.config.get_num_unspent_kickoff_sigs(),
880                sigs.len()
881            )
882            .into());
883        }
884        Ok(sigs)
885    }

915    pub async fn handle_finalized_payout<'a>(
916        &'a self,
917        dbtx: DatabaseTransaction<'a, '_>,
918        deposit_outpoint: OutPoint,
919        payout_tx_blockhash: BlockHash,
920    ) -> Result<bitcoin::Txid, BridgeError> {
921        let (deposit_id, deposit_data) = self
922            .db
923            .get_deposit_data(Some(dbtx), deposit_outpoint)
924            .await?
925            .ok_or(BridgeError::DatabaseError(sqlx::Error::RowNotFound))?;
926
927        // get unused kickoff connector
928        let (round_idx, kickoff_idx) = self
929            .db
930            .get_unused_and_signed_kickoff_connector(
931                Some(dbtx),
932                deposit_id,
933                self.signer.xonly_public_key,
934            )
935            .await?
936            .ok_or(BridgeError::DatabaseError(sqlx::Error::RowNotFound))?;
937
938        let current_round_index = self.db.get_current_round_index(Some(dbtx)).await?;
939        #[cfg(feature = "automation")]
940        if current_round_index != round_idx {
941            // we currently have no free kickoff connectors in the current round, so we need to end round first
942            // if current_round_index should only be smaller than round_idx, and should not be smaller by more than 1
943            // so sanity check:
944            if current_round_index.next_round() != round_idx {
945                return Err(eyre::eyre!(
946                    "Internal error: Expected the current round ({:?}) to be equal to or 1 less than the round of the first available kickoff for deposit reimbursement ({:?}) for deposit {:?}. If the round is less than the current round, there is an issue with the logic of the fn that gets the first available kickoff. If the round is greater, that means the next round do not have any kickoff connectors available for reimbursement, which should not be possible.",
947                    current_round_index, round_idx, deposit_outpoint
948                ).into());
949            }
950            // start the next round to be able to get reimbursement for the payout
951            self.end_round(dbtx).await?;
952        }
953
954        // get signed txs,
955        let kickoff_data = KickoffData {
956            operator_xonly_pk: self.signer.xonly_public_key,
957            round_idx,
958            kickoff_idx,
959        };
960
961        let payout_tx_blockhash = payout_tx_blockhash.as_byte_array().last_20_bytes();
962
963        #[cfg(test)]
964        let payout_tx_blockhash = self
965            .config
966            .test_params
967            .maybe_disrupt_payout_tx_block_hash_commit(payout_tx_blockhash);
968
969        let context = ContractContext::new_context_for_kickoff(
970            kickoff_data,
971            deposit_data,
972            self.config.protocol_paramset(),
973        );
974
975        let signed_txs = create_and_sign_txs(
976            self.db.clone(),
977            &self.signer,
978            self.config.clone(),
979            context,
980            Some(payout_tx_blockhash),
981            Some(dbtx),
982        )
983        .await?;
984
985        let tx_metadata = Some(TxMetadata {
986            tx_type: TransactionType::Dummy, // will be replaced in add_tx_to_queue
987            operator_xonly_pk: Some(self.signer.xonly_public_key),
988            round_idx: Some(round_idx),
989            kickoff_idx: Some(kickoff_idx),
990            deposit_outpoint: Some(deposit_outpoint),
991        });
992
993        // try to send them
994        for (tx_type, signed_tx) in &signed_txs {
995            match *tx_type {
996                TransactionType::Kickoff
997                | TransactionType::OperatorChallengeAck(_)
998                | TransactionType::WatchtowerChallengeTimeout(_)
999                | TransactionType::ChallengeTimeout
1000                | TransactionType::DisproveTimeout
1001                | TransactionType::Reimburse => {
1002                    #[cfg(feature = "automation")]
1003                    self.tx_sender
1004                        .add_tx_to_queue(
1005                            dbtx,
1006                            *tx_type,
1007                            signed_tx,
1008                            &signed_txs,
1009                            tx_metadata,
1010                            &self.config,
1011                            None,
1012                        )
1013                        .await?;
1014                }
1015                _ => {}
1016            }
1017        }
1018
1019        let kickoff_txid = signed_txs
1020            .iter()
1021            .find_map(|(tx_type, tx)| {
1022                if let TransactionType::Kickoff = tx_type {
1023                    Some(tx.compute_txid())
1024                } else {
1025                    None
1026                }
1027            })
1028            .ok_or(eyre::eyre!(
1029                "Couldn't find kickoff tx in signed_txs".to_string(),
1030            ))?;
1031
1032        // mark the kickoff connector as used
1033        self.db
1034            .mark_kickoff_connector_as_used(Some(dbtx), round_idx, kickoff_idx, Some(kickoff_txid))
1035            .await?;
1036
1037        Ok(kickoff_txid)
1038    }

1089    pub async fn end_round<'a>(
1090        &'a self,
1091        dbtx: DatabaseTransaction<'a, '_>,
1092    ) -> Result<(), BridgeError> {
1093        // get current round index
1094        let current_round_index = self.db.get_current_round_index(Some(dbtx)).await?;
1095
1096        let mut activation_prerequisites = Vec::new();
1097

1105            self.config.protocol_paramset().num_round_txs,
1106        )?;

1108        // if we are at round 0, which is just the collateral, we need to start the first round
1109        if current_round_index == RoundIndex::Collateral {
1110            return self.start_first_round(dbtx, kickoff_wpks).await;

1112
1113        let (current_round_txhandler, mut ready_to_reimburse_txhandler) =
1114            create_round_nth_txhandler(

1116                self.collateral_funding_outpoint,
1117                self.config.protocol_paramset().collateral_funding_amount,
1118                current_round_index,
1119                &kickoff_wpks,
1120                self.config.protocol_paramset(),
1121            )?;
1122
1123        let (mut next_round_txhandler, _) = create_round_nth_txhandler(
1124            self.signer.xonly_public_key,
1125            self.collateral_funding_outpoint,
1126            self.config.protocol_paramset().collateral_funding_amount,
1127            current_round_index.next_round(),
1128            &kickoff_wpks,
1129            self.config.protocol_paramset(),
1130        )?;
1131
1132        let mut tweak_cache = TweakCache::default();
1133
1134        // sign ready to reimburse tx
1135        self.signer
1136            .tx_sign_and_fill_sigs(
1137                &mut ready_to_reimburse_txhandler,
1138                &[],
1139                Some(&mut tweak_cache),
1140            )
1141            .wrap_err("Failed to sign ready to reimburse tx")?;
1142
1143        // sign next round tx
1144        self.signer
1145            .tx_sign_and_fill_sigs(&mut next_round_txhandler, &[], Some(&mut tweak_cache))
1146            .wrap_err("Failed to sign next round tx")?;
1147
1148        let current_round_txid = current_round_txhandler.get_cached_tx().compute_txid();
1149        let ready_to_reimburse_tx = ready_to_reimburse_txhandler.get_cached_tx();
1150        let next_round_tx = next_round_txhandler.get_cached_tx();
1151
1152        let ready_to_reimburse_txid = ready_to_reimburse_tx.compute_txid();
1153
1154        let mut unspent_kickoff_connector_indices = Vec::new();
1155
1156        // get kickoff txid for used kickoff connector
1157        for kickoff_connector_idx in
1158            0..self.config.protocol_paramset().num_kickoffs_per_round as u32
1159        {
1160            let kickoff_txid = self
1161                .db
1162                .get_kickoff_txid_for_used_kickoff_connector(
1163                    Some(dbtx),
1164                    current_round_index,
1165                    kickoff_connector_idx,
1166                )
1167                .await?;
1168            match kickoff_txid {
1169                Some(kickoff_txid) => {
1170                    activation_prerequisites.push(ActivatedWithOutpoint {
1171                        outpoint: OutPoint {
1172                            txid: kickoff_txid,
1173                            vout: UtxoVout::KickoffFinalizer.get_vout(), // Kickoff finalizer output index
1174                        },
1175                        relative_block_height: self.config.protocol_paramset().finality_depth - 1,
1176                    });
1177                }
1178                None => {
1179                    let unspent_kickoff_connector = OutPoint {
1180                        txid: current_round_txid,
1181                        vout: UtxoVout::Kickoff(kickoff_connector_idx as usize).get_vout(),
1182                    };
1183                    unspent_kickoff_connector_indices.push(kickoff_connector_idx as usize);
1184                    self.db
1185                        .mark_kickoff_connector_as_used(
1186                            Some(dbtx),
1187                            current_round_index,
1188                            kickoff_connector_idx,
1189                            None,
1190                        )
1191                        .await?;
1192                    activation_prerequisites.push(ActivatedWithOutpoint {
1193                        outpoint: unspent_kickoff_connector,
1194                        relative_block_height: self.config.protocol_paramset().finality_depth - 1,
1195                    });
1196                }
1197            }
1198        }
1199
1200        // Burn unused kickoff connectors
1201        let mut burn_unspent_kickoff_connectors_tx =
1202            create_burn_unused_kickoff_connectors_txhandler(
1203                &current_round_txhandler,
1204                &unspent_kickoff_connector_indices,
1205                &self.signer.address,
1206                self.config.protocol_paramset(),
1207            )?;
1208
1209        // sign burn unused kickoff connectors tx
1210        self.signer
1211            .tx_sign_and_fill_sigs(
1212                &mut burn_unspent_kickoff_connectors_tx,
1213                &[],
1214                Some(&mut tweak_cache),
1215            )
1216            .wrap_err("Failed to sign burn unused kickoff connectors tx")?;
1217
1218        self.tx_sender
1219            .insert_try_to_send(
1220                dbtx,
1221                Some(TxMetadata {
1222                    tx_type: TransactionType::BurnUnusedKickoffConnectors,
1223                    operator_xonly_pk: Some(self.signer.xonly_public_key),
1224                    round_idx: Some(current_round_index),
1225                    kickoff_idx: None,
1226                    deposit_outpoint: None,
1227                }),
1228                burn_unspent_kickoff_connectors_tx.get_cached_tx(),
1229                FeePayingType::CPFP,
1230                None,
1231                &[],
1232                &[],
1233                &[],
1234                &[],
1235            )
1236            .await?;
1237
1238        // send ready to reimburse tx
1239        self.tx_sender
1240            .insert_try_to_send(
1241                dbtx,
1242                Some(TxMetadata {
1243                    tx_type: TransactionType::ReadyToReimburse,
1244                    operator_xonly_pk: Some(self.signer.xonly_public_key),
1245                    round_idx: Some(current_round_index),
1246                    kickoff_idx: None,

1778        // by the PayoutCheckerTask, meaning kickoff_txid is set

1780            payer_xonly_pk,
1781            payout_blockhash,

1795            _ => {

1797                    "Payer info not found for deposit, payout blockhash: {:?}, kickoff txid: {:?}",

1799                    kickoff_txid

1802            }

1806            "Payer xonly pk, payout blockhash and kickoff txid found and valid for own operator for the requested deposit id: {}, payer xonly pk: {:?}, payout blockhash: {:?}, kickoff txid: {:?}",
1807            deposit_id,
1808            payer_xonly_pk,
1809            payout_blockhash,

1860                tracing::info!("We need to advance the round manually to be able to start the kickoff, current round idx: {:?}, kickoff round idx: {:?}", current_round_idx, kickoff_round_idx);
1861                let txs = self.advance_round_manually(dbtx, current_round_idx).await?;
1862                txs_to_send.extend(txs);
1863            }
1864            std::cmp::Ordering::Greater => {
1865                tracing::info!("We are at least on the next round, meaning we can get the reimbursement as reimbursement utxos are in the next round, current round idx: {:?}, kickoff round idx: {:?}", current_round_idx, kickoff_round_idx);
1866                // we are at least on the next round, meaning we can get the reimbursement as reimbursement utxos are in the next round
1867                let reimbursement_tx = kickoff_txs
1868                    .iter()
1869                    .find(|(tx_type, _)| tx_type == &TransactionType::Reimburse)
1870                    .ok_or(eyre::eyre!("Reimburse tx not found in kickoff txs"))?;
1871                txs_to_send.push(reimbursement_tx.clone());
1872            }
1873            std::cmp::Ordering::Equal => {
1874                // first check if the kickoff is in chain
1875                if !self.rpc.is_tx_on_chain(&kickoff_txid).await? {
1876                    tracing::info!(
1877                        "Kickoff tx is not on chain, can send it, kickoff txid: {:?}",
1878                        kickoff_txid
1879                    );
1880                    let kickoff_tx = kickoff_txs
1881                        .iter()
1882                        .find(|(tx_type, _)| tx_type == &TransactionType::Kickoff)
1883                        .ok_or(eyre::eyre!("Kickoff tx not found in kickoff txs"))?;
1884
1885                    // fetch and save the LCP for if we get challenged and need to provide proof of payout later
1886                    let (_, payout_block_height) = self
1887                        .db
1888                        .get_block_info_from_hash(dbtx.as_deref_mut(), payout_blockhash)
1889                        .await?
1890                        .ok_or_eyre("Couldn't find payout blockhash in bitcoin sync")?;
1891
1892                    let move_txid = deposit_data.get_move_txid(self.config.protocol_paramset())?;
1893
1894                    let (_, _, _, citrea_idx) = self
1895                        .db
1896                        .get_payout_info_from_move_txid(dbtx.as_deref_mut(), move_txid)
1897                        .await?
1898                        .ok_or_eyre("Couldn't find payout info from move txid")?;
1899
1900                    let _ = self
1901                        .citrea_client
1902                        .fetch_validate_and_store_lcp(
1903                            payout_block_height as u64,
1904                            citrea_idx as u32,
1905                            &self.db,
1906                            dbtx.as_deref_mut(),
1907                            self.config.protocol_paramset(),
1908                        )
1909                        .await?;
1910
1911                    // sanity check
1912                    if kickoff_tx.1.compute_txid() != kickoff_txid {
1913                        return Err(eyre::eyre!("Kickoff txid mismatch for deposit outpoint: {}, kickoff txid: {:?}, computed txid: {:?}",
1914                        deposit_data.get_deposit_outpoint(), kickoff_txid, kickoff_tx.1.compute_txid()).into());
1915                    }
1916                    txs_to_send.push(kickoff_tx.clone());
1917                }
1918                // kickoff tx is on chain, check if kickoff finalizer is spent
1919                else if !self
1920                    .rpc
1921                    .is_utxo_spent(&OutPoint {
1922                        txid: kickoff_txid,
1923                        vout: UtxoVout::KickoffFinalizer.get_vout(),
1924                    })
1925                    .await?
1926                {
1927                    // kickoff finalizer is not spent, we need to send challenge timeout
1928                    tracing::info!(
1929                        "Kickoff finalizer is not spent, can send challenge timeout, kickoff txid: {:?}",
1930                        kickoff_txid
1931                    );
1932                    // first check if challenge tx was sent, then we need automation enabled to be able to answer the challenge
1933                    if self
1934                        .rpc
1935                        .is_utxo_spent(&OutPoint {
1936                            txid: kickoff_txid,
1937                            vout: UtxoVout::Challenge.get_vout(),
1938                        })
1939                        .await?
1940                    {
1941                        // challenge tx was sent, we need automation enabled to be able to answer the challenge
1942                        tracing::warn!(
1943                            "Challenge tx was sent for deposit outpoint: {:?}, but automation is not enabled, enable automation!",
1944                            deposit_data.get_deposit_outpoint()
1945                        );
1946                        return Err(eyre::eyre!("WARNING: Challenge tx was sent to kickoff connector {:?}, but automation is not enabled, enable automation!", kickoff_txid).into());
1947                    }
1948                    let challenge_timeout_tx = kickoff_txs
1949                        .iter()
1950                        .find(|(tx_type, _)| tx_type == &TransactionType::ChallengeTimeout)
1951                        .ok_or(eyre::eyre!("Challenge timeout tx not found in kickoff txs"))?;
1952                    txs_to_send.push(challenge_timeout_tx.clone());
1953                } else {
1954                    // if kickoff finalizer is spent, it is time to get the reimbursement
1955                    tracing::info!(
1956                        "Kickoff finalizer is spent, can advance the round manually to get the reimbursement, current round idx: {:?}, kickoff round idx: {:?}",
1957                        current_round_idx,
1958                        kickoff_round_idx
1959                    );
1960                    let txs = self.advance_round_manually(dbtx, current_round_idx).await?;
1961                    txs_to_send.extend(txs);
1962                }
1963            }
1964        }
1965        Ok(txs_to_send)

1968    /// For a given deposit outpoint, get the txs that are needed to reimburse the deposit.
1969    /// To avoid operator getting slashed, this function only returns the next tx that needs to be sent
1970    /// This fn can track and enable sending of these transactions during a normal reimbursement process.
1971    ///
1972    /// - First, if the current round is less than the round of the kickoff assigned to the deposit by PayoutCheckerTask, it returns the Round TX.
1973    /// - After Round tx is sent, it returns the Kickoff tx.
1974    /// - After Kickoff tx is sent, it returns the challenge timeout tx.
1975    /// - After challenge timeout tx is sent, it returns BurnUnusedKickoffConnectors tx. If challenge timeout tx is not sent, and but challenge utxo was spent, it means the kickoff was challenged, thus the fn returns an error as it cannot handle the challenge process. Automation is required to answer the challenge.
1976    /// - After all kickoff utxos are spent, and for any live kickoff, all kickoff finalizers are spent, it returns the ReadyToReimburse tx.
1977    /// - After ReadyToReimburse tx is sent, it returns the next Round tx to generate reimbursement utxos.
1978    /// - Finally, after the next round tx is sent, it returns the Reimburse tx.
1979    pub async fn get_reimbursement_txs(
1980        &self,
1981        deposit_outpoint: OutPoint,
1982    ) -> Result<Vec<(TransactionType, Transaction)>, BridgeError> {
1983        let mut dbtx = self.db.begin_transaction().await?;
1984        // first check if the deposit is in the database
1985        let (deposit_id, mut deposit_data) = self
1986            .db
1987            .get_deposit_data(Some(&mut dbtx), deposit_outpoint)
1988            .await?
1989            .ok_or_eyre(format!(
1990                "Deposit data not found for the requested deposit outpoint: {deposit_outpoint:?}, make sure you send the deposit outpoint, not the move txid."
1991            ))?;
1992
1993        tracing::info!(
1994            "Deposit data found for the requested deposit outpoint: {deposit_outpoint:?}, deposit id: {deposit_id:?}",
1995        );
1996
1997        // validate payer is operator and get payer xonly pk, payout blockhash and kickoff txid
1998        let (payout_blockhash, kickoff_txid) = self
1999            .validate_payer_is_operator(Some(&mut dbtx), deposit_id)
2000            .await?;
2001
2002        let mut current_round_idx = self.db.get_current_round_index(Some(&mut dbtx)).await?;
2003
2004        let mut txs_to_send: Vec<(TransactionType, Transaction)>;
2005
2006        loop {
2007            txs_to_send = self
2008                .get_next_txs_to_send(
2009                    Some(&mut dbtx),
2010                    &mut deposit_data,
2011                    payout_blockhash,
2012                    kickoff_txid,
2013                    current_round_idx,
2014                )
2015                .await?;
2016            if txs_to_send.is_empty() {
2017                // if no txs were returned, and we advanced the round in the db, ask for the next txs again
2018                // with the new round index
2019                let round_idx_after_operations =
2020                    self.db.get_current_round_index(Some(&mut dbtx)).await?;
2021                if round_idx_after_operations != current_round_idx {
2022                    current_round_idx = round_idx_after_operations;
2023                    continue;
2024                }
2025            }
2026            break;
2027        }
2028
2029        dbtx.commit().await?;
2030        Ok(txs_to_send)
2031    }

2033    /// Checks the current round status, and returns the next txs that are safe to send to be
2034    /// able to advance to the next round.
2035    async fn advance_round_manually(
2036        &self,
2037        mut dbtx: Option<DatabaseTransaction<'_, '_>>,
2038        round_idx: RoundIndex,
2039    ) -> Result<Vec<(TransactionType, Transaction)>, BridgeError> {
2040        if round_idx == RoundIndex::Collateral {
2041            // if current round is collateral, nothing to do except send the first round tx
2042            return self.send_next_round_tx(dbtx, round_idx).await;
2043        }
2044
2045        // get round txhandlers
2046        let context = ContractContext::new_context_for_round(
2047            self.signer.xonly_public_key,
2048            round_idx,
2049            self.config.protocol_paramset(),
2050        );
2051
2052        let txs = create_and_sign_txs(
2053            self.db.clone(),
2054            &self.signer,
2055            self.config.clone(),
2056            context,
2057            None,
2058            dbtx.as_deref_mut(),
2059        )
2060        .await?;
2061
2062        let round_tx = txs
2063            .iter()
2064            .find(|(tx_type, _)| tx_type == &TransactionType::Round)
2065            .ok_or(eyre::eyre!("Round tx not found in txs"))?;
2066
2067        if !self.rpc.is_tx_on_chain(&round_tx.1.compute_txid()).await? {
2068            return Err(eyre::eyre!("Round tx for round {:?} is not on chain, but the database shows we are on this round, error", round_idx).into());
2069        }
2070
2071        // check if ready to reimburse tx was sent
2072        let ready_to_reimburse_tx = txs
2073            .iter()
2074            .find(|(tx_type, _)| tx_type == &TransactionType::ReadyToReimburse)
2075            .ok_or(eyre::eyre!("Ready to reimburse tx not found in txs"))?;
2076
2077        let mut txs_to_send = Vec::new();
2078
2079        // to be able to send ready to reimburse tx, we need to make sure, all kickoff utxos are spent, and for all kickoffs, all kickoff finalizers are spent
2080        if !self
2081            .rpc
2082            .is_tx_on_chain(&ready_to_reimburse_tx.1.compute_txid())
2083            .await?
2084        {
2085            tracing::info!("Ready to reimburse tx for round {:?} is not on chain, checking prerequisites to see if we are able to send it
2086            Prerequisites:
2087            - all kickoff utxos are spent
2088            - for all kickoffs, all kickoff finalizers are spent
2089            ", round_idx);
2090            // get max height saved in bitcoin syncer
2091            let current_chain_height = self
2092                .db
2093                .get_max_height(dbtx.as_deref_mut())
2094                .await?
2095                .ok_or_eyre("Max block height is not found in the btc syncer database")?;
2096
2097            let round_txid = round_tx.1.compute_txid();
2098            let (unspent_kickoff_utxos, are_all_utxos_spent_finalized) = self
2099                .find_and_mark_unspent_kickoff_utxos(
2100                    dbtx.as_deref_mut(),
2101                    round_idx,
2102                    round_txid,
2103                    current_chain_height,
2104                )
2105                .await?;
2106
2107            if !unspent_kickoff_utxos.is_empty() {
2108                let burn_txs = self
2109                    .create_burn_unused_kickoff_connectors_tx(round_idx, &unspent_kickoff_utxos)
2110                    .await?;
2111                txs_to_send.extend(burn_txs);
2112            } else if !are_all_utxos_spent_finalized {
2113                // if some utxos are not spent, we need to wait until they are spent
2114                return Err(eyre::eyre!(format!(
2115                    "The transactions that spend the kickoff utxos are not yet finalized, wait until they are finalized. Finality depth: {}
2116                    If they are actually finalized, but this error is returned, it means internal bitcoin syncer is slow or stopped.",
2117                    self.config.protocol_paramset().finality_depth
2118                ))
2119                .into());
2120            } else {
2121                // every kickoff utxo is spent, but we need to check if all kickoff finalizers are spent
2122                // if not, we return and error and wait until they are spent
2123                // if all finalizers are spent, it is safe to send ready to reimburse tx
2124                self.validate_all_kickoff_finalizers_spent(
2125                    dbtx.as_deref_mut(),
2126                    round_idx,
2127                    current_chain_height,
2128                )
2129                .await?;
2130                // all finalizers and kickoff utxos are spent, it is safe to send ready to reimburse tx
2131                txs_to_send.push(ready_to_reimburse_tx.clone());
2132            }
2133        } else {
2134            // ready to reimburse tx is on chain, we need to wait for the timelock to send the next round tx
2135            // first check if next round tx is already sent, that means we can update the database
2136            txs_to_send.extend(self.send_next_round_tx(dbtx, round_idx).await?);
2137        }
2138
2139        Ok(txs_to_send)
2140    }

2376                    self.signer.xonly_public_key, round_idx, operator_xonly_pk, used_kickoffs);
2377                    Ok(DutyResult::Handled)
2378                }
2379                Duty::WatchtowerChallenge { .. } => Ok(DutyResult::Handled),
2380                Duty::SendOperatorAsserts {
2381                    kickoff_data,
2382                    deposit_data,
2383                    watchtower_challenges,
2384                    payout_blockhash,
2385                    latest_blockhash,
2386                } => {

2389                    self.send_asserts(
2390                        dbtx,

2396                    )
2397                    .await?;

2399                }
2400                Duty::VerifierDisprove { .. } => Ok(DutyResult::Handled),
2401                Duty::SendLatestBlockhash {
2402                    kickoff_data,
2403                    deposit_data,
2404                    latest_blockhash,
2405                } => {
2406                    tracing::info!("Operator {:?} called send latest blockhash with kickoff_id: {:?}, deposit_data: {:?}, latest_blockhash: {:?}", self.signer.xonly_public_key, kickoff_data, deposit_data, latest_blockhash);
2407                    self.send_latest_blockhash(dbtx, kickoff_data, deposit_data, latest_blockhash)
2408                        .await?;
2409                    Ok(DutyResult::Handled)
2410                }
2411                Duty::CheckIfKickoff {
2412                    txid,
2413                    block_height,
2414                    witness,
2415                    challenged_before: _,
2416                } => {
2417                    tracing::debug!(
2418                        "Operator {:?} called check if kickoff with txid: {:?}, block_height: {:?}",
2419                        self.signer.xonly_public_key,
2420                        txid,
2421                        block_height,
2422                    );
2423
2424                    let kickoff_data = self
2425                        .db
2426                        .get_deposit_data_with_kickoff_txid(Some(dbtx), txid)
2427                        .await?;
2428                    if let Some((deposit_data, kickoff_data)) = kickoff_data {
2429                        StateManager::<Self>::dispatch_new_kickoff_machine(
2430                            self.db.clone(),
2431                            dbtx,
2432                            kickoff_data,
2433                            block_height,
2434                            deposit_data.clone(),
2435                            witness,
2436                        )
2437                        .await?;
2438
2439                        // resend relevant txs
2440                        let context = ContractContext::new_context_for_kickoff(
2441                            kickoff_data,
2442                            deposit_data.clone(),
2443                            self.config.protocol_paramset(),
2444                        );
2445                        let signed_txs = create_and_sign_txs(
2446                            self.db.clone(),
2447                            &self.signer,
2448                            self.config.clone(),
2449                            context,
2450                            Some([0u8; 20]),
2451                            Some(dbtx),
2452                        )
2453                        .await?;
2454                        let tx_metadata = Some(TxMetadata {
2455                            tx_type: TransactionType::Dummy,
2456                            operator_xonly_pk: Some(self.signer.xonly_public_key),
2457                            round_idx: Some(kickoff_data.round_idx),
2458                            kickoff_idx: Some(kickoff_data.kickoff_idx),
2459                            deposit_outpoint: Some(deposit_data.get_deposit_outpoint()),
2460                        });
2461                        for (tx_type, signed_tx) in &signed_txs {
2462                            match *tx_type {
2463                                TransactionType::OperatorChallengeAck(_)
2464                                | TransactionType::WatchtowerChallengeTimeout(_)
2465                                | TransactionType::ChallengeTimeout
2466                                | TransactionType::DisproveTimeout
2467                                | TransactionType::Reimburse => {
2468                                    self.tx_sender
2469                                        .add_tx_to_queue(
2470                                            dbtx,
2471                                            *tx_type,
2472                                            signed_tx,
2473                                            &signed_txs,
2474                                            tx_metadata,
2475                                            &self.config,
2476                                            None,
2477                                        )
2478                                        .await?;
2479                                }
2480                                _ => {}
2481                            }
2482                        }
2483                    }
2484
2485                    Ok(DutyResult::Handled)
2486                }
2487            }
2488        }
