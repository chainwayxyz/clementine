26/// Aggregator struct.
27/// This struct is responsible for aggregating partial signatures from the verifiers.
28/// It will have in total 3 * num_operator + 1 aggregated nonces.
29/// [0] -> Aggregated nonce for the move transaction.
30/// [1..num_operator + 1] -> Aggregated nonces for the operator_takes transactions.
31/// [num_operator + 1..2 * num_operator + 1] -> Aggregated nonces for the slash_or_take transactions.
32/// [2 * num_operator + 1..3 * num_operator + 1] -> Aggregated nonces for the burn transactions.
33/// For now, we do not have the last bit.
34#[derive(Debug, Clone)]
35pub struct Aggregator {
36    pub(crate) rpc: ExtendedRpc,
37    pub(crate) db: Database,
38    pub(crate) config: BridgeConfig,
39    pub(crate) tx_sender: TxSenderClient,
40    operator_clients: Vec<ClementineOperatorClient<tonic::transport::Channel>>,
41    verifier_clients: Vec<ClementineVerifierClient<tonic::transport::Channel>>,
42    verifier_keys: Vec<PublicKey>,
43    operator_keys: Vec<XOnlyPublicKey>,
44}

47    pub async fn new(config: BridgeConfig) -> Result<Self, BridgeError> {
48        let db = Database::new(&config).await?;
49
50        let rpc = ExtendedRpc::connect(
51            config.bitcoin_rpc_url.clone(),
52            config.bitcoin_rpc_user.clone(),
53            config.bitcoin_rpc_password.clone(),
54        )
55        .await?;
56
57        let verifier_endpoints =
58            config
59                .verifier_endpoints
60                .clone()
61                .ok_or(BridgeError::ConfigError(
62                    "No verifier endpoints provided in config".into(),
63                ))?;
64
65        let operator_endpoints =
66            config
67                .operator_endpoints
68                .clone()
69                .ok_or(BridgeError::ConfigError(
70                    "No operator endpoints provided in config".into(),
71                ))?;
72
73        // Create clients to connect to all verifiers
74        let verifier_clients = rpc::get_clients(
75            verifier_endpoints,
76            ClementineVerifierClient::new,
77            &config,
78            true,
79        )
80        .await?;
81
82        // Create clients to connect to all operators
83        let operator_clients = rpc::get_clients(
84            operator_endpoints,
85            ClementineOperatorClient::new,
86            &config,
87            true,
88        )
89        .await?;
90
91        let tx_sender = TxSenderClient::new(db.clone(), "aggregator".to_string());
92
93        tracing::info!(
94            "Aggregator created with {} verifiers and {} operators",
95            verifier_clients.len(),
96            operator_clients.len(),
97        );
98
99        let operator_keys =
100            Aggregator::collect_operator_xonly_public_keys_with_clients(&operator_clients).await?;
101
102        let (_, verifier_keys) =
103            Aggregator::collect_verifier_public_keys_with_clients(&verifier_clients).await?;
104
105        Ok(Aggregator {
106            rpc,
107            db,
108            config,
109            tx_sender,
110            verifier_clients,
111            operator_clients,
112            verifier_keys,
113            operator_keys,
114        })
115    }

133    /// Collects and distributes keys to verifiers from operators and watchtowers for the new deposit
134    /// for operators: get bitvm assert winternitz public keys and watchtower challenge ack hashes
135    /// for watchtowers: get winternitz public keys for watchtower challenges
136    pub async fn collect_and_distribute_keys(
137        &self,
138        deposit_params: &DepositParams,
139    ) -> Result<(), Status> {
140        use tokio::time::{timeout, Duration};
141        const OPERATION_TIMEOUT: Duration = Duration::from_secs(500);
142
143        tracing::info!("Starting collect_and_distribute_keys");
144        let start_time = std::time::Instant::now();
145
146        let deposit_data: DepositData = deposit_params.clone().try_into()?;
147
148        // Create channels with larger capacity to prevent blocking
149        let (operator_keys_tx, operator_keys_rx) = tokio::sync::broadcast::channel(
150            deposit_data.get_num_operators() * deposit_data.get_num_verifiers(),
151        );
152        let operator_rx_handles = (0..deposit_data.get_num_verifiers())
153            .map(|_| operator_keys_rx.resubscribe())
154            .collect::<Vec<_>>();
155
156        let mut operators = self.get_participating_operators(&deposit_data).await?;
157        let operator_xonly_pks = deposit_data.get_operators();
158        let num_operators = deposit_data.get_num_operators();
159        let deposit = deposit_params.clone();
160
161        tracing::info!("Starting operator key collection");
162        let get_operators_keys_handle = tokio::spawn(async move {
163            let operator_futures = operators.iter_mut().zip(operator_xonly_pks.iter()).map(
164                |(operator_client, operator_xonly_pk)| {
165                    let deposit_params = deposit.clone();
166                    let tx = operator_keys_tx.clone();
167                    async move {
168                        tracing::debug!("Requesting keys from operator {}", operator_xonly_pk);
169                        let start = std::time::Instant::now();
170                        let operator_keys = timeout(
171                            OPERATION_TIMEOUT,
172                            operator_client.get_deposit_keys(deposit_params.clone()),
173                        )
174                        .await
175                        .map_err(|_| {
176                            Status::deadline_exceeded("Operator key retrieval timed out")
177                        })??
178                        .into_inner();
179                        tracing::debug!(
180                            "Got keys from operator {} in {:?}",
181                            operator_xonly_pk,
182                            start.elapsed()
183                        );
184
185                        tx.send(OperatorKeysWithDeposit {
186                            deposit_params: Some(deposit_params),
187                            operator_keys: Some(operator_keys),
188                            operator_xonly_pk: operator_xonly_pk.serialize().to_vec(),
189                        })
190                        .map_err(|_| Status::internal("Failed to send operator keys"))?;
191                        Ok::<_, Status>(())
192                    }
193                },
194            );
195            try_join_all(operator_futures).await
196        });
197
198        tracing::info!("Starting operator key distribution to verifiers");
199        let mut verifiers = self.get_participating_verifiers(&deposit_data).await?;
200        let distribute_operators_keys_handle = tokio::spawn(async move {
201            let distribution_futures = verifiers.iter_mut().zip(operator_rx_handles).map(
202                |(verifier, mut rx)| async move {
203                    // Only wait for expected number of messages
204                    let mut received_keys = std::collections::HashSet::new();
205                    while received_keys.len() < num_operators {
206                        tracing::debug!(
207                            "Waiting for operator key (received {}/{})",
208                            received_keys.len(),
209                            num_operators
210                        );
211                        let start = std::time::Instant::now();
212                        match timeout(OPERATION_TIMEOUT, rx.recv()).await {
213                            Ok(Ok(operator_keys)) => {
214                                let operator_xonly_pk = operator_keys.operator_xonly_pk.clone();
215                                if received_keys.insert(operator_xonly_pk.clone()) {
216                                    tracing::debug!(
217                                        "Received operator key {:?} in {:?}",
218                                        operator_xonly_pk,
219                                        start.elapsed()
220                                    );
221                                    timeout(
222                                        OPERATION_TIMEOUT,
223                                        verifier.set_operator_keys(operator_keys),
224                                    )
225                                    .await
226                                    .map_err(|_| {
227                                        Status::deadline_exceeded("Setting operator keys timed out")
228                                    })??;
229                                }
230                            }
231                            Ok(Err(_)) => break, // Channel closed
232                            Err(_) => {
233                                return Err(Status::deadline_exceeded(
234                                    "Timeout waiting for operator keys",
235                                ))
236                            }
237                        }
238                    }
239                    Ok::<_, Status>(())
240                },
241            );
242            try_join_all(distribution_futures).await
243        });
244
245        // Wait for all tasks with a timeout
246        let result = match timeout(
247            OPERATION_TIMEOUT,
248            try_join_all(vec![
249                get_operators_keys_handle,
250                distribute_operators_keys_handle,
251            ]),
252        )
253        .await
254        {
255            Ok(Ok(_)) => Ok(()),
256            Ok(Err(e)) => Err(Status::internal(format!(
257                "Failed to collect and distribute keys: {:?}",
258                e
259            ))),
260            Err(_) => Err(Status::deadline_exceeded(
261                "Overall key collection and distribution timed out",
262            )),
263        };
264
265        tracing::info!(
266            "collect_and_distribute_keys completed in {:?}",
267            start_time.elapsed()
268        );
269
270        result
271    }

273    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
274    async fn aggregate_move_partial_sigs(
275        &self,
276        deposit_data: &mut DepositData,
277        agg_nonce: &MusigAggNonce,
278        partial_sigs: Vec<MusigPartialSignature>,
279    ) -> Result<schnorr::Signature, BridgeError> {
280        let tx = builder::transaction::create_move_to_vault_txhandler(
281            deposit_data,
282            self.config.protocol_paramset(),
283        )?;
284
285        let message = Message::from_digest(
286            tx.calculate_script_spend_sighash_indexed(0, 0, bitcoin::TapSighashType::Default)?
287                .to_byte_array(),
288        );
289
290        let verifiers_public_keys = deposit_data.get_verifiers();
291
292        let final_sig = aggregate_partial_signatures(
293            &verifiers_public_keys,
294            None,
295            *agg_nonce,
296            &partial_sigs,
297            message,
298        )?;
299
300        Ok(final_sig)
301    }

303    /// Returns a list of verifier clients that are participating in the deposit.
304    pub async fn get_participating_verifiers(
305        &self,
306        deposit_data: &DepositData,
307    ) -> Result<Vec<ClementineVerifierClient<tonic::transport::Channel>>, BridgeError> {
308        let verifier_keys = self.get_verifier_keys();
309        let mut participating_verifiers = Vec::new();
310
311        let verifiers = deposit_data.get_verifiers();
312
313        for verifier_pk in verifiers {
314            if let Some(pos) = verifier_keys.iter().position(|key| key == &verifier_pk) {
315                participating_verifiers.push(self.verifier_clients[pos].clone());
316            } else {
317                return Err(BridgeError::VerifierNotFound(verifier_pk));
318            }
319        }
320
321        Ok(participating_verifiers)
322    }

324    /// Returns a list of operator clients that are participating in the deposit.
325    pub async fn get_participating_operators(
326        &self,
327        deposit_data: &DepositData,
328    ) -> Result<Vec<ClementineOperatorClient<tonic::transport::Channel>>, BridgeError> {
329        let operator_keys = self.get_operator_keys();
330        let mut participating_operators = Vec::new();
331
332        let operators = deposit_data.get_operators();
333
334        for operator_pk in operators {
335            if let Some(pos) = operator_keys.iter().position(|key| key == &operator_pk) {
336                participating_operators.push(self.operator_clients[pos].clone());
337            } else {
338                return Err(BridgeError::OperatorNotFound(operator_pk));
339            }
340        }
341
342        Ok(participating_operators)
343    }
