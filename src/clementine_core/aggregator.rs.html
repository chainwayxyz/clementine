43/// Aggregator struct.
44/// This struct is responsible for aggregating partial signatures from the verifiers.
45/// It will have in total 3 * num_operator + 1 aggregated nonces.
46/// \[0\] -> Aggregated nonce for the move transaction.
47/// [1..num_operator + 1] -> Aggregated nonces for the operator_takes transactions.
48/// [num_operator + 1..2 * num_operator + 1] -> Aggregated nonces for the slash_or_take transactions.
49/// [2 * num_operator + 1..3 * num_operator + 1] -> Aggregated nonces for the burn transactions.
50/// For now, we do not have the last bit.
51#[derive(Debug, Clone)]
52pub struct Aggregator {
53    pub(crate) rpc: ExtendedBitcoinRpc,
54    pub(crate) db: Database,
55    pub(crate) config: BridgeConfig,
56    #[cfg(feature = "automation")]
57    pub(crate) tx_sender: TxSenderClient,
58    operator_clients: Vec<ClementineOperatorClient<tonic::transport::Channel>>,
59    verifier_clients: Vec<ClementineVerifierClient<tonic::transport::Channel>>,
60    verifier_keys: Arc<RwLock<Vec<Option<PublicKey>>>>,
61    operator_keys: Arc<RwLock<Vec<Option<XOnlyPublicKey>>>>,
62}

230    /// Generic helper function to fetch keys from clients
231    async fn fetch_pubkeys_from_entities<T, C, F, Fut>(
232        &self,
233        clients: &[C],
234        keys_storage: &RwLock<Vec<Option<T>>>,
235        pubkey_fetcher: F,
236        key_type_name: &str,
237    ) -> Result<Vec<T>, BridgeError>
238    where
239        T: Clone + Send + Sync + Eq + StdHash + std::fmt::Debug,
240        C: Clone + Send + Sync,
241        F: Fn(C) -> Fut + Send + Sync,
242        Fut: Future<Output = Result<T, BridgeError>> + Send,
243    {
244        // Check if all keys are collected
245        let all_collected = {
246            let keys = keys_storage.read().await;
247            keys.iter().all(|key| key.is_some())
248        };
249
250        if !all_collected {
251            // get a write lock early, so that only one thread can try to collect keys
252            let mut keys = keys_storage.write().await;
253
254            // sanity check because we directly use indexes below
255            if keys.len() != clients.len() {
256                return Err(eyre::eyre!(
257                    "Keys storage length does not match clients length, should not happen, keys length: {}, clients length: {}",
258                    keys.len(),
259                    clients.len()
260                )
261                .into());
262            }
263
264            let key_collection_futures = clients
265                .iter()
266                .zip(keys.iter().enumerate())
267                .filter_map(|(client, (idx, key))| {
268                    if key.is_none() {
269                        Some((idx, pubkey_fetcher(client.clone())))
270                    } else {
271                        None
272                    }
273                })
274                .map(|(idx, fut)| async move { (idx, fut.await) });
275
276            let collected_keys = join_all(key_collection_futures).await;
277            let mut missing_keys = Vec::new();
278
279            // Fill in keys with the results of the futures
280            for (idx, new_key) in collected_keys {
281                match new_key {
282                    Ok(new_key) => keys[idx] = Some(new_key),
283                    Err(e) => {
284                        tracing::debug!(
285                            "Failed to collect {} {} (order in config) key: {}",
286                            key_type_name,
287                            idx,
288                            e
289                        );
290                        missing_keys.push(idx);
291                    }
292                }
293            }
294
295            // if keys are not unique, return an error if so
296            let non_none_keys: Vec<_> = keys.iter().filter_map(|key| key.as_ref()).collect();
297            let unique_keys: HashSet<_> = non_none_keys.iter().cloned().collect();
298
299            if unique_keys.len() != non_none_keys.len() {
300                let reason = format!("{key_type_name} keys are not unique: {keys:?}");
301                // reset all keys to None so that faulty keys are not used
302                for key in keys.iter_mut() {
303                    *key = None;
304                }
305                return Err(eyre::eyre!(reason).into());
306            }
307
308            // if not all keys were collected, return an error
309            if keys.iter().any(|key| key.is_none()) {
310                return Err(eyre::eyre!(
311                    "Not all {} keys were able to be collected, missing keys at indices: {:?}",
312                    key_type_name,
313                    missing_keys
314                )
315                .into());
316            }
317        }
318
319        // return all keys if they were all collected
320        Ok(keys_storage
321            .read()
322            .await
323            .iter()
324            .map(|key| key.clone().expect("should all be collected"))
325            .collect())
326    }

328    /// If all verifier keys are already collected, returns them.
329    /// Otherwise, it tries to collect them from the verifiers, saves them and returns them.
330    pub async fn fetch_verifier_keys(&self) -> Result<Vec<PublicKey>, BridgeError> {
331        self.fetch_pubkeys_from_entities(
332            &self.verifier_clients,
333            &self.verifier_keys,
334            |mut client| async move {
335                let mut request = Request::new(Empty {});
336                request.set_timeout(PUBLIC_KEY_COLLECTION_TIMEOUT);
337                let verifier_params = client.get_params(request).await?.into_inner();
338                let public_key = PublicKey::from_slice(&verifier_params.public_key)
339                    .map_err(|e| eyre::eyre!("Failed to parse verifier public key: {}", e))?;
340                Ok::<_, BridgeError>(public_key)
341            },
342            "verifier",
343        )
344        .await
345    }

372    /// Collects and distributes keys to verifiers from operators and watchtowers for the new deposit
373    /// for operators: get bitvm assert winternitz public keys and watchtower challenge ack hashes
374    /// for watchtowers: get winternitz public keys for watchtower challenges
375    pub async fn collect_and_distribute_keys(
376        &self,
377        deposit_params: &DepositParams,
378    ) -> Result<(), BridgeError> {
379        tracing::info!("Starting collect_and_distribute_keys");
380
381        let start_time = std::time::Instant::now();
382
383        let deposit_data: DepositData = deposit_params.clone().try_into()?;
384
385        // Create channels with larger capacity to prevent blocking
386        let (operator_keys_tx, operator_keys_rx) =
387            tokio::sync::broadcast::channel::<clementine::OperatorKeysWithDeposit>(
388                deposit_data.get_num_operators() * deposit_data.get_num_verifiers(),
389            );
390        let operator_rx_handles = (0..deposit_data.get_num_verifiers())
391            .map(|_| operator_keys_rx.resubscribe())
392            .collect::<Vec<_>>();
393
394        let operators = self.get_participating_operators(&deposit_data).await?;
395        let operator_clients = operators.clients();
396
397        let operator_xonly_pks = deposit_data.get_operators();
398        let deposit = deposit_params.clone();
399
400        tracing::info!("Starting operator key collection");
401        #[cfg(test)]
402        let timeout_params = self.config.test_params.timeout_params;
403        #[allow(clippy::unused_enumerate_index)]
404        let get_operators_keys_handle = tokio::spawn(timed_try_join_all(
405            OPERATOR_GET_KEYS_TIMEOUT,
406            "Operator key collection",
407            Some(operators.ids()),
408            operator_clients
409                .into_iter()
410                .zip(operator_xonly_pks.into_iter())
411                .enumerate()
412                .map(move |(_idx, (mut operator_client, operator_xonly_pk))| {
413                    let deposit_params = deposit.clone();
414                    let tx = operator_keys_tx.clone();
415                    async move {
416                        #[cfg(test)]
417                        timeout_params
418                            .hook_timeout_key_collection_operator(_idx)
419                            .await;
420
421                        let operator_keys = operator_client
422                            .get_deposit_keys(deposit_params.clone())
423                            .instrument(
424                                debug_span!("get_deposit_keys", id=%OperatorId(operator_xonly_pk)),
425                            )
426                            .await
427                            .wrap_err(Status::internal("Operator key retrieval failed"))?
428                            .into_inner();
429
430                        // A send error means that all receivers are closed,
431                        // receivers only close if they have an error (while
432                        // loop condition)
433                        // We don't care about the result of the send, we
434                        // only care about the error on the other side.
435                        // Ignore this error, and let the other side's error
436                        // propagate.
437                        let _ = tx.send(OperatorKeysWithDeposit {
438                            deposit_params: Some(deposit_params),
439                            operator_keys: Some(operator_keys),
440                            operator_xonly_pk: operator_xonly_pk.serialize().to_vec(),
441                        });
442
443                        Ok(())
444                    }
445                }),
446        ));
447
448        tracing::info!("Starting operator key distribution to verifiers");
449        let verifiers = self.get_participating_verifiers(&deposit_data).await?;
450
451        let verifier_clients = verifiers.clients();
452        let num_operators = deposit_data.get_num_operators();
453
454        let verifier_ids = verifiers.ids();
455
456        #[cfg(test)]
457        let timeout_params = self.config.test_params.timeout_params;
458        #[allow(clippy::unused_enumerate_index)]
459        let distribute_operators_keys_handle = tokio::spawn(timed_try_join_all(
460            VERIFIER_SEND_KEYS_TIMEOUT,
461            "Verifier key distribution",
462            Some(verifier_ids.clone()),
463            verifier_clients
464                .into_iter()
465                .zip(operator_rx_handles)
466                .zip(verifier_ids)
467                .enumerate()
468                .map(
469                    move |(_idx, ((mut verifier, mut rx), verifier_id))| async move {
470                        #[cfg(test)]
471                        timeout_params
472                            .hook_timeout_key_distribution_verifier(_idx)
473                            .await;
474
475                        // Only wait for expected number of messages
476                        let mut received_keys = std::collections::HashSet::new();
477                        while received_keys.len() < num_operators {
478                            tracing::debug!(
479                                "Waiting for operator key (received {}/{})",
480                                received_keys.len(),
481                                num_operators
482                            );
483
484                            // This will not block forever because of the timeout on the join all.
485                            let operator_keys = rx
486                            .recv()
487                            .instrument(debug_span!("operator_keys_recv"))
488                            .await
489                            .wrap_err(Status::internal(
490                                "Operator broadcast channels closed before all keys were received",
491                            ))?;
492
493                            let operator_xonly_pk = operator_keys.operator_xonly_pk.clone();
494
495                            if !received_keys.insert(operator_xonly_pk.clone()) {
496                                continue;
497                            }
498
499                            timed_request(
500                                VERIFIER_SEND_KEYS_TIMEOUT,
501                                &format!("Setting operator keys for {verifier_id}"),
502                                async {
503                                    Ok(verifier
504                                        .set_operator_keys(operator_keys)
505                                        .await
506                                        .wrap_err_with(|| {
507                                            Status::internal(format!(
508                                                "Failed to set operator keys for {verifier_id}",
509                                            ))
510                                        }))
511                                },
512                            )
513                            .await??;
514                        }
515                        Ok::<_, BridgeError>(())
516                    },
517                ),
518        ));
519
520        // Wait for all tasks to complete
521        let (get_operators_keys_result, distribute_operators_keys_result) =
522            tokio::try_join!(get_operators_keys_handle, distribute_operators_keys_handle)
523                .wrap_err(Status::internal("Task join error in key distribution"))?;
524
525        get_operators_keys_result?;
526        distribute_operators_keys_result?;
527
528        tracing::info!(
529            "collect_and_distribute_keys completed in {:?}",
530            start_time.elapsed()
531        );
532
533        Ok(())
534    }

536    /// Returns a list of verifier clients that are participating in the deposit.
537    pub async fn get_participating_verifiers(
538        &self,
539        deposit_data: &DepositData,
540    ) -> Result<ParticipatingVerifiers, BridgeError> {
541        let verifier_keys = self.fetch_verifier_keys().await?;
542        let mut participating_verifiers = Vec::new();
543
544        let verifiers = deposit_data.get_verifiers();
545
546        for verifier_pk in verifiers {
547            if let Some(pos) = verifier_keys.iter().position(|key| key == &verifier_pk) {
548                participating_verifiers
549                    .push((self.verifier_clients[pos].clone(), VerifierId(verifier_pk)));
550            } else {
551                tracing::error!(
552                    "Verifier public key not found. Deposit data verifier keys: {:?}, self verifier keys: {:?}",
553                    deposit_data.get_verifiers(),
554                    self.verifier_keys
555                );
556                return Err(BridgeError::VerifierNotFound(verifier_pk));
557            }
558        }
559
560        Ok(ParticipatingVerifiers::new(participating_verifiers))
561    }

585    /// Retrieves the status of all entities (operators and verifiers) and restarts background tasks if needed.
586    /// Returns a vector of EntityStatusWithId. Only returns an error if restarting tasks fails when requested.
587    pub async fn get_entity_statuses(
588        &self,
589        restart_tasks: bool,
590    ) -> Result<Vec<EntityStatusWithId>, BridgeError> {
591        tracing::debug!("Getting entities status");
592
593        let operator_clients = self.get_operator_clients();
594        let verifier_clients = self.get_verifier_clients();
595        tracing::debug!("Operator clients: {:?}", operator_clients.len());
596
597        // try to reach all operators and verifiers to collect keys, but do not return err if some of them can't be reached
598        let _ = self.fetch_operator_keys().await;
599        let _ = self.fetch_verifier_keys().await;
600
601        let operator_keys = self.operator_keys.read().await.clone();
602        let verifier_keys = self.verifier_keys.read().await.clone();
603
604        // we will only ask status of entities that we could collect keys for, others are unreachable
605
606        let operator_status = join_all(
607            operator_clients
608                .iter()
609                .zip(operator_keys.iter())
610                // filter out operators that have None as key
611                .filter_map(|(client, key)| key.map(|key| (client, key)))
612                .map(|(client, key)| {
613                    let mut client = client.clone();
614                    async move {
615                        tracing::debug!("Getting operator status for {:?}", key);
616                        let mut request = Request::new(Empty {});
617                        request.set_timeout(ENTITY_STATUS_POLL_TIMEOUT);
618                        let response = client.get_current_status(request).await;
619
620                        EntityStatusWithId {
621                            entity_id: Some(RPCEntityId {
622                                kind: EntityType::Operator as i32,
623                                id: key.to_string(),
624                            }),
625                            status_result: match response {
626                                Ok(response) => Some(StatusResult::Status(response.into_inner())),
627                                Err(e) => Some(StatusResult::Err(clementine::EntityError {
628                                    error: e.to_string(),
629                                })),
630                            },
631                        }
632                    }
633                }),
634        )
635        .await;
636
637        let verifier_status = join_all(
638            verifier_clients
639                .iter()
640                .zip(verifier_keys.iter())
641                .filter_map(|(client, key)| key.map(|key| (client, key)))
642                .map(|(client, key)| {
643                    let mut client = client.clone();
644                    async move {
645                        tracing::debug!("Getting verifier status for {:?}", key);
646                        let mut request = Request::new(Empty {});
647                        request.set_timeout(ENTITY_STATUS_POLL_TIMEOUT);
648                        let response = client.get_current_status(request).await;
649
650                        EntityStatusWithId {
651                            entity_id: Some(RPCEntityId {
652                                kind: EntityType::Verifier as i32,
653                                id: key.to_string(),
654                            }),
655                            status_result: match response {
656                                Ok(response) => Some(StatusResult::Status(response.into_inner())),
657                                Err(e) => Some(StatusResult::Err(clementine::EntityError {
658                                    error: e.to_string(),
659                                })),
660                            },
661                        }
662                    }
663                }),
664        )
665        .await;
666
667        // Combine operator and verifier status into a single vector
668        let mut entity_statuses = operator_status;
669        entity_statuses.extend(verifier_status);
670
671        // try to restart background tasks if needed, with a timeout
672        if restart_tasks {
673            let operator_tasks = operator_clients
674                .iter()
675                .zip(operator_keys.iter())
676                .filter_map(|(client, key)| key.map(|key| (client, key)))
677                .map(|(client, key)| {
678                    let mut client = client.clone();
679                    async move {
680                        let mut request = Request::new(Empty {});
681                        request.set_timeout(RESTART_BACKGROUND_TASKS_TIMEOUT);
682                        client
683                            .restart_background_tasks(Request::new(Empty {}))
684                            .await
685                            .wrap_err_with(|| {
686                                Status::internal(format!(
687                                    "Failed to restart background tasks for operator {key}"
688                                ))
689                            })
690                    }
691                });
692
693            let verifier_tasks = verifier_clients
694                .iter()
695                .zip(verifier_keys.iter())
696                .filter_map(|(client, key)| key.map(|key| (client, key)))
697                .map(|(client, key)| {
698                    let mut client = client.clone();
699                    async move {
700                        let mut request = Request::new(Empty {});
701                        request.set_timeout(RESTART_BACKGROUND_TASKS_TIMEOUT);
702                        client
703                            .restart_background_tasks(Request::new(Empty {}))
704                            .await
705                            .wrap_err_with(|| {
706                                Status::internal(format!(
707                                    "Failed to restart background tasks for verifier {key}"
708                                ))
709                            })
710                    }
711                });
712
713            let (operator_results, verifier_results) = futures::join!(
714                futures::future::join_all(operator_tasks),
715                futures::future::join_all(verifier_tasks)
716            );
717            // Log any errors from restart_background_tasks rpc call
718            for result in operator_results
719                .into_iter()
720                .chain(verifier_results.into_iter())
721            {
722                if let Err(e) = result {
723                    tracing::error!("{e:?}");
724                }
725            }
726        }
727        Ok(entity_statuses)
728    }
729}

731/// Aggregator server wrapper that manages background tasks.
732#[derive(Debug)]
733pub struct AggregatorServer {
734    pub aggregator: Aggregator,
735    background_tasks: crate::task::manager::BackgroundTaskManager,
736}
737
738impl AggregatorServer {
739    pub async fn new(config: BridgeConfig) -> Result<Self, BridgeError> {
740        let aggregator = Aggregator::new(config.clone()).await?;
741        let background_tasks = crate::task::manager::BackgroundTaskManager::default();
742
743        Ok(Self {
744            aggregator,
745            background_tasks,
746        })
747    }
748
749    /// Starts the background tasks for the aggregator.
750    /// If called multiple times, it will restart only the tasks that are not already running.
751    pub async fn start_background_tasks(&self) -> Result<(), BridgeError> {
752        // Start the aggregator metric publisher task
753        self.background_tasks
754            .ensure_task_looping(
755                crate::task::aggregator_metric_publisher::AggregatorMetricPublisher::new(
756                    self.aggregator.clone(),
757                )
758                .await?
759                .with_delay(AGGREGATOR_METRIC_PUBLISHER_POLL_DELAY),
760            )
761            .await;
762
763        tracing::info!("Aggregator metric publisher task started");
764
765        Ok(())
766    }
767
768    pub async fn shutdown(&mut self) {
769        self.background_tasks.graceful_shutdown().await;
770    }
771}
772
773impl Deref for AggregatorServer {
774    type Target = Aggregator;
775
776    fn deref(&self) -> &Self::Target {
777        &self.aggregator
778    }
