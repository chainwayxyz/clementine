15    /// Returns the last deposit index.
16    /// If no deposits exist, returns None
17    pub async fn get_last_deposit_idx(
18        &self,
19        tx: Option<DatabaseTransaction<'_, '_>>,
20    ) -> Result<Option<u32>, BridgeError> {
21        let query = sqlx::query_as::<_, (i32,)>("SELECT COALESCE(MAX(idx), -1) FROM withdrawals");
22        let result = execute_query_with_tx!(self.connection, tx, query, fetch_one)?;
23        if result.0 == -1 {
24            Ok(None)
25        } else {
26            Ok(Some(result.0 as u32))
27        }
28    }

30    /// Returns the last withdrawal index where withdrawal_utxo_txid exists.
31    /// If no withdrawals with UTXOs exist, returns None.
32    pub async fn get_last_withdrawal_idx(
33        &self,
34        tx: Option<DatabaseTransaction<'_, '_>>,
35    ) -> Result<Option<u32>, BridgeError> {
36        let query = sqlx::query_as::<_, (i32,)>(
37            "SELECT COALESCE(MAX(idx), -1) FROM withdrawals WHERE withdrawal_utxo_txid IS NOT NULL",
38        );
39        let result = execute_query_with_tx!(self.connection, tx, query, fetch_one)?;
40        if result.0 == -1 {
41            Ok(None)
42        } else {
43            Ok(Some(result.0 as u32))
44        }
45    }

47    pub async fn set_move_to_vault_txid_from_citrea_deposit(
48        &self,
49        tx: Option<DatabaseTransaction<'_, '_>>,
50        citrea_idx: u32,
51        move_to_vault_txid: &Txid,
52    ) -> Result<(), BridgeError> {
53        let query = sqlx::query(
54            "INSERT INTO withdrawals (idx, move_to_vault_txid)
55             VALUES ($1, $2)
56             ON CONFLICT (idx) DO UPDATE
57             SET move_to_vault_txid = $2",
58        )
59        .bind(i32::try_from(citrea_idx).wrap_err("Failed to convert citrea index to i32")?)
60        .bind(TxidDB(*move_to_vault_txid));
61
62        execute_query_with_tx!(self.connection, tx, query, execute)?;
63        Ok(())
64    }

66    pub async fn set_replacement_deposit_move_txid(
67        &self,
68        tx: Option<DatabaseTransaction<'_, '_>>,
69        old_move_txid: Txid,
70        new_move_txid: Txid,
71    ) -> Result<(), BridgeError> {
72        let query = sqlx::query(
73            "UPDATE withdrawals
74             SET move_to_vault_txid = $2
75             WHERE move_to_vault_txid = $1
76             RETURNING idx",
77        )
78        .bind(TxidDB(old_move_txid))
79        .bind(TxidDB(new_move_txid));
80
81        let result = execute_query_with_tx!(self.connection, tx, query, fetch_optional)?;
82
83        if result.is_none() {
84            return Err(BridgeError::Error(format!(
85                "No withdrawal found with move_to_vault_txid: {}",
86                old_move_txid
87            )));
88        }
89
90        Ok(())
91    }

109        .bind(
110            i32::try_from(withdrawal_utxo.vout)

112        )
113        .bind(
114            i32::try_from(withdrawal_batch_proof_bitcoin_block_height)
115                .wrap_err("Failed to convert withdrawal batch proof bitcoin block height to i32")?,
116        );
117
118        execute_query_with_tx!(self.connection, tx, query, execute)?;
119        Ok(())

122    pub async fn get_withdrawal_utxo_from_citrea_withdrawal(
123        &self,
124        tx: Option<DatabaseTransaction<'_, '_>>,
125        citrea_idx: u32,
126    ) -> Result<Option<OutPoint>, BridgeError> {
127        let query = sqlx::query_as::<_, (Option<TxidDB>, Option<i32>)>(
128            "SELECT w.withdrawal_utxo_txid, w.withdrawal_utxo_vout
129             FROM withdrawals w
130             WHERE w.idx = $1",
131        )
132        .bind(i32::try_from(citrea_idx).wrap_err("Failed to convert citrea index to i32")?);
133
134        let results = execute_query_with_tx!(self.connection, tx, query, fetch_optional)?;
135
136        results
137            .map(|(txid, vout)| match (txid, vout) {
138                (Some(txid), Some(vout)) => Ok(OutPoint {
139                    txid: txid.0,
140                    vout: u32::try_from(vout)
141                        .wrap_err("Failed to convert withdrawal utxo vout to u32")?,
142                }),
143                _ => Err(BridgeError::Error("Unexpected null value".to_string())),
144            })
145            .transpose()
146    }

148    /// Returns the withdrawal indexes and their spending txid for the given
149    /// block id.
150    pub async fn get_payout_txs_for_withdrawal_utxos(
151        &self,
152        tx: Option<DatabaseTransaction<'_, '_>>,
153        block_id: u32,
154    ) -> Result<Vec<(u32, Txid)>, BridgeError> {
155        let query = sqlx::query_as::<_, (i32, TxidDB)>(
156            "SELECT w.idx, bsu.spending_txid
157             FROM withdrawals w
158             JOIN bitcoin_syncer_spent_utxos bsu
159                ON bsu.txid = w.withdrawal_utxo_txid
160                AND bsu.vout = w.withdrawal_utxo_vout
161             WHERE bsu.block_id = $1",
162        )
163        .bind(i32::try_from(block_id).wrap_err("Failed to convert block id to i32")?);
164
165        let results = execute_query_with_tx!(self.connection, tx, query, fetch_all)?;
166
167        results
168            .into_iter()
169            .map(|(idx, txid)| {
170                Ok((
171                    u32::try_from(idx).wrap_err("Failed to convert withdrawal index to u32")?,
172                    txid.0,
173                ))
174            })
175            .collect()
176    }

178    /// Sets the given payout txs' txid and operator index for the given index.
179    pub async fn set_payout_txs_and_payer_operator_xonly_pk(
180        &self,
181        tx: Option<DatabaseTransaction<'_, '_>>,
182        payout_txs_and_payer_operator_xonly_pk: Vec<(
183            u32,
184            Txid,
185            XOnlyPublicKey,
186            bitcoin::BlockHash,
187        )>,
188    ) -> Result<(), BridgeError> {
189        if payout_txs_and_payer_operator_xonly_pk.is_empty() {
190            return Ok(());
191        }
192        // Convert all values first, propagating any errors
193        let converted_values: Result<Vec<_>, BridgeError> = payout_txs_and_payer_operator_xonly_pk
194            .iter()
195            .map(|(idx, txid, operator_xonly_pk, block_hash)| {
196                Ok((
197                    i32::try_from(*idx).wrap_err("Failed to convert payout index to i32")?,
198                    TxidDB(*txid),
199                    XOnlyPublicKeyDB(*operator_xonly_pk),
200                    BlockHashDB(*block_hash),
201                ))
202            })
203            .collect();
204        let converted_values = converted_values?;
205
206        let mut query_builder = QueryBuilder::new(
207            "UPDATE withdrawals AS w SET
208                payout_txid = c.payout_txid,
209                payout_payer_operator_xonly_pk = c.payout_payer_operator_xonly_pk,
210                payout_tx_blockhash = c.payout_tx_blockhash
211                FROM (",
212        );
213
214        query_builder.push_values(
215            converted_values.into_iter(),
216            |mut b, (idx, txid, operator_xonly_pk, block_hash)| {
217                b.push_bind(idx)
218                    .push_bind(txid)
219                    .push_bind(operator_xonly_pk)
220                    .push_bind(block_hash);
221            },
222        );
223
224        query_builder
225            .push(") AS c(idx, payout_txid, payout_payer_operator_xonly_pk, payout_tx_blockhash) WHERE w.idx = c.idx");
226
227        let query = query_builder.build();
228        execute_query_with_tx!(self.connection, tx, query, execute)?;
229
230        Ok(())
231    }

253    pub async fn get_first_unhandled_payout_by_operator_xonly_pk(
254        &self,
255        tx: Option<DatabaseTransaction<'_, '_>>,
256        operator_xonly_pk: XOnlyPublicKey,
257    ) -> Result<Option<(u32, Txid, BlockHash)>, BridgeError> {
258        let query = sqlx::query_as::<_, (i32, Option<TxidDB>, Option<BlockHashDB>)>(
259            "SELECT w.idx, w.move_to_vault_txid, w.payout_tx_blockhash
260             FROM withdrawals w
261             WHERE w.payout_txid IS NOT NULL
262                AND w.is_payout_handled = FALSE
263                AND w.payout_payer_operator_xonly_pk = $1
264                ORDER BY w.idx ASC
265             LIMIT 1",
266        )
267        .bind(XOnlyPublicKeyDB(operator_xonly_pk));
268
269        let results = execute_query_with_tx!(self.connection, tx, query, fetch_optional)?;
270
271        results
272            .map(|(citrea_idx, move_to_vault_txid, payout_tx_blockhash)| {
273                Ok((
274                    u32::try_from(citrea_idx).wrap_err("Failed to convert citrea index to u32")?,
275                    move_to_vault_txid
276                        .expect("move_to_vault_txid Must be Some")
277                        .0,
278                    payout_tx_blockhash
279                        .expect("payout_tx_blockhash Must be Some")
280                        .0,
281                ))
282            })
283            .transpose()
284    }

320#[cfg(test)]
321mod tests {
322    use crate::{
323        database::Database,
324        test::common::{create_test_config_with_thread_name, generate_random_xonly_pk},
325    };
326    use bitcoin::{hashes::Hash, BlockHash, Txid};
327
328    #[tokio::test]
329    async fn set_get_payout_txs_from_citrea_withdrawal() {
330        let config = create_test_config_with_thread_name().await;
331        let db = Database::new(&config).await.unwrap();
332
333        let txid = Txid::from_byte_array([0x45; 32]);
334        let index = 0x1F;
335        let operator_xonly_pk = generate_random_xonly_pk();
336        let utxo = bitcoin::OutPoint {
337            txid: bitcoin::Txid::from_byte_array([0x45; 32]),
338            vout: 0,
339        };
340
341        let mut dbtx = db.begin_transaction().await.unwrap();
342
343        let block_id = db
344            .add_block_info(
345                Some(&mut dbtx),
346                &BlockHash::all_zeros(),
347                &BlockHash::all_zeros(),
348                utxo.vout,
349            )
350            .await
351            .unwrap();
352        db.add_txid_to_block(&mut dbtx, block_id, &txid)
353            .await
354            .unwrap();
355        db.insert_spent_utxo(&mut dbtx, block_id, &txid, &utxo.txid, utxo.vout.into())
356            .await
357            .unwrap();
358
359        assert!(db
360            .get_withdrawal_utxo_from_citrea_withdrawal(Some(&mut dbtx), index)
361            .await
362            .unwrap()
363            .is_none());
364        db.set_move_to_vault_txid_from_citrea_deposit(Some(&mut dbtx), index, &txid)
365            .await
366            .unwrap();
367        db.set_withdrawal_utxo_from_citrea_withdrawal(Some(&mut dbtx), index, utxo, block_id)
368            .await
369            .unwrap();
370
371        let block_hash = BlockHash::all_zeros();
372
373        db.set_payout_txs_and_payer_operator_xonly_pk(
374            Some(&mut dbtx),
375            vec![(index, txid, operator_xonly_pk, block_hash)],
376        )
377        .await
378        .unwrap();
379
380        let txs = db
381            .get_payout_txs_for_withdrawal_utxos(Some(&mut dbtx), block_id)
382            .await
383            .unwrap();
384
385        assert_eq!(txs.len(), 1);
386        assert_eq!(txs[0].0, index);
387        assert_eq!(txs[0].1, txid);
388
389        let withdrawal_utxo = db
390            .get_withdrawal_utxo_from_citrea_withdrawal(Some(&mut dbtx), index)
391            .await
392            .unwrap()
393            .unwrap();
394        assert_eq!(withdrawal_utxo, utxo);
395    }
