1//! # State Machine Related Database Operations
2//!
3//! This module includes database functions for persisting and loading state machines.
4
5use bitcoin::XOnlyPublicKey;
6
7use super::{wrapper::XOnlyPublicKeyDB, Database, DatabaseTransaction};
8use crate::errors::BridgeError;
9use crate::execute_query_with_tx;
10
11impl Database {
12    /// Saves state machines to the database with the current block height
13    ///
14    /// # Arguments
15    ///
16    /// * `tx` - Optional database transaction
17    /// * `kickoff_machines` - Vector of (state_json, kickoff_id, owner_type, dirty) tuples for kickoff machines
18    /// * `round_machines` - Vector of (state_json, operator_xonly_pk, owner_type, dirty) tuples for round machines
19    /// * `block_height` - Current block height
20    ///
21    /// # Errors
22    ///
23    /// Returns a `BridgeError` if the database operation fails
24    pub async fn save_state_machines(
25        &self,
26        mut tx: Option<DatabaseTransaction<'_, '_>>,
27        kickoff_machines: Vec<(String, String, bool)>,
28        round_machines: Vec<(String, XOnlyPublicKey, bool)>,
29        block_height: i32,
30        owner_type: &str,
31    ) -> Result<(), BridgeError> {
32        // Save kickoff machines that are dirty
33        for (state_json, kickoff_id, dirty) in kickoff_machines {
34            // Skip machines that are not dirty
35            if !dirty {
36                continue;
37            }
38
39            let query = sqlx::query(
40                "INSERT INTO state_machines (
41                    machine_type,
42                    state_json,
43                    kickoff_id,
44                    owner_type,
45                    block_height,
46                    created_at,
47                    updated_at
48                ) VALUES ($1, $2, $3, $4, $5, NOW(), NOW())
49                ON CONFLICT (machine_type, kickoff_id, owner_type)
50                DO UPDATE SET
51                    state_json = EXCLUDED.state_json,
52                    block_height = EXCLUDED.block_height,
53                    updated_at = NOW()",
54            )
55            .bind("kickoff")
56            .bind(&state_json)
57            .bind(kickoff_id)
58            .bind(owner_type)
59            .bind(block_height);
60
61            execute_query_with_tx!(self.connection, tx.as_deref_mut(), query, execute)?;
62        }
63
64        // Save round machines that are dirty
65        for (state_json, operator_xonly_pk, dirty) in round_machines {
66            // Skip machines that are not dirty
67            if !dirty {
68                continue;
69            }
70
71            let query = sqlx::query(
72                "INSERT INTO state_machines (
73                    machine_type,
74                    state_json,
75                    operator_xonly_pk,
76                    owner_type,
77                    block_height,
78                    created_at,
79                    updated_at
80                ) VALUES ($1, $2, $3, $4, $5, NOW(), NOW())
81                ON CONFLICT (machine_type, operator_xonly_pk, owner_type)
82                DO UPDATE SET
83                    state_json = EXCLUDED.state_json,
84                    block_height = EXCLUDED.block_height,
85                    updated_at = NOW()",
86            )
87            .bind("round")
88            .bind(&state_json)
89            .bind(XOnlyPublicKeyDB(operator_xonly_pk))
90            .bind(owner_type)
91            .bind(block_height);
92
93            execute_query_with_tx!(self.connection, tx.as_deref_mut(), query, execute)?;
94        }
95
96        // Update state manager status
97        let query = sqlx::query(
98            "INSERT INTO state_manager_status (
99                owner_type,
100                next_height_to_process,
101                updated_at
102            ) VALUES ($1, $2, NOW())
103            ON CONFLICT (owner_type)
104            DO UPDATE SET
105                next_height_to_process = EXCLUDED.next_height_to_process,
106                updated_at = NOW()",
107        )
108        .bind(owner_type)
109        .bind(block_height);
110
111        execute_query_with_tx!(self.connection, tx, query, execute)?;
112
113        Ok(())
114    }

116    /// Gets the last processed block height
117    ///
118    /// # Arguments
119    ///
120    /// * `tx` - Optional database transaction
121    ///
122    /// # Errors
123    ///
124    /// Returns a `BridgeError` if the database operation fails
125    pub async fn get_next_height_to_process(
126        &self,
127        tx: Option<DatabaseTransaction<'_, '_>>,
128        owner_type: &str,
129    ) -> Result<Option<i32>, BridgeError> {
130        let query = sqlx::query_as(
131            "SELECT next_height_to_process FROM state_manager_status WHERE owner_type = $1",
132        )
133        .bind(owner_type);
134
135        let result: Option<(i32,)> =
136            execute_query_with_tx!(self.connection, tx, query, fetch_optional)?;
137
138        Ok(result.map(|(height,)| height))
139    }
140
141    /// Loads kickoff machines from the database
142    ///
143    /// # Arguments
144    ///
145    /// * `tx` - Optional database transaction
146    /// * `owner_type` - The owner type to filter by
147    ///
148    /// # Errors
149    ///
150    /// Returns a `BridgeError` if the database operation fails
151    pub async fn load_kickoff_machines(
152        &self,
153        tx: Option<DatabaseTransaction<'_, '_>>,
154        owner_type: &str,
155    ) -> Result<Vec<(String, String, i32)>, BridgeError> {
156        let query = sqlx::query_as(
157            "SELECT
158                state_json,
159                kickoff_id,
160                block_height
161            FROM state_machines
162            WHERE machine_type = 'kickoff' AND owner_type = $1",
163        )
164        .bind(owner_type);
165
166        let results = execute_query_with_tx!(self.connection, tx, query, fetch_all)?;
167
168        Ok(results)
169    }
170
171    /// Loads round machines from the database
172    ///
173    /// # Arguments
174    ///
175    /// * `tx` - Optional database transaction
176    /// * `owner_type` - The owner type to filter by
177    ///
178    /// # Errors
179    ///
180    /// Returns a `BridgeError` if the database operation fails
181    pub async fn load_round_machines(
182        &self,
183        tx: Option<DatabaseTransaction<'_, '_>>,
184        owner_type: &str,
185    ) -> Result<Vec<(String, XOnlyPublicKey, i32)>, BridgeError> {
186        let query = sqlx::query_as(
187            "SELECT
188                state_json,
189                operator_xonly_pk,
190                block_height
191            FROM state_machines
192            WHERE machine_type = 'round' AND owner_type = $1",
193        )
194        .bind(owner_type);
195
196        let results: Vec<(String, XOnlyPublicKeyDB, i32)> =
197            execute_query_with_tx!(self.connection, tx, query, fetch_all)?;
198
199        Ok(results
200            .into_iter()
201            .map(|(state_json, operator_xonly_pk, block_height)| {
202                (state_json, operator_xonly_pk.0, block_height)
203            })
204            .collect())
205    }
206}

208#[cfg(test)]
209mod tests {
210    use super::*;
211    use crate::test::common::*;
212
213    #[tokio::test]
214    async fn test_save_and_load_state_machines() {
215        let config = create_test_config_with_thread_name().await;
216        let db = Database::new(&config).await.unwrap();
217
218        let xonly_pk1 = generate_random_xonly_pk();
219        let xonly_pk2 = generate_random_xonly_pk();
220
221        // Create test data with owner_type
222        let owner_type = "test_owner";
223        let kickoff_machines = vec![
224            (
225                "kickoff_state_1".to_string(),
226                "kickoff_id_1".to_string(),
227                true, // dirty
228            ),
229            (
230                "kickoff_state_2".to_string(),
231                "kickoff_id_2".to_string(),
232                true, // dirty
233            ),
234        ];
235
236        let round_machines = vec![
237            (
238                "round_state_1".to_string(),
239                xonly_pk1,
240                true, // dirty
241            ),
242            (
243                "round_state_2".to_string(),
244                xonly_pk2,
245                true, // dirty
246            ),
247        ];
248
249        // Save state machines
250        db.save_state_machines(
251            None,
252            kickoff_machines.clone(),
253            round_machines.clone(),
254            123,
255            owner_type,
256        )
257        .await
258        .unwrap();
259
260        // Check last processed block height
261        let block_height = db
262            .get_next_height_to_process(None, owner_type)
263            .await
264            .unwrap();
265        assert_eq!(block_height, Some(123));
266
267        // Load kickoff machines
268        let loaded_kickoff = db.load_kickoff_machines(None, owner_type).await.unwrap();
269        assert_eq!(loaded_kickoff.len(), 2);
270        assert_eq!(loaded_kickoff[0].0, "kickoff_state_1");
271        assert_eq!(loaded_kickoff[0].1, "kickoff_id_1");
272        assert_eq!(loaded_kickoff[0].2, 123);
273
274        // Load round machines
275        let loaded_round = db.load_round_machines(None, owner_type).await.unwrap();
276        assert_eq!(loaded_round.len(), 2);
277        assert_eq!(loaded_round[0].0, "round_state_1");
278        assert_eq!(loaded_round[0].1, xonly_pk1);
279        assert_eq!(loaded_round[0].2, 123);
280
281        // Test dirty flag by updating only one machine
282        let kickoff_machines_update = vec![
283            (
284                "kickoff_state_1_updated".to_string(),
285                "kickoff_id_1".to_string(),
286                true, // dirty - will be updated
287            ),
288            (
289                "kickoff_state_2".to_string(),
290                "kickoff_id_2".to_string(),
291                false, // not dirty - won't be updated
292            ),
293        ];
294
295        db.save_state_machines(None, kickoff_machines_update, vec![], 124, owner_type)
296            .await
297            .unwrap();
298
299        // Load kickoff machines again to verify only one was updated
300        let loaded_kickoff = db.load_kickoff_machines(None, owner_type).await.unwrap();
301        assert_eq!(loaded_kickoff.len(), 2);
302
303        // Find the machine with kickoff_id_1 - it should be updated
304        let machine1 = loaded_kickoff
305            .iter()
306            .find(|m| m.1 == "kickoff_id_1")
307            .unwrap();
308        assert_eq!(machine1.0, "kickoff_state_1_updated");
309
310        // Find the machine with kickoff_id_2 - it should not be updated
311        let machine2 = loaded_kickoff
312            .iter()
313            .find(|m| m.1 == "kickoff_id_2")
314            .unwrap();
315        assert_eq!(machine2.0, "kickoff_state_2");
316    }
