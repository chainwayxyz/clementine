25/// Macro to reduce boilerplate for [`impl_text_wrapper_custom`].
26///
27/// Implements the Type, Encode and Decode traits for a wrapper type.
28/// Assumes the type is declared.
29macro_rules! impl_text_wrapper_base {
30    ($wrapper:ident, $inner:ty, $encode:expr, $decode:expr) => {
31        impl sqlx::Type<sqlx::Postgres> for $wrapper {
32            fn type_info() -> sqlx::postgres::PgTypeInfo {
33                sqlx::postgres::PgTypeInfo::with_name("TEXT")
34            }
35        }
36
37        impl Encode<'_, Postgres> for $wrapper {
38            fn encode_by_ref(
39                &self,
40                buf: &mut PgArgumentBuffer,
41            ) -> Result<sqlx::encode::IsNull, sqlx::error::BoxDynError> {
42                let s = $encode(&self.0);
43                <&str as Encode<Postgres>>::encode_by_ref(&s.as_str(), buf)
44            }
45        }
46
47        impl<'r> Decode<'r, Postgres> for $wrapper {
48            fn decode(value: PgValueRef<'r>) -> Result<Self, sqlx::error::BoxDynError> {
49                let s = <&str as Decode<Postgres>>::decode(value)?;
50                Ok(Self($decode(s)?))
51            }
52        }
53    };
54}

56/// Macro for implementing text-based SQL wrapper types with custom encoding/decoding
57///
58/// # Parameters
59/// - `$wrapper`: The name of the wrapper type to create
60/// - `$inner`: The inner type being wrapped
61/// - `$encode`: Expression for converting inner type to string
62/// - `$decode`: Expression for converting string back to inner type
63///
64/// The macro creates a new type that wraps the inner type and implements:
65/// - SQLx Type trait to indicate TEXT column type
66/// - SQLx Encode trait for converting to database format
67/// - SQLx Decode trait for converting from database format
68macro_rules! impl_text_wrapper_custom {
69    // Default case (include serde)
70    ($wrapper:ident, $inner:ty, $encode:expr, $decode:expr) => {
71        impl_text_wrapper_custom!($wrapper, $inner, $encode, $decode, true);
72    };
73
74    // true case - with serde
75    ($wrapper:ident, $inner:ty, $encode:expr, $decode:expr, true) => {
76        #[derive(sqlx::FromRow, Debug, Clone, Serialize, Deserialize, PartialEq)]
77        pub struct $wrapper(pub $inner);
78
79        impl_text_wrapper_base!($wrapper, $inner, $encode, $decode);
80    };
81
82    // false case - without serde
83    ($wrapper:ident, $inner:ty, $encode:expr, $decode:expr, false) => {
84        #[derive(sqlx::FromRow, Debug, Clone, PartialEq)]
85        pub struct $wrapper(pub $inner);
86
87        impl_text_wrapper_base!($wrapper, $inner, $encode, $decode);
88    };
89}

91/// Macro for implementing BYTEA-based SQL wrapper types with custom encoding/decoding
92///
93/// # Parameters
94/// - `$wrapper`: The name of the wrapper type to create
95/// - `$inner`: The inner type being wrapped
96/// - `$encode`: Expression for converting inner type to bytes
97/// - `$decode`: Expression for converting bytes back to inner type
98///
99/// The macro creates a new type that wraps the inner type and implements:
100/// - SQLx Type trait to indicate BYTEA column type
101/// - SQLx Encode trait for converting to database format
102/// - SQLx Decode trait for converting from database format
103macro_rules! impl_bytea_wrapper_custom {
104    ($wrapper:ident, $inner:ty, $encode:expr, $decode:expr) => {
105        #[derive(sqlx::FromRow, Debug, Clone, PartialEq)]
106        pub struct $wrapper(pub $inner);
107
108        impl sqlx::Type<sqlx::Postgres> for $wrapper {
109            fn type_info() -> sqlx::postgres::PgTypeInfo {
110                sqlx::postgres::PgTypeInfo::with_name("BYTEA")
111            }
112        }
113
114        impl Encode<'_, Postgres> for $wrapper {
115            fn encode_by_ref(
116                &self,
117                buf: &mut PgArgumentBuffer,
118            ) -> Result<sqlx::encode::IsNull, sqlx::error::BoxDynError> {
119                let bytes = $encode(&self.0);
120                <&[u8] as Encode<Postgres>>::encode(bytes.as_ref(), buf)
121            }
122        }
123
124        impl<'r> Decode<'r, Postgres> for $wrapper {
125            fn decode(value: PgValueRef<'r>) -> Result<Self, sqlx::error::BoxDynError> {
126                let bytes = <Vec<u8> as Decode<Postgres>>::decode(value)?;
127                Ok(Self($decode(&bytes)?))
128            }
129        }
130    };
131}

133/// Macro for implementing BYTEA-based SQL wrapper types using standard serialization
134///
135/// This macro creates a wrapper type that uses the inner type's default serialization
136/// methods (`serialize()` and `from_slice()`) for encoding/decoding to/from BYTEA columns.
137///
138/// # Parameters
139/// - `$wrapper`: The name of the wrapper type to create
140/// - `$inner`: The inner type being wrapped
141///
142/// The macro creates a new type that wraps the inner type and implements:
143/// - SQLx Type trait to indicate BYTEA column type
144/// - SQLx Encode trait for converting to database format
145/// - SQLx Decode trait for converting from database format
146macro_rules! impl_bytea_wrapper_default {
147    ($wrapper:ident, $inner:ty) => {
148        impl_bytea_wrapper_custom!(
149            $wrapper,
150            $inner,
151            |x: &$inner| x.serialize(),
152            |x: &[u8]| -> Result<$inner, BoxDynError> {
153                <$inner>::from_slice(x).map_err(|e| Box::new(e) as sqlx::error::BoxDynError)
154            }
155        );
156    };
157}

159/// Macro for implementing text-based SQL wrapper types using standard string conversion
160///
161/// This macro creates a wrapper type that uses the inner type's default string conversion
162/// methods (`to_string()` and `from_str()`) for encoding/decoding to/from TEXT columns.
163///
164/// # Parameters
165/// - `$wrapper`: The name of the wrapper type to create
166/// - `$inner`: The inner type being wrapped
167///
168/// The macro creates a new type that wraps the inner type and implements:
169/// - SQLx Type trait to indicate TEXT column type
170/// - SQLx Encode trait for converting to database format
171/// - SQLx Decode trait for converting from database format
172macro_rules! impl_text_wrapper_default {
173    ($wrapper:ident, $inner:ty) => {
174        impl_text_wrapper_custom!(
175            $wrapper,
176            $inner,
177            <$inner as ToString>::to_string,
178            <$inner as FromStr>::from_str
179        );
180    };
181}

247impl_text_wrapper_custom!(
248    BlockHeaderDB,
249    block::Header,
250    |header: &block::Header| {
251        let mut bytes = Vec::new();
252        header
253            .consensus_encode(&mut bytes)
254            .expect("exceeded max Vec size or ran out of memory");
255        bytes.to_hex_string(bitcoin::hex::Case::Lower)
256    },
257    |s: &str| -> Result<block::Header, BoxDynError> {
258        let bytes = hex::decode(s)?;
259        block::Header::consensus_decode(&mut bytes.as_slice()).map_err(Into::into)
260    }

297    macro_rules! test_encode_decode_invariant {
298        ($db_type:ty, $inner:ty, $db_wrapper:expr, $table_name:expr, $column_type:expr) => {
299            let db_wrapper = $db_wrapper;
300
301            let config = create_test_config_with_thread_name().await;
302            let database = Database::new(&config).await.unwrap();
303
304            // Create table if it doesn't exist
305            database
306                .connection
307                .execute(sqlx::query(&format!(
308                    "CREATE TABLE IF NOT EXISTS {} ({} {} PRIMARY KEY)",
309                    $table_name, $table_name, $column_type
310                )))
311                .await
312                .unwrap();
313
314            // Insert the value
315            database
316                .connection
317                .execute(
318                    sqlx::query(&format!(
319                        "INSERT INTO {} ({}) VALUES ($1)",
320                        $table_name, $table_name
321                    ))
322                    .bind(db_wrapper.clone()),
323                )
324                .await
325                .unwrap();
326
327            // Retrieve the value
328            let retrieved: $db_type = sqlx::query_scalar(&format!(
329                "SELECT {} FROM {} WHERE {} = $1",
330                $table_name, $table_name, $table_name
331            ))
332            .bind(db_wrapper.clone())
333            .fetch_one(&database.connection)
334            .await
335            .unwrap();
336
337            // Verify the retrieved value matches the original
338            assert_eq!(retrieved, db_wrapper);
339
340            // Clean up
341            database
342                .connection
343                .execute(sqlx::query(&format!("DROP TABLE {}", $table_name)))
344                .await
345                .unwrap();
346        };
347    }
