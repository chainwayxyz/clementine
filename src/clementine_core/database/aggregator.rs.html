1//! # Verifier Related Database Operations
2//!
3//! This module includes database functions which are mainly used by a verifier.
4
5use super::{wrapper::TxidDB, Database, DatabaseTransaction};
6use crate::{errors::BridgeError, execute_query_with_tx};
7use bitcoin::{consensus, Transaction, Txid};
8use eyre;
9use sqlx::QueryBuilder;
10
11impl Database {
12    /// Sets a signed emergency stop transaction for a given move transaction ID
13    pub async fn set_signed_emergency_stop_tx(
14        &self,
15        tx: Option<DatabaseTransaction<'_, '_>>,
16        move_txid: &Txid,
17        emergency_stop_tx: &Transaction,
18    ) -> Result<(), BridgeError> {
19        let query = sqlx::query(
20            "INSERT INTO emergency_stop_sigs (move_txid, emergency_stop_tx) VALUES ($1, $2)
21             ON CONFLICT (move_txid) DO NOTHING;",
22        )
23        .bind(TxidDB(*move_txid))
24        .bind(consensus::serialize(emergency_stop_tx));
25
26        execute_query_with_tx!(self.connection, tx, query, execute)?;
27
28        Ok(())
29    }
30
31    /// Gets emergency stop transactions for a list of move transaction IDs
32    pub async fn get_emergency_stop_txs(
33        &self,
34        tx: Option<DatabaseTransaction<'_, '_>>,
35        move_txids: Vec<Txid>,
36    ) -> Result<Vec<(Txid, Transaction)>, BridgeError> {
37        if move_txids.is_empty() {
38            return Ok(Vec::new());
39        }
40
41        let mut query_builder = QueryBuilder::new(
42            "SELECT move_txid, emergency_stop_tx FROM emergency_stop_sigs WHERE move_txid IN (",
43        );
44
45        let mut separated = query_builder.separated(", ");
46        for txid in &move_txids {
47            separated.push_bind(TxidDB(*txid));
48        }
49        query_builder.push(")");
50
51        let query = query_builder.build_query_as::<(TxidDB, Vec<u8>)>();
52
53        let results: Vec<(TxidDB, Vec<u8>)> =
54            execute_query_with_tx!(self.connection, tx, query, fetch_all)?;
55
56        Ok(results
57            .into_iter()
58            .map(|(txid, tx_data)| {
59                let tx = consensus::deserialize(&tx_data)
60                    .map_err(|e| eyre::eyre!("Failed to deserialize emergency stop tx: {e}"))?;
61                Ok((txid.0, tx))
62            })
63            .collect::<Result<_, eyre::Report>>()?)
64    }
65}

68mod tests {
69    use super::*;

74    use bitcoin::{hashes::Hash, Transaction, Txid};

76    fn create_test_transaction() -> Transaction {
77        let tx_handler = TxHandlerBuilder::new(TransactionType::Dummy).finalize();
78        tx_handler.get_cached_tx().clone()
79    }
80
81    #[tokio::test]
82    async fn test_set_get_emergency_stop_tx() {
83        let config = create_test_config_with_thread_name().await;
84        let database = Database::new(&config).await.unwrap();
85
86        let move_txid = Txid::from_byte_array([1u8; 32]);
87        let emergency_stop_tx = create_test_transaction();
88        database
89            .set_signed_emergency_stop_tx(None, &move_txid, &emergency_stop_tx)
90            .await
91            .unwrap();
92
93        let results = database
94            .get_emergency_stop_txs(None, vec![move_txid])
95            .await
96            .unwrap();
97
98        assert_eq!(results.len(), 1);
99        assert_eq!(results[0].0, move_txid);
100        assert_eq!(results[0].1, emergency_stop_tx);
101
102        // Test getting non-existent tx
103        let non_existent_txid = Txid::from_byte_array([2u8; 32]);
104        let results = database
105            .get_emergency_stop_txs(None, vec![non_existent_txid])
106            .await
107            .unwrap();
108        assert!(results.is_empty());
109
110        // Test getting multiple txs
111        let move_txid2 = Txid::from_byte_array([3u8; 32]);
112        let emergency_stop_tx2 = create_test_transaction();
113        database
114            .set_signed_emergency_stop_tx(None, &move_txid2, &emergency_stop_tx2)
115            .await
116            .unwrap();
117
118        let results = database
119            .get_emergency_stop_txs(None, vec![move_txid, move_txid2])
120            .await
121            .unwrap();
122
123        assert_eq!(results.len(), 2);
124        let mut results = results;
125        results.sort_by(|a, b| a.0.cmp(&b.0));
126        assert_eq!(results[0].0, move_txid);
127        assert_eq!(results[0].1, emergency_stop_tx);
128        assert_eq!(results[1].0, move_txid2);
129        assert_eq!(results[1].1, emergency_stop_tx2);
130
131        // Test updating existing tx
132        let updated_tx = create_test_transaction();
133        database
134            .set_signed_emergency_stop_tx(None, &move_txid, &updated_tx)
135            .await
136            .unwrap();
137
138        let results = database
139            .get_emergency_stop_txs(None, vec![move_txid])
140            .await
141            .unwrap();
142
143        assert_eq!(results.len(), 1);
144        assert_eq!(results[0].0, move_txid);
145        assert_eq!(results[0].1, updated_tx);
146    }
