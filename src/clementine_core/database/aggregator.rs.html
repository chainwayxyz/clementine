1//! # Verifier Related Database Operations
2//!
3//! This module includes database functions which are mainly used by a verifier.

5use super::{wrapper::TxidDB, Database, DatabaseTransaction};
6use crate::execute_query_with_tx;
7use bitcoin::Txid;
8use clementine_errors::BridgeError;
9use eyre;
10use sqlx::QueryBuilder;
11
12impl Database {
13    /// Sets a signed emergency stop transaction for a given move transaction ID
14    pub async fn insert_signed_emergency_stop_tx_if_not_exists(
15        &self,
16        tx: Option<DatabaseTransaction<'_, '_>>,
17        move_txid: &Txid,
18        encrypted_emergency_stop_tx: &[u8],
19    ) -> Result<(), BridgeError> {
20        let query = sqlx::query(
21            "INSERT INTO emergency_stop_sigs (move_txid, emergency_stop_tx) VALUES ($1, $2)
22             ON CONFLICT (move_txid) DO NOTHING;",
23        )
24        .bind(TxidDB(*move_txid))
25        .bind(encrypted_emergency_stop_tx);
26
27        execute_query_with_tx!(self.connection, tx, query, execute)?;
28
29        Ok(())
30    }
31
32    /// Gets emergency stop transactions for a list of move transaction IDs
33    pub async fn get_emergency_stop_txs(
34        &self,
35        tx: Option<DatabaseTransaction<'_, '_>>,
36        move_txids: Vec<Txid>,
37    ) -> Result<Vec<(Txid, Vec<u8>)>, BridgeError> {
38        if move_txids.is_empty() {
39            return Ok(Vec::new());
40        }
41
42        let mut query_builder = QueryBuilder::new(
43            "SELECT move_txid, emergency_stop_tx FROM emergency_stop_sigs WHERE move_txid IN (",
44        );
45
46        let mut separated = query_builder.separated(", ");
47        for txid in &move_txids {
48            separated.push_bind(TxidDB(*txid));
49        }
50        query_builder.push(")");
51
52        let query = query_builder.build_query_as::<(TxidDB, Vec<u8>)>();
53
54        let results: Vec<(TxidDB, Vec<u8>)> =
55            execute_query_with_tx!(self.connection, tx, query, fetch_all)?;
56
57        Ok(results
58            .into_iter()
59            .map(|(txid, tx_data)| Ok((txid.0, tx_data)))
60            .collect::<Result<_, eyre::Report>>()?)
61    }
62}

79    }

81    #[tokio::test]
82    async fn test_set_get_emergency_stop_tx() {
83        let config = create_test_config_with_thread_name().await;
84        let database = Database::new(&config).await.unwrap();
85
86        let move_txid = Txid::from_byte_array([1u8; 32]);
87        let emergency_stop_tx = create_test_transaction();
88        database
89            .insert_signed_emergency_stop_tx_if_not_exists(
90                None,
91                &move_txid,
92                &consensus::serialize(&emergency_stop_tx),
93            )
94            .await
95            .unwrap();
96
97        let results = database
98            .get_emergency_stop_txs(None, vec![move_txid])
99            .await
100            .unwrap();
101
102        assert_eq!(results.len(), 1);
103        assert_eq!(results[0].0, move_txid);
104        assert_eq!(results[0].1, consensus::serialize(&emergency_stop_tx));
105
106        // Test getting non-existent tx
107        let non_existent_txid = Txid::from_byte_array([2u8; 32]);
108        let results = database
109            .get_emergency_stop_txs(None, vec![non_existent_txid])
110            .await
111            .unwrap();
112        assert!(results.is_empty());
113
114        // Test getting multiple txs
115        let move_txid2 = Txid::from_byte_array([3u8; 32]);
116        let emergency_stop_tx2 = create_test_transaction();
117        database
118            .insert_signed_emergency_stop_tx_if_not_exists(
119                None,
120                &move_txid2,
121                &consensus::serialize(&emergency_stop_tx2),
122            )
123            .await
124            .unwrap();
125
126        let results = database
127            .get_emergency_stop_txs(None, vec![move_txid, move_txid2])
128            .await
129            .unwrap();
130
131        assert_eq!(results.len(), 2);
132        let mut results = results;
133        results.sort_by(|a, b| a.0.cmp(&b.0));
134        assert_eq!(results[0].0, move_txid);
135        assert_eq!(results[0].1, consensus::serialize(&emergency_stop_tx));
136        assert_eq!(results[1].0, move_txid2);
137        assert_eq!(results[1].1, consensus::serialize(&emergency_stop_tx2));
138
139        // Test updating existing tx
140        let updated_tx = create_test_transaction();
141        database
142            .insert_signed_emergency_stop_tx_if_not_exists(
143                None,
144                &move_txid,
145                &consensus::serialize(&updated_tx),
146            )
147            .await
148            .unwrap();
149
150        let results = database
151            .get_emergency_stop_txs(None, vec![move_txid])
152            .await
153            .unwrap();
154
155        assert_eq!(results.len(), 1);
156        assert_eq!(results[0].0, move_txid);
157        assert_eq!(results[0].1, consensus::serialize(&updated_tx));
158    }
