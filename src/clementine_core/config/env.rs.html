1//! # Environment Variable Support For [`BridgeConfig`]
2
3use super::BridgeConfig;
4use crate::{builder::transaction::SecurityCouncil, errors::BridgeError};
5use bitcoin::{secp256k1::SecretKey, Amount};
6use std::{path::PathBuf, str::FromStr};
7
8pub(crate) fn read_string_from_env(env_var: &'static str) -> Result<String, BridgeError> {
9    std::env::var(env_var).map_err(|e| BridgeError::EnvVarNotSet(e, env_var))
10}
11
12pub(crate) fn read_string_from_env_then_parse<T: std::str::FromStr>(
13    env_var: &'static str,
14) -> Result<T, BridgeError>
15where
16    <T as FromStr>::Err: std::fmt::Debug,
17{
18    read_string_from_env(env_var)?
19        .parse::<T>()
20        .map_err(|e| BridgeError::EnvVarMalformed(env_var, format!("{:?}", e)))
21}
22
23impl BridgeConfig {
24    pub fn from_env() -> Result<Self, BridgeError> {
25        let verifier_endpoints =
26            std::env::var("VERIFIER_ENDPOINTS")
27                .ok()
28                .map(|verifier_endpoints| {
29                    verifier_endpoints
30                        .split(",")
31                        .collect::<Vec<&str>>()
32                        .iter()
33                        .map(|x| x.to_string())
34                        .collect::<Vec<String>>()
35                });
36        let operator_endpoints =
37            std::env::var("OPERATOR_ENDPOINTS")
38                .ok()
39                .map(|operator_endpoints| {
40                    operator_endpoints
41                        .split(",")
42                        .collect::<Vec<&str>>()
43                        .iter()
44                        .map(|x| x.to_string())
45                        .collect::<Vec<String>>()
46                });
47
48        let all_verifiers_secret_keys =
49            if let Ok(all_verifiers_secret_keys) = std::env::var("ALL_VERIFIERS_SECRET_KEYS") {
50                Some(
51                    all_verifiers_secret_keys
52                        .split(",")
53                        .collect::<Vec<&str>>()
54                        .iter()
55                        .map(|x| SecretKey::from_str(x))
56                        .collect::<Result<Vec<SecretKey>, _>>()
57                        .map_err(|e| {
58                            BridgeError::EnvVarMalformed("ALL_VERIFIERS_SECRET_KEYS", e.to_string())
59                        })?,
60                )
61            } else {
62                None
63            };
64        let all_operators_secret_keys =
65            if let Ok(all_operators_secret_keys) = std::env::var("ALL_OPERATORS_SECRET_KEYS") {
66                Some(
67                    all_operators_secret_keys
68                        .split(",")
69                        .collect::<Vec<&str>>()
70                        .iter()
71                        .map(|x| SecretKey::from_str(x))
72                        .collect::<Result<Vec<SecretKey>, _>>()
73                        .map_err(|e| {
74                            BridgeError::EnvVarMalformed("ALL_OPERATORS_SECRET_KEYS", e.to_string())
75                        })?,
76                )
77            } else {
78                None
79            };
80
81        let winternitz_secret_key = if let Ok(sk) = std::env::var("WINTERNITZ_SECRET_KEY") {
82            Some(sk.parse::<SecretKey>().map_err(|e| {
83                BridgeError::EnvVarMalformed("WINTERNITZ_SECRET_KEY", e.to_string())
84            })?)
85        } else {
86            None
87        };
88
89        let operator_withdrawal_fee_sats = if let Ok(operator_withdrawal_fee_sats) =
90            std::env::var("OPERATOR_WITHDRAWAL_FEE_SATS")
91        {
92            Some(Amount::from_sat(
93                operator_withdrawal_fee_sats.parse::<u64>().map_err(|e| {
94                    BridgeError::EnvVarMalformed("OPERATOR_WITHDRAWAL_FEE_SATS", e.to_string())
95                })?,
96            ))
97        } else {
98            None
99        };
100
101        let header_chain_proof_path =
102            if let Ok(header_chain_proof_path) = std::env::var("HEADER_CHAIN_PROOF_PATH") {
103                Some(PathBuf::from(header_chain_proof_path))
104            } else {
105                None
106            };
107
108        // TLS certificate and key paths
109        let server_cert_path = read_string_from_env("SERVER_CERT_PATH").map(PathBuf::from)?;
110        let server_key_path = read_string_from_env("SERVER_KEY_PATH").map(PathBuf::from)?;
111        let client_cert_path = read_string_from_env("CLIENT_CERT_PATH").map(PathBuf::from)?;
112        let ca_cert_path = read_string_from_env("CA_CERT_PATH").map(PathBuf::from)?;
113        let client_key_path = read_string_from_env("CLIENT_KEY_PATH").map(PathBuf::from)?;
114        let aggregator_cert_path =
115            read_string_from_env("AGGREGATOR_CERT_PATH").map(PathBuf::from)?;
116        let client_verification =
117            read_string_from_env("CLIENT_VERIFICATION").is_ok_and(|s| s == "true" || s == "1");
118
119        let security_council_string = read_string_from_env("SECURITY_COUNCIL")?;
120
121        let security_council = SecurityCouncil::from_str(&security_council_string)?;
122
123        let config = BridgeConfig {
124            // Protocol paramset's source is independently defined
125            protocol_paramset: Default::default(),
126            host: read_string_from_env("HOST")?,
127            port: read_string_from_env_then_parse::<u16>("PORT")?,
128            secret_key: read_string_from_env_then_parse::<SecretKey>("SECRET_KEY")?,
129            winternitz_secret_key,
130            operator_withdrawal_fee_sats,
131            bitcoin_rpc_url: read_string_from_env("BITCOIN_RPC_URL")?,
132            bitcoin_rpc_user: read_string_from_env("BITCOIN_RPC_USER")?,
133            bitcoin_rpc_password: read_string_from_env("BITCOIN_RPC_PASSWORD")?,
134            db_host: read_string_from_env("DB_HOST")?,
135            db_port: read_string_from_env_then_parse::<usize>("DB_PORT")?,
136            db_user: read_string_from_env("DB_USER")?,
137            db_password: read_string_from_env("DB_PASSWORD")?,
138            db_name: read_string_from_env("DB_NAME")?,
139            citrea_rpc_url: read_string_from_env("CITREA_RPC_URL")?,
140            citrea_light_client_prover_url: read_string_from_env("CITREA_LIGHT_CLIENT_PROVER_URL")?,
141            citrea_chain_id: read_string_from_env_then_parse::<u32>("CITREA_CHAIN_ID")?,
142            bridge_contract_address: read_string_from_env("BRIDGE_CONTRACT_ADDRESS")?,
143            header_chain_proof_path,
144            verifier_endpoints,
145            operator_endpoints,
146            all_verifiers_secret_keys,
147            all_operators_secret_keys,
148            security_council,
149
150            client_verification,
151            server_cert_path,
152            server_key_path,
153            ca_cert_path,
154            client_cert_path,
155            client_key_path,
156            aggregator_cert_path,
157
158            #[cfg(test)]
159            test_params: super::TestParams::default(),
160        };
161
162        tracing::debug!("BridgeConfig from env: {:?}", config);
163        Ok(config)
164    }
165}

189            );
190        }
191        if let Some(ref operator_withdrawal_fee_sats) = default_config.operator_withdrawal_fee_sats
192        {
193            std::env::set_var(
194                "OPERATOR_WITHDRAWAL_FEE_SATS",
195                operator_withdrawal_fee_sats.to_sat().to_string(),
196            );
197        }
198        std::env::set_var("BITCOIN_RPC_URL", &default_config.bitcoin_rpc_url);
199        std::env::set_var("BITCOIN_RPC_USER", &default_config.bitcoin_rpc_user);
200        std::env::set_var("BITCOIN_RPC_PASSWORD", &default_config.bitcoin_rpc_password);
201        std::env::set_var("DB_HOST", default_config.db_host.clone());
202        std::env::set_var("DB_PORT", default_config.db_port.to_string());
203        std::env::set_var("DB_USER", default_config.db_user.clone());
204        std::env::set_var("DB_PASSWORD", &default_config.db_password);
205        std::env::set_var("DB_NAME", &default_config.db_name);
206        std::env::set_var("CITREA_RPC_URL", &default_config.citrea_rpc_url);
207        std::env::set_var(
208            "CITREA_LIGHT_CLIENT_PROVER_URL",
209            &default_config.citrea_light_client_prover_url,

211        std::env::set_var(
212            "CITREA_CHAIN_ID",
213            default_config.citrea_chain_id.to_string(),
214        );
215        std::env::set_var(
216            "BRIDGE_CONTRACT_ADDRESS",
217            &default_config.bridge_contract_address,
218        );
219        std::env::set_var(
220            "AGGREGATOR_CERT_PATH",

232
233        std::env::set_var(
234            "SECURITY_COUNCIL",
235            default_config.security_council.to_string(),

239            std::env::set_var("HEADER_CHAIN_PROOF_PATH", header_chain_proof_path);
240        }
241        if let Some(ref verifier_endpoints) = default_config.verifier_endpoints {
242            std::env::set_var("VERIFIER_ENDPOINTS", verifier_endpoints.join(","));
243        }

248            std::env::set_var(
249                "ALL_VERIFIERS_SECRET_KEYS",

253                    .collect::<Vec<String>>()
254                    .join(","),
255            );
256        }
257        if let Some(ref all_operators_secret_keys) = default_config.all_operators_secret_keys {
258            std::env::set_var(
259                "ALL_OPERATORS_SECRET_KEYS",

261                    .iter()
262                    .map(|sk| sk.display_secret().to_string())
263                    .collect::<Vec<String>>()
264                    .join(","),
265            );
266        }

268        assert_eq!(super::BridgeConfig::from_env().unwrap(), default_config);
269    }
