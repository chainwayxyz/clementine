23/// A pre-defined paramset name that can be converted into a
24/// [`ProtocolParamset`] reference.
25///
26/// See: [`REGTEST_PARAMSET`]
27pub enum ProtocolParamsetName {
28    // Pre-defined paramsets
29    Regtest,
30}
31
32impl FromStr for ProtocolParamsetName {
33    type Err = BridgeError;
34
35    fn from_str(s: &str) -> Result<Self, Self::Err> {
36        match s {
37            "regtest" => Ok(ProtocolParamsetName::Regtest),
38            _ => Err(BridgeError::ConfigError(format!(
39                "Unknown paramset name: {}",
40                s
41            ))),
42        }
43    }
44}
45
46impl Display for ProtocolParamsetName {
47    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
48        match self {
49            ProtocolParamsetName::Regtest => write!(f, "regtest"),
50        }
51    }
52}
53
54impl From<ProtocolParamsetName> for &'static ProtocolParamset {
55    fn from(name: ProtocolParamsetName) -> Self {
56        match name {
57            ProtocolParamsetName::Regtest => &REGTEST_PARAMSET,
58        }
59    }
60}

63/// Protocol parameters that affect the transactions in the contract (which also
64/// change the pre-calculated txids and sighashes).
65///
66/// These parameters are used when generating the transactions and changing them
67/// will break compatibility between actors, making deposits impossible.  A
68/// paramset is chosen by the actor by choosing a ParamsetName inside the
69/// [`crate::config::BridgeConfig`].
70pub struct ProtocolParamset {
71    /// Bitcoin network to work on (mainnet, testnet, regtest).
72    pub network: Network,
73    /// Number of round transactions that the operator will create.
74    pub num_round_txs: usize,
75    /// Number of kickoff UTXOs per round transaction.
76    pub num_kickoffs_per_round: usize,
77    /// Number of kickoffs that are signed per round and deposit.
78    /// There are num_kickoffs_per_round utxo's, but only num_signed_kickoffs are signed.
79    pub num_signed_kickoffs: usize,
80    /// Bridge deposit amount that users can deposit.
81    pub bridge_amount: Amount,
82    /// Amount allocated for each kickoff UTXO.
83    pub kickoff_amount: Amount,
84    /// Amount allocated for operator challenge transactions.
85    pub operator_challenge_amount: Amount,
86    /// Collateral funding amount for operators used to fund the round transaction chain.
87    pub collateral_funding_amount: Amount,
88    /// Length of the blockhash commitment in kickoff transactions.
89    pub kickoff_blockhash_commit_length: u32,
90    /// Total number of bytes of a watchtower challenge.
91    pub watchtower_challenge_bytes: usize,
92    /// Winternitz derivation log_d (shared for all WOTS commitments)
93    /// Currently used in statics and thus cannot be different from [`WINTERNITZ_LOG_D`].
94    pub winternitz_log_d: u32,
95    /// Number of blocks after which user can take deposit back if deposit request fails.
96    pub user_takes_after: u16,
97    /// Number of blocks for operator challenge timeout timelock (currently BLOCKS_PER_WEEK)
98    pub operator_challenge_timeout_timelock: u16,
99    /// Number of blocks for operator challenge NACK timelock (currently BLOCKS_PER_WEEK * 3)
100    pub operator_challenge_nack_timelock: u16,
101    /// Number of blocks for disprove timeout timelock (currently BLOCKS_PER_WEEK * 5)
102    pub disprove_timeout_timelock: u16,
103    /// Number of blocks for assert timeout timelock (currently BLOCKS_PER_WEEK * 4)
104    pub assert_timeout_timelock: u16,
105    /// Number of blocks for latest blockhash timeout timelock (currently BLOCKS_PER_WEEK * 2.5)
106    pub latest_blockhash_timeout_timelock: u16,
107    /// Number of blocks for operator reimburse timelock (currently BLOCKS_PER_DAY * 2)
108    /// Timelocks operator from sending the next Round Tx after the Ready to Reimburse Tx.
109    pub operator_reimburse_timelock: u16,
110    /// Number of blocks for watchtower challenge timeout timelock (currently BLOCKS_PER_WEEK * 2)
111    pub watchtower_challenge_timeout_timelock: u16,
112    /// Time to wait after a kickoff to send a watchtower challenge
113    pub time_to_send_watchtower_challenge: u16,
114    /// Time to wait before trying to disprove (so that you collect all operator challenge acks before disproving)
115    pub time_to_disprove: u16,
116    /// Amount of depth a block should have from the current head to be considered finalized
117    pub finality_depth: u32,
118    /// start height to sync the chain from, i.e. the height bridge was deployed
119    pub start_height: u32,
120    /// Batch size of the header chain proofs
121    pub header_chain_proof_batch_size: u32,
122}

125    pub fn from_toml_file(path: &Path) -> Result<Self, BridgeError> {
126        let contents = fs::read_to_string(path)
127            .map_err(|e| BridgeError::Error(format!("Failed to read config file: {}", e)))?;
128
129        let paramset: Self = toml::from_str(&contents)
130            .map_err(|e| BridgeError::Error(format!("Failed to parse TOML: {}", e)))?;
131
132        Ok(paramset)
133    }
