51        let start = Instant::now();

53        let bitvm_cache = {
54            let cache_path = std::env::var("BITVM_CACHE_PATH").unwrap_or_else(|_| "bitvm_cache.bin".to_string());
55            tracing::info!("Loading BitVM cache from file: {}", cache_path);
56            match BitvmCache::load_from_file(&cache_path) {
57                Ok(cache) => {
58                    tracing::info!("Loaded BitVM cache from file");
59                    cache
60                }
61                Err(_) => {
62                    tracing::info!("No BitVM cache found, generating fresh data");
63                    let fresh_data = generate_fresh_data();
64                    if let Err(e) = fresh_data.save_to_file(&cache_path) {
65                        tracing::error!("Failed to save BitVM cache to file: {}", e);
66                    }
67                    fresh_data
68                }
69            }
70        };
71        println!("BitVM initialization took: {:?}", start.elapsed());
72        bitvm_cache
73    };

82impl BitvmCache {
83    fn save_to_file(&self, path: &str) -> Result<(), BridgeError> {
84        let serialized = borsh::to_vec(self).map_err(|e| {
85            tracing::error!("Failed to serialize BitVM cache: {}", e);
86            BridgeError::ConfigError("Failed to serialize BitVM cache".to_string())
87        })?;
88
89        fs::write(path, serialized).map_err(|e| {
90            tracing::error!("Failed to save BitVM cache: {}", e);
91            BridgeError::ConfigError("Failed to save BitVM cache".to_string())
92        })
93    }
94
95    fn load_from_file(path: &str) -> Result<Self, BridgeError> {
96        let bytes = fs::read(path).map_err(|e| {
97            tracing::error!("Failed to read BitVM cache: {}", e);
98            BridgeError::ConfigError("No BitVM cache found".to_string())
99        })?;
100
101        tracing::info!("Loaded BitVM cache from file, read {} bytes", bytes.len());
102
103        Self::try_from_slice(&bytes).map_err(|e| {
104            tracing::error!("Failed to deserialize BitVM cache: {}", e);
105            BridgeError::ConfigError("Failed to deserialize BitVM cache".to_string())
106        })
107    }
108}

110fn generate_fresh_data() -> BitvmCache {
111    let vk = get_ark_verifying_key();
112
113    let dummy_pks = ClementineBitVMPublicKeys::create_replacable();
114
115    let partial_scripts = api_generate_partial_script(&vk);
116
117    let scripts = partial_scripts
118        .iter()
119        .map(|s| s.clone().to_bytes())
120        .collect::<Vec<_>>();
121
122    for (script_idx, script) in scripts.iter().enumerate() {
123        let mut pos = 0;
124        while pos + 20 <= script.len() {
125            // Check if this window matches our pattern (255u8 in the end)
126            if script[pos + 4..pos + 20] == [255u8; 16] {
127                panic!("Dummy value found in script {}", script_idx);
128            }
129            pos += 1;
130        }
131    }
132
133    let disprove_scripts = api_generate_full_tapscripts(dummy_pks.bitvm_pks, &partial_scripts);
134
135    let scripts: Vec<Vec<u8>> = disprove_scripts
136        .iter()
137        .map(|s| s.clone().to_bytes())
138        .collect();
139
140    // Build mapping of dummy keys to their positions
141    let mut replacement_places: ClementineBitVMReplacementData = Default::default();
142    // For each script
143    for (script_idx, script) in scripts.iter().enumerate() {
144        let mut pos = 0;
145        while pos + 20 <= script.len() {
146            // Check if this window matches our pattern (255u8 in the end)
147            if script[pos + 4..pos + 20] == [255u8; 16] {
148                let pk_type_idx = script[pos];
149                let pk_idx = u16::from_be_bytes([script[pos + 1], script[pos + 2]]);
150                let digit_idx = script[pos + 3];
151
152                match pk_type_idx {
153                    0 => {
154                        replacement_places.payout_tx_blockhash_pk[digit_idx as usize]
155                            .push((script_idx, pos));
156                    }
157                    1 => {
158                        replacement_places.latest_blockhash_pk[digit_idx as usize]
159                            .push((script_idx, pos));
160                    }
161                    2 => {
162                        replacement_places.challenge_sending_watchtowers_pk[digit_idx as usize]
163                            .push((script_idx, pos));
164                    }
165                    3 => {
166                        replacement_places.bitvm_pks.0[pk_idx as usize][digit_idx as usize]
167                            .push((script_idx, pos));
168                    }
169                    4 => {
170                        replacement_places.bitvm_pks.1[pk_idx as usize][digit_idx as usize]
171                            .push((script_idx, pos));
172                    }
173                    5 => {
174                        replacement_places.bitvm_pks.2[pk_idx as usize][digit_idx as usize]
175                            .push((script_idx, pos));
176                    }
177                    _ => {
178                        panic!("Invalid pk type index: {}", pk_type_idx);
179                    }
180                }
181                pos += 20;
182            } else {
183                pos += 1;
184            }
185        }
186    }
187
188    BitvmCache {
189        disprove_scripts: scripts,
190        replacement_places,
191    }
192}

352    pub fn vec_to_array<const N: usize>(vec: &[[u8; 20]]) -> [[u8; 20]; N] {
353        let mut result = [[255u8; 20]; N];
354        for (i, item) in vec.iter().enumerate() {
355            if i < N {
356                result[i] = *item;
357            }
358        }

383    pub fn get_assert_scripts(
384        &self,
385        xonly_public_key: XOnlyPublicKey,
386    ) -> Vec<std::sync::Arc<dyn SpendableScript>> {
387        let mut scripts = Vec::new();
388        let first_script: Arc<dyn SpendableScript> = Arc::new(WinternitzCommit::new(
389            vec![
390                (self.challenge_sending_watchtowers_pk.to_vec(), 40),
391                (self.bitvm_pks.0[0].to_vec(), 64),
392                (self.bitvm_pks.1[NUM_U256 - 2].to_vec(), 64),
393                (self.bitvm_pks.1[NUM_U256 - 1].to_vec(), 64),
394                (self.bitvm_pks.2[NUM_HASH - 3].to_vec(), 32),
395                (self.bitvm_pks.2[NUM_HASH - 2].to_vec(), 32),
396                (self.bitvm_pks.2[NUM_HASH - 1].to_vec(), 32),
397            ],
398            xonly_public_key,
399            4,
400        ));
401        scripts.push(first_script);
402        // iterate NUM_U256 6 by 6
403        for i in (0..NUM_U256 - 2).step_by(6) {
404            let last_idx = std::cmp::min(i + 6, NUM_U256);
405            let script: Arc<dyn SpendableScript> = Arc::new(WinternitzCommit::new(
406                self.bitvm_pks.1[i..last_idx]
407                    .iter()
408                    .map(|x| (x.to_vec(), 64))
409                    .collect::<Vec<_>>(),
410                xonly_public_key,
411                4,
412            ));
413            scripts.push(script);
414        }
415        // iterate NUM_HASH 12 by 12
416        for i in (0..NUM_HASH - 3).step_by(12) {
417            let last_idx = std::cmp::min(i + 12, NUM_HASH);
418            let script: Arc<dyn SpendableScript> = Arc::new(WinternitzCommit::new(
419                self.bitvm_pks.2[i..last_idx]
420                    .iter()
421                    .map(|x| (x.to_vec(), 32))
422                    .collect::<Vec<_>>(),
423                xonly_public_key,
424                4,
425            ));
426            scripts.push(script);
427        }
428        scripts
429    }

490        deposit_outpoint: bitcoin::OutPoint,
491        paramset: &'static ProtocolParamset,
492    ) -> Vec<WinternitzDerivationPath> {
493        if mini_assert_idx == 0 {
494            vec![

501                WinternitzDerivationPath::BitvmAssert(16 * 2, 5, 362, deposit_outpoint, paramset),
502            ]
503        } else if (1..=2).contains(&mini_assert_idx) {
504            // for 1, we will have 6 derivations index starting from 0 to 5
505            // for 2, we will have 6 derivations index starting from 6 to 11
506            let derivations: u32 = (mini_assert_idx as u32 - 1) * 6;
507
508            let mut derivations_vec = vec![];
509            for i in 0..6 {
510                if derivations + i < NUM_U256 as u32 - 2 {
511                    derivations_vec.push(WinternitzDerivationPath::BitvmAssert(
512                        32 * 2,
513                        4,
514                        derivations + i,
515                        deposit_outpoint,
516                        paramset,
517                    ));
518                }
519            }
520            derivations_vec
521        } else {
522            let derivations: u32 = (mini_assert_idx as u32 - 3) * 12;
523            let mut derivations_vec = vec![];
524            for i in 0..12 {
525                if derivations + i < NUM_HASH as u32 - 3 {
526                    derivations_vec.push(WinternitzDerivationPath::BitvmAssert(
527                        16 * 2,
528                        5,
529                        derivations + i,
530                        deposit_outpoint,
531                        paramset,
532                    ));
533                }
534            }
535            derivations_vec
536        }

556        if cfg!(debug_assertions) {
557            vec![ScriptBuf::from_bytes(vec![0x51])] // OP_TRUE
558        } else {
559            replace_disprove_scripts(self)
560        }
561    }
562}

564pub fn replace_disprove_scripts(pks: &ClementineBitVMPublicKeys) -> Vec<ScriptBuf> {
565    let start = Instant::now();
566    tracing::info!("Starting script replacement");
567
568    let cache = &*BITVM_CACHE;
569    let mut result: Vec<Vec<u8>> = cache.disprove_scripts.clone();
570    let replacement_places = &cache.replacement_places;
571
572    for (digit, places) in replacement_places.payout_tx_blockhash_pk.iter().enumerate() {
573        for (script_idx, pos) in places.iter() {
574            result[*script_idx][*pos..*pos + 20]
575                .copy_from_slice(&pks.payout_tx_blockhash_pk[digit]);
576        }
577    }
578
579    for (digit, places) in replacement_places.latest_blockhash_pk.iter().enumerate() {
580        for (script_idx, pos) in places.iter() {
581            result[*script_idx][*pos..*pos + 20].copy_from_slice(&pks.latest_blockhash_pk[digit]);
582        }
583    }
584
585    for (digit, places) in replacement_places
586        .challenge_sending_watchtowers_pk
587        .iter()
588        .enumerate()
589    {
590        for (script_idx, pos) in places.iter() {
591            result[*script_idx][*pos..*pos + 20]
592                .copy_from_slice(&pks.challenge_sending_watchtowers_pk[digit]);
593        }
594    }
595
596    for (digit, places) in replacement_places.bitvm_pks.0.iter().enumerate() {
597        for (pk_idx, places) in places.iter().enumerate() {
598            for (script_idx, pos) in places.iter() {
599                result[*script_idx][*pos..*pos + 20]
600                    .copy_from_slice(&pks.bitvm_pks.0[digit][pk_idx]);
601            }
602        }
603    }
604
605    for (digit, places) in replacement_places.bitvm_pks.1.iter().enumerate() {
606        for (pk_idx, places) in places.iter().enumerate() {
607            for (script_idx, pos) in places.iter() {
608                result[*script_idx][*pos..*pos + 20]
609                    .copy_from_slice(&pks.bitvm_pks.1[digit][pk_idx]);
610            }
611        }
612    }
613
614    for (digit, places) in replacement_places.bitvm_pks.2.iter().enumerate() {
615        for (pk_idx, places) in places.iter().enumerate() {
616            for (script_idx, pos) in places.iter() {
617                result[*script_idx][*pos..*pos + 20]
618                    .copy_from_slice(&pks.bitvm_pks.2[digit][pk_idx]);
619            }
620        }
621    }
622
623    let result: Vec<ScriptBuf> = result.into_iter().map(ScriptBuf::from_bytes).collect();
624
625    let elapsed = start.elapsed();
626    tracing::info!("Script replacement completed in {:?}", elapsed);
627
628    result
629}
