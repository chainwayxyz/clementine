53        let start = Instant::now();
54
55        let bitvm_cache = {
56            let cache_path = std::env::var("BITVM_CACHE_PATH").unwrap_or_else(|_| "bitvm_cache.bin".to_string());
57            tracing::info!("Loading BitVM cache from file: {}", cache_path);
58            match BitvmCache::load_from_file(&cache_path) {
59                Ok(cache) => {
60                    tracing::info!("Loaded BitVM cache from file");
61                    cache
62                }
63                Err(_) => {
64                    tracing::info!("No BitVM cache found, generating fresh data");
65                    let fresh_data = generate_fresh_data();
66                    if let Err(e) = fresh_data.save_to_file(&cache_path) {
67                        tracing::error!("Failed to save BitVM cache to file: {}", e);
68                    }
69                    fresh_data
70                }
71            }
72        };
73        println!("BitVM initialization took: {:?}", start.elapsed());
74        bitvm_cache
75    };

84impl BitvmCache {
85    fn save_to_file(&self, path: &str) -> Result<(), BridgeError> {
86        let serialized = borsh::to_vec(self).map_err(|e| {
87            tracing::error!("Failed to serialize BitVM cache: {}", e);
88            BridgeError::ConfigError("Failed to serialize BitVM cache".to_string())
89        })?;
90
91        fs::write(path, serialized).map_err(|e| {
92            tracing::error!("Failed to save BitVM cache: {}", e);
93            BridgeError::ConfigError("Failed to save BitVM cache".to_string())
94        })
95    }
96
97    fn load_from_file(path: &str) -> Result<Self, BridgeError> {
98        let bytes = fs::read(path).map_err(|e| {
99            tracing::error!("Failed to read BitVM cache: {}", e);
100            BridgeError::ConfigError("No BitVM cache found".to_string())
101        })?;
102
103        tracing::info!("Loaded BitVM cache from file, read {} bytes", bytes.len());
104
105        Self::try_from_slice(&bytes).map_err(|e| {
106            tracing::error!("Failed to deserialize BitVM cache: {}", e);
107            BridgeError::ConfigError("Failed to deserialize BitVM cache".to_string())
108        })
109    }
110}

357            deposit_constant,

390    pub fn vec_to_array<const N: usize>(vec: &[[u8; 20]]) -> [[u8; 20]; N] {
391        let mut result = [[255u8; 20]; N];
392        for (i, item) in vec.iter().enumerate() {
393            if i < N {
394                result[i] = *item;
395            }
396        }

402        slice: &[Vec<[u8; 20]>],
403    ) -> [[[u8; 20]; INNER_LEN]; OUTER_LEN] {
404        let mut result = [[[255u8; 20]; INNER_LEN]; OUTER_LEN];

409        }
410        result

421    pub fn get_assert_scripts(
422        &self,
423        xonly_public_key: XOnlyPublicKey,
424    ) -> Vec<std::sync::Arc<dyn SpendableScript>> {
425        let mut scripts = Vec::new();
426        let first_script: Arc<dyn SpendableScript> = Arc::new(WinternitzCommit::new(
427            vec![
428                (self.latest_blockhash_pk.to_vec(), 40),
429                (self.challenge_sending_watchtowers_pk.to_vec(), 40),
430                (self.bitvm_pks.0[0].to_vec(), 64),
431                (self.bitvm_pks.1[NUM_U256 - 2].to_vec(), 64),
432                (self.bitvm_pks.1[NUM_U256 - 1].to_vec(), 64),
433                (self.bitvm_pks.2[NUM_HASH - 3].to_vec(), 32),
434                (self.bitvm_pks.2[NUM_HASH - 2].to_vec(), 32),
435                (self.bitvm_pks.2[NUM_HASH - 1].to_vec(), 32),
436            ],
437            xonly_public_key,
438            4,
439        ));
440        scripts.push(first_script);
441        // iterate NUM_U256 6 by 6
442        for i in (0..NUM_U256 - 2).step_by(6) {
443            let last_idx = std::cmp::min(i + 6, NUM_U256);
444            let script: Arc<dyn SpendableScript> = Arc::new(WinternitzCommit::new(
445                self.bitvm_pks.1[i..last_idx]
446                    .iter()
447                    .map(|x| (x.to_vec(), 64))
448                    .collect::<Vec<_>>(),
449                xonly_public_key,
450                4,
451            ));
452            scripts.push(script);
453        }
454        // iterate NUM_HASH 12 by 12
455        for i in (0..NUM_HASH - 3).step_by(12) {
456            let last_idx = std::cmp::min(i + 12, NUM_HASH);
457            let script: Arc<dyn SpendableScript> = Arc::new(WinternitzCommit::new(
458                self.bitvm_pks.2[i..last_idx]
459                    .iter()
460                    .map(|x| (x.to_vec(), 32))
461                    .collect::<Vec<_>>(),
462                xonly_public_key,
463                4,
464            ));
465            scripts.push(script);
466        }
467        scripts
468    }

495    ) -> Vec<WinternitzDerivationPath> {
496        if mini_assert_idx == 0 {
497            vec![
498                Self::get_latest_blockhash_derivation(deposit_outpoint, paramset),
499                Self::get_challenge_sending_watchtowers_derivation(deposit_outpoint, paramset),
500                WinternitzDerivationPath::BitvmAssert(32 * 2, 3, 0, deposit_outpoint, paramset),
501                WinternitzDerivationPath::BitvmAssert(32 * 2, 4, 12, deposit_outpoint, paramset),
502                WinternitzDerivationPath::BitvmAssert(32 * 2, 4, 13, deposit_outpoint, paramset),
503                WinternitzDerivationPath::BitvmAssert(16 * 2, 5, 360, deposit_outpoint, paramset),
504                WinternitzDerivationPath::BitvmAssert(16 * 2, 5, 361, deposit_outpoint, paramset),
505                WinternitzDerivationPath::BitvmAssert(16 * 2, 5, 362, deposit_outpoint, paramset),
506            ]
507        } else if (1..=2).contains(&mini_assert_idx) {
508            // for 1, we will have 6 derivations index starting from 0 to 5
509            // for 2, we will have 6 derivations index starting from 6 to 11
510            let derivations: u32 = (mini_assert_idx as u32 - 1) * 6;
511
512            let mut derivations_vec = vec![];
513            for i in 0..6 {
514                if derivations + i < NUM_U256 as u32 - 2 {
515                    derivations_vec.push(WinternitzDerivationPath::BitvmAssert(
516                        32 * 2,
517                        4,
518                        derivations + i,
519                        deposit_outpoint,
520                        paramset,
521                    ));
522                }
523            }
524            derivations_vec
525        } else {
526            let derivations: u32 = (mini_assert_idx as u32 - 3) * 12;
527            let mut derivations_vec = vec![];
528            for i in 0..12 {
529                if derivations + i < NUM_HASH as u32 - 3 {
530                    derivations_vec.push(WinternitzDerivationPath::BitvmAssert(
531                        16 * 2,
532                        5,
533                        derivations + i,
534                        deposit_outpoint,
535                        paramset,
536                    ));
537                }
538            }
539            derivations_vec
540        }
541    }
542    pub fn get_assert_taproot_leaf_hashes(
543        &self,
544        xonly_public_key: XOnlyPublicKey,
545    ) -> Vec<bitcoin::TapNodeHash> {
546        let assert_scripts = self.get_assert_scripts(xonly_public_key);
547        assert_scripts
548            .iter()
549            .map(|script| {
550                let taproot_builder = taproot_builder_with_scripts(&[script.to_script_buf()]);
551                taproot_builder
552                    .try_into_taptree()
553                    .expect("taproot builder always builds a full taptree")
554                    .root_hash()
555            })
556            .collect::<Vec<_>>()
557    }

568pub fn replace_disprove_scripts(pks: &ClementineBitVMPublicKeys) -> Vec<ScriptBuf> {
569    let start = Instant::now();
570    tracing::info!("Starting script replacement");
571
572    let cache = &*BITVM_CACHE;
573    let mut result: Vec<Vec<u8>> = cache.disprove_scripts.clone();
574    let replacement_places = &cache.replacement_places;
575
576    for (digit, places) in replacement_places.payout_tx_blockhash_pk.iter().enumerate() {
577        for (script_idx, pos) in places.iter() {
578            result[*script_idx][*pos..*pos + 20]
579                .copy_from_slice(&pks.payout_tx_blockhash_pk[digit]);
580        }
581    }
582
583    for (digit, places) in replacement_places.latest_blockhash_pk.iter().enumerate() {
584        for (script_idx, pos) in places.iter() {
585            result[*script_idx][*pos..*pos + 20].copy_from_slice(&pks.latest_blockhash_pk[digit]);
586        }
587    }
588
589    for (digit, places) in replacement_places
590        .challenge_sending_watchtowers_pk
591        .iter()
592        .enumerate()
593    {
594        for (script_idx, pos) in places.iter() {
595            result[*script_idx][*pos..*pos + 20]
596                .copy_from_slice(&pks.challenge_sending_watchtowers_pk[digit]);
597        }
598    }
599
600    for (digit, places) in replacement_places.bitvm_pks.0.iter().enumerate() {
601        for (pk_idx, places) in places.iter().enumerate() {
602            for (script_idx, pos) in places.iter() {
603                result[*script_idx][*pos..*pos + 20]
604                    .copy_from_slice(&pks.bitvm_pks.0[digit][pk_idx]);
605            }
606        }
607    }
608
609    for (digit, places) in replacement_places.bitvm_pks.1.iter().enumerate() {
610        for (pk_idx, places) in places.iter().enumerate() {
611            for (script_idx, pos) in places.iter() {
612                result[*script_idx][*pos..*pos + 20]
613                    .copy_from_slice(&pks.bitvm_pks.1[digit][pk_idx]);
614            }
615        }
616    }
617
618    for (digit, places) in replacement_places.bitvm_pks.2.iter().enumerate() {
619        for (pk_idx, places) in places.iter().enumerate() {
620            for (script_idx, pos) in places.iter() {
621                result[*script_idx][*pos..*pos + 20]
622                    .copy_from_slice(&pks.bitvm_pks.2[digit][pk_idx]);
623            }
624        }
625    }
626
627    let result: Vec<ScriptBuf> = result.into_iter().map(ScriptBuf::from_bytes).collect();
628
629    let elapsed = start.elapsed();
630    tracing::info!("Script replacement completed in {:?}", elapsed);
631
632    result
633}
