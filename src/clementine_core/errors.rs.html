1//! # Errors
2//!
3//! This module provides the manual `From<TxError>` implementation for `BridgeError`.
4//! Error types are now in the `clementine-errors` crate and should be imported
5//! directly from there (e.g., `use clementine_errors::BridgeError;`).
6//!
7//! This file will be removed once `TxError` is moved to `clementine-errors`.
8
9use clementine_errors::BridgeError;
10
11// Re-export TxError from its original location for backwards compatibility
12pub use crate::builder::transaction::TxError;
13
14/// Manual implementation to convert TxError into BridgeError.
15///
16/// This is implemented manually because `TxError` contains `TransactionType`,
17/// a domain-specific type that can't be moved to `clementine-errors` until
18/// `clementine-types` crate exists.
19impl From<TxError> for BridgeError {
20    fn from(err: TxError) -> Self {
21        BridgeError::Transaction(eyre::eyre!(err))
22    }
23}

35            BridgeError::IntConversionError
36                .into_eyre()

38                .into_eyre()
39                .wrap_err("some other")
40                .downcast_ref::<BridgeError>()

47    #[test]
48    fn test_status_shows_all_errors_in_chain() {
49        let err: BridgeError = Err::<(), BridgeError>(BridgeError::BitcoinRPC(
50            BitcoinRPCError::TransactionNotConfirmed,
51        ))
52        .wrap_err(tonic::Status::deadline_exceeded("Error A"))
53        .wrap_err("Error B")
54        .expect_err("should be error")
55        .into();
56
57        let status: tonic::Status = err.into();
58        assert!(status.message().contains("Error A"));
59        assert!(status.message().contains("Error B"));
60        assert!(status.message().contains("Bitcoin"));
61    }
