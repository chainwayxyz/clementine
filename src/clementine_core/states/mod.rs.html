27#[derive(Debug, Error)]
28pub enum StateMachineError {
29    #[error("State machine received event that it doesn't know how to handle: {0}")]
30    UnhandledEvent(String),
31
32    #[error(transparent)]
33    Other(#[from] eyre::Report),
34}
35pub(crate) enum ContextProcessResult<
36    T: Owner,
37    M: IntoStateMachine,
38    Fut: Future<Output = (InitializedStateMachine<M>, context::StateContext<T>)> + Send,
39> {
40    Unchanged(InitializedStateMachine<M>),
41    Processing(Fut),
42}

44/// Utility trait to make processing generic
45pub(crate) trait ContextProcessor<T: Owner, M: IntoStateMachine> {
46    /// Processes the machine with the given state context (which contains the block cache)
47    /// If the machine is unchanged, it is returned as is. Otherwise, the machine is processed
48    /// and the result is returned as a future that processes the new events.
49    fn process_with_ctx(
50        self,
51        block: &context::StateContext<T>,
52    ) -> ContextProcessResult<
53        T,
54        M,
55        impl Future<Output = (InitializedStateMachine<M>, context::StateContext<T>)> + Send,
56    >;
57}
58
59/// Generic implementation for all state machines
60impl<T, M> ContextProcessor<T, M> for InitializedStateMachine<M>
61where
62    T: Owner,
63    for<'evt, 'ctx> M: IntoStateMachine<Event<'evt> = M::StateEvent, Context<'ctx> = context::StateContext<T>>
64        + Send
65        + BlockMatcher
66        + Clone,
67    M::State: awaitable::State<M> + 'static + Send,
68    for<'sub> M::Superstate<'sub>: awaitable::Superstate<M> + Send,
69    for<'evt> M::Event<'evt>: Send + Sync,
70{
71    fn process_with_ctx(
72        mut self,
73        block: &context::StateContext<T>,
74    ) -> ContextProcessResult<T, M, impl Future<Output = (Self, context::StateContext<T>)> + Send>
75    {
76        let events = self.match_block(&block.cache);
77        if events.is_empty() {
78            ContextProcessResult::Unchanged(self)
79        } else {
80            let mut ctx = block.clone();
81            ContextProcessResult::Processing(async move {
82                for event in events {
83                    self.handle_with_context(&event, &mut ctx).await;
84                }
85                (self, ctx)
86            })
87        }
88    }
89}

109    pub async fn new(
110        db: Database,
111        owner: T,
112        paramset: &'static ProtocolParamset,
113    ) -> eyre::Result<Self> {
114        let queue = PGMQueueExt::new_with_pool(db.get_pool()).await;
115
116        queue.create(&Self::queue_name()).await.wrap_err_with(|| {
117            format!("Error creating pqmq queue with name {}", Self::queue_name())
118        })?;
119
120        let mut mgr = Self {
121            context: context::StateContext::new(
122                db.clone(),
123                Arc::new(owner.clone()),
124                Default::default(),
125                paramset,
126            ),
127            db,
128            owner,
129            paramset,
130            round_machines: Vec::new(),
131            kickoff_machines: Vec::new(),
132            queue,
133            next_height_to_process: paramset.start_height,
134        };
135
136        mgr.load_from_db().await?;
137        Ok(mgr)
138    }

140    /// Loads the state machines from the database.
141    /// This method should be called when initializing the StateManager.
142    ///
143    /// # Errors
144    /// Returns a `BridgeError` if the database operation fails
145    pub async fn load_from_db(&mut self) -> Result<(), BridgeError> {
146        // Start a transaction
147        let mut tx = self.db.begin_transaction().await?;
148
149        // Get the owner type from the context
150        let owner_type = &self.context.owner_type;
151
152        // First, check if we have any state saved
153        let status = self
154            .db
155            .get_next_height_to_process(Some(&mut tx), owner_type)
156            .await?;
157
158        // If no state is saved, return early
159        let Some(block_height) = status else {
160            tracing::info!("No state machines found in the database");
161            tx.commit().await?;
162            return Ok(());
163        };
164
165        tracing::warn!("Loading state machines from block height {}", block_height);
166
167        // Load kickoff machines
168        let kickoff_machines = self
169            .db
170            .load_kickoff_machines(Some(&mut tx), owner_type)
171            .await?;
172
173        // Process and recreate kickoff machines
174        for (state_json, kickoff_id, saved_block_height) in &kickoff_machines {
175            tracing::debug!(
176                "Loaded kickoff machine: state={}, block_height={}",
177                state_json,
178                saved_block_height
179            );
180
181            // Deserialize the machine state from JSON
182            let machine: Result<UninitializedStateMachine<kickoff::KickoffStateMachine<T>>, _> =
183                serde_json::from_str(state_json);
184
185            match machine {
186                Ok(uninitialized) => {
187                    // Create a context for initialization
188                    let mut ctx = context::StateContext::new(
189                        self.db.clone(),
190                        Arc::new(self.owner.clone()),
191                        Default::default(),
192                        self.paramset,
193                    );
194
195                    // Initialize the machine with the context
196                    let initialized = uninitialized.init_with_context(&mut ctx).await;
197                    self.kickoff_machines.push(initialized);
198                }
199                Err(e) => {
200                    tracing::warn!(
201                        "Failed to deserialize kickoff machine with ID {}: {}",
202                        kickoff_id,
203                        e
204                    );
205                }
206            }
207        }
208
209        // Load round machines
210        let round_machines = self
211            .db
212            .load_round_machines(Some(&mut tx), owner_type)
213            .await?;
214
215        // Process and recreate round machines
216        for (state_json, operator_xonly_pk, saved_block_height) in &round_machines {
217            tracing::debug!(
218                "Loaded round machine: state={}, block_height={}",
219                state_json,
220                saved_block_height
221            );
222
223            // Deserialize the machine state from JSON
224            let machine: Result<UninitializedStateMachine<round::RoundStateMachine<T>>, _> =
225                serde_json::from_str(state_json);
226
227            match machine {
228                Ok(uninitialized) => {
229                    // Create a context for initialization
230                    let mut ctx = context::StateContext::new(
231                        self.db.clone(),
232                        Arc::new(self.owner.clone()),
233                        Default::default(),
234                        self.paramset,
235                    );
236
237                    // Initialize the machine with the context
238                    let initialized = uninitialized.init_with_context(&mut ctx).await;
239                    self.round_machines.push(initialized);
240                }
241                Err(e) => {
242                    tracing::warn!(
243                        "Failed to deserialize round machine with operator index {:?}: {}",
244                        operator_xonly_pk,
245                        e
246                    );
247                }
248            }
249        }
250
251        tracing::info!(
252            "Loaded {} kickoff machines and {} round machines from the database",
253            kickoff_machines.len(),
254            round_machines.len()
255        );
256
257        tx.commit().await?;
258        self.next_height_to_process =
259            u32::try_from(block_height).wrap_err(BridgeError::IntConversionError)?;
260        Ok(())
261    }

276    /// Saves the state machines to the database. Resets the dirty flag for all machines after successful save.
277    ///
278    /// # Errors
279    /// Returns a `BridgeError` if the database operation fails.
280    ///
281    /// # TODO
282    /// We should only save `dirty` machines but we currently save all of them.
283    pub async fn save_state_to_db(
284        &mut self,
285        block_height: u32,
286        dbtx: Option<DatabaseTransaction<'_, '_>>,
287    ) -> eyre::Result<()> {
288        // Get the owner type from the context
289        let owner_type = &self.context.owner_type;
290
291        // Prepare kickoff machines data with direct serialization
292        let kickoff_machines: eyre::Result<Vec<_>> = self
293            .kickoff_machines
294            .iter()
295            .map(|machine| -> eyre::Result<_> {
296                // Directly serialize the machine
297                let state_json = serde_json::to_string(&machine).wrap_err_with(|| {
298                    format!("Failed to serialize kickoff machine: {:?}", machine)
299                })?;
300                let kickoff_id =
301                    serde_json::to_string(&machine.kickoff_data).wrap_err_with(|| {
302                        format!("Failed to serialize kickoff id for machine: {:?}", machine)
303                    })?;
304
305                // Use the machine's dirty flag to determine if it needs updating
306                Ok((state_json, (kickoff_id), machine.dirty))
307            })
308            .collect();
309
310        // Prepare round machines data with direct serialization
311        let round_machines: eyre::Result<Vec<_>> = self
312            .round_machines
313            .iter()
314            .map(|machine| -> eyre::Result<_> {
315                let state_json = serde_json::to_string(machine).wrap_err_with(|| {
316                    format!("Failed to serialize round machine: {:?}", machine)
317                })?;
318                let operator_xonly_pk = machine.operator_data.xonly_pk;
319
320                // Use the machine's dirty flag to determine if it needs updating
321                Ok((state_json, (operator_xonly_pk), machine.dirty))
322            })
323            .collect();
324
325        // Use the database function to save the state machines
326        self.db
327            .save_state_machines(
328                dbtx,
329                kickoff_machines?,
330                round_machines?,
331                block_height as i32,
332                owner_type,
333            )
334            .await?;
335
336        // Reset the dirty flag for all machines after successful save
337        for machine in &mut self.kickoff_machines {
338            if machine.dirty {
339                machine
340                    .handle_with_context(&KickoffEvent::SavedToDb, &mut self.context)
341                    .await;
342            }
343        }
344
345        for machine in &mut self.round_machines {
346            if machine.dirty {
347                machine
348                    .handle_with_context(&RoundEvent::SavedToDb, &mut self.context)
349                    .await;
350            }
351        }
352
353        Ok(())
354    }

360    /// Updates the machines using the context and returns machines without
361    /// events and futures that process new events for machines that changed.
362    /// Empties the `machines` vector.
363    ///
364    /// # Parameters
365    /// * `machines`: A mutable reference to the vector of state machines to update.
366    /// * `base_context`: A reference to the base state context.
367    ///
368    /// # Returns
369    /// A tuple of the unchanged machines and the futures that process new
370    /// events for machines that generated events.
371    ///
372    /// # Type Parameters
373    /// * `M`: The type of the state machine.
374    /// * `a`: The lifetime of the state context reference (the future captures the context by reference).
375    #[allow(clippy::type_complexity)]
376    fn update_machines<'a, M>(
377        machines: &mut Vec<InitializedStateMachine<M>>,
378        base_context: &'a context::StateContext<T>,
379    ) -> (
380        Vec<InitializedStateMachine<M>>,
381        Vec<
382            impl Future<Output = (InitializedStateMachine<M>, context::StateContext<T>)> + Send + 'a,
383        >,
384    )
385    where
386        M: IntoStateMachine + Send + Sync + 'static,
387        M::State: Send + Sync + 'static,
388        InitializedStateMachine<M>: ContextProcessor<T, M>,
389    {
390        let mut unchanged_machines = Vec::new();
391        let mut processing_futures = Vec::new();
392
393        for machine in std::mem::take(machines).into_iter() {
394            match machine.process_with_ctx(base_context) {
395                ContextProcessResult::Processing(future) => processing_futures.push(future),
396                ContextProcessResult::Unchanged(machine) => unchanged_machines.push(machine),
397            }
398        }
399
400        (unchanged_machines, processing_futures)
401    }

403    pub async fn process_and_add_new_states_from_height(
404        &mut self,
405        new_round_machines: Vec<InitializedStateMachine<round::RoundStateMachine<T>>>,
406        new_kickoff_machines: Vec<InitializedStateMachine<kickoff::KickoffStateMachine<T>>>,
407        start_height: u32,
408    ) -> Result<(), eyre::Report> {
409        // save old round machines, only process new ones and then append old machines back
410        let saved_round_machines = std::mem::take(&mut self.round_machines);
411        let saved_kickoff_machines = std::mem::take(&mut self.kickoff_machines);
412
413        self.round_machines = new_round_machines;
414        self.kickoff_machines = new_kickoff_machines;
415
416        for block_height in start_height..self.next_height_to_process {
417            let block = self.db.get_full_block(None, block_height).await?;
418            if let Some(block) = block {
419                self.update_block_cache(&block, block_height);
420                self.process_block_parallel(block_height).await?;
421            } else {
422                return Err(eyre::eyre!(
423                    "Block at height {} not found in process_and_add_new_states_from_height",
424                    block_height
425                ));
426            }
427        }
428
429        // append saved states
430        self.round_machines.extend(saved_round_machines);
431        self.kickoff_machines.extend(saved_kickoff_machines);
432
433        Ok(())
434    }

436    /// It requires that the block cache is updated before calling this function.
437    /// Moves all state machines forward in parallel.
438    /// The state machines are updated until all of them stabilize in their state (ie.
439    /// the block does not generate any new events)
440    ///
441    /// # Errors
442    /// If the state machines do not stabilize after some iterations, we return an error.
443    pub async fn process_block_parallel(&mut self, block_height: u32) -> Result<(), eyre::Report> {
444        eyre::ensure!(
445            self.context.cache.block_height == block_height,
446            "Block cache is not updated"
447        );
448
449        let kickoff_machines_checkpoint = self.kickoff_machines.clone();
450        let round_machines_checkpoint = self.round_machines.clone();
451
452        // Process all machines, for those unaffected collect them them, otherwise return
453        // a future that processes the new events.
454        let (mut final_kickoff_machines, mut kickoff_futures) =
455            Self::update_machines(&mut self.kickoff_machines, &self.context);
456        let (mut final_round_machines, mut round_futures) =
457            Self::update_machines(&mut self.round_machines, &self.context);
458
459        let mut iterations = 0;
460
461        // On each iteration, we'll update the changed machines until all machines
462        // stabilize in their state.
463        while !kickoff_futures.is_empty() || !round_futures.is_empty() {
464            // Execute all futures in parallel
465            let (kickoff_results, round_results) =
466                join(join_all(kickoff_futures), join_all(round_futures)).await;
467
468            // Unzip the results into updated machines and state contexts
469            let (mut changed_kickoff_machines, mut kickoff_contexts): (Vec<_>, Vec<_>) =
470                kickoff_results.into_iter().unzip();
471            let (mut changed_round_machines, mut round_contexts): (Vec<_>, Vec<_>) =
472                round_results.into_iter().unzip();
473
474            // Merge and handle errors
475            let mut all_errors = Vec::new();
476            for ctx in kickoff_contexts.iter_mut().chain(round_contexts.iter_mut()) {
477                all_errors.extend(std::mem::take(&mut ctx.errors));
478            }
479
480            if !all_errors.is_empty() {
481                tracing::warn!(
482                    "Multiple errors occurred during state processing: owner: {:?}, errors: {:?}",
483                    self.context.owner_type,
484                    all_errors
485                );
486                // revert state machines to the saved state as the content of the machines might be changed before the error occurred
487                self.kickoff_machines = kickoff_machines_checkpoint;
488                self.round_machines = round_machines_checkpoint;
489                // Return first error or create a combined error
490                return Err(BridgeError::Error(format!(
491                    "Multiple errors occurred during state processing: {:?}",
492                    all_errors
493                ))
494                .into());
495            }
496
497            // Append the newly generated state machines into the changed machines list
498            for ctx in kickoff_contexts.iter_mut().chain(round_contexts.iter_mut()) {
499                #[cfg(debug_assertions)]
500                for machine in &ctx.new_round_machines {
501                    if !machine.dirty {
502                        panic!(
503                            "Round machine not dirty despite having been newly created: {:?}",
504                            machine.state()
505                        );
506                    }
507                }
508                #[cfg(debug_assertions)]
509                for machine in &ctx.new_kickoff_machines {
510                    if !machine.dirty {
511                        panic!(
512                            "Kickoff machine not dirty despite having been newly created: {:?}",
513                            machine.state()
514                        );
515                    }
516                }
517                changed_round_machines.extend(std::mem::take(&mut ctx.new_round_machines));
518                changed_kickoff_machines.extend(std::mem::take(&mut ctx.new_kickoff_machines));
519            }
520
521            // If the machines do not stabilize after a while, we return an error
522            // TODO: https://github.com/chainwayxyz/clementine/issues/675
523            // I think something like max(2 * num_kickoffs_per_round, number of utxos in a kickoff * 2) is a safe value
524            if iterations > 100000 {
525                return Err(eyre::eyre!(
526                    r#"{}/{} kickoff and {}/{} round state machines did not stabilize after 100000 iterations, debug repr of changed machines:
527                        ---- Kickoff machines ----
528                        {:?}
529                        ---- Round machines ----
530                        {:?}
531                        "#,
532                    changed_kickoff_machines.len(),
533                    final_kickoff_machines.len() + changed_kickoff_machines.len(),
534                    changed_round_machines.len(),
535                    final_round_machines.len() + changed_round_machines.len(),
536                    changed_kickoff_machines
537                        .iter()
538                        .map(|m| m.state())
539                        .collect::<Vec<_>>(),
540                    changed_round_machines
541                        .iter()
542                        .map(|m| m.state())
543                        .collect::<Vec<_>>(),
544                ));
545            }
546
547            // Reprocess changed machines and commit these futures to be handled
548            // in the next round If they're empty, we'll exit the loop.
549            let (finalized_kickoff_machines, new_kickoff_futures) =
550                Self::update_machines(&mut changed_kickoff_machines, &self.context);
551            let (finalized_round_machines, new_round_futures) =
552                Self::update_machines(&mut changed_round_machines, &self.context);
553            final_kickoff_machines.extend(finalized_kickoff_machines);
554            final_round_machines.extend(finalized_round_machines);
555
556            // Update the futures to be processed
557            kickoff_futures = new_kickoff_futures;
558            round_futures = new_round_futures;
559            iterations += 1;
560        }
561
562        drop(kickoff_futures);
563        drop(round_futures);
564
565        // Set back the original machines
566        self.round_machines = final_round_machines;
567        self.kickoff_machines = final_kickoff_machines;
568        self.next_height_to_process = max(block_height + 1, self.next_height_to_process);
569
570        Ok(())
571    }
