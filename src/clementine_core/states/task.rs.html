26/// A task that fetches new blocks from Bitcoin and adds them to the state management queue
27#[derive(Debug)]
28pub struct BlockFetcherTask<T: Owner + std::fmt::Debug + 'static> {
29    /// The database to fetch events from
30    db: Database,
31    /// Queue for sending events
32    queue: PGMQueueExt,
33    /// Queue name for this owner type
34    queue_name: String,
35    /// The next height to process
36    next_height: u32,
37    /// Protocol parameters
38    paramset: &'static ProtocolParamset,
39    /// Owner type marker
40    _phantom: std::marker::PhantomData<T>,
41}
42
43impl<T: Owner + std::fmt::Debug + 'static> BlockFetcherTask<T> {
44    /// Creates a new block fetcher task
45    pub async fn new(
46        next_height: u32,
47        db: Database,
48        paramset: &'static ProtocolParamset,
49    ) -> Result<Self, BridgeError> {
50        let queue = PGMQueueExt::new_with_pool(db.get_pool()).await;
51        let queue_name = StateManager::<T>::queue_name();
52
53        tracing::info!(
54            "Creating block fetcher task for owner type {} starting from height {}",
55            T::OWNER_TYPE,
56            next_height
57        );
58
59        Ok(Self {
60            db,
61            queue,
62            queue_name,
63            next_height,
64            paramset,
65            _phantom: std::marker::PhantomData,
66        })
67    }
68}

70#[async_trait]
71impl<T: Owner + std::fmt::Debug + 'static> Task for BlockFetcherTask<T> {
72    type Output = bool;
73
74    async fn run_once(&mut self) -> Result<Self::Output, BridgeError> {
75        let mut dbtx = self.db.begin_transaction().await?;
76
77        // Poll for the next bitcoin syncer event
78        let Some(event) = self
79            .db
80            .fetch_next_bitcoin_syncer_evt(&mut dbtx, &self.queue_name)
81            .await?
82        else {
83            // No event found, we can safely commit the transaction and return
84            dbtx.commit().await?;
85            return Ok(false);
86        };
87
88        // Process the event
89        let did_find_new_block = match event {
90            BitcoinSyncerEvent::NewBlock(block_id) => {
91                let current_tip_height = self
92                    .db
93                    .get_block_info_from_id(Some(&mut dbtx), block_id)
94                    .await?
95                    .ok_or(BridgeError::Error(
96                        "Block not found in BlockFetcherTask".to_string(),
97                    ))?
98                    .1;
99                let mut new_tip = false;
100
101                // Update states to catch up to finalized chain
102                while current_tip_height >= self.paramset.finality_depth
103                    && self.next_height <= current_tip_height - self.paramset.finality_depth
104                {
105                    new_tip = true;
106
107                    let block = self
108                        .db
109                        .get_full_block(Some(&mut dbtx), self.next_height)
110                        .await?
111                        .ok_or(BridgeError::Error(format!(
112                            "Block at height {} not found in BlockFetcherTask, current tip height is {}",
113                            self.next_height, current_tip_height
114                        )))?;
115
116                    let new_block_id = self
117                        .db
118                        .get_canonical_block_id_from_height(Some(&mut dbtx), self.next_height)
119                        .await?;
120
121                    if new_block_id.is_none() {
122                        tracing::error!("Block at height {} not found in BlockFetcherTask, current tip height is {}", self.next_height, current_tip_height);
123                        return Err(BridgeError::Error(format!(
124                            "Block at height {} not found in BlockFetcherTask, current tip height is {}",
125                            self.next_height, current_tip_height
126                        )));
127                    }
128
129                    let event = SystemEvent::NewBlock {
130                        block_id: new_block_id.expect("it must be some"),
131                        block,
132                        height: self.next_height,
133                    };
134
135                    self.queue
136                        .send_with_cxn(&self.queue_name, &event, &mut *dbtx)
137                        .await
138                        .map_err(|e| {
139                            BridgeError::Error(format!(
140                                "Error sending new block event to queue: {:?}",
141                                e
142                            ))
143                        })?;
144
145                    self.next_height += 1;
146                }
147
148                new_tip
149            }
150            BitcoinSyncerEvent::ReorgedBlock(_) => false,
151        };
152
153        dbtx.commit().await?;
154        // Return whether we found new blocks
155        Ok(did_find_new_block)
156    }
157}

159#[derive(Debug)]
160pub struct MessageConsumerTask<T: Owner + std::fmt::Debug + 'static> {
161    db: Database,
162    inner: StateManager<T>,
163    /// Queue name for this owner type (cached)
164    queue_name: String,
165}
166
167#[async_trait]
168impl<T: Owner + std::fmt::Debug + 'static> Task for MessageConsumerTask<T> {
169    type Output = bool;
170
171    async fn run_once(&mut self) -> Result<Self::Output, BridgeError> {
172        let new_event_received = async {
173            let mut dbtx = self.db.begin_transaction().await?;
174
175            // Poll new event
176            let Some(Message {
177                msg_id, message, ..
178            }): Option<Message<SystemEvent>> = self
179                .inner
180                .queue
181                // 2nd param of read_with_cxn is the visibility timeout, set to 0 as we only have 1 consumer of the queue, which is the state machine
182                .read_with_cxn(&self.queue_name, 0, &mut *dbtx)
183                .await
184                .wrap_err("Reading event from queue")?
185            else {
186                dbtx.commit().await?;
187                return Ok::<_, BridgeError>(false);
188            };
189
190            self.inner.handle_event(message, &mut dbtx).await?;
191
192            // Delete event from queue
193            self.inner
194                .queue
195                .archive_with_cxn(&self.queue_name, msg_id, &mut *dbtx)
196                .await
197                .wrap_err("Deleting event from queue")?;
198
199            dbtx.commit().await?;
200            Ok(true)
201        }
202        .await?;
203
204        Ok(new_event_received)
205    }
206}

208impl<T: Owner + std::fmt::Debug + 'static> IntoTask for StateManager<T> {
209    type Task = WithDelay<BufferedErrors<MessageConsumerTask<T>>>;
210
211    /// Converts the StateManager into the consumer task with a delay
212    fn into_task(self) -> Self::Task {
213        MessageConsumerTask {
214            db: self.db.clone(),
215            inner: self,
216            queue_name: StateManager::<T>::queue_name(),
217        }
218        .into_buffered_errors(50)
219        .with_delay(POLL_DELAY)
220    }
221}

223impl<T: Owner + std::fmt::Debug + 'static> StateManager<T> {
224    pub async fn block_fetcher_task(&self) -> Result<WithDelay<BlockFetcherTask<T>>, BridgeError> {
225        Ok(
226            BlockFetcherTask::<T>::new(self.next_height_to_process, self.db.clone(), self.paramset)
227                .await?
228                .with_delay(POLL_DELAY),
229        )
230    }
