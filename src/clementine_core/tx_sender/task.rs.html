23#[derive(Debug)]
24pub struct TxSenderTask {
25    db: Database,
26    current_tip_height: u32,
27    inner: TxSender,
28}
29
30#[async_trait]
31impl Task for TxSenderTask {
32    type Output = bool;
33
34    async fn run_once(&mut self) -> std::result::Result<Self::Output, BridgeError> {
35        let mut dbtx = self.db.begin_transaction().await.map_to_eyre()?;
36
37        let is_block_update = async {
38            let Some(event) = self
39                .db
40                .fetch_next_bitcoin_syncer_evt(&mut dbtx, &self.inner.btc_syncer_consumer_id)
41                .await?
42            else {
43                return Ok(false);
44            };
45
46            tracing::info!("TXSENDER: Event: {:?}", event);
47            Ok::<_, BridgeError>(match event {
48                BitcoinSyncerEvent::NewBlock(block_id) => {
49                    self.db.confirm_transactions(&mut dbtx, block_id).await?;
50                    self.current_tip_height = self
51                        .db
52                        .get_block_info_from_id(Some(&mut dbtx), block_id)
53                        .await?
54                        .ok_or(BridgeError::Error(
55                            "Block not found in TxSenderTask".to_string(),
56                        ))?
57                        .1;
58
59                    tracing::info!("TXSENDER: Confirmed transactions for block {}", block_id);
60                    dbtx.commit().await?;
61                    true
62                }
63                BitcoinSyncerEvent::ReorgedBlock(block_id) => {
64                    tracing::info!("TXSENDER: Unconfirming transactions for block {}", block_id);
65                    self.db.unconfirm_transactions(&mut dbtx, block_id).await?;
66                    dbtx.commit().await?;
67                    false
68                }
69            })
70        }
71        .await?;
72
73        if is_block_update {
74            // Pull in all block updates before trying to send.
75            return Ok(true);
76        }
77
78        // tracing::info!("TXSENDER: Getting fee rate");
79        // let fee_rate_result = self.inner.get_fee_rate().await;
80        // tracing::info!("TXSENDER: Fee rate result: {:?}", fee_rate_result);
81        // let fee_rate = fee_rate_result?;
82        // tracing::info!("TXSENDER: Trying to send unconfirmed txs");
83        let fee_rate = FeeRate::from_sat_per_vb_unchecked(1);
84
85        self.inner
86            .try_to_send_unconfirmed_txs(fee_rate, self.current_tip_height)
87            .await?;
88
89        Ok(false)
90    }
91}
92
93impl IntoTask for TxSender {
94    type Task = WithDelay<IgnoreError<TxSenderTask>>;
95
96    fn into_task(self) -> Self::Task {
97        TxSenderTask {
98            db: self.db.clone(),
99            current_tip_height: 0,
100            inner: self,
101        }
102        .ignore_error()
103        .with_delay(POLL_DELAY)
104    }
105}
