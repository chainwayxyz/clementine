47/// Manages the process of sending Bitcoin transactions, including handling fee bumping
48/// strategies like Replace-By-Fee (RBF) and Child-Pays-For-Parent (CPFP).
49///
50/// It interacts with a Bitcoin Core RPC endpoint (`ExtendedRpc`) to query network state
51/// (like fee rates) and submit transactions. It uses a `Database` to persist transaction

82    /// from the original "parent" transaction. The child pays a high fee, sufficient

87    /// version that includes a higher fee. The original transaction must signal

171    /// Gets the current recommended fee rate from the Bitcoin Core node.
172    ///
173    /// Uses the `estimatesmartfee` RPC call with a confirmation target of 1 block
174    /// and conservative estimation mode.
175    ///
176    /// If fee estimation is unavailable (e.g., node is warming up), it returns
177    /// an error, except in Regtest mode where it defaults to 1 sat/vB for testing convenience.
178    ///
179    /// TODO: Implement more sophisticated fee estimation (e.g., mempool.space API).
180    async fn _get_fee_rate(&self) -> Result<FeeRate> {
181        tracing::info!("Getting fee rate");
182        let fee_rate = self
183            .rpc
184            .client
185            .estimate_smart_fee(1, Some(EstimateMode::Conservative))
186            .await;
187
188        match fee_rate {
189            Ok(fee_rate) => match fee_rate.fee_rate {
190                Some(fee_rate) => Ok(FeeRate::from_sat_per_kwu(fee_rate.to_sat())),
191                None => {
192                    if self.network == bitcoin::Network::Regtest {
193                        tracing::debug!("Using fee rate of 1 sat/vb (Regtest mode)");
194                        return Ok(FeeRate::from_sat_per_vb_unchecked(1));
195                    }
196
197                    tracing::error!("TXSENDER: Fee estimation error: {:?}", fee_rate.errors);
198                    Ok(FeeRate::from_sat_per_vb_unchecked(1))
199                }
200            },
201            Err(e) => {
202                tracing::error!("TXSENDER: Error getting fee rate: {:?}", e);
203                Ok(FeeRate::from_sat_per_vb_unchecked(1))
204            }
205        }
206    }

208    /// Calculates the total fee required for a transaction package based on the fee bumping strategy.
209    ///
210    /// # Arguments
211    /// * `parent_tx_weight` - The weight of the main transaction being bumped.
212    /// * `num_fee_payer_utxos` - The number of fee payer UTXOs used (relevant for child tx size in CPFP).
213    /// * `fee_rate` - The target fee rate (sat/kwu or similar).
214    /// * `fee_paying_type` - The strategy being used (CPFP or RBF).
215    ///
216    /// # Calculation Logic
217    /// *   **CPFP:** Calculates the weight of the hypothetical child transaction based on the
218    ///     number of fee payer inputs and standard P2TR output sizes. It then calculates the
219    ///     fee based on the *combined virtual size* (vbytes) of the parent and child transactions,
220    ///     as miners evaluate the package deal.
221    /// *   **RBF:** Calculates the weight of the replacement transaction itself (assuming inputs
222    ///     and potentially outputs change slightly). The fee is calculated based on the weight
223    ///     of this single replacement transaction.
224    ///
225    /// Reference for weight estimates: <https://bitcoin.stackexchange.com/a/116959>
226    fn calculate_required_fee(
227        parent_tx_weight: Weight,
228        num_fee_payer_utxos: usize,
229        fee_rate: FeeRate,
230        fee_paying_type: FeePayingType,
231    ) -> Result<Amount> {
232        tracing::info!(
233            "Calculating required fee for {} fee payer utxos",
234            num_fee_payer_utxos
235        );
236        // Estimate the weight of the child transaction (for CPFP) or the RBF replacement.
237        // P2TR input witness adds ~57.5vbytes (230 WU). P2TR output adds 43 vbytes (172 WU).
238        // Base transaction overhead (version, locktime, input/output counts) ~ 10.5 vBytes (42 WU)
239        // Anchor input marker (OP_FALSE OP_RETURN ..) adds overhead. Exact WU TBD.
240        // For CPFP child: (N fee payer inputs) + (1 anchor input) + (1 change output)
241        // For RBF replacement: (N fee payer inputs) + (1 change output) - assuming it replaces a tx with an anchor.
242        // TODO: Refine these weight estimations.
243        let child_tx_weight = match fee_paying_type {
244            // CPFP Child: N fee payer inputs + 1 anchor input + 1 change output + base overhead.
245            // Approx WU: (230 * num_fee_payer_utxos) + 230 + 172 + base_overhead_wu
246            // Simplified calculation used here needs verification.
247            FeePayingType::CPFP => Weight::from_wu_usize(230 * num_fee_payer_utxos + 207 + 172), // TODO: Verify 207 WU for anchor input + overhead
248            // RBF Replacement: N fee payer inputs + 1 change output + base overhead.
249            // Assumes it replaces a tx of similar structure but potentially different inputs/fees.
250            // Simplified calculation used here needs verification.
251            FeePayingType::RBF => Weight::from_wu_usize(230 * num_fee_payer_utxos + 172), // TODO: Verify WU for RBF structure
252        };
253
254        // Calculate total weight for fee calculation.
255        // For CPFP, miners consider the effective fee rate over the combined *vbytes* of parent + child.
256        // For RBF, miners consider the fee rate of the single replacement transaction's weight.
257        let total_weight = match fee_paying_type {
258            FeePayingType::CPFP => Weight::from_vb_unchecked(
259                child_tx_weight.to_vbytes_ceil() + parent_tx_weight.to_vbytes_ceil(),
260            ),
261            FeePayingType::RBF => child_tx_weight + parent_tx_weight, // Should likely just be the RBF tx weight? Check RBF rules.
262        };
263
264        fee_rate
265            .checked_mul_by_weight(total_weight)
266            .ok_or_eyre("Fee calculation overflow")
267            .map_err(Into::into)
268    }

295    /// The main loop for processing transactions requiring fee bumps or initial sending.
296    ///
297    /// Fetches transactions from the database that are eligible to be sent or bumped
298    /// based on the `new_fee_rate` and `current_tip_height`.
299    ///
300    /// For each eligible transaction (`id`):
301    /// 1.  **Bump Fee Payers:** Calls `bump_fees_of_fee_payer_txs` to ensure any associated,
302    ///     unconfirmed fee payer UTXOs (used for CPFP) are themselves confirmed.
303    /// 2.  **Send/Bump Main Tx:** Calls `send_tx` to either perform RBF or CPFP on the main
304    ///     transaction (`id`) using the `new_fee_rate`.
305    /// 3.  **Handle Errors:**
306    ///     - `UnconfirmedFeePayerUTXOsLeft`: Skips the current tx, waiting for fee payers to confirm.
307    ///     - `InsufficientFeePayerAmount`: Calls `create_fee_payer_utxo` to provision more funds
308    ///       for a future CPFP attempt.
309    ///     - Other errors are logged.
310    ///
311    /// # Arguments
312    /// * `new_fee_rate` - The current target fee rate based on network conditions.
313    /// * `current_tip_height` - The current blockchain height, used for time-lock checks.
314    #[tracing::instrument(skip_all, fields(sender = self.btc_syncer_consumer_id, new_fee_rate, current_tip_height))]
315    async fn try_to_send_unconfirmed_txs(
316        &self,
317        new_fee_rate: FeeRate,
318        current_tip_height: u32,
319    ) -> Result<()> {
320        // tracing::info!("Trying to send unconfirmed txs with new fee rate: {new_fee_rate:?}, current tip height: {current_tip_height:?}");
321        let txs = self
322            .db
323            .get_sendable_txs(None, new_fee_rate, current_tip_height)
324            .await
325            .map_to_eyre()?;
326
327        if !txs.is_empty() {
328            tracing::debug!("Trying to send {} sendable txs ", txs.len());
329        }
330
331        #[cfg(test)]
332        {
333            if env::var("TXSENDER_DBG_INACTIVE_TXS").is_ok() {
334                self.db
335                    .debug_inactive_txs(new_fee_rate, current_tip_height)
336                    .await;
337            }
338        }
339
340        for id in txs {
341            // Update debug state
342            tracing::debug!(
343                try_to_send_id = id,
344                "Processing TX in try_to_send_unconfirmed_txs with fee rate {new_fee_rate}",
345            );
346
347            let (tx_metadata, tx, fee_paying_type, seen_block_id, rbf_signing_info) =
348                match self.db.get_try_to_send_tx(None, id).await {
349                    Ok(res) => res,
350                    Err(e) => {
351                        log_error_for_tx!(self.db, id, format!("Failed to get tx details: {}", e));
352                        continue;
353                    }
354                };
355
356            // Check if the transaction is already confirmed (only happens if it was confirmed after this loop started)
357            if let Some(block_id) = seen_block_id {
358                tracing::debug!(
359                    try_to_send_id = id,
360                    "Transaction confirmed in block {}",
361                    block_id
362                );
363
364                // Update sending state
365                let _ = self
366                    .db
367                    .update_tx_debug_sending_state(id, "confirmed", true)
368                    .await;
369
370                continue;
371            }
372
373            let result = match fee_paying_type {
374                FeePayingType::CPFP => self.send_cpfp_tx(id, tx, tx_metadata, new_fee_rate).await,
375                FeePayingType::RBF => {
376                    self.send_rbf_tx(id, tx, tx_metadata, new_fee_rate, rbf_signing_info)
377                        .await
378                }
379            };
380
381            if let Err(e) = result {
382                log_error_for_tx!(self.db, id, format!("Failed to send tx: {:?}", e));
383            }
384        }
385
386        Ok(())
387    }

494    async fn create_bumpable_tx(
495        rpc: &ExtendedRpc,
496        signer: &Actor,
497        network: bitcoin::Network,
498        fee_paying_type: FeePayingType,
499        requires_rbf_signing_info: bool,
500    ) -> Result<Transaction, BridgeError> {
501        let (address, spend_info) =
502            builder::address::create_taproot_address(&[], Some(signer.xonly_public_key), network);
503
504        let amount = Amount::from_sat(100000);
505        let outpoint = rpc.send_to_address(&address, amount).await?;
506        rpc.mine_blocks(1).await?;
507
508        let version = match fee_paying_type {
509            FeePayingType::CPFP => Version::non_standard(3),
510            FeePayingType::RBF => Version::TWO,
511        };
512
513        let mut txhandler = TxHandlerBuilder::new(TransactionType::Dummy)
514            .with_version(version)
515            .add_input(
516                match fee_paying_type {
517                    FeePayingType::CPFP => {
518                        SignatureId::from(NormalSignatureKind::OperatorSighashDefault)
519                    }
520                    FeePayingType::RBF if !requires_rbf_signing_info => {
521                        NormalSignatureKind::Challenge.into()
522                    }
523                    FeePayingType::RBF => (NumberedSignatureKind::WatchtowerChallenge, 0i32).into(),
524                },
525                SpendableTxIn::new(
526                    outpoint,
527                    TxOut {
528                        value: amount,
529                        script_pubkey: address.script_pubkey(),
530                    },
531                    vec![],
532                    Some(spend_info),
533                ),
534                SpendPath::KeySpend,
535                DEFAULT_SEQUENCE,
536            )
537            .add_output(UnspentTxOut::from_partial(TxOut {
538                value: amount
539                    - builder::transaction::anchor_output().value
540                    - MIN_TAPROOT_AMOUNT * 3, // buffer so that rbf works without adding inputs
541                script_pubkey: address.script_pubkey(), // In practice, should be the wallet address, not the signer address
542            }))
543            .add_output(UnspentTxOut::from_partial(
544                builder::transaction::anchor_output(),
545            ))
546            .finalize();
547
548        signer
549            .tx_sign_and_fill_sigs(&mut txhandler, &[], None)
550            .unwrap();
551
552        let tx = txhandler.get_cached_tx().clone();
553        Ok(tx)
554    }

654            signer.clone(),
655            rpc.clone(),
656            db,
657            "tx_sender".into(),
658            config.protocol_paramset().network,
659        );
660
661        let scripts: Vec<Arc<dyn SpendableScript>> =
662            vec![Arc::new(CheckSig::new(xonly_pk)).clone()];
663        let (taproot_address, taproot_spend_info) = builder::address::create_taproot_address(

665                .iter()
