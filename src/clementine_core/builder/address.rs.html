24/// A helper to construct a `TaprootBuilder` from a slice of script buffers, forming the script tree.
25/// Finds the needed depth the script tree needs to be to fit all the scripts and inserts the scripts.
26pub fn taproot_builder_with_scripts(scripts: impl Into<Vec<ScriptBuf>>) -> TaprootBuilder {
27    // doesn't clone if its already an owned Vec
28    let mut scripts: Vec<ScriptBuf> = scripts.into();
29    let builder = TaprootBuilder::new();
30    let num_scripts = scripts.len();
31
32    // Special return cases for n = 0 or n = 1
33    match num_scripts {
34        0 => return builder,
35        1 => {
36            return builder
37                .add_leaf(0, scripts.remove(0))
38                .expect("one root leaf added on empty builder")
39        }
40        _ => {}
41    }
42
43    let deepest_layer_depth: u8 = ((num_scripts - 1).ilog2() + 1) as u8;
44
45    let num_empty_nodes_in_final_depth = 2_usize.pow(deepest_layer_depth.into()) - num_scripts;
46    let num_nodes_in_final_depth = num_scripts - num_empty_nodes_in_final_depth;
47
48    scripts
49        .into_iter()
50        .enumerate()
51        .fold(builder, |acc, (i, script)| {
52            let is_node_in_last_minus_one_depth = (i >= num_nodes_in_final_depth) as u8;
53
54            acc.add_leaf(
55                deepest_layer_depth - is_node_in_last_minus_one_depth,
56                script,
57            )
58            .expect("algorithm tested to be correct")
59        })
60}

62/// Calculates the depth of each leaf in a balanced Taproot tree structure.
63/// The returned Vec contains the depth for each script at the corresponding index.
64pub fn calculate_taproot_leaf_depths(num_scripts: usize) -> Vec<u8> {
65    match num_scripts {
66        0 => return vec![],
67        1 => return vec![0],
68        _ => {}
69    }
70
71    let deepest_layer_depth: u8 = ((num_scripts - 1).ilog2() + 1) as u8;
72
73    let num_empty_nodes_in_final_depth = 2_usize.pow(deepest_layer_depth.into()) - num_scripts;
74    let num_nodes_in_final_depth = num_scripts - num_empty_nodes_in_final_depth;
75
76    (0..num_scripts)
77        .map(|i| {
78            let is_node_in_last_minus_one_depth = (i >= num_nodes_in_final_depth) as u8;
79            deepest_layer_depth - is_node_in_last_minus_one_depth
80        })
81        .collect()
82}

84/// Creates a taproot address with given scripts and internal key.
85///
86/// # Arguments
87///
88/// - `scripts`: If empty, it is most likely a key path spend address
89/// - `internal_key`: If not given, will be defaulted to an unspendable x-only public key
90/// - `network`: Bitcoin network
91/// - If both `scripts` and `internal_key` are given, it means one can spend using both script and key path.
92/// - If none given, it is an unspendable address.
93///
94/// # Returns
95///
96/// - [`Address`]: Generated taproot address
97/// - [`TaprootSpendInfo`]: Taproot spending information
98///
99/// # Panics
100///
101/// Will panic if some of the operations have invalid parameters.
102pub fn create_taproot_address(
103    scripts: &[ScriptBuf],
104    internal_key: Option<XOnlyPublicKey>,
105    network: bitcoin::Network,
106) -> (Address, TaprootSpendInfo) {
107    // Build script tree
108    let taproot_builder = taproot_builder_with_scripts(scripts);
109    // Finalize the tree
110    let tree_info = match internal_key {
111        Some(xonly_pk) => taproot_builder
112            .finalize(&SECP, xonly_pk)
113            .expect("builder return is finalizable"),
114        None => taproot_builder
115            .finalize(&SECP, *bitvm_client::UNSPENDABLE_XONLY_PUBKEY)
116            .expect("builder return is finalizable"),
117    };
118
119    // Create the address
120    let taproot_address: Address = Address::p2tr_tweaked(tree_info.output_key(), network);
121
122    (taproot_address, tree_info)
123}

125/// Generates a deposit address for the user. Funds can be spent by N-of-N or
126/// user can take after specified time should the deposit fail.
127///
128/// # Parameters
129///
130/// - `nofn_xonly_pk`: N-of-N x-only public key of the depositor
131/// - `recovery_taproot_address`: User's x-only public key that can be used to
132///   take funds after some time
133/// - `user_evm_address`: User's EVM address.
134/// - `amount`: Amount to deposit
135/// - `network`: Bitcoin network to work on
136/// - `user_takes_after`: User can take the funds back, after this amounts of
137///   blocks have passed
138///
139/// # Returns
140///
141/// - [`Address`]: Deposit taproot Bitcoin address
142/// - [`TaprootSpendInfo`]: Deposit address's taproot spending information
143///
144/// # Panics
145///
146/// Panics if given parameters are malformed.
147pub fn generate_deposit_address(
148    nofn_xonly_pk: XOnlyPublicKey,
149    recovery_taproot_address: &Address<NetworkUnchecked>,
150    user_evm_address: EVMAddress,
151    network: bitcoin::Network,
152    user_takes_after: u16,
153) -> Result<(Address, TaprootSpendInfo), BridgeError> {
154    let deposit_script = BaseDepositScript::new(nofn_xonly_pk, user_evm_address).to_script_buf();
155
156    let recovery_script_pubkey = recovery_taproot_address
157        .clone()
158        .assume_checked()
159        .script_pubkey();
160
161    let recovery_extracted_xonly_pk = recovery_script_pubkey
162        .try_get_taproot_pk()
163        .wrap_err("Recovery taproot address is not a valid taproot address")?;
164
165    let script_timelock =
166        TimelockScript::new(Some(recovery_extracted_xonly_pk), user_takes_after).to_script_buf();
167
168    let (addr, spend) = create_taproot_address(&[deposit_script, script_timelock], None, network);
169    Ok((addr, spend))
170}

172/// Builds a Taproot address specifically for replacement deposits.
173/// Replacement deposits are to replace old move_to_vault transactions in case any issue is found on the bridge.
174/// This address incorporates a script committing to an old move transaction ID
175/// and a multisig script for the security council.
176/// This replacement deposit address will be used to create a new deposit transaction, which will then be used to
177/// sign the new related bridge deposit tx's.
178///
179/// # Parameters
180///
181/// - `old_move_txid`: The `Txid` of the old move_to_vault transaction that is being replaced.
182/// - `nofn_xonly_pk`: The N-of-N XOnlyPublicKey for the deposit.
183/// - `network`: The Bitcoin network on which the address will be used.
184/// - `security_council`: The `SecurityCouncil` configuration for the multisig script.
185///
186/// # Returns
187///
188/// - `Ok((Address, TaprootSpendInfo))` containing the new replacement deposit address
189///   and its associated `TaprootSpendInfo` if successful.
190/// - `Err(BridgeError)` if any error occurs during address generation.
191pub fn generate_replacement_deposit_address(
192    old_move_txid: bitcoin::Txid,
193    nofn_xonly_pk: XOnlyPublicKey,
194    network: bitcoin::Network,
195    security_council: SecurityCouncil,
196) -> Result<(Address, TaprootSpendInfo), BridgeError> {
197    let deposit_script =
198        ReplacementDepositScript::new(nofn_xonly_pk, old_move_txid).to_script_buf();
199
200    let security_council_script = Multisig::from_security_council(security_council).to_script_buf();
201
202    let (addr, spend) =
203        create_taproot_address(&[deposit_script, security_council_script], None, network);
204    Ok((addr, spend))
205}

306    #[test]
307    fn test_calculate_taproot_leaf_depths() {
308        // Test case 1: 0 scripts
309        let expected: Vec<u8> = vec![];
310        assert_eq!(calculate_taproot_leaf_depths(0), expected);
311
312        // Test case 2: 1 script
313        assert_eq!(calculate_taproot_leaf_depths(1), vec![0]);
314
315        // Test case 3: 2 scripts (balanced tree, depth 1 for both)
316        assert_eq!(calculate_taproot_leaf_depths(2), vec![1, 1]);
317
318        // Test case 4: 3 scripts (unbalanced)
319        // The first two scripts are at depth 2, the last is promoted to depth 1.
320        assert_eq!(calculate_taproot_leaf_depths(3), vec![2, 2, 1]);
321
322        // Test case 5: 4 scripts (perfectly balanced tree, all at depth 2)
323        assert_eq!(calculate_taproot_leaf_depths(4), vec![2, 2, 2, 2]);
324
325        // Test case 6: 5 scripts (unbalanced)
326        // num_nodes_in_final_depth is 2, so first two are at depth 3, rest are at depth 2.
327        // deepest_layer_depth = ilog2(4) + 1 = 3
328        // num_empty_nodes = 2^3 - 5 = 3
329        // num_nodes_in_final_depth = 5 - 3 = 2
330        // Depths: (3, 3, 2, 2, 2)
331        assert_eq!(calculate_taproot_leaf_depths(5), vec![3, 3, 2, 2, 2]);
332
333        // Test case 7: 8 scripts (perfectly balanced tree, all at depth 3)
334        assert_eq!(
335            calculate_taproot_leaf_depths(8),
336            vec![3, 3, 3, 3, 3, 3, 3, 3]
337        );
338    }
