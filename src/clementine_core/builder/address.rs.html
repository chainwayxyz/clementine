22pub fn taproot_builder_with_scripts(scripts: &[ScriptBuf]) -> TaprootBuilder {
23    let builder = TaprootBuilder::new();
24    let num_scripts = scripts.len();
25
26    // Special return cases for n = 0 or n = 1
27    match num_scripts {
28        0 => return builder,
29        1 => {
30            return builder
31                .add_leaf(0, scripts[0].clone())
32                .expect("one root leaf added on empty builder")
33        }
34        _ => {}
35    }
36
37    let deepest_layer_depth: u8 = ((num_scripts - 1).ilog2() + 1) as u8;
38
39    let num_empty_nodes_in_final_depth = 2_usize.pow(deepest_layer_depth.into()) - num_scripts;
40    let num_nodes_in_final_depth = num_scripts - num_empty_nodes_in_final_depth;
41
42    (0..num_scripts).fold(builder, |acc, i| {
43        let is_node_in_last_minus_one_depth = (i >= num_nodes_in_final_depth) as u8;
44
45        acc.add_leaf(
46            deepest_layer_depth - is_node_in_last_minus_one_depth,
47            scripts[i].clone(),
48        )
49        .expect("algorithm tested to be correct")
50    })
51}

53/// Creates a taproot address with either key path spend or script spend path
54/// addresses. This depends on given arguments.
55///
56/// # Arguments
57///
58/// - `scripts`: If empty, it is most likely a key path spend address
59/// - `internal_key`: If not given, will be defaulted to an unspendable x-only public key
60/// - `network`: Bitcoin network
61/// - If both `scripts` and `internal_key` are given, it means one can spend using both script and key path.
62/// - If none given, it is an unspendable address.
63///
64/// # Returns
65///
66/// - [`Address`]: Generated taproot address
67/// - [`TaprootSpendInfo`]: Taproot spending information
68///
69/// # Panics
70///
71/// Will panic if some of the operations have invalid paramaters.
72pub fn create_taproot_address(
73    scripts: &[ScriptBuf],
74    internal_key: Option<XOnlyPublicKey>,
75    network: bitcoin::Network,
76) -> (Address, TaprootSpendInfo) {
77    // Build script tree
78    let taproot_builder = taproot_builder_with_scripts(scripts);
79    // Finalize the tree
80    let tree_info = match internal_key {
81        Some(xonly_pk) => taproot_builder
82            .finalize(&SECP, xonly_pk)
83            .expect("builder return is finalizable"),
84        None => taproot_builder
85            .finalize(&SECP, *bitvm_client::UNSPENDABLE_XONLY_PUBKEY)
86            .expect("builder return is finalizable"),
87    };
88    // Create the address
89    let taproot_address = match internal_key {
90        Some(xonly_pk) => Address::p2tr(&SECP, xonly_pk, tree_info.merkle_root(), network),
91        None => Address::p2tr(
92            &SECP,
93            *bitvm_client::UNSPENDABLE_XONLY_PUBKEY,
94            tree_info.merkle_root(),
95            network,
96        ),
97    };
98
99    (taproot_address, tree_info)
100}

102/// Generates a deposit address for the user. Funds can be spend by N-of-N or
103/// user can take after specified time should the deposit fail.
104///
105/// # Parameters
106///
107/// - `nofn_xonly_pk`: N-of-N x-only public key of the depositor
108/// - `recovery_taproot_address`: User's x-only public key that can be used to
109///   take funds after some time
110/// - `user_evm_address`: User's EVM address.
111/// - `amount`: Amount to deposit
112/// - `network`: Bitcoin network to work on
113/// - `user_takes_after`: User can take the funds back, after this amounts of
114///   blocks have passed
115///
116/// # Returns
117///
118/// - [`Address`]: Deposit taproot Bitcoin address
119/// - [`TaprootSpendInfo`]: Deposit address's taproot spending information
120///
121/// # Panics
122///
123/// Panics if given parameters are malformed.
124pub fn generate_deposit_address(
125    nofn_xonly_pk: XOnlyPublicKey,
126    recovery_taproot_address: &Address<NetworkUnchecked>,
127    user_evm_address: EVMAddress,
128    network: bitcoin::Network,
129    user_takes_after: u16,
130) -> Result<(Address, TaprootSpendInfo), BridgeError> {
131    let deposit_script = BaseDepositScript::new(nofn_xonly_pk, user_evm_address).to_script_buf();
132
133    let recovery_script_pubkey = recovery_taproot_address
134        .clone()
135        .assume_checked()
136        .script_pubkey();
137
138    let recovery_extracted_xonly_pk =
139        XOnlyPublicKey::from_slice(&recovery_script_pubkey.as_bytes()[2..34])
140            .wrap_err("Failed to extract xonly public key from recovery taproot address")?;
141
142    let script_timelock =
143        TimelockScript::new(Some(recovery_extracted_xonly_pk), user_takes_after).to_script_buf();
144
145    let (addr, spend) = create_taproot_address(&[deposit_script, script_timelock], None, network);
146    Ok((addr, spend))
147}

149pub fn generate_replacement_deposit_address(
150    old_move_txid: bitcoin::Txid,
151    nofn_xonly_pk: XOnlyPublicKey,
152    network: bitcoin::Network,
153    security_council: SecurityCouncil,
154) -> Result<(Address, TaprootSpendInfo), BridgeError> {
155    let deposit_script =
156        ReplacementDepositScript::new(nofn_xonly_pk, old_move_txid).to_script_buf();
157
158    let security_council_script = Multisig::from_security_council(security_council).to_script_buf();
159
160    let (addr, spend) =
161        create_taproot_address(&[deposit_script, security_council_script], None, network);
162    Ok((addr, spend))
163}

165/// Shorthand function for creating a checksig taproot address: A single checksig script with the given xonly PK and no internal key.
166///
167/// # Returns
168///
169/// See [`create_taproot_address`].
170///
171/// - [`Address`]: Checksig taproot Bitcoin address
172/// - [`TaprootSpendInfo`]: Checksig address's taproot spending information
173pub fn create_checksig_address(
174    xonly_pk: XOnlyPublicKey,
175    network: bitcoin::Network,
176) -> (Address, TaprootSpendInfo) {
177    let script = CheckSig::new(xonly_pk);
178    create_taproot_address(&[script.to_script_buf()], None, network)
179}

198        let secret_key = SecretKey::new(&mut rand::thread_rng());
199        let internal_key =
200            XOnlyPublicKey::from_keypair(&Keypair::from_secret_key(&SECP, &secret_key)).0;

202        // No internal key or scripts (key path spend).

204            builder::address::create_taproot_address(&[], None, bitcoin::Network::Regtest);
205        assert_eq!(address.address_type().unwrap(), AddressType::P2tr);
206        assert!(address.is_related_to_xonly_pubkey(

208                .tap_tweak(&SECP, spend_info.merkle_root())
209                .0
210                .to_x_only_public_key()
211        ));

217
218        // Key path spend.

310    #[test]
311    pub fn test_taproot_builder_with_scripts() {
312        for i in [0, 1, 10, 50, 100, 1000].into_iter() {
313            let scripts = (0..i)
314                .map(|k| ScriptBuf::builder().push_int(k).into_script())
315                .collect::<Vec<_>>();
316            let builder = super::taproot_builder_with_scripts(&scripts);
317            let tree_info = builder
318                .finalize(&SECP, *bitvm_client::UNSPENDABLE_XONLY_PUBKEY)
319                .unwrap();
320
321            assert_eq!(tree_info.script_map().len(), i as usize);
322        }
323    }
