1//! # Collaterals
2//!
3//! This module contains the logic for creating the `round_tx`, `ready_to_reimburse_tx`,
4//! and `unspent_kickoff_tx` transactions. These transactions are used to control the sequence of transactions
5//! in the withdrawal process and limits the number of withdrawals the operator can make in a given time period.
6//!
7//! The flow is as follows:
8//! `round_tx -> ready_to_reimburse_tx -> round_tx -> ...`
9//!
10//! The `round_tx` is used to create a collateral for the withdrawal, kickoff utxos for the current
11//! round and the reimburse connectors for the previous round.
12

35/// Creates a [`TxHandler`] for `round_tx`.
36///
37/// This transaction is used to create a collateral for the withdrawal, kickoff UTXOs for the current round, and the reimburse connectors for the previous round.
38/// It always uses the first output of the previous `ready_to_reimburse_tx` as the input, chaining rounds together.
39///
40/// `round tx` inputs:
41/// 1. Either the first collateral utxo of operator, or operators collateral in the previous rounds ready to reimburse tx.
42///
43/// `round tx` outputs:
44/// 1. Operator's Burn Connector
45/// 2. Kickoff utxo(s): the utxos will be used as the input for the kickoff transactions
46/// 3. Reimburse utxo(s): the utxo(s) will be used as an input to Reimburse TX
47/// 4. P2Anchor: Anchor output for CPFP
48///
49/// # Arguments
50/// * `operator_xonly_pk` - The operator's x-only public key.
51/// * `txin` - The input to the round transaction (either a previous output or the first collateral).
52/// * `pubkeys` - Winternitz public keys for the round's kickoff UTXOs.
53/// * `paramset` - Protocol parameter set.
54///
55/// # Returns
56/// A [`TxHandler`] for the round transaction, or a [`BridgeError`] if construction fails.
57pub fn create_round_txhandler(
58    operator_xonly_pk: XOnlyPublicKey,
59    txin: RoundTxInput,
60    pubkeys: &[bitvm::signatures::winternitz::PublicKey],
61    paramset: &'static ProtocolParamset,
62) -> Result<TxHandler, BridgeError> {
63    let mut builder = TxHandlerBuilder::new(TransactionType::Round).with_version(NON_STANDARD_V3);
64    let input_amount;
65    match txin {
66        RoundTxInput::Prevout(prevout) => {
67            input_amount = prevout.get_prevout().value;
68            builder = builder.add_input(
69                NormalSignatureKind::OperatorSighashDefault,
70                *prevout,
71                SpendPath::KeySpend,
72                Sequence::from_height(paramset.operator_reimburse_timelock),
73            );
74        }
75        RoundTxInput::Collateral(outpoint, amount) => {
76            let (op_address, op_spend) =
77                create_taproot_address(&[], Some(operator_xonly_pk), paramset.network);
78            input_amount = amount;
79            builder = builder.add_input(
80                NormalSignatureKind::OperatorSighashDefault,
81                SpendableTxIn::new(
82                    outpoint,
83                    TxOut {
84                        value: input_amount,
85                        script_pubkey: op_address.script_pubkey(),
86                    },
87                    vec![],
88                    Some(op_spend.clone()),
89                ),
90                SpendPath::KeySpend,
91                DEFAULT_SEQUENCE,
92            );
93        }
94    }
95
96    // This 1 block is to enforce that operator has to put a sequence number in the input
97    // so this spending path can't be used to send kickoff tx
98    let timeout_block_count_locked_script =
99        Arc::new(TimelockScript::new(Some(operator_xonly_pk), 1));
100
101    let total_required = (paramset.kickoff_amount + paramset.default_utxo_amount())
102        .checked_mul(paramset.num_kickoffs_per_round as u64)
103        .and_then(|kickoff_total| kickoff_total.checked_add(paramset.anchor_amount()))
104        .ok_or_else(|| {
105            BridgeError::ArithmeticOverflow("Total required amount calculation overflow")
106        })?;
107
108    let remaining_amount = input_amount.checked_sub(total_required).ok_or_else(|| {
109        BridgeError::InsufficientFunds("Input amount insufficient for required outputs")
110    })?;
111
112    builder = builder.add_output(UnspentTxOut::from_scripts(
113        remaining_amount,
114        vec![],
115        Some(operator_xonly_pk),
116        paramset.network,
117    ));
118
119    // add kickoff utxos
120    for pubkey in pubkeys.iter().take(paramset.num_kickoffs_per_round) {
121        let blockhash_commit = Arc::new(WinternitzCommit::new(
122            vec![(pubkey.clone(), paramset.kickoff_blockhash_commit_length)],
123            operator_xonly_pk,
124            paramset.winternitz_log_d,
125        ));
126        builder = builder.add_output(UnspentTxOut::from_scripts(
127            paramset.kickoff_amount,
128            vec![blockhash_commit, timeout_block_count_locked_script.clone()],
129            None,
130            paramset.network,
131        ));
132    }
133    // Create reimburse utxos
134    for _ in 0..paramset.num_kickoffs_per_round {
135        builder = builder.add_output(UnspentTxOut::from_scripts(
136            paramset.default_utxo_amount(),
137            vec![],
138            Some(operator_xonly_pk),
139            paramset.network,
140        ));
141    }
142    Ok(builder
143        .add_output(UnspentTxOut::from_partial(
144            builder::transaction::anchor_output(paramset.anchor_amount()),
145        ))
146        .finalize())
147}

149/// Creates a vector of [`TxHandler`] for `assert_timeout_tx` transactions.
150///
151/// These transactions can be sent by anyone if the operator did not send their asserts in time, burning their burn connector and kickoff finalizer.
152///
153/// # Inputs
154/// 1. KickoffTx: Assert utxo (corresponding to the assert)
155/// 2. KickoffTx: KickoffFinalizer utxo
156/// 3. RoundTx: BurnConnector utxo
157///
158/// # Outputs
159/// 1. Anchor output for CPFP
160///
161/// # Arguments
162/// * `kickoff_txhandler` - The kickoff transaction handler providing the input.
163/// * `round_txhandler` - The round transaction handler providing an additional input.
164/// * `num_asserts` - Number of assert timeout transactions to create.
165/// * `paramset` - Protocol parameter set.
166///
167/// # Returns
168/// A vector of [`TxHandler`] for all assert timeout transactions, or a [`BridgeError`] if construction fails.
169pub fn create_assert_timeout_txhandlers(
170    kickoff_txhandler: &TxHandler,
171    round_txhandler: &TxHandler,
172    num_asserts: usize,
173    paramset: &'static ProtocolParamset,
174) -> Result<Vec<TxHandler>, BridgeError> {
175    let mut txhandlers = Vec::new();
176    for idx in 0..num_asserts {
177        txhandlers.push(
178            TxHandlerBuilder::new(TransactionType::AssertTimeout(idx))
179                .with_version(NON_STANDARD_V3)
180                .add_input(
181                    (NumberedSignatureKind::AssertTimeout1, idx as i32),
182                    kickoff_txhandler.get_spendable_output(UtxoVout::Assert(idx))?,
183                    SpendPath::ScriptSpend(0),
184                    Sequence::from_height(paramset.assert_timeout_timelock),
185                )
186                .add_input(
187                    (NumberedSignatureKind::AssertTimeout2, idx as i32),
188                    kickoff_txhandler.get_spendable_output(UtxoVout::KickoffFinalizer)?,
189                    SpendPath::ScriptSpend(0),
190                    DEFAULT_SEQUENCE,
191                )
192                .add_input(
193                    (NumberedSignatureKind::AssertTimeout3, idx as i32),
194                    round_txhandler.get_spendable_output(UtxoVout::CollateralInRound)?,
195                    SpendPath::KeySpend,
196                    DEFAULT_SEQUENCE,
197                )
198                .add_output(UnspentTxOut::from_partial(
199                    builder::transaction::anchor_output(paramset.anchor_amount()),
200                ))
201                .finalize(),
202        );
203    }
204    Ok(txhandlers)
205}

207/// Creates the nth (1-indexed) `round_txhandler` and `reimburse_generator_txhandler` pair for a specific operator.
208///
209/// # Arguments
210/// * `operator_xonly_pk` - The operator's x-only public key.
211/// * `input_outpoint` - The outpoint to use as input for the first round.
212/// * `input_amount` - The amount for the input outpoint.
213/// * `index` - The index of the round to create.
214/// * `pubkeys` - Winternitz keys for all rounds.
215/// * `paramset` - Protocol parameter set.
216///
217/// # Returns
218/// A tuple of (`TxHandler` for the round, `TxHandler` for ready-to-reimburse), or a [`BridgeError`] if construction fails.
219pub fn create_round_nth_txhandler(
220    operator_xonly_pk: XOnlyPublicKey,
221    input_outpoint: OutPoint,
222    input_amount: Amount,
223    index: RoundIndex,
224    pubkeys: &KickoffWinternitzKeys,
225    paramset: &'static ProtocolParamset,
226) -> Result<(TxHandler, TxHandler), BridgeError> {
227    // 0th round is the collateral, there are no keys for the 0th round
228    // Additionally there are no keys after num_rounds + 1, +1 is because we need additional round to generate
229    // reimbursement connectors of previous round
230
231    if index == RoundIndex::Collateral
232        || index.to_index() > RoundIndex::Round(paramset.num_round_txs).to_index()
233    {
234        return Err(TxError::InvalidRoundIndex(index).into());
235    }
236
237    // create the first round txhandler
238    let mut round_txhandler = create_round_txhandler(
239        operator_xonly_pk,
240        RoundTxInput::Collateral(input_outpoint, input_amount),
241        pubkeys.get_keys_for_round(RoundIndex::Round(0))?,
242        paramset,
243    )?;
244    let mut ready_to_reimburse_txhandler =
245        create_ready_to_reimburse_txhandler(&round_txhandler, operator_xonly_pk, paramset)?;
246
247    // get which round index we are creating txhandlers for
248    let round_idx = match index {
249        RoundIndex::Collateral => 0, // impossible, checked before
250        RoundIndex::Round(idx) => idx,
251    };
252    // iterate starting from second round to the requested round
253    for round_idx in RoundIndex::iter_rounds_range(1, round_idx + 1) {
254        round_txhandler = create_round_txhandler(
255            operator_xonly_pk,
256            RoundTxInput::Prevout(Box::new(
257                ready_to_reimburse_txhandler
258                    .get_spendable_output(UtxoVout::CollateralInReadyToReimburse)?,
259            )),
260            pubkeys.get_keys_for_round(round_idx)?,
261            paramset,
262        )?;
263        ready_to_reimburse_txhandler =
264            create_ready_to_reimburse_txhandler(&round_txhandler, operator_xonly_pk, paramset)?;
265    }
266    Ok((round_txhandler, ready_to_reimburse_txhandler))
267}

269/// Creates a [`TxHandler`] for the `ready_to_reimburse_tx`.
270///
271/// # Inputs
272/// 1. RoundTx: BurnConnector utxo
273///
274/// # Outputs
275/// 1. Operator's collateral
276/// 2. Anchor output for CPFP
277///
278/// # Arguments
279/// * `round_txhandler` - The round transaction handler providing the input.
280/// * `operator_xonly_pk` - The operator's x-only public key.
281/// * `paramset` - Protocol parameter set.
282///
283/// # Returns
284/// A [`TxHandler`] for the ready-to-reimburse transaction, or a [`BridgeError`] if construction fails.
285pub fn create_ready_to_reimburse_txhandler(
286    round_txhandler: &TxHandler,
287    operator_xonly_pk: XOnlyPublicKey,
288    paramset: &'static ProtocolParamset,
289) -> Result<TxHandler, BridgeError> {
290    let prevout = round_txhandler.get_spendable_output(UtxoVout::CollateralInRound)?;
291    let prev_value = prevout.get_prevout().value;
292
293    Ok(TxHandlerBuilder::new(TransactionType::ReadyToReimburse)
294        .with_version(NON_STANDARD_V3)
295        .add_input(
296            NormalSignatureKind::OperatorSighashDefault,
297            prevout,
298            SpendPath::KeySpend,
299            DEFAULT_SEQUENCE,
300        )
301        .add_output(UnspentTxOut::from_scripts(
302            prev_value.checked_sub(paramset.anchor_amount()).ok_or(
303                BridgeError::ArithmeticOverflow(
304                    "Insufficient funds while creating ready to reimburse tx",
305                ),
306            )?,
307            vec![],
308            Some(operator_xonly_pk),
309            paramset.network,
310        ))
311        .add_output(UnspentTxOut::from_partial(
312            builder::transaction::anchor_output(paramset.anchor_amount()),
313        ))
314        .finalize())
315}

317/// Creates a vector of [`TxHandler`] for `unspent_kickoff_tx` transactions.
318/// These transactions can be sent if an operator sends ReadyToReimburse transaction without spending all the kickoff utxos of the round.
319///
320/// # Inputs
321/// 1. ReadyToReimburseTx: BurnConnector utxo
322/// 2. RoundTx: Any kickoff utxo of the same round
323///
324/// # Outputs
325/// 1. Anchor output for CPFP
326///
327/// # Arguments
328/// * `round_txhandler` - The round transaction handler providing the kickoff utxos.
329/// * `ready_to_reimburse_txhandler` - The ready-to-reimburse transaction handler providing the collateral.
330/// * `paramset` - Protocol parameter set.
331///
332/// # Returns
333/// A vector of [`TxHandler`] for unspent kickoff transactions, or a [`BridgeError`] if construction fails.
334pub fn create_unspent_kickoff_txhandlers(
335    round_txhandler: &TxHandler,
336    ready_to_reimburse_txhandler: &TxHandler,
337    paramset: &'static ProtocolParamset,
338) -> Result<Vec<TxHandler>, BridgeError> {
339    let mut txhandlers = Vec::new();
340    for idx in 0..paramset.num_kickoffs_per_round {
341        txhandlers.push(
342            TxHandlerBuilder::new(TransactionType::UnspentKickoff(idx))
343                .with_version(NON_STANDARD_V3)
344                .add_input(
345                    (NumberedSignatureKind::UnspentKickoff1, idx as i32),
346                    ready_to_reimburse_txhandler
347                        .get_spendable_output(UtxoVout::CollateralInReadyToReimburse)?,
348                    SpendPath::KeySpend,
349                    DEFAULT_SEQUENCE,
350                )
351                .add_input(
352                    (NumberedSignatureKind::UnspentKickoff2, idx as i32),
353                    round_txhandler.get_spendable_output(UtxoVout::Kickoff(idx))?,
354                    SpendPath::ScriptSpend(1),
355                    Sequence::from_height(1),
356                )
357                .add_output(UnspentTxOut::from_partial(
358                    builder::transaction::anchor_output(paramset.anchor_amount()),
359                ))
360                .finalize(),
361        );
362    }
363    Ok(txhandlers)
364}

366/// Creates a [`TxHandler`] for burning unused kickoff connectors.
367///
368/// # Inputs
369/// 1. RoundTx: Kickoff utxo(s) (per unused connector)
370///
371/// # Outputs
372/// 1. Change output to the provided address
373/// 2. Anchor output for CPFP
374///
375/// # Arguments
376/// * `round_txhandler` - The round transaction handler providing the input.
377/// * `unused_kickoff_connectors_indices` - Indices of the unused kickoff connectors (0-indexed).
378/// * `change_address` - The address to send the change to.
379///
380/// # Returns
381/// A [`TxHandler`] for burning unused kickoff connectors, or a [`BridgeError`] if construction fails or no unused kickoff connectors are provided.
382pub fn create_burn_unused_kickoff_connectors_txhandler(
383    round_txhandler: &TxHandler,
384    unused_kickoff_connectors_indices: &[usize],
385    change_address: &Address,
386    paramset: &'static ProtocolParamset,
387) -> Result<TxHandler, BridgeError> {
388    if unused_kickoff_connectors_indices.is_empty() {
389        return Err(eyre::eyre!(
390            "create_burn_unused_kickoff_connectors_txhandler called with no unused kickoff connectors"
391        )
392        .into());
393    }
394    let mut tx_handler_builder =
395        TxHandlerBuilder::new(TransactionType::BurnUnusedKickoffConnectors)
396            .with_version(NON_STANDARD_V3);
397    let mut input_amount = Amount::ZERO;
398    for &idx in unused_kickoff_connectors_indices {
399        let txin = round_txhandler.get_spendable_output(UtxoVout::Kickoff(idx))?;
400        input_amount = input_amount.checked_add(txin.get_prevout().value).ok_or(
401            BridgeError::ArithmeticOverflow("Amount overflow in burn unused kickoff connectors tx"),
402        )?;
403        tx_handler_builder = tx_handler_builder.add_input(
404            NormalSignatureKind::OperatorSighashDefault,
405            txin,
406            SpendPath::ScriptSpend(1),
407            Sequence::from_height(1),
408        );
409    }
410    if !paramset.bridge_nonstandard && input_amount >= paramset.anchor_amount() {
411        // if we use standard tx's, kickoff utxo's will hold some sats so we can return the change to the change address
412        // but if we use nonstandard tx's with 0 sat values then the change is 0 anyway, no need to add an output
413        tx_handler_builder = tx_handler_builder.add_output(UnspentTxOut::from_partial(TxOut {
414            // In a standard bridge, a kickoff UTXO will hold on the order of 10^4 sats, so this UTXO will not be considered dust
415            value: input_amount - paramset.anchor_amount(),
416            script_pubkey: change_address.script_pubkey(),
417        }));
418    }
419    tx_handler_builder = tx_handler_builder.add_output(UnspentTxOut::from_partial(
420        builder::transaction::anchor_output(paramset.anchor_amount()),
421    ));
422    Ok(tx_handler_builder.finalize())
423}

428    use crate::config::protocol::REGTEST_PARAMSET;

431    #[tokio::test]
432    async fn test_create_round_nth_txhandler_and_round_txhandlers() {
433        // check if round_nth_txhandler and round_txhandlers are consistent with each other
434        let op_xonly_pk = XOnlyPublicKey::from_str(
435            "50929b74c1a04954b78b4b6035e97a5e078a5a0f28ec96d547bfee9ace803ac0",
436        )
437        .expect("this key is valid");
438        let paramset = &REGTEST_PARAMSET;
439        let input_outpoint = OutPoint::new(bitcoin::Txid::all_zeros(), 0);
440        let input_amount = Amount::from_sat(10000000000);
441        let pubkeys = KickoffWinternitzKeys::new(
442            vec![
443                vec![[0u8; 20]; 44];
444                (paramset.num_round_txs + 1) * paramset.num_kickoffs_per_round
445            ],
446            paramset.num_kickoffs_per_round,
447            paramset.num_round_txs,
448        )
449        .unwrap();
450
451        let mut round_tx_input = RoundTxInput::Collateral(input_outpoint, input_amount);
452
453        for i in 0..paramset.num_round_txs {
454            let (round_nth_txhandler, ready_to_reimburse_nth_txhandler) =
455                create_round_nth_txhandler(
456                    op_xonly_pk,
457                    input_outpoint,
458                    input_amount,
459                    RoundIndex::Round(i),
460                    &pubkeys,
461                    paramset,
462                )
463                .unwrap();
464
465            let round_txhandler = create_round_txhandler(
466                op_xonly_pk,
467                round_tx_input,
468                pubkeys.get_keys_for_round(RoundIndex::Round(i)).unwrap(),
469                paramset,
470            )
471            .unwrap();
472
473            let ready_to_reimburse_txhandler =
474                create_ready_to_reimburse_txhandler(&round_txhandler, op_xonly_pk, paramset)
475                    .unwrap();
476
477            assert_eq!(round_nth_txhandler.get_txid(), round_txhandler.get_txid());
478            assert_eq!(
479                ready_to_reimburse_nth_txhandler.get_txid(),
480                ready_to_reimburse_txhandler.get_txid()
481            );
482
483            let prev_ready_to_reimburse_txhandler = ready_to_reimburse_txhandler;
484            round_tx_input = RoundTxInput::Prevout(Box::new(
485                prev_ready_to_reimburse_txhandler
486                    .get_spendable_output(UtxoVout::CollateralInReadyToReimburse)
487                    .unwrap(),
488            ));
489        }
490    }
