1//! # Collaterals
2//!
3//! This module contains the logic for creating the `round_tx`, `ready_to_reimburse_tx`,
4//! and `unspent_kickoff_tx` transactions. These transactions are used to control the sequence of transactions
5//! in the withdrawal process and limits the number of withdrawals the operator can make in a given time period.
6//!
7//! The flow is as follows:
8//! `round_tx -> ready_to_reimburse_tx -> round_tx -> ...`
9//!
10//! The `round_tx` is used to create a collateral for the withdrawal, kickoff utxos for the current
11//! round and the reimburse connectors for the previous round.

36/// Creates a [`TxHandler`] for `round_tx`. It will always use the first
37/// output of the  previous `ready_to_reimburse_tx` as the input. The flow is as follows:
38/// `round_tx -> ready_to_reimburse_tx -> round_tx -> ...`
39///
40/// # Returns
41///
42/// A `round_tx` that has outputs of:
43///
44/// 1. Operator's Burn Connector
45/// 2. Kickoff input utxo(s): the utxo(s) will be used as the input(s) for the kickoff_tx(s)
46/// 3. Reimburse utxo(s): the utxo(s) will be used as an input to Reimburse TX
47/// 4. P2Anchor: Anchor output for CPFP
48pub fn create_round_txhandler(
49    operator_xonly_pk: XOnlyPublicKey,
50    txin: RoundTxInput,
51    pubkeys: &[bitvm::signatures::winternitz::PublicKey],
52    paramset: &'static ProtocolParamset,
53) -> Result<TxHandler, BridgeError> {
54    let mut builder =
55        TxHandlerBuilder::new(TransactionType::Round).with_version(Version::non_standard(3));
56    let input_amount;
57    match txin {
58        RoundTxInput::Prevout(prevout) => {
59            input_amount = prevout.get_prevout().value;
60            builder = builder.add_input(
61                NormalSignatureKind::OperatorSighashDefault,
62                *prevout,
63                SpendPath::KeySpend,
64                Sequence::from_height(paramset.operator_reimburse_timelock),
65            );
66        }
67        RoundTxInput::Collateral(outpoint, amount) => {
68            let (op_address, op_spend) =
69                create_taproot_address(&[], Some(operator_xonly_pk), paramset.network);
70            input_amount = amount;
71            builder = builder.add_input(
72                NormalSignatureKind::OperatorSighashDefault,
73                SpendableTxIn::new(
74                    outpoint,
75                    TxOut {
76                        value: input_amount,
77                        script_pubkey: op_address.script_pubkey(),
78                    },
79                    vec![],
80                    Some(op_spend.clone()),
81                ),
82                SpendPath::KeySpend,
83                DEFAULT_SEQUENCE,
84            );
85        }
86    }
87
88    // This 1 block is to enforce that operator has to put a sequence number in the input
89    // so this spending path can't be used to send kickoff tx
90    let timeout_block_count_locked_script =
91        Arc::new(TimelockScript::new(Some(operator_xonly_pk), 1));
92
93    builder = builder.add_output(UnspentTxOut::from_scripts(
94        input_amount
95            - (paramset.kickoff_amount + MIN_TAPROOT_AMOUNT)
96                * (paramset.num_kickoffs_per_round as u64)
97            - ANCHOR_AMOUNT,
98        vec![],
99        Some(operator_xonly_pk),
100        paramset.network,
101    ));
102
103    // add kickoff utxos
104    for pubkey in pubkeys.iter().take(paramset.num_kickoffs_per_round) {
105        let blockhash_commit = Arc::new(WinternitzCommit::new(
106            vec![(pubkey.clone(), paramset.kickoff_blockhash_commit_length)],
107            operator_xonly_pk,
108            paramset.winternitz_log_d,
109        ));
110        builder = builder.add_output(UnspentTxOut::from_scripts(
111            paramset.kickoff_amount,
112            vec![blockhash_commit, timeout_block_count_locked_script.clone()],
113            None,
114            paramset.network,
115        ));
116    }
117    // Create reimburse utxos
118    for _ in 0..paramset.num_kickoffs_per_round {
119        builder = builder.add_output(UnspentTxOut::from_scripts(
120            MIN_TAPROOT_AMOUNT,
121            vec![],
122            Some(operator_xonly_pk),
123            paramset.network,
124        ));
125    }
126    Ok(builder
127        .add_output(UnspentTxOut::from_partial(
128            builder::transaction::anchor_output(),
129        ))
130        .finalize())
131}

133/// Creates a [`TxHandler`] for the `assert_timeout_tx`. This transaction will be sent by anyone
134/// in case the operator did not send any of their asserts in time, burning their burn connector
135/// and kickoff finalizer.
136pub fn create_assert_timeout_txhandlers(
137    kickoff_txhandler: &TxHandler,
138    round_txhandler: &TxHandler,
139    num_asserts: usize,
140    paramset: &'static ProtocolParamset,
141) -> Result<Vec<TxHandler>, BridgeError> {
142    let mut txhandlers = Vec::new();
143    for idx in 0..num_asserts {
144        txhandlers.push(
145            TxHandlerBuilder::new(TransactionType::AssertTimeout(idx))
146                .with_version(Version::non_standard(3))
147                .add_input(
148                    (NumberedSignatureKind::AssertTimeout1, idx as i32),
149                    kickoff_txhandler.get_spendable_output(UtxoVout::Assert(idx))?,
150                    SpendPath::ScriptSpend(0),
151                    Sequence::from_height(paramset.assert_timeout_timelock),
152                )
153                .add_input(
154                    (NumberedSignatureKind::AssertTimeout2, idx as i32),
155                    kickoff_txhandler.get_spendable_output(UtxoVout::KickoffFinalizer)?,
156                    SpendPath::ScriptSpend(0),
157                    DEFAULT_SEQUENCE,
158                )
159                .add_input(
160                    (NumberedSignatureKind::AssertTimeout3, idx as i32),
161                    round_txhandler.get_spendable_output(UtxoVout::BurnConnector)?,
162                    SpendPath::KeySpend,
163                    DEFAULT_SEQUENCE,
164                )
165                .add_output(UnspentTxOut::from_partial(
166                    builder::transaction::anchor_output(),
167                ))
168                .finalize(),
169        );
170    }
171    Ok(txhandlers)
172}

174/// Creates the nth (0-indexed) `round_txhandler` and `reimburse_generator_txhandler` pair
175/// for a specific operator.
176pub fn create_round_nth_txhandler(
177    operator_xonly_pk: XOnlyPublicKey,
178    input_outpoint: OutPoint,
179    input_amount: Amount,
180    index: usize,
181    pubkeys: &KickoffWinternitzKeys,
182    paramset: &'static ProtocolParamset,
183) -> Result<(TxHandler, TxHandler), BridgeError> {
184    let mut round_txhandler = create_round_txhandler(
185        operator_xonly_pk,
186        RoundTxInput::Collateral(input_outpoint, input_amount),
187        pubkeys.get_keys_for_round(0),
188        paramset,
189    )?;
190    let mut ready_to_reimburse_txhandler =
191        create_ready_to_reimburse_txhandler(&round_txhandler, operator_xonly_pk, paramset)?;
192    for idx in 1..=index {
193        round_txhandler = create_round_txhandler(
194            operator_xonly_pk,
195            RoundTxInput::Prevout(Box::new(
196                ready_to_reimburse_txhandler.get_spendable_output(UtxoVout::BurnConnector)?,
197            )),
198            pubkeys.get_keys_for_round(idx),
199            paramset,
200        )?;
201        ready_to_reimburse_txhandler =
202            create_ready_to_reimburse_txhandler(&round_txhandler, operator_xonly_pk, paramset)?;
203    }
204    Ok((round_txhandler, ready_to_reimburse_txhandler))
205}

207pub fn create_ready_to_reimburse_txhandler(
208    round_txhandler: &TxHandler,
209    operator_xonly_pk: XOnlyPublicKey,
210    paramset: &'static ProtocolParamset,
211) -> Result<TxHandler, BridgeError> {
212    let prevout = round_txhandler.get_spendable_output(UtxoVout::BurnConnector)?;
213    let prev_value = prevout.get_prevout().value;
214
215    Ok(TxHandlerBuilder::new(TransactionType::ReadyToReimburse)
216        .with_version(Version::non_standard(3))
217        .add_input(
218            NormalSignatureKind::OperatorSighashDefault,
219            prevout,
220            SpendPath::KeySpend,
221            DEFAULT_SEQUENCE,
222        )
223        .add_output(UnspentTxOut::from_scripts(
224            prev_value - ANCHOR_AMOUNT,
225            vec![],
226            Some(operator_xonly_pk),
227            paramset.network,
228        ))
229        .add_output(UnspentTxOut::from_partial(
230            builder::transaction::anchor_output(),
231        ))
232        .finalize())
233}

235pub fn create_unspent_kickoff_txhandlers(
236    round_txhandler: &TxHandler,
237    ready_to_reimburse_txhandler: &TxHandler,
238    paramset: &'static ProtocolParamset,
239) -> Result<Vec<TxHandler>, BridgeError> {
240    let mut txhandlers = Vec::new();
241    for idx in 0..paramset.num_kickoffs_per_round {
242        txhandlers.push(
243            TxHandlerBuilder::new(TransactionType::UnspentKickoff(idx))
244                .with_version(Version::non_standard(3))
245                .add_input(
246                    (NumberedSignatureKind::UnspentKickoff1, idx as i32),
247                    ready_to_reimburse_txhandler.get_spendable_output(UtxoVout::BurnConnector)?,
248                    SpendPath::KeySpend,
249                    DEFAULT_SEQUENCE,
250                )
251                .add_input(
252                    (NumberedSignatureKind::UnspentKickoff2, idx as i32),
253                    round_txhandler.get_spendable_output(UtxoVout::Kickoff(idx))?,
254                    SpendPath::ScriptSpend(1),
255                    Sequence::from_height(1),
256                )
257                .add_output(UnspentTxOut::from_partial(
258                    builder::transaction::anchor_output(),
259                ))
260                .finalize(),
261        );
262    }
263    Ok(txhandlers)
264}

266pub fn create_burn_unused_kickoff_connectors_txhandler(
267    round_txhandler: &TxHandler,
268    unused_kickoff_connectors_indices: &[usize], // indices of the kickoff connectors that are not used, 0 indexed, 0 => first kickoff connector
269    change_address: &Address,
270) -> Result<TxHandler, BridgeError> {
271    let mut tx_handler_builder =
272        TxHandlerBuilder::new(TransactionType::BurnUnusedKickoffConnectors)
273            .with_version(Version::non_standard(3));
274    for &idx in unused_kickoff_connectors_indices {
275        tx_handler_builder = tx_handler_builder.add_input(
276            NormalSignatureKind::OperatorSighashDefault,
277            round_txhandler.get_spendable_output(UtxoVout::Kickoff(idx))?,
278            SpendPath::ScriptSpend(1),
279            Sequence::from_height(1),
280        );
281    }
282    tx_handler_builder = tx_handler_builder.add_output(UnspentTxOut::from_partial(TxOut {
283        value: MIN_TAPROOT_AMOUNT,
284        script_pubkey: change_address.script_pubkey(),
285    }));
286    tx_handler_builder = tx_handler_builder.add_output(UnspentTxOut::from_partial(
287        builder::transaction::anchor_output(),
288    ));
289    Ok(tx_handler_builder.finalize())
