15/// Creates a [`TxHandler`] for the `watchtower_challenge_tx`. This transaction
16/// is sent by the watchtowers to reveal their Groth16 proofs with their public
17/// inputs for the longest chain proof. The data is encoded as 32 byte script pubkeys
18/// of taproot utxos, and a single at max 80 byte OP_RETURN utxo.
19pub fn create_watchtower_challenge_txhandler(
20    kickoff_txhandler: &TxHandler,
21    watchtower_idx: usize,
22    commit_data: &[u8],
23    paramset: &'static ProtocolParamset,
24) -> Result<TxHandler, BridgeError> {
25    if commit_data.len() != paramset.watchtower_challenge_bytes {
26        return Err(TxError::IncorrectWatchtowerChallengeDataLength.into());
27    }
28    let mut builder = TxHandlerBuilder::new(TransactionType::WatchtowerChallenge(watchtower_idx))
29        .with_version(Version::non_standard(3))
30        .add_input(
31            (
32                NumberedSignatureKind::WatchtowerChallenge,
33                watchtower_idx as i32,
34            ),
35            kickoff_txhandler
36                .get_spendable_output(UtxoVout::WatchtowerChallenge(watchtower_idx))?,
37            SpendPath::KeySpend,
38            DEFAULT_SEQUENCE,
39        );
40    let mut current_idx = 0;
41    while current_idx + 80 < paramset.watchtower_challenge_bytes {
42        // encode next 32 bytes of data as script pubkey of taproot utxo
43        let data = PushBytesBuf::try_from(commit_data[current_idx..current_idx + 32].to_vec())
44            .map_err(|e| {
45                BridgeError::Error(format!(
46                    "Failed to create pushbytesbuf for watchtower challenge op_return: {}",
47                    e
48                ))
49            })?;
50
51        let data_encoded_scriptbuf = Builder::new()
52            .push_opcode(WitnessVersion::V1.into())
53            .push_slice(data)
54            .into_script();
55
56        builder = builder.add_output(UnspentTxOut::from_partial(TxOut {
57            value: MIN_TAPROOT_AMOUNT,
58            script_pubkey: data_encoded_scriptbuf,
59        }));
60        current_idx += 32;
61    }
62
63    // add the remaining data as an op_return output
64    if current_idx < paramset.watchtower_challenge_bytes {
65        let remaining_data =
66            PushBytesBuf::try_from(commit_data[current_idx..].to_vec()).map_err(|e| {
67                BridgeError::Error(format!(
68                    "Failed to create pushbytesbuf for watchtower challenge op_return: {}",
69                    e
70                ))
71            })?;
72        builder = builder.add_output(UnspentTxOut::from_partial(op_return_txout(remaining_data)));
73    }
74
75    Ok(builder
76        .add_output(UnspentTxOut::from_partial(
77            builder::transaction::anchor_output(),
78        ))
79        .finalize())
80}

82/// Creates the watchtower challenge timeout txhandler.
83/// This tx needs to be sent by operators when a watchtower doesn't send a challenge,
84/// otherwise operator will be forced to reveal their preimage.
85pub fn create_watchtower_challenge_timeout_txhandler(
86    kickoff_txhandler: &TxHandler,
87    watchtower_idx: usize,
88    paramset: &'static ProtocolParamset,
89) -> Result<TxHandler, BridgeError> {
90    let watchtower_challenge_vout = UtxoVout::WatchtowerChallenge(watchtower_idx);
91    let challenge_ack_vout = UtxoVout::WatchtowerChallengeAck(watchtower_idx);
92    Ok(
93        TxHandlerBuilder::new(TransactionType::WatchtowerChallengeTimeout(watchtower_idx))
94            .with_version(Version::non_standard(3))
95            .add_input(
96                (
97                    NumberedSignatureKind::WatchtowerChallengeTimeout1,
98                    watchtower_idx as i32,
99                ),
100                kickoff_txhandler.get_spendable_output(watchtower_challenge_vout)?,
101                SpendPath::ScriptSpend(0),
102                Sequence::from_height(paramset.watchtower_challenge_timeout_timelock),
103            )
104            .add_input(
105                (
106                    NumberedSignatureKind::WatchtowerChallengeTimeout2,
107                    watchtower_idx as i32,
108                ),
109                kickoff_txhandler.get_spendable_output(challenge_ack_vout)?,
110                SpendPath::ScriptSpend(1),
111                Sequence::from_height(paramset.watchtower_challenge_timeout_timelock),
112            )
113            .add_output(UnspentTxOut::from_partial(
114                builder::transaction::anchor_output(),
115            ))
116            .finalize(),
117    )
118}

120/// Creates a [`TxHandler`] for the `operator_challenge_NACK_tx`. This transaction will force
121/// the operator to reveal the preimage for the corresponding watchtower since if they do not
122/// reveal the preimage, the NofN will be able to spend the output after 0.5 week, which will
123/// prevent the operator from sending `assert_begin_tx`.
124pub fn create_operator_challenge_nack_txhandler(
125    kickoff_txhandler: &TxHandler,
126    watchtower_idx: usize,
127    round_txhandler: &TxHandler,
128    paramset: &'static ProtocolParamset,
129) -> Result<TxHandler, BridgeError> {
130    Ok(
131        TxHandlerBuilder::new(TransactionType::OperatorChallengeNack(watchtower_idx))
132            .with_version(Version::non_standard(3))
133            .add_input(
134                (
135                    NumberedSignatureKind::OperatorChallengeNack1,
136                    watchtower_idx as i32,
137                ),
138                kickoff_txhandler
139                    .get_spendable_output(UtxoVout::WatchtowerChallengeAck(watchtower_idx))?,
140                SpendPath::ScriptSpend(0),
141                Sequence::from_height(paramset.operator_challenge_nack_timelock),
142            )
143            .add_input(
144                (
145                    NumberedSignatureKind::OperatorChallengeNack2,
146                    watchtower_idx as i32,
147                ),
148                kickoff_txhandler.get_spendable_output(UtxoVout::KickoffFinalizer)?,
149                SpendPath::ScriptSpend(0),
150                DEFAULT_SEQUENCE,
151            )
152            .add_input(
153                (
154                    NumberedSignatureKind::OperatorChallengeNack3,
155                    watchtower_idx as i32,
156                ),
157                round_txhandler.get_spendable_output(UtxoVout::BurnConnector)?,
158                SpendPath::KeySpend,
159                DEFAULT_SEQUENCE,

161            .add_output(UnspentTxOut::from_partial(
162                builder::transaction::anchor_output(),
163            ))
164            .finalize(),
165    )
166}

168/// Creates a [`TxHandler`] for the `operator_challenge_ACK_tx`. This transaction will is used so that
169/// the operator can acknowledge the challenge and reveal the preimage for the corresponding watchtower.
170/// If the operator does not reveal the preimage, the NofN will be able to spend the output after 0.5 week using
171/// `operator_challenge_NACK_tx`.
172pub fn create_operator_challenge_ack_txhandler(
173    kickoff_txhandler: &TxHandler,
174    watchtower_idx: usize,
175    _paramset: &'static ProtocolParamset,
176) -> Result<TxHandler, BridgeError> {
177    Ok(
178        TxHandlerBuilder::new(TransactionType::OperatorChallengeAck(watchtower_idx))
179            .with_version(Version::non_standard(3))
180            .add_input(
181                NormalSignatureKind::OperatorChallengeAck1,
182                kickoff_txhandler
183                    .get_spendable_output(UtxoVout::WatchtowerChallengeAck(watchtower_idx))?,
184                SpendPath::ScriptSpend(2),
185                DEFAULT_SEQUENCE,
186            )
187            .add_output(UnspentTxOut::from_partial(
188                builder::transaction::anchor_output(),
189            ))
190            .add_output(UnspentTxOut::from_partial(op_return_txout(b"PADDING")))
191            .finalize(),
192    )
193}

195/// Creates a [`TxHandler`] for the `disprove_tx`. This transaction will be sent by NofN, meaning
196/// that the operator was malicious. This transaction burns the operator's burn connector, kicking the
197/// operator out of the system.
198pub fn create_disprove_txhandler(
199    kickoff_txhandler: &TxHandler,
200    round_txhandler: &TxHandler,
201) -> Result<TxHandler, BridgeError> {
202    Ok(TxHandlerBuilder::new(TransactionType::Disprove)
203        .add_input(
204            NormalSignatureKind::NoSignature,
205            kickoff_txhandler.get_spendable_output(UtxoVout::Disprove)?,
206            SpendPath::Unknown,
207            DEFAULT_SEQUENCE,
208        )
209        .add_input(
210            NormalSignatureKind::Disprove2,
211            round_txhandler.get_spendable_output(UtxoVout::BurnConnector)?,
212            SpendPath::KeySpend,
213            DEFAULT_SEQUENCE,
214        )
215        .add_output(UnspentTxOut::from_partial(
216            builder::transaction::anchor_output(),
217        ))
218        .finalize())
219}

221/// Creates a [`TxHandler`] for the `challenge`. This transaction is for covering
222/// the operators' cost for a challenge to prevent people from maliciously
223/// challenging them and causing them to lose money.
224pub fn create_challenge_txhandler(
225    kickoff_txhandler: &TxHandler,
226    operator_reimbursement_address: &bitcoin::Address,
227    paramset: &'static ProtocolParamset,
228) -> Result<TxHandler, BridgeError> {
229    Ok(TxHandlerBuilder::new(TransactionType::Challenge)
230        .with_version(Version::non_standard(3))
231        .add_input(
232            NormalSignatureKind::Challenge,
233            kickoff_txhandler.get_spendable_output(UtxoVout::Challenge)?,
234            SpendPath::ScriptSpend(0),
235            DEFAULT_SEQUENCE,
236        )
237        .add_output(UnspentTxOut::from_partial(TxOut {
238            value: paramset.operator_challenge_amount,
239            script_pubkey: operator_reimbursement_address.script_pubkey(),
240        }))
241        .add_output(UnspentTxOut::from_partial(op_return_txout(b"TODO")))
242        .finalize())
243}

247pub fn create_challenge_timeout_txhandler(
