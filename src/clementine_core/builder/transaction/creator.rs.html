26// helper function to get a txhandler from a hashmap
27fn get_txhandler(
28    txhandlers: &BTreeMap<TransactionType, TxHandler>,
29    tx_type: TransactionType,
30) -> Result<&TxHandler, TxError> {
31    txhandlers
32        .get(&tx_type)

37/// Helper struct to get specific kickoff winternitz keys for a sequential collateral tx
38pub struct KickoffWinternitzKeys {
39    pub keys: Vec<bitvm::signatures::winternitz::PublicKey>,
40    num_kickoffs_per_round: usize,

44    pub fn new(
45        keys: Vec<bitvm::signatures::winternitz::PublicKey>,
46        num_kickoffs_per_round: usize,
47    ) -> Self {
48        Self {
49            keys,
50            num_kickoffs_per_round,
51        }

54    /// Get the winternitz keys for a specific sequential collateral tx
55    pub fn get_keys_for_round(
56        &self,

344/// Struct to store common txhandlers for kickoffs
345pub struct TxHandlerCache {
346    pub prev_ready_to_reimburse: Option<TxHandler>,
347    pub saved_txs: BTreeMap<TransactionType, TxHandler>,
348}
349
350impl Default for TxHandlerCache {
351    fn default() -> Self {
352        Self::new()
353    }
354}
355
356impl TxHandlerCache {
357    pub fn new() -> Self {
358        Self {
359            saved_txs: BTreeMap::new(),
360            prev_ready_to_reimburse: None,
361        }
362    }
363    pub fn store_for_next_kickoff(
364        &mut self,
365        txhandlers: &mut BTreeMap<TransactionType, TxHandler>,
366    ) -> Result<(), BridgeError> {
367        // can possibly cache next round tx too, as next round has the needed reimburse utxos
368        // but need to implement a new TransactionType for that
369        for tx_type in [
370            TransactionType::MoveToVault,
371            TransactionType::Round,
372            TransactionType::ReadyToReimburse,
373        ]
374        .iter()
375        {
376            let txhandler = txhandlers
377                .remove(tx_type)
378                .ok_or(TxError::TxHandlerNotFound(*tx_type))?;
379            self.saved_txs.insert(*tx_type, txhandler);
380        }
381        Ok(())
382    }
383    /// store move_to_vault and previous ready to reimburse
384    pub fn store_for_next_round(&mut self) -> Result<(), BridgeError> {
385        let move_to_vault =
386            remove_txhandler_from_map(&mut self.saved_txs, TransactionType::MoveToVault)?;
387        self.prev_ready_to_reimburse = Some(remove_txhandler_from_map(
388            &mut self.saved_txs,
389            TransactionType::ReadyToReimburse,
390        )?);
391        self.saved_txs = BTreeMap::new();
392        self.saved_txs
393            .insert(move_to_vault.get_transaction_type(), move_to_vault);
394        Ok(())
395    }
396    pub fn get_prev_ready_to_reimburse(&self) -> Option<&TxHandler> {
397        self.prev_ready_to_reimburse.as_ref()
398    }
399    pub fn get_cached_txs(&mut self) -> BTreeMap<TransactionType, TxHandler> {
400        std::mem::take(&mut self.saved_txs)
401    }
402}

404pub async fn create_txhandlers(
405    transaction_type: TransactionType,
406    context: ContractContext,
407    txhandler_cache: &mut TxHandlerCache,
408    db_cache: &mut ReimburseDbCache,
409) -> Result<BTreeMap<TransactionType, TxHandler>, BridgeError> {
410    let ReimburseDbCache { paramset, .. } = db_cache.clone();
411
412    let operator_data = db_cache.get_operator_data().await?.clone();
413    let kickoff_winternitz_keys = db_cache.get_kickoff_winternitz_keys().await?;
414
415    let ContractContext {
416        operator_xonly_pk,
417        round_idx,
418        ..
419    } = context;
420
421    let mut txhandlers = txhandler_cache.get_cached_txs();
422
423    if !txhandlers.contains_key(&TransactionType::Round) {
424        // create round tx, ready to reimburse tx, and unspent kickoff txs if not in cache
425        let round_txhandlers = create_round_txhandlers(
426            paramset,
427            round_idx as usize,
428            &operator_data,
429            kickoff_winternitz_keys,
430            txhandler_cache.get_prev_ready_to_reimburse(),
431        )?;
432        for round_txhandler in round_txhandlers.into_iter() {
433            txhandlers.insert(round_txhandler.get_transaction_type(), round_txhandler);
434        }
435    }
436
437    if matches!(
438        transaction_type,
439        TransactionType::Round
440            | TransactionType::ReadyToReimburse
441            | TransactionType::UnspentKickoff(_)
442    ) {
443        // return if only one of the collateral tx's were requested
444        // do not continue as we might not have the necessary context for the remaining tx's
445        return Ok(txhandlers);
446    }
447
448    // get the next round txhandler (because reimburse connectors will be in it)
449    let next_round_txhandler = create_round_txhandler(
450        operator_data.xonly_pk,
451        RoundTxInput::Prevout(Box::new(
452            get_txhandler(&txhandlers, TransactionType::ReadyToReimburse)?
453                .get_spendable_output(UtxoVout::BurnConnector)?,
454        )),
455        kickoff_winternitz_keys.get_keys_for_round(round_idx as usize + 1),
456        paramset,
457    )?;
458
459    let mut deposit_data = context.deposit_data.ok_or(TxError::InsufficientContext)?;
460    let kickoff_data = KickoffData {
461        operator_xonly_pk,
462        round_idx,
463        kickoff_idx: context.kickoff_idx.ok_or(TxError::InsufficientContext)?,
464    };
465
466    if !txhandlers.contains_key(&TransactionType::MoveToVault) {
467        // if not cached create move_txhandler
468        let move_txhandler =
469            builder::transaction::create_move_to_vault_txhandler(&mut deposit_data, paramset)?;
470        txhandlers.insert(move_txhandler.get_transaction_type(), move_txhandler);
471    }
472
473    let num_asserts = ClementineBitVMPublicKeys::number_of_assert_txs();
474    let public_hashes = db_cache.get_challenge_ack_hashes().await?.to_vec();
475
476    let disprove_root_hash = *db_cache.get_bitvm_disprove_root_hash().await?;
477    let latest_blockhash_root_hash = *db_cache.get_latest_blockhash_root_hash().await?;
478
479    let kickoff_txhandler = if matches!(
480        transaction_type,
481        TransactionType::LatestBlockhash | TransactionType::MiniAssert(_)
482    ) {
483        // create scripts if any mini assert tx or latest blockhash tx is specifically requested as it needs
484        // the actual scripts to be able to spend
485        let actor = context.signer.clone().ok_or(TxError::InsufficientContext)?;
486
487        // deposit_data.deposit_outpoint.txid
488
489        let bitvm_pks =
490            actor.generate_bitvm_pks_for_deposit(deposit_data.get_deposit_outpoint(), paramset)?;
491
492        let assert_scripts = bitvm_pks.get_assert_scripts(operator_data.xonly_pk);
493
494        let latest_blockhash_script = Arc::new(WinternitzCommit::new(
495            vec![(bitvm_pks.latest_blockhash_pk.to_vec(), 40)],
496            operator_data.xonly_pk,
497            context.paramset.winternitz_log_d,
498        ));
499
500        let kickoff_txhandler = create_kickoff_txhandler(
501            kickoff_data,
502            get_txhandler(&txhandlers, TransactionType::Round)?,
503            get_txhandler(&txhandlers, TransactionType::MoveToVault)?,
504            &mut deposit_data,
505            operator_data.xonly_pk,
506            AssertScripts::AssertSpendableScript(assert_scripts),
507            &disprove_root_hash,
508            AssertScripts::AssertSpendableScript(vec![latest_blockhash_script]),
509            &public_hashes,
510            paramset,
511        )?;
512
513        // Create and insert mini_asserts into return Vec
514        let mini_asserts = create_mini_asserts(&kickoff_txhandler, num_asserts)?;
515
516        for mini_assert in mini_asserts.into_iter() {
517            txhandlers.insert(mini_assert.get_transaction_type(), mini_assert);
518        }
519
520        let latest_blockhash_txhandler = create_latest_blockhash_txhandler(&kickoff_txhandler)?;
521        txhandlers.insert(
522            latest_blockhash_txhandler.get_transaction_type(),
523            latest_blockhash_txhandler,
524        );
525
526        kickoff_txhandler
527    } else {
528        // use db data for scripts
529        create_kickoff_txhandler(
530            kickoff_data,
531            get_txhandler(&txhandlers, TransactionType::Round)?,
532            get_txhandler(&txhandlers, TransactionType::MoveToVault)?,
533            &mut deposit_data,
534            operator_data.xonly_pk,
535            AssertScripts::AssertScriptTapNodeHash(db_cache.get_bitvm_assert_hash().await?),
536            &disprove_root_hash,
537            AssertScripts::AssertScriptTapNodeHash(&[latest_blockhash_root_hash]),
538            &public_hashes,
539            paramset,
540        )?
541    };
542    txhandlers.insert(kickoff_txhandler.get_transaction_type(), kickoff_txhandler);
543
544    // Creates the challenge_tx handler.
545    let challenge_txhandler = builder::transaction::create_challenge_txhandler(
546        get_txhandler(&txhandlers, TransactionType::Kickoff)?,
547        &operator_data.reimburse_addr,
548        paramset,
549    )?;
550    txhandlers.insert(
551        challenge_txhandler.get_transaction_type(),
552        challenge_txhandler,
553    );
554
555    // Creates the challenge timeout txhandler
556    let challenge_timeout_txhandler = create_challenge_timeout_txhandler(
557        get_txhandler(&txhandlers, TransactionType::Kickoff)?,
558        paramset,
559    )?;
560
561    txhandlers.insert(
562        challenge_timeout_txhandler.get_transaction_type(),
563        challenge_timeout_txhandler,
564    );
565
566    let kickoff_not_finalized_txhandler =
567        builder::transaction::create_kickoff_not_finalized_txhandler(
568            get_txhandler(&txhandlers, TransactionType::Kickoff)?,
569            get_txhandler(&txhandlers, TransactionType::ReadyToReimburse)?,
570        )?;
571    txhandlers.insert(
572        kickoff_not_finalized_txhandler.get_transaction_type(),
573        kickoff_not_finalized_txhandler,
574    );
575
576    let latest_blockhash_timeout_txhandler = create_latest_blockhash_timeout_txhandler(
577        get_txhandler(&txhandlers, TransactionType::Kickoff)?,
578        get_txhandler(&txhandlers, TransactionType::Round)?,
579        paramset,
580    )?;
581    txhandlers.insert(
582        latest_blockhash_timeout_txhandler.get_transaction_type(),
583        latest_blockhash_timeout_txhandler,
584    );
585
586    // create watchtower tx's except WatchtowerChallenges
587    for watchtower_idx in 0..deposit_data.get_num_watchtowers() {
588        // Each watchtower will sign their Groth16 proof of the header chain circuit. Then, the operator will either
589        // - acknowledge the challenge by sending the operator_challenge_ACK_tx, otherwise their burn connector
590        // will get burned by operator_challenge_nack
591        let watchtower_challenge_timeout_txhandler =
592            builder::transaction::create_watchtower_challenge_timeout_txhandler(
593                get_txhandler(&txhandlers, TransactionType::Kickoff)?,
594                watchtower_idx,
595                paramset,
596            )?;
597        txhandlers.insert(
598            watchtower_challenge_timeout_txhandler.get_transaction_type(),
599            watchtower_challenge_timeout_txhandler,
600        );
601
602        let operator_challenge_nack_txhandler =
603            builder::transaction::create_operator_challenge_nack_txhandler(
604                get_txhandler(&txhandlers, TransactionType::Kickoff)?,
605                watchtower_idx,
606                get_txhandler(&txhandlers, TransactionType::Round)?,
607                paramset,
608            )?;
609        txhandlers.insert(
610            operator_challenge_nack_txhandler.get_transaction_type(),
611            operator_challenge_nack_txhandler,
612        );
613
614        let operator_challenge_ack_txhandler =
615            builder::transaction::create_operator_challenge_ack_txhandler(
616                get_txhandler(&txhandlers, TransactionType::Kickoff)?,
617                watchtower_idx,
618                paramset,
619            )?;
620        txhandlers.insert(
621            operator_challenge_ack_txhandler.get_transaction_type(),
622            operator_challenge_ack_txhandler,
623        );
624    }
625
626    if let TransactionType::WatchtowerChallenge(_) = transaction_type {
627        return Err(BridgeError::Error(
628            "Cant directly create a watchtower challenge in create_txhandlers as it needs commit data".to_string(),
629        ));
630    }
631
632    let assert_timeouts = create_assert_timeout_txhandlers(
633        get_txhandler(&txhandlers, TransactionType::Kickoff)?,
634        get_txhandler(&txhandlers, TransactionType::Round)?,
635        num_asserts,
636        paramset,
637    )?;
638
639    for assert_timeout in assert_timeouts.into_iter() {
640        txhandlers.insert(assert_timeout.get_transaction_type(), assert_timeout);
641    }
642
643    // Creates the disprove_timeout_tx handler.
644    let disprove_timeout_txhandler = builder::transaction::create_disprove_timeout_txhandler(
645        get_txhandler(&txhandlers, TransactionType::Kickoff)?,
646        paramset,
647    )?;
648
649    txhandlers.insert(
650        disprove_timeout_txhandler.get_transaction_type(),
651        disprove_timeout_txhandler,
652    );
653
654    // Creates the reimburse_tx handler.
655    let reimburse_txhandler = builder::transaction::create_reimburse_txhandler(
656        get_txhandler(&txhandlers, TransactionType::MoveToVault)?,
657        &next_round_txhandler,
658        get_txhandler(&txhandlers, TransactionType::Kickoff)?,
659        kickoff_data.kickoff_idx as usize,
660        paramset,
661        &operator_data.reimburse_addr,
662    )?;
663
664    txhandlers.insert(
665        reimburse_txhandler.get_transaction_type(),
666        reimburse_txhandler,
667    );
668
669    match transaction_type {
670        TransactionType::AllNeededForDeposit => {
671            let disprove_txhandler = builder::transaction::create_disprove_txhandler(
672                get_txhandler(&txhandlers, TransactionType::Kickoff)?,
673                get_txhandler(&txhandlers, TransactionType::Round)?,
674            )?;
675            txhandlers.insert(
676                disprove_txhandler.get_transaction_type(),
677                disprove_txhandler,
678            );
679        }
680        TransactionType::Disprove => {
681            // TODO: if TransactionType::Disprove, we need to add the actual disprove script here because requester wants to disprove the withdrawal
682        }
683        _ => {}
684    }
685
686    Ok(txhandlers)
687}

689/// Function to create the round txhandler and ready to reimburse txhandler for a specific operator and round index
690pub fn create_round_txhandlers(
691    paramset: &'static ProtocolParamset,
692    round_idx: usize,
693    operator_data: &OperatorData,
694    kickoff_winternitz_keys: &KickoffWinternitzKeys,
695    prev_ready_to_reimburse: Option<&TxHandler>,
696) -> Result<Vec<TxHandler>, BridgeError> {
697    let mut txhandlers = Vec::with_capacity(2 + paramset.num_kickoffs_per_round);
698
699    let (round_txhandler, ready_to_reimburse_txhandler) = match prev_ready_to_reimburse {
700        Some(prev_ready_to_reimburse_txhandler) => {
701            if round_idx == 0 {
702                return Err(
703                    eyre::eyre!("Round 0 cannot be created from prev_ready_to_reimburse").into(),
704                );
705            }
706            let round_txhandler = builder::transaction::create_round_txhandler(
707                operator_data.xonly_pk,
708                RoundTxInput::Prevout(Box::new(
709                    prev_ready_to_reimburse_txhandler
710                        .get_spendable_output(UtxoVout::BurnConnector)?,
711                )),
712                kickoff_winternitz_keys.get_keys_for_round(round_idx),
713                paramset,
714            )?;
715
716            let ready_to_reimburse_txhandler =
717                builder::transaction::create_ready_to_reimburse_txhandler(
718                    &round_txhandler,
719                    operator_data.xonly_pk,
720                    paramset,
721                )?;
722            (round_txhandler, ready_to_reimburse_txhandler)
723        }
724        None => {
725            // create nth sequential collateral tx and reimburse generator tx for the operator
726            builder::transaction::create_round_nth_txhandler(
727                operator_data.xonly_pk,
728                operator_data.collateral_funding_outpoint,
729                paramset.collateral_funding_amount,
730                round_idx,
731                kickoff_winternitz_keys,
732                paramset,
733            )?
734        }
735    };
736
737    let unspent_kickoffs = create_unspent_kickoff_txhandlers(
738        &round_txhandler,
739        &ready_to_reimburse_txhandler,
740        paramset,
741    )?;
742
743    txhandlers.push(round_txhandler);
744    txhandlers.push(ready_to_reimburse_txhandler);
745
746    for unspent_kickoff in unspent_kickoffs.into_iter() {
747        txhandlers.push(unspent_kickoff);
748    }
749
750    Ok(txhandlers)
751}

1092        txhandlers.insert(
1093            TransactionType::Kickoff,
1094            TxHandlerBuilder::new(TransactionType::Kickoff).finalize(),
1095        );
1096

1098        assert!(cache.store_for_next_kickoff(&mut txhandlers).is_ok());

1100        assert!(cache.saved_txs.len() == 3);

1103        assert!(cache
1104            .saved_txs

1106        // prev_ready_to_reimburse should be None as it is the first iteration
1107        assert!(cache.prev_ready_to_reimburse.is_none());

1110        txhandlers = cache.get_cached_txs();

1112        assert!(txhandlers.contains_key(&TransactionType::MoveToVault));
1113        assert!(txhandlers.contains_key(&TransactionType::Round));
1114        assert!(txhandlers.contains_key(&TransactionType::ReadyToReimburse));
1115        assert!(cache.store_for_next_kickoff(&mut txhandlers).is_ok());
1116        // prev ready to reimburse still none as we didnt go to next round
1117        assert!(cache.prev_ready_to_reimburse.is_none());
1118
1119        // should delete saved txs and store prev ready to reimburse, but it should keep movetovault
1120        assert!(cache.store_for_next_round().is_ok());
1121        assert!(cache.saved_txs.len() == 1);
1122        assert!(cache.prev_ready_to_reimburse.is_some());
1123        assert!(cache.saved_txs.contains_key(&TransactionType::MoveToVault));
1124
1125        // retrieve cached movetovault
1126        txhandlers = cache.get_cached_txs();
1127
1128        // create new round txs
1129        txhandlers.insert(
1130            TransactionType::ReadyToReimburse,
1131            TxHandlerBuilder::new(TransactionType::ReadyToReimburse).finalize(),

1133        txhandlers.insert(
1134            TransactionType::Round,
1135            TxHandlerBuilder::new(TransactionType::Round).finalize(),
1136        );
1137        // add not relevant tx
