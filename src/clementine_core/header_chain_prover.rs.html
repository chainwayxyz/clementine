87    /// Creates a new [`HeaderChainProver`] instance. Also saves a proof
88    /// assumption if specified in the config.
89    pub async fn new(
90        config: &BridgeConfig,
91        rpc: ExtendedBitcoinRpc,
92    ) -> Result<Self, HeaderChainProverError> {
93        let db = Database::new(config).await.map_to_eyre()?;
94        let tip_height = rpc.get_current_chain_height().await.map_to_eyre()?;
95        if tip_height
96            < config.protocol_paramset().start_height + config.protocol_paramset().finality_depth
97                - 1
98        {
99            return Err(eyre::eyre!(
100                "Start height is not finalized, reduce start height: {} < {}",
101                tip_height,
102                config.protocol_paramset().start_height + config.protocol_paramset().finality_depth
103                    - 1
104            )
105            .into());
106        }
107        db.fetch_and_save_missing_blocks(
108            None,
109            &rpc,
110            config.protocol_paramset().genesis_height,
111            config.protocol_paramset().start_height,
112        )
113        .await
114        .wrap_err("Failed to save initial block infos")?;
115
116        if let Some(proof_file) = &config.header_chain_proof_path {
117            tracing::info!("Starting prover with assumption file {:?}.", proof_file);
118            let file = File::open(proof_file)
119                .wrap_err_with(|| format!("Failed to open proof assumption file {proof_file:?}"))?;
120
121            let mut reader = BufReader::new(file);
122            let mut assumption = Vec::new();
123            reader
124                .read_to_end(&mut assumption)
125                .wrap_err("Can't read assumption file")?;
126
127            let proof: Receipt = borsh::from_slice(&assumption)
128                .wrap_err(HeaderChainProverError::ProverDeSerializationError)?;
129            let proof_output: BlockHeaderCircuitOutput = borsh::from_slice(&proof.journal.bytes)
130                .wrap_err(HeaderChainProverError::ProverDeSerializationError)?;
131
132            let network = config.protocol_paramset().network;
133
134            if proof_output.method_id != get_hcp_method_id(network)? {
135                return Err(eyre::eyre!(
136                    "Header chain proof assumption file Method ID mismatch for our current network ({:?}): got {:?}, expected {:?}",
137                    network,
138                    proof_output.method_id,
139                    get_hcp_method_id(network)?
140                )
141                .into());
142            }
143
144            if proof.verify(proof_output.method_id).is_err() {
145                return Err(eyre::eyre!(
146                    "Header chain proof assumption file verification failed for our current network ({:?})",
147                    network
148                )
149                .into());
150            }
151
152            // Create block entry, if not exists.
153            let block_hash = BlockHash::from_raw_hash(
154                Hash::from_slice(&proof_output.chain_state.best_block_hash).map_to_eyre()?,
155            );
156            let block_header = rpc.get_block_header(&block_hash).await.wrap_err(format!(
157                "Failed to get block header with block hash {block_hash} (retrieved from assumption file)",
158            ))?;
159            let block_height = rpc
160                .get_block_info(&block_hash)
161                .await
162                .map(|info| info.height)
163                .wrap_err(format!(
164                    "Failed to get block info with block hash {block_hash} (retrieved from assumption file)"
165                ))?;
166            tracing::info!(
167                "Adding proof assumption for a block with hash of {:?}, header of {:?} and height of {}",
168                block_hash,
169                block_header,
170                block_height
171            );
172
173            // If an unproven block in database already exists, it shouldn't
174            // effect anything.
175            // PS: This also ignores other db errors but there are other places
176            // where we check for those errors.
177            let _ = db
178                .save_unproven_finalized_block(
179                    None,
180                    block_hash,
181                    block_header,
182                    proof_output.chain_state.block_height.into(),
183                )
184                .await
185                .inspect_err(|e| {
186                    tracing::debug!("Can't set initial block info for header chain prover, because: {e}. Doesn't affect anything, continuing...");
187                });
188
189            db.set_block_proof(None, block_hash, proof)
190                .await
191                .map_to_eyre()?;
192        } else {
193            tracing::info!("Starting prover without assumption, proving genesis block");
194
195            let genesis_block_hash = rpc
196                .get_block_hash(config.protocol_paramset().genesis_height.into())
197                .await
198                .wrap_err(format!(
199                    "Failed to get genesis block hash at height {}",
200                    config.protocol_paramset().genesis_height
201                ))?;
202
203            tracing::debug!(
204                "Genesis block hash: {}, height: {}",
205                genesis_block_hash,
206                config.protocol_paramset().genesis_height
207            ); // Should be debug
208
209            let genesis_block_header =
210                rpc.get_block_header(&genesis_block_hash)
211                    .await
212                    .wrap_err(format!(
213                        "Failed to get genesis block header at height {}",
214                        config.protocol_paramset().genesis_height
215                    ))?;
216
217            let genesis_chain_state = HeaderChainProver::get_chain_state_from_height(
218                rpc.clone(),
219                config.protocol_paramset().genesis_height.into(),
220                config.protocol_paramset().network,
221            )
222            .await
223            .map_to_eyre()?;
224            tracing::debug!("Genesis chain state (verbose): {:?}", genesis_chain_state);
225
226            let genesis_chain_state_hash = genesis_chain_state.to_hash();
227            if genesis_chain_state_hash != config.protocol_paramset().genesis_chain_state_hash {
228                return Err(eyre::eyre!(
229                    "Genesis chain state hash mismatch: {} != {}",
230                    hex::encode(genesis_chain_state_hash),
231                    hex::encode(config.protocol_paramset().genesis_chain_state_hash)
232                )
233                .into());
234            }
235
236            let proof = HeaderChainProver::prove_genesis_block(
237                genesis_chain_state,
238                config.protocol_paramset().network,
239            )
240            .map_to_eyre()?;
241
242            let _ = db
243                .save_unproven_finalized_block(
244                    None,
245                    genesis_block_hash,
246                    genesis_block_header,
247                    config.protocol_paramset().genesis_height.into(),
248                )
249                .await;
250
251            db.set_block_proof(None, genesis_block_hash, proof)
252                .await
253                .map_to_eyre()?;
254        }
255
256        Ok(HeaderChainProver {
257            db,
258            batch_size: config
259                .protocol_paramset()
260                .header_chain_proof_batch_size
261                .into(),
262            network: config.protocol_paramset().network,
263        })
264    }

266    pub async fn get_chain_state_from_height(
267        rpc: ExtendedBitcoinRpc,
268        height: u64,
269        network: Network,
270    ) -> Result<ChainState, HeaderChainProverError> {
271        let block_hash = rpc
272            .get_block_hash(height)
273            .await
274            .wrap_err(format!("Failed to get block hash at height {height}"))?;
275
276        let block_header = rpc.get_block_header(&block_hash).await.wrap_err(format!(
277            "Failed to get block header with block hash {block_hash}"
278        ))?;
279
280        let mut last_11_block_timestamps: [u32; 11] = [0; 11];
281        let mut last_block_hash = block_hash;
282        let mut last_block_height = height;
283        for _ in 0..11 {
284            let block_header = rpc
285                .get_block_header(&last_block_hash)
286                .await
287                .wrap_err(format!(
288                    "Failed to get block header with block hash {last_block_hash}"
289                ))?;
290
291            last_11_block_timestamps[last_block_height as usize % 11] = block_header.time;
292
293            last_block_hash = block_header.prev_blockhash;
294            last_block_height = last_block_height.wrapping_sub(1);
295
296            if last_block_hash.to_byte_array() == [0u8; 32] {
297                break;
298            }
299        }
300
301        let epoch_start_block_height = height / 2016 * 2016;
302
303        let (epoch_start_timestamp, expected_bits) = if network == Network::Regtest {
304            (0, block_header.bits.to_consensus())
305        } else {
306            let epoch_start_block_hash = rpc
307                .get_block_hash(epoch_start_block_height)
308                .await
309                .wrap_err(format!(
310                    "Failed to get block hash at height {epoch_start_block_height}"
311                ))?;
312            let epoch_start_block_header = rpc
313                .get_block_header(&epoch_start_block_hash)
314                .await
315                .wrap_err(format!(
316                    "Failed to get block header with block hash {epoch_start_block_hash}"
317                ))?;
318            let bits = if network == Network::Testnet4 {
319                // Real difficulty will show up at epoch start block no matter what
320                epoch_start_block_header.bits.to_consensus()
321            } else {
322                block_header.bits.to_consensus()
323            };
324
325            (epoch_start_block_header.time, bits)
326        };
327
328        let block_info = rpc.get_block_info(&block_hash).await.wrap_err(format!(
329            "Failed to get block info with block hash {block_hash}"
330        ))?;
331
332        let total_work = block_info.chainwork;
333
334        let total_work: [u8; 32] = total_work.try_into().expect("Total work is 32 bytes");
335
336        let mut block_hashes_mmr = MMRGuest::new();
337        block_hashes_mmr.append(block_hash.to_byte_array());
338
339        let chain_state = ChainState {
340            block_height: height as u32,
341            total_work,
342            best_block_hash: block_hash.to_byte_array(),
343            current_target_bits: expected_bits,
344            epoch_start_time: epoch_start_timestamp,
345            prev_11_timestamps: last_11_block_timestamps,
346            block_hashes_mmr,
347        };
348        Ok(chain_state)
349    }

351    /// Proves the work only proof for the given HCP receipt.
352    pub fn prove_work_only(
353        &self,
354        hcp_receipt: Receipt,
355    ) -> Result<(Receipt, WorkOnlyCircuitOutput), HeaderChainProverError> {
356        let block_header_circuit_output: BlockHeaderCircuitOutput =
357            borsh::from_slice(&hcp_receipt.journal.bytes)
358                .wrap_err(HeaderChainProverError::ProverDeSerializationError)?;
359        let input = WorkOnlyCircuitInput {
360            header_chain_circuit_output: block_header_circuit_output,
361        };
362        let mut env = ExecutorEnv::builder();
363
364        env.write_slice(&borsh::to_vec(&input).wrap_err(BridgeError::BorshError)?);
365
366        env.add_assumption(hcp_receipt);
367
368        let env = env
369            .build()
370            .map_err(|e| eyre::eyre!(e))
371            .wrap_err("Failed to build environment")?;
372
373        let prover = risc0_zkvm::default_prover();
374
375        let elf = match self.network {
376            Network::Bitcoin => MAINNET_WORK_ONLY_ELF,
377            Network::Testnet4 => TESTNET4_WORK_ONLY_ELF,
378            Network::Signet => SIGNET_WORK_ONLY_ELF,
379            Network::Regtest => REGTEST_WORK_ONLY_ELF,
380            _ => Err(BridgeError::UnsupportedNetwork.into_eyre())?,
381        };
382
383        tracing::warn!("Starting proving HCP work only proof for creating a watchtower challenge");
384        let receipt = if !is_dev_mode() {
385            prover
386                .prove_with_opts(env, elf, &ProverOpts::groth16())
387                .map_err(|e| eyre::eyre!(e))?
388                .receipt
389        } else {
390            let stark_receipt = prover
391                .prove_with_opts(env, elf, &ProverOpts::succinct())
392                .map_err(|e| eyre::eyre!(e))?
393                .receipt;
394            let journal = stark_receipt.journal.bytes.clone();
395            dev_stark_to_risc0_g16(stark_receipt, &journal)?
396        };
397        tracing::warn!("HCP work only proof proof generated for creating a watchtower challenge");
398        let work_output: WorkOnlyCircuitOutput = borsh::from_slice(&receipt.journal.bytes)
399            .wrap_err(HeaderChainProverError::ProverDeSerializationError)?;
400
401        Ok((receipt, work_output))
402    }

404    /// Proves blocks till the block with hash `current_block_hash`.
405    ///
406    /// # Parameters
407    ///
408    /// - `current_block_hash`: Hash of the target block
409    /// - `block_headers`: Previous block headers before the target block
410    /// - `previous_proof`: Previous proof's receipt
411    #[tracing::instrument(skip_all)]
412    async fn prove_and_save_block(
413        &self,
414        current_block_hash: BlockHash,
415        block_headers: Vec<Header>,
416        previous_proof: Receipt,
417    ) -> Result<Receipt, BridgeError> {
418        tracing::debug!(
419            "Prover starts proving {} blocks ending with block with hash {}",
420            block_headers.len(),
421            current_block_hash
422        );
423
424        let headers: Vec<CircuitBlockHeader> = block_headers.into_iter().map(Into::into).collect();
425        let network = self.network;
426        let receipt = tokio::task::spawn_blocking(move || {
427            Self::prove_block_headers(network, previous_proof, headers)
428        })
429        .await
430        .wrap_err("Failed to join the prove_block_headers task")?
431        .wrap_err("Failed to prove block headers")?;
432
433        self.db
434            .set_block_proof(None, current_block_hash, receipt.clone())
435            .await?;
436
437        Ok(receipt)
438    }

440    /// Proves given block headers.
441    ///
442    /// # Parameters
443    ///
444    /// - `prev_receipt`: Previous proof's receipt, if not genesis block
445    /// - `block_headers`: Block headers to prove
446    ///
447    /// # Returns
448    ///
449    /// - [`Receipt`]: Proved block headers' proof receipt.
450    fn prove_block_headers(
451        network: Network,
452        prev_receipt: Receipt,
453        block_headers: Vec<CircuitBlockHeader>,
454    ) -> Result<Receipt, HeaderChainProverError> {
455        // Prepare proof input.
456        let prev_output: BlockHeaderCircuitOutput = borsh::from_slice(&prev_receipt.journal.bytes)
457            .wrap_err(HeaderChainProverError::ProverDeSerializationError)?;
458        let method_id = prev_output.method_id;
459
460        let prev_proof = HeaderChainPrevProofType::PrevProof(prev_output);
461
462        let input = HeaderChainCircuitInput {
463            method_id,
464            prev_proof,
465            block_headers,
466        };
467        Self::prove_with_input(input, Some(prev_receipt), network)
468    }

470    pub fn prove_genesis_block(
471        genesis_chain_state: ChainState,
472        network: Network,
473    ) -> Result<Receipt, HeaderChainProverError> {
474        let image_id = get_hcp_method_id(network)?;
475        let header_chain_circuit_type = HeaderChainPrevProofType::GenesisBlock(genesis_chain_state);
476        let input = HeaderChainCircuitInput {
477            method_id: image_id,
478            prev_proof: header_chain_circuit_type,
479            block_headers: vec![],
480        };
481
482        Self::prove_with_input(input, None, network)
483    }

485    fn prove_with_input(
486        input: HeaderChainCircuitInput,
487        prev_receipt: Option<Receipt>,
488        network: Network,
489    ) -> Result<Receipt, HeaderChainProverError> {
490        let mut env = ExecutorEnv::builder();
491
492        env.write_slice(&borsh::to_vec(&input).wrap_err(BridgeError::BorshError)?);
493
494        if let Some(prev_receipt) = prev_receipt {
495            env.add_assumption(prev_receipt);
496        }
497
498        let env = env
499            .build()
500            .map_err(|e| eyre::eyre!(e))
501            .wrap_err("Failed to build environment")?;
502
503        let prover = risc0_zkvm::default_prover();
504
505        let elf = match network {
506            Network::Bitcoin => MAINNET_HEADER_CHAIN_ELF,
507            Network::Testnet4 => TESTNET4_HEADER_CHAIN_ELF,
508            Network::Signet => SIGNET_HEADER_CHAIN_ELF,
509            Network::Regtest => REGTEST_HEADER_CHAIN_ELF,
510            _ => Err(BridgeError::UnsupportedNetwork.into_eyre())?,
511        };
512
513        let receipt = prover.prove(env, elf).map_err(|e| eyre::eyre!(e))?.receipt;
514        tracing::debug!(
515            "Proof receipt for header chain circuit input {:?}: {:?}",
516            input,
517            receipt
518        );
519
520        Ok(receipt)
521    }

523    /// Produces a proof for the chain up to the block with the given hash.
524    ///
525    /// # Returns
526    ///
527    /// - [`Receipt`]: Specified block's proof receipt
528    /// - [`u64`]: Height of the proven header chain
529    pub async fn prove_till_hash(
530        &self,
531        block_hash: BlockHash,
532    ) -> Result<(Receipt, u64), BridgeError> {
533        let (_, _, height) = self
534            .db
535            .get_block_info_from_hash_hcp(None, block_hash)
536            .await?
537            .ok_or(eyre::eyre!("Block not found in prove_till_hash"))?;
538
539        let latest_proven_block = self
540            .db
541            .get_latest_proven_block_info_until_height(None, height)
542            .await?
543            .ok_or_eyre("No proofs found before the given block hash")?;
544
545        if latest_proven_block.2 == height as u64 {
546            let receipt = self
547                .db
548                .get_block_proof_by_hash(None, latest_proven_block.0)
549                .await
550                .wrap_err("Failed to get block proof")?
551                .ok_or(eyre!("Failed to get block proof"))?;
552            return Ok((receipt, height as u64));
553        }
554
555        let block_headers = self
556            .db
557            .get_block_info_from_range(None, latest_proven_block.2 + 1, height.into())
558            .await?
559            .into_iter()
560            .map(|(_hash, header)| header)
561            .collect::<Vec<_>>();
562
563        let previous_proof = self
564            .db
565            .get_block_proof_by_hash(None, latest_proven_block.0)
566            .await?
567            .ok_or(eyre::eyre!("No proven block found"))?;
568        let receipt = self
569            .prove_and_save_block(block_hash, block_headers, previous_proof)
570            .await?;
571        tracing::info!("Generated new proof for height {}", height);
572        Ok((receipt, height as u64))
573    }

575    /// Gets the proof of the latest finalized blockchain tip. If the finalized
576    /// blockchain tip isn't yet proven, it will be proven first in batches
577    /// (last proven block in database to finalized blockchain tip).
578    ///
579    /// # Returns
580    ///
581    /// - [`Receipt`]: Specified block's proof receipt
582    /// - [`u64`]: Height of the proven header chain
583    pub async fn get_tip_header_chain_proof(&self) -> Result<(Receipt, u64), BridgeError> {
584        let max_height = self.db.get_latest_finalized_block_height(None).await?;
585
586        if let Some(max_height) = max_height {
587            let block_hash = self
588                .db
589                .get_block_info_from_range(None, max_height, max_height)
590                .await?
591                .into_iter()
592                .next()
593                .expect("Block should be in table")
594                .0;
595            Ok(self.prove_till_hash(block_hash).await?)
596        } else {
597            Err(eyre::eyre!("No finalized blocks in header chain proofs table").into())
598        }
599    }

623    /// Checks if there are enough blocks to prove.
624    #[tracing::instrument(skip_all)]
625    async fn is_batch_ready(&self) -> Result<bool, BridgeError> {
626        let non_proven_block = if let Some(block) = self.db.get_next_unproven_block(None).await? {
627            block
628        } else {
629            return Ok(false);
630        };
631        let tip_height = self
632            .db
633            .get_latest_finalized_block_height(None)
634            .await?
635            .ok_or(eyre::eyre!("No tip block found"))?;
636
637        tracing::debug!(
638            "Tip height: {}, non proven block height: {}, {}",
639            tip_height,
640            non_proven_block.2,
641            self.batch_size
642        );
643        if tip_height - non_proven_block.2 + 1 >= self.batch_size {
644            return Ok(true);
645        }
646        tracing::debug!(
647            "Batch not ready: {} - {} < {}",
648            tip_height,
649            non_proven_block.2,
650            self.batch_size
651        );
652
653        Ok(false)
654    }

656    /// Proves blocks if the batch is ready. If not, skips.
657    pub async fn prove_if_ready(&self) -> Result<Option<Receipt>, BridgeError> {
658        if !self.is_batch_ready().await? {
659            return Ok(None);
660        }
661
662        let unproven_blocks = self
663            .db
664            .get_next_n_non_proven_block(
665                self.batch_size
666                    .try_into()
667                    .wrap_err("Can't convert u64 to u32")?,
668            )
669            .await?;
670        let (unproven_blocks, prev_proof) = match unproven_blocks {
671            Some(unproven_blocks) => unproven_blocks,
672            None => {
673                tracing::debug!("No unproven blocks found");
674                return Ok(None);
675            }
676        };
677
678        let current_block_hash = unproven_blocks.iter().next_back().expect("Exists").0;
679        let current_block_height = unproven_blocks.iter().next_back().expect("Exists").2;
680        let block_headers = unproven_blocks
681            .iter()
682            .map(|(_, header, _)| *header)
683            .collect::<Vec<_>>();
684
685        let receipt = self
686            .prove_and_save_block(current_block_hash, block_headers, prev_proof)
687            .await?;
688        tracing::info!(
689            "Header chain proof generated for block with hash {:?} and height {}",
690            current_block_hash,
691            current_block_height,
692        );
693
694        Ok(Some(receipt))
695    }

722    /// Mines `block_num` amount of blocks (if not already mined) and returns
723    /// the first `block_num` block headers in blockchain.
724    async fn mine_and_get_first_n_block_headers(
725        rpc: ExtendedBitcoinRpc,
726        db: Database,
727        block_num: u64,
728    ) -> Vec<Header> {
729        let height = rpc.get_block_count().await.unwrap();
730        tracing::debug!(
731            "Current tip height: {}, target block height: {}",
732            height,
733            block_num
734        );
735        if height < block_num {
736            tracing::debug!(
737                "Mining {} blocks to reach block number {}",
738                block_num - height,
739                block_num
740            );
741            rpc.mine_blocks(block_num - height + 1).await.unwrap();
742        }
743
744        tracing::debug!("Getting first {} block headers from blockchain", block_num);
745        let mut headers = Vec::new();
746        for i in 0..block_num + 1 {
747            let hash = rpc.get_block_hash(i).await.unwrap();
748            let header = rpc.get_block_header(&hash).await.unwrap();
749
750            headers.push(header);
751
752            let _ignore_errors = db
753                .save_unproven_finalized_block(None, hash, header, i)
754                .await;
755        }
756
757        headers
758    }
