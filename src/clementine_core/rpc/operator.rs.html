62    #[tracing::instrument(skip_all, err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
63    async fn get_params(
64        &self,
65        _request: Request<Empty>,
66    ) -> Result<Response<Self::GetParamsStream>, Status> {
67        tracing::info!("Get params rpc called");
68        let operator = self.operator.clone();
69        let (tx, rx) = mpsc::channel(DEFAULT_CHANNEL_SIZE);
70        let out_stream: Self::GetParamsStream = ReceiverStream::new(rx);
71
72        let (mut wpk_receiver, mut signature_receiver) = operator.get_params().await?;
73
74        let handle = tokio::spawn(async move {
75            let operator_config: OperatorParams = operator.clone().into();
76            tx.send(Ok(operator_config))
77                .await
78                .map_err(output_stream_ended_prematurely)?;
79
80            while let Some(winternitz_public_key) = wpk_receiver.recv().await {
81                let operator_winternitz_pubkey: OperatorParams = winternitz_public_key.into();
82                tx.send(Ok(operator_winternitz_pubkey))
83                    .await
84                    .map_err(output_stream_ended_prematurely)?;
85            }
86
87            while let Some(operator_sig) = signature_receiver.recv().await {
88                let unspent_kickoff_sig: OperatorParams = operator_sig.into();
89                tx.send(Ok(unspent_kickoff_sig))
90                    .await
91                    .map_err(output_stream_ended_prematurely)?;
92            }
93
94            Ok::<(), Status>(())
95        });
96        monitor_standalone_task(handle, "Operator get_params");
97
98        Ok(Response::new(out_stream))
99    }

101    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
102    async fn deposit_sign(
103        &self,
104        request: Request<DepositSignSession>,
105    ) -> Result<Response<Self::DepositSignStream>, Status> {
106        tracing::info!("Deposit sign rpc called");
107        let (tx, rx) = mpsc::channel(DEFAULT_CHANNEL_SIZE);
108
109        let deposit_sign_session = request.into_inner();
110        let deposit_params: DepositParams = deposit_sign_session.try_into()?;
111        let deposit_data: DepositData = deposit_params.try_into()?;
112        tracing::info!(
113            "Parsed deposit sign rpc params, deposit data: {:?}",
114            deposit_data
115        );
116
117        let expected_sigs = self
118            .operator
119            .config
120            .get_num_required_operator_sigs(&deposit_data);
121
122        let mut deposit_signatures_rx = self.operator.deposit_sign(deposit_data).await?;
123
124        tokio::spawn(async move {
125            let mut sent_sigs = 0;
126            while let Some(sig) = deposit_signatures_rx.recv().await {
127                let operator_burn_sig = SchnorrSig {
128                    schnorr_sig: sig.serialize().to_vec(),
129                };
130
131                if tx
132                    .send(Ok(operator_burn_sig))
133                    .inspect_ok(|_| {
134                        sent_sigs += 1;
135                        tracing::debug!(
136                            "Sent signature {}/{} in deposit_sign()",
137                            sent_sigs,
138                            expected_sigs
139                        );
140                    })
141                    .await
142                    .is_err()
143                {
144                    break;
145                }
146            }
147        });
148
149        Ok(Response::new(ReceiverStream::new(rx)))
150    }

176    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
177    async fn withdraw(
178        &self,
179        request: Request<WithdrawParamsWithSig>,
180    ) -> Result<Response<RawSignedTx>, Status> {
181        tracing::info!("Withdraw rpc called");
182        let params = request.into_inner();
183        let withdraw_params = params.withdrawal.ok_or(Status::invalid_argument(
184            "Withdrawal params not found for withdrawal",
185        ))?;
186        let (withdrawal_id, input_signature, input_outpoint, output_script_pubkey, output_amount) =
187            parser::operator::parse_withdrawal_sig_params(withdraw_params)?;
188
189        tracing::warn!(
190            "Parsed withdraw rpc params, withdrawal id: {:?}, input signature: {:?}, input outpoint: {:?}, output script pubkey: {:?}, output amount: {:?}, verification signature: {:?}", withdrawal_id, input_signature, input_outpoint, output_script_pubkey, output_amount, params.verification_signature
191        );
192
193        // if verification address is set in config, check if verification signature is valid
194        if let Some(address_in_config) = self.operator.config.aggregator_verification_address {
195            let verification_signature = params
196                .verification_signature
197                .map(|sig| {
198                    PrimitiveSignature::from_str(&sig).map_err(|e| {
199                        Status::invalid_argument(format!("Invalid verification signature: {e}"))
200                    })
201                })
202                .transpose()?;
203            // check if verification signature is provided by aggregator
204            if let Some(verification_signature) = verification_signature {
205                let address_from_sig =
206                    recover_address_from_ecdsa_signature::<OperatorWithdrawalMessage>(
207                        withdrawal_id,
208                        input_signature,
209                        input_outpoint,
210                        output_script_pubkey.clone(),
211                        output_amount,
212                        verification_signature,
213                    )?;
214
215                // check if verification signature is signed by the address in config
216                if address_from_sig != address_in_config {
217                    return Err(BridgeError::InvalidECDSAVerificationSignature).map_to_status();
218                }
219            } else {
220                // if verification signature is not provided, but verification address is set in config, return error
221                return Err(BridgeError::ECDSAVerificationSignatureMissing).map_to_status();
222            }
223        }
224
225        let payout_tx = self
226            .operator
227            .withdraw(
228                withdrawal_id,
229                input_signature,
230                input_outpoint,
231                output_script_pubkey,
232                output_amount,
233            )
234            .await?;
235
236        tracing::info!(
237            "Withdraw rpc completed successfully for withdrawal id: {:?}",
238            withdrawal_id
239        );
240
241        Ok(Response::new(RawSignedTx::from(&payout_tx)))
242    }

356    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
357    async fn internal_finalized_payout(
358        &self,
359        request: Request<FinalizedPayoutParams>,
360    ) -> Result<Response<clementine::Txid>, Status> {
361        if !cfg!(test) {
362            return Err(Status::permission_denied(
363                "This method is only available in tests",
364            ));
365        }
366
367        tracing::info!(
368            "Internal finalized payout rpc called with finalized payout params: {:?}",
369            request.get_ref()
370        );
371
372        let payout_blockhash: [u8; 32] = request
373            .get_ref()
374            .payout_blockhash
375            .clone()
376            .try_into()
377            .map_err(|e| {
378                Status::invalid_argument(format!(
379                    "Failed to convert payout blockhash to [u8; 32]: {e:?}"
380                ))
381            })?;
382        let deposit_outpoint: OutPoint = request
383            .get_ref()
384            .deposit_outpoint
385            .clone()
386            .ok_or(Status::invalid_argument("Failed to get deposit outpoint"))?
387            .try_into()?;
388
389        let mut dbtx = self.operator.db.begin_transaction().await?;
390        let kickoff_txid = self
391            .operator
392            .handle_finalized_payout(
393                &mut dbtx,
394                deposit_outpoint,
395                BlockHash::from_byte_array(payout_blockhash),
396            )
397            .await?;
398        dbtx.commit()
399            .await
400            .wrap_err("Failed to commit transaction")
401            .map_to_status()?;
402
403        Ok(Response::new(kickoff_txid.into()))
404    }

407    async fn internal_end_round(
408        &self,
409        _request: Request<Empty>,
410    ) -> Result<Response<Empty>, Status> {
411        tracing::warn!("Internal end round rpc called");
412        #[cfg(feature = "automation")]
413        {
414            use eyre::Context;
415
416            let mut dbtx = self.operator.db.begin_transaction().await?;
417
418            self.operator.end_round(&mut dbtx).await?;
419
420            dbtx.commit()
421                .await
422                .wrap_err("Failed to commit transaction")
423                .map_to_status()?;
424            Ok(Response::new(Empty {}))
425        }
426
427        #[cfg(not(feature = "automation"))]
428        Err(Status::unimplemented(
429            "Automation is not enabled. Operator does not manage its rounds",
430        ))
431    }
