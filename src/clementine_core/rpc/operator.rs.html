38    #[tracing::instrument(skip_all, err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
39    async fn get_params(
40        &self,
41        _request: Request<Empty>,
42    ) -> Result<Response<Self::GetParamsStream>, Status> {
43        let operator = self.operator.clone();
44        let (tx, rx) = mpsc::channel(1280);
45        let out_stream: Self::GetParamsStream = ReceiverStream::new(rx);
46
47        let (mut wpk_receiver, mut signature_receiver) = operator.get_params().await?;
48
49        tokio::spawn(async move {
50            let operator_config: OperatorParams = operator.clone().into();
51            tx.send(Ok(operator_config))
52                .await
53                .map_err(output_stream_ended_prematurely)?;
54
55            while let Some(winternitz_public_key) = wpk_receiver.recv().await {
56                let operator_winternitz_pubkey: OperatorParams = winternitz_public_key.into();
57                tx.send(Ok(operator_winternitz_pubkey))
58                    .await
59                    .map_err(output_stream_ended_prematurely)?;
60            }
61
62            while let Some(operator_sig) = signature_receiver.recv().await {
63                let unspent_kickoff_sig: OperatorParams = operator_sig.into();
64                tx.send(Ok(unspent_kickoff_sig))
65                    .await
66                    .map_err(output_stream_ended_prematurely)?;
67            }
68
69            Ok::<(), Status>(())
70        });
71
72        Ok(Response::new(out_stream))
73    }

75    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
76    async fn deposit_sign(
77        &self,
78        request: Request<DepositSignSession>,
79    ) -> Result<Response<Self::DepositSignStream>, Status> {
80        let (tx, rx) = mpsc::channel(1280);
81
82        let deposit_sign_session = request.into_inner();
83        let deposit_params: DepositParams = deposit_sign_session.try_into()?;
84        let deposit_data: DepositData = deposit_params.try_into()?;
85
86        let expected_sigs = self
87            .operator
88            .config
89            .get_num_required_operator_sigs(&deposit_data);
90
91        let mut deposit_signatures_rx = self.operator.deposit_sign(deposit_data).await?;
92
93        tokio::spawn(async move {
94            let mut sent_sigs = 0;
95            while let Some(sig) = deposit_signatures_rx.recv().await {
96                let operator_burn_sig = SchnorrSig {
97                    schnorr_sig: sig.serialize().to_vec(),
98                };
99
100                if tx
101                    .send(Ok(operator_burn_sig))
102                    .inspect_ok(|_| {
103                        sent_sigs += 1;
104                        tracing::debug!(
105                            "Sent signature {}/{} in deposit_sign()",
106                            sent_sigs,
107                            expected_sigs
108                        );
109                    })
110                    .await
111                    .is_err()
112                {
113                    break;
114                }
115            }
116        });
117
118        Ok(Response::new(ReceiverStream::new(rx)))
119    }

200    async fn get_deposit_keys(
201        &self,
202        request: Request<DepositParams>,
203    ) -> Result<Response<OperatorKeys>, Status> {
204        let start = std::time::Instant::now();
205        let deposit_params = request.into_inner();
206        let deposit_data: DepositData = deposit_params.try_into()?;
207
208        let winternitz_keys = self
209            .operator
210            .generate_assert_winternitz_pubkeys(deposit_data.get_deposit_outpoint())?;
211        let hashes = self
212            .operator
213            .generate_challenge_ack_preimages_and_hashes(&deposit_data)?;
214        tracing::info!("Generated deposit keys in {:?}", start.elapsed());
215
216        Ok(Response::new(OperatorKeys {
217            winternitz_pubkeys: winternitz_keys
218                .into_iter()
219                .map(|pubkey| pubkey.into())
220                .collect(),
221            challenge_ack_digests: hashes
222                .into_iter()
223                .map(|hash| ChallengeAckDigest { hash: hash.into() })
224                .collect(),

254    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
255    async fn internal_finalized_payout(
256        &self,
257        request: Request<FinalizedPayoutParams>,
258    ) -> Result<Response<clementine::Txid>, Status> {
259        if !cfg!(test) {
260            return Err(Status::permission_denied(
261                "This method is only available in tests",
262            ));
263        }
264
265        let payout_blockhash: [u8; 32] = request
266            .get_ref()
267            .payout_blockhash
268            .clone()
269            .try_into()
270            .expect("Failed to convert payout blockhash to [u8; 32]");
271        let deposit_outpoint = request
272            .get_ref()
273            .deposit_outpoint
274            .clone()
275            .expect("Failed to get deposit outpoint");
276        let deposit_outpoint: OutPoint = deposit_outpoint
277            .try_into()
278            .expect("Failed to convert deposit outpoint to OutPoint");
279
280        let mut dbtx = self.operator.db.begin_transaction().await?;
281        let kickoff_txid = self
282            .operator
283            .handle_finalized_payout(
284                &mut dbtx,
285                deposit_outpoint,
286                BlockHash::from_byte_array(payout_blockhash),
287            )
288            .await?;
289        dbtx.commit().await.expect("Failed to commit transaction");
290
291        Ok(Response::new(kickoff_txid.into()))
292    }
