43/// Returns gRPC clients.
44///
45/// # Parameters
46///
47/// - `endpoints`: URIs for clients (can be http/https URLs or unix:// paths)
48/// - `connect`: Function that will be used to initiate gRPC connection
49/// - `config`: Configuration containing TLS certificate paths
50///
51/// # Returns
52///
53/// - [`CLIENT`]: [`tonic`] gRPC client.
54pub async fn get_clients<CLIENT, F>(
55    endpoints: Vec<String>,
56    connect: F,
57    config: &crate::config::BridgeConfig,
58    use_client_cert: bool,
59) -> Result<Vec<CLIENT>, BridgeError>
60where
61    F: FnOnce(Channel) -> CLIENT + Copy,
62{
63    // Ensure certificates exist in test mode
64    #[cfg(test)]
65    {
66        ensure_test_certificates().map_err(|e| {
67            BridgeError::ConfigError(format!("Failed to ensure test certificates: {}", e))
68        })?;
69    }
70
71    // Get certificate paths from config or use defaults
72    let client_ca_cert = tokio::fs::read(&config.ca_cert_path)
73        .await
74        .wrap_err(format!(
75            "Failed to read CA certificate from {}",
76            config.ca_cert_path.display()
77        ))?;
78
79    let client_ca = Certificate::from_pem(client_ca_cert);
80
81    // Get certificate paths from config or use defaults
82    let client_cert_path = &config.client_cert_path.clone();
83    let client_key_path = &config.client_key_path.clone();
84
85    // Load client certificate and key
86    let client_cert = tokio::fs::read(&client_cert_path).await.map_err(|e| {
87        BridgeError::ConfigError(format!(
88            "Failed to read client certificate from {}: {}",
89            client_cert_path.display(),
90            e
91        ))
92    })?;
93
94    let client_key = tokio::fs::read(&client_key_path).await.map_err(|e| {
95        BridgeError::ConfigError(format!(
96            "Failed to read client key from {}: {}",
97            client_key_path.display(),
98            e
99        ))
100    })?;
101
102    futures::future::try_join_all(
103        endpoints
104            .into_iter()
105            .map(|endpoint| {
106                let client_cert = client_cert.clone();
107                let client_key = client_key.clone();
108                let client_ca = client_ca.clone();
109
110                let tls_config = if use_client_cert {
111                    let client_identity = Identity::from_pem(client_cert, client_key);
112                    ClientTlsConfig::new()
113                        .identity(client_identity)
114                        .ca_certificate(client_ca)
115                } else {
116                    ClientTlsConfig::new().ca_certificate(client_ca)
117                };
118
119                async move {
120                    let channel = if endpoint.starts_with("unix://") {
121                        #[cfg(unix)]
122                        {
123                            // Handle Unix socket (only available on Unix platforms)
124                            let path = endpoint.trim_start_matches("unix://").to_string();
125                            Channel::from_static("lttp://[::]:50051")
126                                .connect_with_connector(tower::service_fn(move |_| {
127                                    let path = PathBuf::from(path.clone());
128                                    async move {
129                                        let unix_stream =
130                                            tokio::net::UnixStream::connect(path).await?;
131                                        Ok::<_, std::io::Error>(TokioIo::new(unix_stream))
132                                    }
133                                }))
134                                .await
135                                .wrap_err_with(|| {
136                                    format!("Failed to connect to Unix socket {}", endpoint)
137                                })?
138                        }
139
140                        #[cfg(not(unix))]
141                        {
142                            // Windows doesn't support Unix sockets
143                            return Err(BridgeError::ConfigError(format!(
144                                "Unix sockets ({}), are not supported on this platform",
145                                endpoint
146                            )));
147                        }
148                    } else {
149                        // Handle TCP/HTTP connection
150                        let uri = Uri::try_from(endpoint.clone()).map_err(|e| {
151                            BridgeError::ConfigError(format!(
152                                "Endpoint {} is malformed: {}",
153                                endpoint, e
154                            ))
155                        })?;
156
157                        Channel::builder(uri)
158                            .tls_config(tls_config)
159                            .wrap_err("Failed to configure TLS")?
160                            .connect()
161                            .await
162                            .wrap_err("Failed to connect to endpoint")?
163                    };
164
165                    Ok(connect(channel))
166                }
167            })
168            .collect::<Vec<_>>(),
169    )
170    .await
171}
