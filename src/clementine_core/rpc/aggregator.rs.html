94/// For each expected sighash, we collect a batch of public nonces from all verifiers. We aggregate and send to the agg_nonce_sender. Then repeat for the next sighash.
95async fn nonce_aggregator(
96    mut nonce_streams: Vec<
97        impl Stream<Item = Result<MusigPubNonce, BridgeError>> + Unpin + Send + 'static,
98    >,
99    mut sighash_stream: impl Stream<Item = Result<(TapSighash, SignatureInfo), BridgeError>>
100        + Unpin
101        + Send
102        + 'static,
103    agg_nonce_sender: Sender<AggNonceQueueItem>,
104) -> Result<(MusigAggNonce, MusigAggNonce), BridgeError> {
105    let mut total_sigs = 0;
106
107    tracing::info!("Starting nonce aggregation");
108
109    // We assume the sighash stream returns the correct number of items.
110    while let Some(msg) = sighash_stream.next().await {
111        let (sighash, siginfo) = msg.wrap_err("Sighash stream failed")?;
112
113        total_sigs += 1;
114
115        let pub_nonces = get_next_pub_nonces(&mut nonce_streams)
116            .await
117            .wrap_err_with(|| {
118                format!("Failed to aggregate nonces for sighash with info: {siginfo:?}")
119            })?;
120
121        tracing::debug!(
122            "Received nonces for signature id {:?} in nonce_aggregator",
123            siginfo.signature_id
124        );
125
126        // TODO: consider spawn_blocking here
127        let agg_nonce = aggregate_nonces(pub_nonces.iter().collect::<Vec<_>>().as_slice());
128
129        agg_nonce_sender
130            .send(AggNonceQueueItem { agg_nonce, sighash })
131            .await
132            .wrap_err_with(|| AggregatorError::OutputStreamEndedEarly {
133                stream_name: "nonce_aggregator".to_string(),
134            })?;
135
136        tracing::debug!(
137            "Sent nonces for signature id {:?} in nonce_aggregator",
138            siginfo.signature_id
139        );
140    }
141
142    if total_sigs == 0 {
143        tracing::warn!("Sighash stream returned 0 signatures");
144    }
145    // aggregate nonces for the movetx signature
146    let pub_nonces = try_join_all(nonce_streams.iter_mut().map(|s| async {
147        s.next()
148            .await
149            .transpose()? // Return the inner error if it exists
150            .ok_or_else(|| -> eyre::Report {
151                AggregatorError::InputStreamEndedEarlyUnknownSize {
152                    // Return an early end error if the stream is empty
153                    stream_name: "Nonce stream".to_string(),
154                }
155                .into()
156            })
157    }))
158    .await
159    .wrap_err("Failed to aggregate nonces for the move tx")?;
160
161    tracing::debug!("Received nonces for movetx in nonce_aggregator");
162
163    // TODO: consider spawn_blocking here
164    let move_tx_agg_nonce = aggregate_nonces(pub_nonces.iter().collect::<Vec<_>>().as_slice());
165
166    let pub_nonces = try_join_all(nonce_streams.iter_mut().map(|s| async {
167        s.next()
168            .await
169            .transpose()? // Return the inner error if it exists
170            .ok_or_else(|| -> eyre::Report {
171                AggregatorError::InputStreamEndedEarlyUnknownSize {
172                    // Return an early end error if the stream is empty
173                    stream_name: "Nonce stream".to_string(),
174                }
175                .into()
176            })
177    }))
178    .await
179    .wrap_err("Failed to aggregate nonces for the emergency stop tx")?;
180
181    let emergency_stop_agg_nonce =
182        aggregate_nonces(pub_nonces.iter().collect::<Vec<_>>().as_slice());
183
184    Ok((move_tx_agg_nonce, emergency_stop_agg_nonce))
185}

187/// Reroutes aggregated nonces to the signature aggregator.
188async fn nonce_distributor(
189    mut agg_nonce_receiver: Receiver<AggNonceQueueItem>,
190    mut partial_sig_streams: Vec<(
191        Streaming<clementine::PartialSig>,
192        Sender<clementine::VerifierDepositSignParams>,
193    )>,
194    partial_sig_sender: Sender<(Vec<MusigPartialSignature>, AggNonceQueueItem)>,
195) -> Result<(), BridgeError> {
196    let mut sig_count = 0;
197    while let Some(queue_item) = agg_nonce_receiver.recv().await {
198        sig_count += 1;
199
200        tracing::debug!(
201            "Received aggregated nonce {} in nonce_distributor",
202            sig_count
203        );
204
205        let agg_nonce_wrapped = clementine::VerifierDepositSignParams {
206            params: Some(clementine::verifier_deposit_sign_params::Params::AggNonce(
207                queue_item.agg_nonce.serialize().to_vec(),
208            )),
209        };
210
211        // Broadcast aggregated nonce to all streams
212        try_join_all(
213            partial_sig_streams
214                .iter_mut()
215                .enumerate()
216                .map(|(idx, (_, tx))| {
217                    let agg_nonce_wrapped = agg_nonce_wrapped.clone();
218                    async move {
219                        tx.send(agg_nonce_wrapped).await.wrap_err_with(|| {
220                            AggregatorError::OutputStreamEndedEarly {
221                                stream_name: format!("Partial sig stream {idx}"),
222                            }
223                        })
224                    }
225                }),
226        )
227        .await
228        .wrap_err("Failed to send aggregated nonces to verifiers")?;
229
230        tracing::debug!(
231            "Sent aggregated nonce {} to verifiers in nonce_distributor",
232            sig_count
233        );
234
235        let partial_sigs = try_join_all(partial_sig_streams.iter_mut().enumerate().map(
236            |(idx, (stream, _))| async move {
237                let partial_sig = stream
238                    .message()
239                    .await
240                    .wrap_err_with(|| AggregatorError::RequestFailed {
241                        request_name: format!("Partial sig stream {idx}"),
242                    })?
243                    .ok_or_eyre(AggregatorError::InputStreamEndedEarlyUnknownSize {
244                        stream_name: format!("Partial sig stream {idx}"),
245                    })?;
246
247                Ok::<_, BridgeError>(
248                    MusigPartialSignature::from_slice(&partial_sig.partial_sig)
249                        .wrap_err("Failed to parse partial signature")?,
250                )
251            },
252        ))
253        .await?;
254
255        tracing::debug!(
256            "Received partial signature {} from verifiers in nonce_distributor",
257            sig_count
258        );
259
260        partial_sig_sender
261            .send((partial_sigs, queue_item))
262            .await
263            .map_err(|_| {
264                eyre::eyre!(AggregatorError::OutputStreamEndedEarly {
265                    stream_name: "partial_sig_sender".into(),
266                })
267            })?;
268
269        tracing::debug!(
270            "Sent partial signature {} to signature_aggregator in nonce_distributor",
271            sig_count
272        );
273    }
274
275    Ok(())
276}

281    verifiers_public_keys: Vec<PublicKey>,
282    final_sig_sender: Sender<FinalSigQueueItem>,

285    while let Some((partial_sigs, queue_item)) = partial_sig_receiver.recv().await {
286        sig_count += 1;
287        tracing::debug!(
288            "Received partial signatures {} in signature_aggregator",
289            sig_count
290        );
291
292        let final_sig = crate::musig2::aggregate_partial_signatures(
293            &verifiers_public_keys,

295            queue_item.agg_nonce,
296            &partial_sigs,
297            Message::from_digest(queue_item.sighash.to_byte_array()),
298        )?;

300        final_sig_sender

306                eyre::eyre!(AggregatorError::OutputStreamEndedEarly {
307                    stream_name: "final_sig_sender".into(),
308                })
309            })?;
310        tracing::debug!(
311            "Sent aggregated signature {} to signature_distributor in signature_aggregator",
312            sig_count
313        );
314    }
315
316    Ok(())
317}

378            params: Some(Params::EmergencyStopAggNonce(
379                emergency_stop_agg_nonce.serialize().to_vec(),

383        .wrap_err_with(|| AggregatorError::OutputStreamEndedEarly {
384            stream_name: "Deposit finalize sender (for emergency stop agg nonce)".to_string(),
385        })?;
386    }
387    tracing::info!("Sent emergency stop aggregated nonce to verifiers in signature distributor");
388
389    Ok(())

392/// Creates a stream of nonces from verifiers.
393/// This will automatically get's the first response from the verifiers.
394///
395/// # Returns
396///
397/// - Vec<[`clementine::NonceGenFirstResponse`]>: First response from each verifier
398/// - Vec<BoxStream<Result<[`MusigPubNonce`], BridgeError>>>: Stream of nonces from each verifier
399async fn create_nonce_streams(
400    verifier_clients: Vec<ClementineVerifierClient<tonic::transport::Channel>>,
401    num_nonces: u32,
402) -> Result<
403    (
404        Vec<clementine::NonceGenFirstResponse>,
405        Vec<BoxStream<'static, Result<MusigPubNonce, BridgeError>>>,
406    ),
407    BridgeError,
408> {
409    let mut nonce_streams = try_join_all(verifier_clients.into_iter().enumerate().map(
410        |(idx, client)| {
411            let mut client = client.clone();
412
413            async move {
414                let response_stream = client
415                    .nonce_gen(tonic::Request::new(clementine::NonceGenRequest {
416                        num_nonces,
417                    }))
418                    .await
419                    .wrap_err_with(|| AggregatorError::RequestFailed {
420                        request_name: format!("Nonce gen stream for verifier {idx}"),
421                    })?;
422
423                Ok::<_, BridgeError>(response_stream.into_inner())
424            }
425        },
426    ))
427    .await?;
428
429    // Get the first responses from verifiers.
430    let first_responses: Vec<clementine::NonceGenFirstResponse> = try_join_all(
431        nonce_streams
432            .iter_mut()
433            .enumerate()
434            .map(|(idx, stream)| async move {
435                parser::verifier::parse_nonce_gen_first_response(stream)
436                    .await
437                    .wrap_err_with(|| format!("Failed to get initial response from verifier {idx}"))
438            }),
439    )
440    .await
441    .wrap_err("Failed to get nonce gen's initial responses from verifiers")?;
442
443    let transformed_streams = nonce_streams
444        .into_iter()
445        .enumerate()
446        .map(|(idx, stream)| {
447            stream
448                .map(move |result| {
449                    Aggregator::extract_pub_nonce(
450                        result
451                            .wrap_err_with(|| AggregatorError::InputStreamEndedEarlyUnknownSize {
452                                stream_name: format!("Nonce gen stream for verifier {idx}"),
453                            })?
454                            .response,
455                    )
456                })
457                .boxed()
458        })
459        .collect::<Vec<_>>();
460
461    Ok((first_responses, transformed_streams))
462}

464/// Use items collected from the broadcast receiver for an async function call.
465///
466/// Handles the boilerplate of managing a receiver of a broadcast channel.
467/// If receiver is lagged at any time (data is lost) an error is returned.
468async fn collect_and_call<R, T, F, Fut>(
469    rx: &mut tokio::sync::broadcast::Receiver<Vec<T>>,
470    mut f: F,
471) -> Result<R, Status>
472where
473    R: Default,
474    T: Clone,
475    F: FnMut(Vec<T>) -> Fut,
476    Fut: Future<Output = Result<R, Status>>,
477{
478    loop {
479        match rx.recv().await {
480            Ok(params) => {
481                f(params).await?;
482            }
483            Err(tokio::sync::broadcast::error::RecvError::Lagged(n)) => {
484                break Err(Status::internal(format!(
485                    "lost {n} items due to lagging receiver"
486                )));
487            }
488            Err(tokio::sync::broadcast::error::RecvError::Closed) => break Ok(R::default()),
489        }
490    }
491}

506    /// For a specific deposit, collects needed signatures from all operators into a [`Vec<Vec<Signature>>`].
507    async fn collect_operator_sigs(
508        operator_clients: Vec<ClementineOperatorClient<tonic::transport::Channel>>,
509        config: BridgeConfig,
510        mut deposit_sign_session: DepositSignSession,
511    ) -> Result<Vec<Vec<Signature>>, BridgeError> {
512        deposit_sign_session.nonce_gen_first_responses = Vec::new(); // not needed for operators
513        let mut operator_sigs_streams =
514            // create deposit sign streams with each operator
515            try_join_all(operator_clients.into_iter().enumerate().map(|(idx, mut operator_client)| {
516                let sign_session = deposit_sign_session.clone();
517                async move {
518                    let stream = operator_client
519                        .deposit_sign(tonic::Request::new(sign_session))
520                        .await.wrap_err_with(|| AggregatorError::RequestFailed {
521                            request_name: format!("Deposit sign stream for operator {idx}"),
522                        })?;
523                    Ok::<_, BridgeError>(stream.into_inner())
524                }
525            }))
526                .await?;
527
528        let deposit_data: DepositData = deposit_sign_session
529            .deposit_params
530            .clone()
531            .ok_or_else(|| eyre::eyre!("No deposit params found in deposit sign session"))?
532            .try_into()
533            .wrap_err("Failed to convert deposit params to deposit data")?;
534
535        // calculate number of signatures needed from each operator
536        let needed_sigs = config.get_num_required_operator_sigs(&deposit_data);
537
538        // get signatures from each operator's signature streams
539        let operator_sigs = try_join_all(operator_sigs_streams.iter_mut().enumerate().map(
540            |(idx, stream)| async move {
541                let mut sigs: Vec<Signature> = Vec::with_capacity(needed_sigs);
542                while let Some(sig) =
543                    stream
544                        .message()
545                        .await
546                        .wrap_err_with(|| AggregatorError::RequestFailed {
547                            request_name: format!("Deposit sign stream for operator {idx}"),
548                        })?
549                {
550                    sigs.push(Signature::from_slice(&sig.schnorr_sig).wrap_err_with(|| {
551                        format!("Failed to parse Schnorr signature from operator {idx}")
552                    })?);
553                }
554                Ok::<_, BridgeError>(sigs)
555            },
556        ))
557        .await?;
558
559        // check if all signatures are received
560        for (idx, sigs) in operator_sigs.iter().enumerate() {
561            if sigs.len() != needed_sigs {
562                return Err(eyre::eyre!(
563                    "Not all operator sigs received from operator {}.\n Expected: {}, got: {}",
564                    idx,
565                    needed_sigs,
566                    sigs.len()
567                )
568                .into());
569            }
570        }
571        Ok(operator_sigs)
572    }

574    async fn send_movetx(
575        &self,
576        partial_sigs: Vec<Vec<u8>>,
577        movetx_agg_nonce: MusigAggNonce,
578        deposit_params: DepositParams,
579    ) -> Result<TxHandler<Signed>, Status> {
580        let mut deposit_data: crate::builder::transaction::DepositData =
581            deposit_params.try_into()?;
582        let musig_partial_sigs = parser::verifier::parse_partial_sigs(partial_sigs)?;
583
584        // create move tx and calculate sighash
585        let mut move_txhandler =
586            create_move_to_vault_txhandler(&mut deposit_data, self.config.protocol_paramset())?;
587
588        let sighash = move_txhandler.calculate_script_spend_sighash_indexed(
589            0,
590            0,
591            bitcoin::TapSighashType::Default,
592        )?;
593
594        // aggregate partial signatures
595        let verifiers_public_keys = deposit_data.get_verifiers();
596        let final_sig = crate::musig2::aggregate_partial_signatures(
597            &verifiers_public_keys,
598            None,
599            movetx_agg_nonce,
600            &musig_partial_sigs,
601            Message::from_digest(sighash.to_byte_array()),
602        )
603        .map_err(|x| BridgeError::Error(format!("Aggregating MoveTx signatures failed {}", x)))?;
604
605        // Put the signature in the tx
606        move_txhandler.set_p2tr_script_spend_witness(&[final_sig.as_ref()], 0, 0)?;
607        // Add fee bumper.
608        let move_tx = move_txhandler.get_cached_tx();
609
610        let mut dbtx = self.db.begin_transaction().await?;
611        self.tx_sender
612            .insert_try_to_send(
613                &mut dbtx,
614                Some(TxMetadata {
615                    deposit_outpoint: Some(deposit_data.get_deposit_outpoint()),
616                    operator_xonly_pk: None,
617                    round_idx: None,
618                    kickoff_idx: None,
619                    tx_type: TransactionType::MoveToVault,
620                }),
621                move_tx,
622                FeePayingType::CPFP,
623                None,
624                &[],
625                &[],
626                &[],
627                &[],
628            )
629            .await
630            .map_err(BridgeError::from)?;
631        dbtx.commit()
632            .await
633            .map_err(|e| Status::internal(format!("Failed to commit db transaction: {}", e)))?;
634
635        tracing::info!(
636            "move_tx: {}",
637            hex::encode(bitcoin::consensus::serialize(&move_tx))
638        );
639
640        // TODO: Sign the transaction correctly after we create taproot witness generation functions
641        Ok(move_txhandler.promote()?)
642    }

651            .try_into()
652            .wrap_err("Failed to convert deposit params to deposit data")?;
653        let musig_partial_sigs = parser::verifier::parse_partial_sigs(emergency_stop_sigs)
654            .wrap_err("Failed to parse emergency stop signatures")?;
655
656        // create move tx and calculate sighash
657        let move_txhandler =
658            create_move_to_vault_txhandler(&mut deposit_data, self.config.protocol_paramset())?;
659
660        let mut emergency_stop_txhandler = create_emergency_stop_txhandler(
661            &mut deposit_data,
662            &move_txhandler,
663            self.config.protocol_paramset(),
664        )?;
665
666        let sighash = emergency_stop_txhandler.calculate_script_spend_sighash_indexed(
667            0,
668            0,
669            bitcoin::TapSighashType::SinglePlusAnyoneCanPay,
670        )?;
671
672        let verifiers_public_keys = deposit_data.get_verifiers();
673
674        let final_sig = crate::musig2::aggregate_partial_signatures(
675            &verifiers_public_keys,
676            None,
677            emergency_stop_agg_nonce,
678            &musig_partial_sigs,
679            Message::from_digest(sighash.to_byte_array()),
680        )
681        .wrap_err("Failed to aggregate emergency stop signatures")?;
682
683        let final_sig = bitcoin::taproot::Signature {
684            signature: final_sig,
685            sighash_type: bitcoin::TapSighashType::SinglePlusAnyoneCanPay,
686        };
687
688        // insert the signature into the tx
689        emergency_stop_txhandler.set_p2tr_script_spend_witness(&[final_sig.serialize()], 0, 0)?;
690
691        let emergency_stop_tx = emergency_stop_txhandler.get_cached_tx();
692        let move_to_vault_txid = move_txhandler.get_txid();
693
694        tracing::debug!("Move to vault tx id: {}", move_to_vault_txid.to_string());
695
696        self.db
697            .set_signed_emergency_stop_tx(None, move_to_vault_txid, emergency_stop_tx)
698            .await?;
699
700        Ok(())
701    }

833    async fn optimistic_payout(
834        &self,
835        request: tonic::Request<super::WithdrawParams>,
836    ) -> std::result::Result<tonic::Response<super::RawSignedTx>, tonic::Status> {
837        let withdraw_params = request.into_inner();
838        let (deposit_id, input_signature, input_outpoint, output_script_pubkey, output_amount) =
839            parser::operator::parse_withdrawal_sig_params(withdraw_params.clone()).await?;
840        // get which deposit the withdrawal belongs to
841        let withdrawal = self
842            .db
843            .get_move_to_vault_txid_from_citrea_deposit(None, deposit_id)
844            .await?;
845        if let Some(move_txid) = withdrawal {
846            // check if withdrawal utxo is correct
847            let withdrawal_utxo = self
848                .db
849                .get_withdrawal_utxo_from_citrea_withdrawal(None, deposit_id)
850                .await?
851                .ok_or(Status::invalid_argument(format!(
852                    "Withdrawal utxo not found for deposit id {}",
853                    deposit_id
854                )))?;
855            if withdrawal_utxo != input_outpoint {
856                return Err(Status::invalid_argument(format!(
857                    "Withdrawal utxo is not correct: {:?} != {:?}",
858                    withdrawal_utxo, input_outpoint
859                )));
860            }
861            let deposit_data = self
862                .db
863                .get_deposit_data_with_move_tx(None, move_txid)
864                .await?;
865            let mut deposit_data = deposit_data.ok_or(BridgeError::Error(format!(
866                "Deposit data not found for move txid {}",
867                move_txid
868            )))?;
869
870            // get which verifiers participated in the deposit to collect the optimistic payout tx signature
871            let verifiers = self.get_participating_verifiers(&deposit_data).await?;
872            let (first_responses, mut nonce_streams) =
873                create_nonce_streams(verifiers.clone(), 1).await?;
874            // collect nonces
875            let pub_nonces = get_next_pub_nonces(&mut nonce_streams)
876                .await
877                .wrap_err("Failed to aggregate nonces for optimistic payout")
878                .map_to_status()
879                .map_err(|e| *e)?;
880            let agg_nonce = aggregate_nonces(pub_nonces.iter().collect::<Vec<_>>().as_slice());
881            // send the agg nonce to the verifiers to sign the optimistic payout tx
882            let verifier_clients = self.get_verifier_clients();
883            let payout_sigs = verifier_clients
884                .iter()
885                .zip(first_responses)
886                .map(|(client, first_response)| {
887                    let mut client = client.clone();
888                    let withdrawal_params = withdraw_params.clone();
889                    let agg_nonce_bytes = agg_nonce.serialize().to_vec();
890                    async move {
891                        client
892                            .optimistic_payout_sign(OptimisticPayoutParams {
893                                withdrawal: Some(withdrawal_params),
894                                agg_nonce: agg_nonce_bytes,
895                                nonce_gen: Some(first_response),
896                            })
897                            .await
898                    }
899                })
900                .collect::<Vec<_>>();
901
902            // Prepare input and output of the payout transaction.
903            let withdrawal_prevout = self
904                .rpc
905                .get_txout_from_outpoint(&input_outpoint)
906                .await
907                .map_to_status()
908                .map_err(|e| *e)?;
909            let withdrawal_utxo = UTXO {
910                outpoint: input_outpoint,
911                txout: withdrawal_prevout,
912            };
913            let output_txout = TxOut {
914                value: output_amount,
915                script_pubkey: output_script_pubkey,
916            };
917
918            let mut opt_payout_txhandler = create_optimistic_payout_txhandler(
919                &mut deposit_data,
920                withdrawal_utxo,
921                output_txout,
922                input_signature,
923                self.config.protocol_paramset(),
924            )?;
925            // txin at index 1 is deposited utxo in movetx
926            let sighash = opt_payout_txhandler.calculate_script_spend_sighash_indexed(
927                1,
928                0,
929                bitcoin::TapSighashType::Default,
930            )?;
931
932            // calculate final sig
933            let payout_sig = try_join_all(payout_sigs).await?;
934            let musig_partial_sigs = payout_sig
935                .iter()
936                .map(|sig| MusigPartialSignature::from_slice(&sig.get_ref().partial_sig))
937                .collect::<Result<Vec<_>, _>>()
938                .map_err(|e| Status::internal(format!("Failed to parse partial sig: {:?}", e)))?;
939
940            let final_sig = bitcoin::taproot::Signature {
941                signature: crate::musig2::aggregate_partial_signatures(
942                    &deposit_data.get_verifiers(),
943                    None,
944                    agg_nonce,
945                    &musig_partial_sigs,
946                    Message::from_digest(sighash.to_byte_array()),
947                )?,
948                sighash_type: bitcoin::TapSighashType::Default,
949            };
950
951            // set witness and send tx
952            opt_payout_txhandler.set_p2tr_script_spend_witness(&[final_sig.serialize()], 1, 0)?;
953            let opt_payout_txhandler = opt_payout_txhandler.promote()?;
954            let opt_payout_tx = opt_payout_txhandler.get_cached_tx();
955            let mut dbtx = self.db.begin_transaction().await?;
956            self.tx_sender
957                .add_tx_to_queue(
958                    &mut dbtx,
959                    TransactionType::OptimisticPayout,
960                    opt_payout_tx,
961                    &[],
962                    None,
963                    &self.config,
964                )
965                .await
966                .map_err(BridgeError::from)?;
967            dbtx.commit().await.map_err(|e| {
968                Status::internal(format!(
969                    "Failed to commit db transaction to send optimistic payout tx: {}",
970                    e
971                ))
972            })?;
973
974            Ok(Response::new(RawSignedTx::from(opt_payout_tx)))
975        } else {
976            Err(Status::not_found(format!(
977                "Withdrawal with index {} not found.",
978                deposit_id
979            )))
980        }
981    }

1093    /// Handles a new deposit request from a user. This function coordinates the signing process
1094    /// between verifiers to create a valid move transaction. It ensures a covenant using pre-signed NofN transactions.
1095    /// It also collects signatures from operators to ensure that the operators can be slashed if they act maliciously.
1096    ///
1097    /// Overview:
1098    /// 1. Receive and parse deposit parameters from user
1099    /// 2. Signs all NofN transactions with verifiers using MuSig2:
1100    ///    - Creates nonce streams with verifiers (get pub nonces for each transaction)
1101    ///    - Opens deposit signing streams with verifiers (sends aggnonces for each transaction, receives partial sigs)
1102    ///    - Opens deposit finalization streams with verifiers (sends final signatures, receives movetx signatures)
1103    /// 3. Collects signatures from operators
1104    /// 4. Waits for all tasks to complete
1105    /// 5. Returns signed move transaction
1106    ///
1107    /// The following pipelines are used to coordinate the signing process, these move the data between the verifiers and the aggregator:
1108    ///    - Nonce aggregation
1109    ///    - Nonce distribution
1110    ///    - Signature aggregation
1111    ///    - Signature distribution
1112    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
1113    async fn new_deposit(
1114        &self,
1115        request: Request<Deposit>,
1116    ) -> Result<Response<clementine::Txid>, Status> {
1117        let deposit_info: DepositInfo = request.into_inner().try_into()?;
1118
1119        let deposit_data = DepositData {
1120            deposit: deposit_info,
1121            nofn_xonly_pk: None,
1122            actors: Actors {
1123                verifiers: self.get_verifier_keys(),
1124                watchtowers: vec![],
1125                operators: self.get_operator_keys(),
1126            },
1127            security_council: self.config.security_council.clone(),
1128        };
1129
1130        let deposit_params = deposit_data.clone().into();
1131
1132        // Collect and distribute keys needed keys from operators and watchtowers to verifiers
1133        let start = std::time::Instant::now();
1134        self.collect_and_distribute_keys(&deposit_params).await?;
1135        tracing::info!("Collected and distributed keys in {:?}", start.elapsed());
1136
1137        let participating_verifiers = self.get_participating_verifiers(&deposit_data).await?;
1138
1139        // Generate nonce streams for all verifiers.
1140        let num_required_sigs = self.config.get_num_required_nofn_sigs(&deposit_data);
1141        let (first_responses, nonce_streams) = create_nonce_streams(
1142            participating_verifiers.clone(),
1143            num_required_sigs as u32 + 2,
1144        )
1145        .await?; // ask for +2 for the final movetx signature + emergency stop signature, but don't send it on deposit_sign stage
1146
1147        let mut partial_sig_streams =
1148            try_join_all(participating_verifiers.iter().map(|verifier_client| {
1149                let mut verifier_client = verifier_client.clone();
1150
1151                async move {
1152                    let (tx, rx) = tokio::sync::mpsc::channel(1280);
1153                    let stream = verifier_client
1154                        .deposit_sign(tokio_stream::wrappers::ReceiverStream::new(rx))
1155                        .await?
1156                        .into_inner();
1157
1158                    Ok::<_, Status>((stream, tx))
1159                }
1160            }))
1161            .await?;
1162
1163        // Create initial deposit session and send to verifiers
1164        let deposit_sign_session = DepositSignSession {
1165            deposit_params: Some(deposit_params.clone()),
1166            nonce_gen_first_responses: first_responses,
1167        };
1168
1169        tracing::debug!("Sending deposit sign session to verifiers");
1170        for (_, tx) in partial_sig_streams.iter_mut() {
1171            let deposit_sign_param: VerifierDepositSignParams = deposit_sign_session.clone().into();
1172
1173            tx.send(deposit_sign_param).await.map_err(|e| {
1174                Status::internal(format!("Failed to send deposit sign session: {:?}", e))
1175            })?;
1176        }
1177
1178        // Set up deposit finalization streams
1179        let deposit_finalize_clients = participating_verifiers.clone();
1180        let deposit_finalize_streams = try_join_all(deposit_finalize_clients.into_iter().map(
1181            |mut verifier_client| async move {
1182                let (tx, rx) = tokio::sync::mpsc::channel(1280);
1183                let receiver_stream = tokio_stream::wrappers::ReceiverStream::new(rx);
1184                // start deposit_finalize with tokio spawn
1185                let deposit_finalize_future =
1186                    tokio::spawn(
1187                        async move { verifier_client.deposit_finalize(receiver_stream).await },
1188                    );
1189
1190                Ok::<_, Status>((deposit_finalize_future, tx))
1191            },
1192        ))
1193        .await?;
1194
1195        tracing::info!("Sending deposit finalize streams to verifiers");
1196
1197        let (mut deposit_finalize_futures, deposit_finalize_sender): (Vec<_>, Vec<_>) =
1198            deposit_finalize_streams.into_iter().unzip();
1199
1200        // Send initial finalization params
1201        let deposit_finalize_first_param: VerifierDepositFinalizeParams =
1202            deposit_sign_session.clone().into();
1203        for tx in deposit_finalize_sender.iter() {
1204            tx.send(deposit_finalize_first_param.clone())
1205                .await
1206                .map_err(|e| {
1207                    Status::internal(format!(
1208                        "Failed to send deposit finalize first param: {:?}",
1209                        e
1210                    ))
1211                })?;
1212        }
1213
1214        let deposit_data: crate::builder::transaction::DepositData =
1215            deposit_params.clone().try_into()?;
1216
1217        let deposit_blockhash = self
1218            .rpc
1219            .get_blockhash_of_tx(&deposit_data.get_deposit_outpoint().txid)
1220            .await
1221            .map_to_status()
1222            .map_err(|e| *e)?;
1223
1224        let verifiers_public_keys = deposit_data.get_verifiers();
1225
1226        // Create sighash stream for transaction signing
1227        let sighash_stream = Box::pin(create_nofn_sighash_stream(
1228            self.db.clone(),
1229            self.config.clone(),
1230            deposit_data.clone(),
1231            deposit_blockhash,
1232            false,
1233        ));
1234
1235        // Create channels for pipeline communication
1236        let (agg_nonce_sender, agg_nonce_receiver) = channel(1280);
1237        let (partial_sig_sender, partial_sig_receiver) = channel(1280);
1238        let (final_sig_sender, final_sig_receiver) = channel(1280);
1239
1240        // Start the nonce aggregation pipe.
1241        let nonce_agg_handle = tokio::spawn(nonce_aggregator(
1242            nonce_streams,
1243            sighash_stream,
1244            agg_nonce_sender,
1245        ));
1246
1247        // Start the nonce distribution pipe.
1248        let nonce_dist_handle = tokio::spawn(nonce_distributor(
1249            agg_nonce_receiver,
1250            partial_sig_streams,
1251            partial_sig_sender,
1252        ));
1253
1254        // Start the signature aggregation pipe.
1255        let sig_agg_handle = tokio::spawn(signature_aggregator(
1256            partial_sig_receiver,
1257            verifiers_public_keys,
1258            final_sig_sender,
1259        ));
1260
1261        tracing::debug!("Getting signatures from operators");
1262        // Get sigs from each operator in background
1263        let operator_sigs_fut = tokio::spawn(Aggregator::collect_operator_sigs(
1264            self.get_participating_operators(&deposit_data).await?,
1265            self.config.clone(),
1266            deposit_sign_session,
1267        ));
1268
1269        // Join the nonce aggregation handle to get the movetx agg nonce.
1270        let nonce_agg_handle = nonce_agg_handle
1271            .map_err(|_| Status::internal("panic when aggregating nonces"))
1272            .map(|res| -> Result<(MusigAggNonce, MusigAggNonce), Status> {
1273                res.and_then(|r| r.map_err(Into::into))
1274            })
1275            .shared();
1276
1277        // Start the deposit finalization pipe.
1278        let sig_dist_handle = tokio::spawn(signature_distributor(
1279            final_sig_receiver,
1280            deposit_finalize_sender.clone(),
1281            nonce_agg_handle.clone(),
1282        ));
1283
1284        tracing::debug!(
1285            "Waiting for pipeline tasks to complete (nonce agg, sig agg, sig dist, operator sigs)"
1286        );
1287
1288        tracing::debug!("Waiting for pipeline tasks to complete");
1289        // Wait for all pipeline tasks to complete
1290        try_join_all([nonce_dist_handle, sig_agg_handle, sig_dist_handle])
1291            .await
1292            .map_err(|_| Status::internal("panic when pipelining"))?;
1293
1294        tracing::debug!("Pipeline tasks completed");
1295
1296        // Right now we collect all operator sigs then start to send them, we can do it simultaneously in the future
1297        // Need to change sig verification ordering in deposit_finalize() in verifiers so that we verify
1298        // 1st signature of all operators, then 2nd of all operators etc.
1299        let operator_sigs = operator_sigs_fut
1300            .await
1301            .map_err(|_| Status::internal("panic when collecting operator signatures"))??;
1302
1303        tracing::debug!("Got all operator signatures");
1304
1305        // send operators sigs to verifiers after all verifiers have signed
1306        let send_operator_sigs: Vec<_> = deposit_finalize_sender
1307            .iter()
1308            .map(|tx| async {
1309                for sigs in operator_sigs.iter() {
1310                    for sig in sigs.iter() {
1311                        let deposit_finalize_param: VerifierDepositFinalizeParams = sig.into();
1312
1313                        tx.send(deposit_finalize_param).await.wrap_err_with(|| {
1314                            eyre::eyre!(AggregatorError::OutputStreamEndedEarly {
1315                                stream_name: "deposit_finalize_sender".into(),
1316                            })
1317                        })?;
1318                    }
1319                }
1320
1321                Ok::<(), BridgeError>(())
1322            })
1323            .collect();
1324
1325        // wait until all operator sigs are sent to every verifier
1326        try_join_all(send_operator_sigs).await?;
1327
1328        tracing::debug!("Waiting for deposit finalization");
1329
1330        // Collect partial signatures for move transaction
1331        let partial_sigs: Vec<(Vec<u8>, Vec<u8>)> =
1332            try_join_all(deposit_finalize_futures.iter_mut().map(|fut| async {
1333                let response = fut
1334                    .await
1335                    .map_err(|_| Status::internal("panic finishing deposit_finalize"))??
1336                    .into_inner();
1337                Ok::<_, Status>((
1338                    response.move_to_vault_partial_sig,
1339                    response.emergency_stop_partial_sig,
1340                ))
1341            }))
1342            .await
1343            .map_err(|e| Status::internal(format!("Failed to finalize deposit: {:?}", e)))?;
1344
1345        let (move_to_vault_sigs, emergency_stop_sigs): (Vec<Vec<u8>>, Vec<Vec<u8>>) =
1346            partial_sigs.into_iter().unzip();
1347
1348        tracing::debug!("Received move tx partial sigs: {:?}", move_to_vault_sigs);
1349
1350        // Create the final move transaction and check the signatures
1351        let (movetx_agg_nonce, emergency_stop_agg_nonce) = nonce_agg_handle.await?;
1352
1353        // Verify emergency stop signatures
1354        self.verify_and_save_emergency_stop_sigs(
1355            emergency_stop_sigs,
1356            emergency_stop_agg_nonce,
1357            deposit_params.clone(),
1358        )
1359        .await?;
1360
1361        let signed_movetx_handler = self
1362            .send_movetx(move_to_vault_sigs, movetx_agg_nonce, deposit_params)
1363            .await?;
1364        let txid = *signed_movetx_handler.get_txid();
1365
1366        Ok(Response::new(txid.into()))
1367    }

1384            withdraw_responses: responses
1385                .into_iter()
1386                .map(|r| clementine::WithdrawResult {
1387                    result: Some(match r {
1388                        Ok(response) => {

1390                        }
1391                        Err(e) => clementine::withdraw_result::Result::Error(
1392                            clementine::WithdrawErrorResponse {
1393                                error: e.to_string(),
1394                            },
1395                        ),
1396                    }),
1397                })
