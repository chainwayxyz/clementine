36    async fn optimistic_payout_sign(
37        &self,
38        request: Request<OptimisticPayoutParams>,
39    ) -> Result<Response<PartialSig>, Status> {
40        let params = request.into_inner();
41        let agg_nonce = MusigAggNonce::from_slice(params.agg_nonce.as_slice())
42            .map_err(|e| Status::invalid_argument(format!("Invalid musigagg nonce: {}", e)))?;
43        let nonce_session_id = params
44            .nonce_gen
45            .ok_or(Status::invalid_argument(
46                "Nonce params not found for optimistic payout",
47            ))?
48            .id;
49        let withdraw_params = params.withdrawal.ok_or(Status::invalid_argument(
50            "Withdrawal params not found for optimistic payout",
51        ))?;
52        let (withdrawal_id, input_signature, input_outpoint, output_script_pubkey, output_amount) =
53            parser::operator::parse_withdrawal_sig_params(withdraw_params).await?;
54        let partial_sig = self
55            .verifier
56            .sign_optimistic_payout(
57                nonce_session_id,
58                agg_nonce,
59                withdrawal_id,
60                input_signature,
61                input_outpoint,
62                output_script_pubkey,
63                output_amount,
64            )
65            .await?;
66        Ok(Response::new(partial_sig.into()))
67    }

69    async fn internal_create_watchtower_challenge(
70        &self,
71        request: tonic::Request<super::TransactionRequest>,
72    ) -> std::result::Result<tonic::Response<super::RawTxWithRbfInfo>, tonic::Status> {
73        let transaction_request = request.into_inner();
74        let transaction_data = parse_transaction_request(transaction_request)?;
75
76        let (_tx_type, signed_tx, rbf_info) = self
77            .verifier
78            .create_watchtower_challenge(
79                transaction_data,
80                &{
81                    let challenge_bytes = self
82                        .verifier
83                        .config
84                        .protocol_paramset()
85                        .watchtower_challenge_bytes;
86                    let mut challenge = vec![0u8; challenge_bytes];
87                    for (step, i) in (0..challenge_bytes).step_by(32).enumerate() {
88                        if i < challenge_bytes {
89                            challenge[i] = step as u8;
90                        }
91                    }
92                    challenge
93                }, // dummy challenge with 1u8, 2u8 every 32 bytes
94            )
95            .await?;
96
97        Ok(Response::new(RawTxWithRbfInfo {
98            raw_tx: bitcoin::consensus::serialize(&signed_tx),
99            rbf_info: Some(rbf_info.into()),
100        }))
101    }

112    #[tracing::instrument(skip_all, err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
113    async fn set_operator(
114        &self,
115        req: Request<Streaming<OperatorParams>>,
116    ) -> Result<Response<Empty>, Status> {
117        let mut in_stream = req.into_inner();
118
119        let (collateral_funding_outpoint, operator_xonly_pk, wallet_reimburse_address) =
120            parser::operator::parse_details(&mut in_stream).await?;
121
122        let mut operator_kickoff_winternitz_public_keys = Vec::new();
123        // we need num_round_txs + 1 because the last round includes reimburse generators of previous round
124        for _ in 0..self.verifier.config.get_num_kickoff_winternitz_pks() {
125            operator_kickoff_winternitz_public_keys
126                .push(parser::operator::parse_winternitz_public_keys(&mut in_stream).await?);
127        }
128
129        let mut unspent_kickoff_sigs =
130            Vec::with_capacity(self.verifier.config.get_num_unspent_kickoff_sigs());
131        for _ in 0..self.verifier.config.get_num_unspent_kickoff_sigs() {
132            unspent_kickoff_sigs.push(parser::operator::parse_schnorr_sig(&mut in_stream).await?);
133        }
134
135        if in_stream.message().await?.is_some() {
136            return Err(Status::invalid_argument(
137                "Expected end of stream, got more messages in set_operator",
138            ));
139        }
140
141        self.verifier
142            .set_operator(
143                collateral_funding_outpoint,
144                operator_xonly_pk,
145                wallet_reimburse_address,
146                operator_kickoff_winternitz_public_keys,
147                unspent_kickoff_sigs,
148            )
149            .await?;
150
151        Ok(Response::new(Empty {}))
152    }

154    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
155    async fn nonce_gen(
156        &self,
157        req: Request<NonceGenRequest>,
158    ) -> Result<Response<Self::NonceGenStream>, Status> {
159        let num_nonces = req.into_inner().num_nonces;
160
161        let (session_id, pub_nonces) = self.verifier.nonce_gen(num_nonces).await?;
162
163        let (tx, rx) = mpsc::channel(pub_nonces.len() + 1);
164
165        tokio::spawn(async move {
166            let nonce_gen_first_response = clementine::NonceGenFirstResponse {
167                id: session_id,
168                num_nonces,
169            };
170            let session_id: NonceGenResponse = nonce_gen_first_response.into();
171            tx.send(Ok(session_id)).await?;
172
173            for pub_nonce in &pub_nonces {
174                let pub_nonce: NonceGenResponse = pub_nonce.into();
175                tx.send(Ok(pub_nonce)).await?;
176            }
177
178            Ok::<(), SendError<_>>(())
179        });
180
181        Ok(Response::new(ReceiverStream::new(rx)))
182    }

184    async fn deposit_sign(
185        &self,
186        req: Request<Streaming<VerifierDepositSignParams>>,
187    ) -> Result<Response<Self::DepositSignStream>, Status> {
188        let mut in_stream = req.into_inner();
189        let verifier = self.verifier.clone();
190
191        let (tx, rx) = mpsc::channel(1280);
192        let out_stream: Self::DepositSignStream = ReceiverStream::new(rx);
193
194        let (param_tx, mut param_rx) = mpsc::channel(1);
195        let (agg_nonce_tx, agg_nonce_rx) = mpsc::channel(1280);
196
197        // Send incoming data to deposit sign job.
198        tokio::spawn(async move {
199            let params = fetch_next_message_from_stream!(in_stream, params)?;
200            let (deposit_data, session_id) = match params {
201                clementine::verifier_deposit_sign_params::Params::DepositSignFirstParam(
202                    deposit_sign_session,
203                ) => parser::verifier::parse_deposit_sign_session(
204                    deposit_sign_session,
205                    &verifier.signer.public_key,
206                )?,
207                _ => return Err(Status::invalid_argument("Expected DepositOutpoint")),
208            };
209            param_tx
210                .send((deposit_data, session_id))
211                .await
212                .map_err(error::output_stream_ended_prematurely)?;
213
214            while let Some(result) =
215                fetch_next_optional_message_from_stream!(&mut in_stream, params)
216            {
217                let agg_nonce = match result {
218                    clementine::verifier_deposit_sign_params::Params::AggNonce(agg_nonce) => {
219                        MusigAggNonce::from_slice(agg_nonce.as_slice())
220                            .map_err(|_| ParserError::RPCParamMalformed("AggNonce".to_string()))?
221                    }
222                    _ => return Err(Status::invalid_argument("Expected AggNonce")),
223                };
224
225                agg_nonce_tx
226                    .send(agg_nonce)
227                    .await
228                    .map_err(error::output_stream_ended_prematurely)?;
229            }
230            Ok(())
231        });
232
233        // Start partial sig job and return partial sig responses.
234        tokio::spawn(async move {
235            let (deposit_data, session_id) = param_rx
236                .recv()
237                .await
238                .ok_or(error::expected_msg_got_none("parameters")())?;
239
240            let mut partial_sig_receiver = verifier
241                .deposit_sign(deposit_data.clone(), session_id, agg_nonce_rx)
242                .await?;
243
244            let mut nonce_idx = 0;
245            let num_required_sigs = verifier.config.get_num_required_nofn_sigs(&deposit_data);
246            while let Some(partial_sig) = partial_sig_receiver.recv().await {
247                tx.send(Ok(PartialSig {
248                    partial_sig: partial_sig.serialize().to_vec(),
249                }))
250                .await
251                .map_err(|e| {
252                    Status::aborted(format!(
253                        "Error sending partial sig, stream ended prematurely: {e}"
254                    ))
255                })?;
256
257                nonce_idx += 1;
258                tracing::trace!(
259                    "Verifier {:?} signed and sent sighash {} of {} through rpc deposit_sign",
260                    verifier.signer.public_key,
261                    nonce_idx,
262                    num_required_sigs
263                );
264                if nonce_idx == num_required_sigs {
265                    break;
266                }
267            }
268
269            Ok::<(), Status>(())
270        });
271
272        Ok(Response::new(out_stream))
273    }

275    /// Function to finalize the deposit. Verifier will check the validity of the both nofn signatures and
276    /// operator signatures. It will receive data from the stream in this order -> nofn sigs, movetx agg nonce, operator sigs.
277    /// If everything is correct, it will partially sign the move tx and send it to aggregator.
278    #[tracing::instrument(skip(self), err(level = tracing::Level::ERROR), ret(level = tracing::Level::TRACE))]
279    async fn deposit_finalize(
280        &self,
281        req: Request<Streaming<VerifierDepositFinalizeParams>>,
282    ) -> Result<Response<VerifierDepositFinalizeResponse>, Status> {
283        let mut in_stream = req.into_inner();
284        tracing::trace!(
285            "In verifier {:?} deposit_finalize()",
286            self.verifier.signer.public_key
287        );
288
289        let (sig_tx, sig_rx) = mpsc::channel(1280);
290        let (agg_nonce_tx, agg_nonce_rx) = mpsc::channel(1);
291        let (operator_sig_tx, operator_sig_rx) = mpsc::channel(1280);
292
293        let params = fetch_next_message_from_stream!(in_stream, params)?;
294        let (deposit_data, session_id) = match params {
295            Params::DepositSignFirstParam(deposit_sign_session) => {
296                parser::verifier::parse_deposit_sign_session(
297                    deposit_sign_session,
298                    &self.verifier.signer.public_key,
299                )?
300            }
301            _ => Err(Status::internal("Expected DepositOutpoint"))?,
302        };
303        tracing::trace!(
304            "Verifier {:?} got DepositSignFirstParam in deposit_finalize()",
305            self.verifier.signer.public_key
306        );
307
308        // Start deposit finalize job.
309        let verifier = self.verifier.clone();
310        let mut dep_data = deposit_data.clone();
311        let deposit_finalize_handle = tokio::spawn(async move {
312            verifier
313                .deposit_finalize(
314                    &mut dep_data,
315                    session_id,
316                    sig_rx,
317                    agg_nonce_rx,
318                    operator_sig_rx,
319                )
320                .await
321        });
322
323        // Start parsing inputs and send them to deposit finalize job.
324        let verifier = self.verifier.clone();
325        tokio::spawn(async move {
326            let num_required_nofn_sigs = verifier.config.get_num_required_nofn_sigs(&deposit_data);
327            let mut nonce_idx = 0;
328            while let Some(sig) =
329                parser::verifier::parse_next_deposit_finalize_param_schnorr_sig(&mut in_stream)
330                    .await?
331            {
332                tracing::debug!(
333                    "Received full nofn sig {} in deposit_finalize()",
334                    nonce_idx + 1
335                );
336                sig_tx
337                    .send(sig)
338                    .await
339                    .map_err(error::output_stream_ended_prematurely)?;
340                tracing::debug!(
341                    "Sent full nofn sig {} to src/verifier in deposit_finalize()",
342                    nonce_idx + 1
343                );
344                nonce_idx += 1;
345                if nonce_idx == num_required_nofn_sigs {
346                    break;
347                }
348            }
349            if nonce_idx < num_required_nofn_sigs {
350                panic!(
351                    "Expected more nofn sigs {} < {}",
352                    nonce_idx, num_required_nofn_sigs
353                )
354            }
355
356            let move_tx_agg_nonce =
357                parser::verifier::parse_deposit_finalize_param_move_tx_agg_nonce(&mut in_stream)
358                    .await?;
359            agg_nonce_tx
360                .send(move_tx_agg_nonce)
361                .await
362                .map_err(error::output_stream_ended_prematurely)?;
363
364            let emergency_stop_agg_nonce =
365                parser::verifier::parse_deposit_finalize_param_emergency_stop_agg_nonce(
366                    &mut in_stream,
367                )
368                .await?;
369            agg_nonce_tx
370                .send(emergency_stop_agg_nonce)
371                .await
372                .map_err(error::output_stream_ended_prematurely)?;
373
374            let num_required_op_sigs = verifier
375                .config
376                .get_num_required_operator_sigs(&deposit_data);
377            let num_operators = deposit_data.get_num_operators();
378            let num_required_total_op_sigs = num_required_op_sigs * num_operators;
379            let mut total_op_sig_count = 0;
380            for _ in 0..num_operators {
381                let mut op_sig_count = 0;
382
383                while let Some(operator_sig) =
384                    parser::verifier::parse_next_deposit_finalize_param_schnorr_sig(&mut in_stream)
385                        .await?
386                {
387                    tracing::debug!(
388                        "Received full operator sig {} in deposit_finalize()",
389                        op_sig_count + 1
390                    );
391                    operator_sig_tx
392                        .send(operator_sig)
393                        .await
394                        .map_err(error::output_stream_ended_prematurely)?;
395                    tracing::debug!(
396                        "Sent full operator sig {} to src/verifier in deposit_finalize()",
397                        op_sig_count + 1
398                    );
399
400                    op_sig_count += 1;
401                    total_op_sig_count += 1;
402                    if op_sig_count == num_required_op_sigs {
403                        break;
404                    }
405                }
406            }
407
408            if total_op_sig_count < num_required_total_op_sigs {
409                panic!(
410                    "Not enough operator signatures. Needed: {}, received: {}",
411                    num_required_total_op_sigs, total_op_sig_count
412                );
413            }
414
415            Ok::<(), Status>(())
416        });
417
418        let partial_sig = deposit_finalize_handle.await.map_err(|e| {
419            Status::internal(format!("Deposit finalize thread failed to finish: {}", e).as_str())
420        })??;
421
422        let response = VerifierDepositFinalizeResponse {
423            move_to_vault_partial_sig: partial_sig.0.serialize().to_vec(),
424            emergency_stop_partial_sig: partial_sig.1.serialize().to_vec(),
425        };
426
427        Ok(Response::new(response))
428    }

469    async fn internal_handle_kickoff(
470        &self,
471        request: Request<clementine::Txid>,
472    ) -> Result<Response<Empty>, Status> {
473        let txid = request.into_inner();
474        let txid = bitcoin::Txid::try_from(txid).expect("Should be able to convert");
475        let mut dbtx = self.verifier.db.begin_transaction().await?;
476        let kickoff_data = self
477            .verifier
478            .db
479            .get_deposit_data_with_kickoff_txid(None, txid)
480            .await?;
481        if let Some((deposit_data, kickoff_id)) = kickoff_data {
482            self.verifier
483                .handle_kickoff(&mut dbtx, Witness::new(), deposit_data, kickoff_id, false)
484                .await?;
485        } else {
486            return Err(Status::not_found("Kickoff txid not found"));
487        }
488        dbtx.commit().await.expect("Failed to commit transaction");
489        Ok(Response::new(Empty {}))
490    }
