1use tonic::{service::Interceptor, transport::CertificateDer, Request, Status};
2
3#[derive(Debug, Clone)]
4pub enum Interceptors {
5    OnlyAggregatorAndSelf {
6        aggregator_cert: CertificateDer<'static>,
7        our_cert: CertificateDer<'static>,
8    },
9    Noop,
10}
11
12fn is_internal(req: &Request<()>) -> bool {
13    // This normally doesn't exist but we add it in the AddMethodMiddleware
14    let Some(path) = req.metadata().get("grpc-method") else {
15        // No grpc method? this should not happen
16        tracing::error!("Missing grpc-method header in request");
17        return false;
18    };
19
20    path.as_bytes().starts_with(b"Internal")
21}
22
23impl Interceptor for Interceptors {
24    #[allow(clippy::result_large_err)]
25    fn call(&mut self, req: Request<()>) -> Result<Request<()>, Status> {
26        match self {
27            Interceptors::OnlyAggregatorAndSelf {
28                our_cert,
29                aggregator_cert,
30            } => only_aggregator_and_self(req, our_cert, aggregator_cert),
31            Interceptors::Noop => Ok(req),
32        }
33    }
34}
35
36#[allow(clippy::result_large_err)]
37fn only_aggregator_and_self(
38    req: Request<()>,
39    our_cert: &CertificateDer<'static>,
40    aggregator_cert: &CertificateDer<'static>,
41) -> Result<Request<()>, Status> {
42    let Some(peer_certs) = req.peer_certs() else {
43        if cfg!(test) {
44            // Test mode, we don't need to verify peer certificates
45            return Ok(req);
46        } else {
47            // If we're not in test mode, we need to check peer certificates
48            return Err(Status::unauthenticated(
49                "Failed to verify peer certificate, is TLS enabled?",
50            ));
51        }
52    };
53
54    if is_internal(&req) {
55        if peer_certs.contains(our_cert) {
56            Ok(req)
57        } else {
58            Err(Status::unauthenticated(
59                "Unauthorized call to internal method (not self)",
60            ))
61        }
62    } else if peer_certs.contains(aggregator_cert) || peer_certs.contains(our_cert) {
63        Ok(req)
64    } else {
65        Err(Status::unauthenticated(
66            "Unauthorized call to method (not aggregator or self)",
67        ))
68    }
69}
