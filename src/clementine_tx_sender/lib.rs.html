12// Define a macro for logging errors and saving them to the database
13#[macro_export]
14macro_rules! log_error_for_tx {
15    ($db:expr, $try_to_send_id:expr, $err:expr) => {{
16        let db = $db.clone();
17        let try_to_send_id = $try_to_send_id;
18        let err = $err.to_string();
19        tracing::warn!(try_to_send_id, "{}", err);
20        tokio::spawn(async move {
21            let _ = db
22                .save_tx_debug_submission_error(None, try_to_send_id, &err)

24        });

99/// Trait for building child transactions in the transaction sender.
100///
101/// This abstraction allows the core crate to provide `SpendableTxIn`-based transaction building
102/// using `TxHandler`, while keeping the tx-sender crate independent of the builder module.
103///
104/// All methods are static - no instance of this trait is stored.
105pub trait TxSenderTxBuilder: Send + Sync + 'static {
106    /// The type representing a spendable transaction input.
107    /// In core, this would be `SpendableTxIn`.
108    type SpendableInput: SpendableInputInfo;
109
110    /// Builds a child transaction for CPFP.
111    ///
112    /// This method constructs a child transaction that spends the P2A anchor output
113    /// and fee payer UTXOs, paying the required fees for the CPFP package.
114    ///
115    /// # Arguments
116    /// * `p2a_anchor` - The P2A anchor output to spend
117    /// * `anchor_sat` - Amount in the anchor output
118    /// * `fee_payer_utxos` - UTXOs to fund the child transaction
119    /// * `change_address` - Address for the change output
120    /// * `required_fee` - The calculated required fee for the package
121    /// * `signer_address` - The signer's address (for script pubkey)
122    /// * `signer` - The signer to sign the transaction inputs
123    ///
124    /// # Returns
125    /// A signed child transaction ready for package submission.
126    fn build_child_tx<S: TxSenderSigner>(
127        p2a_anchor: OutPoint,
128        anchor_sat: Amount,
129        fee_payer_utxos: Vec<Self::SpendableInput>,
130        change_address: Address,
131        required_fee: Amount,
132        signer: &S,
133    ) -> Result<Transaction, BridgeError>;
134
135    /// Converts database UTXOs into the builder's SpendableInput type.
136    ///
137    /// # Arguments
138    /// * `utxos` - Vector of (txid, vout, amount) tuples from the database
139    /// * `signer_address` - The signer's address (for script pubkey generation)
140    ///
141    /// # Returns
142    /// Vector of SpendableInput instances ready for use in transaction building.
143    fn utxos_to_spendable_inputs(
144        utxos: Vec<(Txid, u32, Amount)>,
145        signer_address: &Address,

398/// Manages the process of sending Bitcoin transactions, including handling fee bumping
399/// strategies like Replace-By-Fee (RBF) and Child-Pays-For-Parent (CPFP).
400///
401/// It interacts with a Bitcoin Core RPC endpoint (`ExtendedBitcoinRpc`) to query network state
402/// (like fee rates) and submit transactions. It uses a `Database` to persist transaction
403/// state, track confirmation status, and manage associated data like fee payer UTXOs.
404/// The `Actor` provides signing capabilities for transactions controlled by this service.
405///
406/// The `TxSenderTxBuilder` type parameter provides static methods for transaction building
407/// capabilities for CPFP child transactions, using `SpendableTxIn` and `TxHandler`.
408#[derive(Clone)]
409pub struct TxSender<S, D, B>
410where
411    S: TxSenderSigner + 'static,
412    D: TxSenderDatabase + 'static,
413    B: TxSenderTxBuilder + 'static,
414{
415    pub signer: S,
416    pub rpc: clementine_extended_rpc::ExtendedBitcoinRpc,
417    pub db: D,
418    pub btc_syncer_consumer_id: String,
419    pub protocol_paramset: &'static ProtocolParamset,
420    pub tx_sender_limits: TxSenderLimits,
421    pub http_client: reqwest::Client,
422    mempool_config: MempoolConfig,
423    /// Phantom data to track the TxBuilder type.
424    /// B provides static methods for transaction building.
425    _tx_builder: std::marker::PhantomData<B>,
426}

491    /// Calculates the total fee required for a transaction package based on the fee bumping strategy.
492    ///
493    /// # Arguments
494    /// * `parent_tx_weight` - The weight of the main transaction being bumped.
495    /// * `num_fee_payer_utxos` - The number of fee payer UTXOs used (relevant for child tx size in CPFP).
496    /// * `fee_rate` - The target fee rate (sat/kwu or similar).
497    /// * `fee_paying_type` - The strategy being used (CPFP or RBF).
498    ///
499    /// # Calculation Logic
500    /// *   **CPFP:** Calculates the weight of the hypothetical child transaction based on the
501    ///     number of fee payer inputs and standard P2TR output sizes. It then calculates the
502    ///     fee based on the *combined virtual size* (vbytes) of the parent and child transactions,
503    ///     as miners evaluate the package deal.
504    /// *   **RBF:** Calculates the weight of the replacement transaction itself (assuming inputs
505    ///     and potentially outputs change slightly). The fee is calculated based on the weight
506    ///     of this single replacement transaction.
507    ///
508    /// Reference for weight estimates: <https://bitcoin.stackexchange.com/a/116959>
509    fn calculate_required_fee(
510        parent_tx_weight: Weight,
511        num_fee_payer_utxos: usize,
512        fee_rate: FeeRate,
513        fee_paying_type: FeePayingType,
514    ) -> Result<Amount> {
515        tracing::info!(
516            "Calculating required fee for {} fee payer utxos",
517            num_fee_payer_utxos
518        );
519        // Estimate the weight of the child transaction (for CPFP) or the RBF replacement.
520        // P2TR input witness adds ~57.5vbytes (230 WU). P2TR output adds 43 vbytes (172 WU).
521        // Base transaction overhead (version, locktime, input/output counts) ~ 10.5 vBytes (42 WU)
522        // Anchor input marker (OP_FALSE OP_RETURN ..) adds overhead. Exact WU TBD.
523        // For CPFP child: (N fee payer inputs) + (1 anchor input) + (1 change output)
524        // For RBF replacement: (N fee payer inputs) + (1 change output) - assuming it replaces a tx with an anchor.
525        let child_tx_weight = match fee_paying_type {
526            // CPFP Child: N fee payer inputs + 1 anchor input + 1 change output + base overhead.
527            // Approx WU: (230 * num_fee_payer_utxos) + 230 + 172 + base_overhead_wu
528            // Simplified calculation used here needs verification.
529            FeePayingType::CPFP => Weight::from_wu_usize(230 * num_fee_payer_utxos + 207 + 172),
530            // RBF Replacement: N fee payer inputs + 1 change output + base overhead.
531            // Assumes it replaces a tx of similar structure but potentially different inputs/fees.
532            // Simplified calculation used here needs verification.
533            FeePayingType::RBF => Weight::from_wu_usize(230 * num_fee_payer_utxos + 172),
534            FeePayingType::NoFunding => Weight::from_wu_usize(0),
535        };
536
537        // Calculate total weight for fee calculation.
538        // For CPFP, miners consider the effective fee rate over the combined *vbytes* of parent + child.
539        // For RBF, miners consider the fee rate of the single replacement transaction's weight.
540        let total_weight = match fee_paying_type {
541            FeePayingType::CPFP => Weight::from_vb_unchecked(
542                child_tx_weight.to_vbytes_ceil() + parent_tx_weight.to_vbytes_ceil(),
543            ),
544            FeePayingType::RBF => child_tx_weight + parent_tx_weight, // Should likely just be the RBF tx weight? Check RBF rules.
545            FeePayingType::NoFunding => parent_tx_weight,
546        };
547
548        fee_rate
549            .checked_mul_by_weight(total_weight)
550            .ok_or_eyre("Fee calculation overflow")
551            .map_err(Into::into)
552    }

566    /// Fetches transactions that are eligible to be sent or bumped from
567    /// database based on the given fee rate and tip height. Then, places a send
568    /// transaction request to the Bitcoin based on the fee strategy.
569    ///
570    /// For each eligible transaction (`id`):
571    ///
572    /// 1.  **Send/Bump Main Tx:** Calls `send_tx` to either perform RBF or CPFP on the main
573    ///     transaction (`id`) using the `new_fee_rate`.
574    /// 2.  **Handle Errors:**
575    ///     - [`SendTxError::UnconfirmedFeePayerUTXOsLeft`]: Skips the current tx, waiting for fee
576    ///       payers to confirm.
577    ///     - [`SendTxError::InsufficientFeePayerAmount`]: Calls `create_fee_payer_utxo` to
578    ///       provision more funds for a future CPFP attempt.
579    ///     - Other errors are logged.
580    ///
581    /// # Arguments
582    /// * `new_fee_rate` - The current target fee rate based on network conditions.
583    /// * `current_tip_height` - The current blockchain height, used for time-lock checks.
584    /// * `is_tip_height_increased` - True if the tip height has increased since the last time we sent unconfirmed transactions.
585    #[tracing::instrument(skip_all, fields(sender = self.btc_syncer_consumer_id, new_fee_rate, current_tip_height))]
586    async fn try_to_send_unconfirmed_txs(
587        &self,
588        new_fee_rate: FeeRate,
589        current_tip_height: u32,
590        is_tip_height_increased: bool,
591    ) -> Result<()> {
592        // get_sendable_txs doesn't return txs that we already sent in the past with >= fee rate to the current fee rate
593        // but if we have a new block height, but the tx is still not confirmed, we want to send it again anyway in case
594        // some error occurred on our bitcoin rpc/our tx got evicted from mempool somehow (for ex: if a fee payer of cpfp tx was reorged,
595        // cpfp tx will get evicted as v3 cpfp cannot have unconfirmed ancestors)
596        // if block height is increased, we use a dummy high fee rate to get all sendable txs
597        let get_sendable_txs_fee_rate = if is_tip_height_increased {
598            FeeRate::from_sat_per_kwu(u32::MAX as u64)
599        } else {
600            new_fee_rate
601        };
602        let txs = self
603            .db
604            .get_sendable_txs(get_sendable_txs_fee_rate, current_tip_height)
605            .await
606            .map_to_eyre()?;
607
608        // bump fees of fee payer transactions that are unconfirmed
609        self.bump_fees_of_unconfirmed_fee_payer_txs(new_fee_rate)
610            .await?;
611
612        if !txs.is_empty() {
613            tracing::debug!("Trying to send {} sendable txs ", txs.len());
614        }
615
616        if std::env::var("TXSENDER_DBG_INACTIVE_TXS").is_ok() {
617            self.db
618                .debug_inactive_txs(get_sendable_txs_fee_rate, current_tip_height)
619                .await;
620        }
621
622        for id in txs {
623            // Update debug state
624            tracing::debug!(
625                try_to_send_id = id,
626                "Processing TX in try_to_send_unconfirmed_txs with fee rate {new_fee_rate}",
627            );
628
629            let (tx_metadata, tx, fee_paying_type, seen_block_id, rbf_signing_info) =
630                match self.db.get_try_to_send_tx(None, id).await {
631                    Ok(res) => res,
632                    Err(e) => {
633                        log_error_for_tx!(self.db, id, format!("Failed to get tx details: {}", e));
634                        continue;
635                    }
636                };
637
638            // Check if the transaction is already confirmed (only happens if it was confirmed after this loop started)
639            if let Some(block_id) = seen_block_id {
640                tracing::debug!(
641                    try_to_send_id = id,
642                    "Transaction already confirmed in block with block id of {}",
643                    block_id
644                );
645
646                // Update sending state
647                let _ = self
648                    .db
649                    .update_tx_debug_sending_state(id, "confirmed", true)
650                    .await;
651
652                continue;
653            }
654
655            // Get effective fee rate and block height to calculate adjusted fee rate
656            let (previous_effective_fee_rate, last_bump_block_height) =
657                match self.db.get_effective_fee_rate(None, id).await {
658                    Ok(res) => res,
659                    Err(e) => {
660                        log_error_for_tx!(
661                            self.db,
662                            id,
663                            format!("Failed to get effective fee rate: {}", e)
664                        );
665                        continue;
666                    }
667                };
668
669            // Calculate adjusted fee rate considering:
670            // 1. If new_fee_rate > previous_effective_fee_rate, use max(new_fee_rate, previous_effective_fee_rate + incremental_fee_rate)
671            // 2. If tx has been stuck for 10+ blocks, bump with incremental fee
672            let adjusted_fee_rate = match self
673                .calculate_target_fee_rate(
674                    previous_effective_fee_rate,
675                    new_fee_rate,
676                    last_bump_block_height,
677                    current_tip_height,
678                )
679                .await
680            {
681                Ok(rate) => rate,
682                Err(e) => {
683                    log_error_for_tx!(
684                        self.db,
685                        id,
686                        format!("Failed to calculate adjusted fee rate: {}", e)
687                    );
688                    continue;
689                }
690            };
691
692            let result = match fee_paying_type {
693                // Send nonstandard transactions to testnet4 using the mempool.space accelerator.
694                // As mempool uses out of band payment, we don't need to do cpfp or rbf.
695                _ if self.protocol_paramset.network == bitcoin::Network::Testnet4
696                    && self.is_bridge_tx_nonstandard(&tx) =>
697                {
698                    self.send_testnet4_nonstandard_tx(&tx, id).await
699                }
700                FeePayingType::CPFP => {
701                    self.send_cpfp_tx(id, tx, tx_metadata, adjusted_fee_rate, current_tip_height)
702                        .await
703                }
704                FeePayingType::RBF => {
705                    self.send_rbf_tx(
706                        id,
707                        tx,
708                        tx_metadata,
709                        adjusted_fee_rate,
710                        rbf_signing_info,
711                        current_tip_height,
712                    )
713                    .await
714                }
715                FeePayingType::NoFunding => self.send_no_funding_tx(id, tx, tx_metadata).await,
716            };
717
718            if let Err(e) = result {
719                log_error_for_tx!(self.db, id, format!("Failed to send tx: {:?}", e));
720            }
721        }
722
723        Ok(())
724    }
725    pub fn client(&self) -> TxSenderClient<D> {
726        TxSenderClient::new(self.db.clone(), self.btc_syncer_consumer_id.clone())
727    }

729    /// Calculates the effective fee rate for a transaction, considering previous effective fee rate
730    /// and minimum incremental fee requirements.
731    ///
732    /// This function implements the logic for fee bumping that ensures:
733    /// 1. If no previous effective fee rate exists, use the new fee rate
734    /// 2. If previous effective fee rate exists, use the maximum of:
735    ///    - The new fee rate
736    ///    - Previous effective fee rate + minimum incremental fee rate
737    ///
738    /// # Arguments
739    /// * `previous_effective_fee_rate` - The previous effective fee rate (if any)
740    /// * `new_fee_rate` - The target fee rate for the new attempt
741    /// * `last_bump_block_height` - The block height when the last fee bump was done (if any)
742    /// * `current_tip_height` - The current blockchain tip height
743    ///
744    /// # Returns
745    /// The effective fee rate to use (in sat/kwu), capped by the hard cap from config
746    pub async fn calculate_target_fee_rate(
747        &self,
748        previous_effective_fee_rate: Option<FeeRate>,
749        new_fee_rate: FeeRate,
750        last_bump_block_height: Option<u32>,
751        current_tip_height: u32,
752    ) -> Result<FeeRate> {
753        // Hard cap from config (in sat/vB), convert to sat/kwu
754        let hard_cap = FeeRate::from_sat_per_vb(self.tx_sender_limits.fee_rate_hard_cap)
755            .expect("fee_rate_hard_cap should be valid");
756
757        let Some(previous_rate) = previous_effective_fee_rate else {
758            // No previous effective fee rate, use the new fee rate (capped)
759            return Ok(std::cmp::min(new_fee_rate, hard_cap));
760        };
761
762        // Check if the tx has been stuck for 10+ blocks
763        let is_stuck = match last_bump_block_height {
764            Some(block_height) => {
765                current_tip_height.saturating_sub(block_height)
766                    >= self.tx_sender_limits.fee_bump_after_blocks
767            }
768            None => false,
769        };
770
771        // Get minimum fee increment rate from node for BIP125 compliance. Returned value is in BTC/kvB
772        let incremental_fee_rate = self
773            .rpc
774            .get_network_info()
775            .await
776            .map_err(|e| eyre::eyre!(e))?
777            .incremental_fee;
778        let incremental_fee_rate_sat_per_kvb = incremental_fee_rate.to_sat();
779        let incremental_fee_rate =
780            FeeRate::from_sat_per_kwu(incremental_fee_rate_sat_per_kvb.div_ceil(4));
781
782        // Minimum bump fee rate required by BIP125
783        let min_bump_feerate =
784            previous_rate.to_sat_per_kwu() + incremental_fee_rate.to_sat_per_kwu();
785
786        // If new fee rate is higher than previous, use max of new_fee_rate and min_bump_feerate
787        if new_fee_rate.to_sat_per_kwu() > previous_rate.to_sat_per_kwu() {
788            let effective_feerate = std::cmp::max(new_fee_rate.to_sat_per_kwu(), min_bump_feerate);
789            let result = FeeRate::from_sat_per_kwu(effective_feerate);
790            return Ok(std::cmp::min(result, hard_cap));
791        }
792
793        // If the tx is stuck for 10+ blocks, force a fee bump (previous + incremental)
794        if is_stuck {
795            let result = FeeRate::from_sat_per_kwu(min_bump_feerate);
796            let capped_result = std::cmp::min(result, hard_cap);
797
798            tracing::debug!(
799                "TX stuck for at least {} blocks, forcing fee bump from {} to {} sat/kwu (hard cap: {} sat/kwu)",
800                self.tx_sender_limits.fee_bump_after_blocks,
801                previous_rate.to_sat_per_kwu(),
802                capped_result.to_sat_per_kwu(),
803                hard_cap.to_sat_per_kwu()
804            );
805
806            return Ok(capped_result);
807        }
808
809        // Neither higher fee rate nor stuck, use previous rate (no change needed)
810        Ok(previous_rate)
811    }

813    /// Sends a transaction that is already fully funded and signed.
814    ///
815    /// This function is used for transactions that do not require fee bumping strategies
816    /// like RBF or CPFP. The transaction is submitted directly to the Bitcoin network
817    /// without any modifications.
818    ///
819    /// # Arguments
820    /// * `try_to_send_id` - The database ID tracking this send attempt.
821    /// * `tx` - The fully funded and signed transaction ready for broadcast.
822    /// * `tx_metadata` - Optional metadata associated with the transaction for debugging.
823    ///
824    /// # Behavior
825    /// 1. Attempts to broadcast the transaction using `send_raw_transaction` RPC.
826    /// 2. Updates the database with success/failure state for debugging purposes.
827    /// 3. Logs appropriate messages for monitoring and troubleshooting.
828    ///
829    /// # Returns
830    /// * `Ok(())` - If the transaction was successfully broadcast.
831    /// * `Err(SendTxError)` - If the broadcast failed.
832    #[tracing::instrument(skip_all, fields(sender = self.btc_syncer_consumer_id, try_to_send_id, tx_meta=?tx_metadata))]
833    pub async fn send_no_funding_tx(
834        &self,
835        try_to_send_id: u32,
836        tx: Transaction,
837        tx_metadata: Option<TxMetadata>,
838    ) -> Result<()> {
839        match self.rpc.send_raw_transaction(&tx).await {
840            Ok(sent_txid) => {
841                tracing::debug!(
842                    try_to_send_id,
843                    "Successfully sent no funding tx with txid {}",
844                    sent_txid
845                );
846                let _ = self
847                    .db
848                    .update_tx_debug_sending_state(try_to_send_id, "no_funding_send_success", true)
849                    .await;
850            }
851            Err(e) => {
852                tracing::error!(
853                    "Failed to send no funding tx with try_to_send_id: {try_to_send_id:?} and metadata: {tx_metadata:?}"
854                );
855                let err_msg = format!("send_raw_transaction error for no funding tx: {e}");
856                log_error_for_tx!(self.db, try_to_send_id, err_msg);
857                let _ = self
858                    .db
859                    .update_tx_debug_sending_state(try_to_send_id, "no_funding_send_failed", true)
860                    .await;
861                return Err(SendTxError::Other(eyre::eyre!(e)));
862            }
863        };
864
865        Ok(())
866    }
