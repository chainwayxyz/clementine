250// Clementine do not use the runtime option to determine dev mode which is newly added in risc0_zkvm.
251// Instead, it uses the environment variable RISC0_DEV_MODE to determine if it is in dev mode.
252// However is_dev_mode function from risc0_zkvm is deprecated.
253// So we implement our own version of is_dev_mode.
254pub fn is_dev_mode() -> bool {
255    std::env::var("RISC0_DEV_MODE")
256        .ok()
257        .map(|x| x.to_lowercase())
258        .filter(|x| x == "1" || x == "true" || x == "yes")
259        .is_some()
260}
261
262pub fn get_verifying_key() -> ark_groth16::VerifyingKey<Bn254> {
263    if is_dev_mode() {
264        get_ark_verifying_key_dev_mode_bridge()
265    } else {
266        get_ark_verifying_key_prod()
267    }
268}

270/// Sha256(control_root, pre_state_digest, post_state_digest, id_bn254_fr)
271pub fn calculate_succinct_output_prefix(method_id: &[u8]) -> [u8; 32] {
272    let succinct_verifier_params = SuccinctReceiptVerifierParameters::default();
273    let succinct_control_root = succinct_verifier_params.control_root;
274    let mut succinct_control_root_bytes: [u8; 32] =
275        succinct_control_root.as_bytes().try_into().unwrap();
276    for byte in succinct_control_root_bytes.iter_mut() {
277        *byte = byte.reverse_bits();
278    }
279    let pre_state_bytes = method_id.to_vec();
280    let control_id_bytes: [u8; 32] = BN254_IDENTITY_CONTROL_ID.into();
281
282    // Expected post state for an execution that halted successfully
283    let post_state: SystemState = risc0_binfmt::SystemState {
284        pc: 0,
285        merkle_root: risc0_zkp::core::digest::Digest::default(),
286    };
287    let post_state_bytes: [u8; 32] = post_state.digest().into();
288
289    let mut hasher = Sha256::new();
290    hasher.update(succinct_control_root_bytes);
291    hasher.update(pre_state_bytes);
292    hasher.update(post_state_bytes);
293    hasher.update(control_id_bytes);
294    let result: [u8; 32] = hasher.finalize().into();
295
296    result
297}

299pub fn total_work_from_wt_tx(wt_tx: &Transaction) -> [u8; 16] {
300    let output = wt_tx.output[2].clone();
301    let mut instructions = output.script_pubkey.instructions();
302    if let Some(Ok(Instruction::Op(opcodes::all::OP_RETURN))) = instructions.next() {
303        if let Some(Ok(Instruction::PushBytes(data))) = instructions.next() {
304            let data_bytes = data.as_bytes();
305            let total_work: [u8; 16] = data_bytes[64..]
306                .try_into()
307                .expect("Expected total work data to be exactly 16 bytes long after OP_RETURN");
308            return total_work;
309        }
310        panic!("Expected OP_RETURN followed by data");
311    }
312    panic!("Expected OP_RETURN instruction in the transaction output script");
313}

315/// Convert a recursion VM seal (i.e. succinct receipt) into a JSON format compatible with the
316/// `stark_verify` witness generator. Taken from risc0-groth16 v2.3.2.
317/// This function will be removed once Risc Zero publishes a new version that exposes this function publicly.
318pub fn to_json<R: Read, W: Write>(mut reader: R, mut writer: W) -> Result<()> {
319    let mut iop = vec![0u32; K_SEAL_WORDS];
320    reader
321        .read_exact(bytemuck::cast_slice_mut(&mut iop))
322        .context("Failed to read seal data from reader")?;
323
324    writeln!(writer, "{{\n  \"iop\" : [").context("Failed to write JSON header")?;
325
326    let mut pos = 0;
327    for (index, seal_type) in K_SEAL_TYPES.iter().take(K_SEAL_ELEMS).enumerate() {
328        if pos != 0 {
329            writeln!(writer, ",").context("Failed to write JSON separator")?;
330        }
331        match seal_type {
332            IopType::Fp => {
333                let value = BabyBearElem::new_raw(iop[pos]).as_u32();
334                pos += 1;
335                writeln!(writer, "    \"{value}\"")
336                    .with_context(|| format!("Failed to write Fp value at index {index}"))?;
337            }
338            _ => {
339                if pos + DIGEST_WORDS > iop.len() {
340                    return Err(eyre!(
341                        "Not enough data for digest at position {}: need {} words, have {} remaining",
342                        pos,
343                        DIGEST_WORDS,
344                        iop.len() - pos
345                    ));
346                }
347                let digest = Risc0Digest::try_from(&iop[pos..pos + DIGEST_WORDS])
348                    .with_context(|| format!("Failed to create digest at position {pos}"))?;
349                let value = digest_to_decimal(&digest).with_context(|| {
350                    format!("Failed to convert digest to decimal at index {index}")
351                })?;
352                pos += 8;
353                writeln!(writer, "    \"{value}\"")
354                    .with_context(|| format!("Failed to write digest value at index {index}",))?;
355            }
356        }
357    }
358    write!(writer, "  ]\n}}").context("Failed to write JSON footer")?;
359
360    Ok(())
361}

363fn digest_to_decimal(digest: &Risc0Digest) -> Result<String> {
364    to_decimal(&format!("{:?}", digest_to_fr(digest)))
365        .ok_or_else(|| eyre!("Failed to convert digest to decimal format"))
366}
367
368fn to_decimal(s: &str) -> Option<String> {
369    s.strip_prefix("Fr(0x")
370        .and_then(|s| s.strip_suffix(')'))
371        .and_then(|stripped| BigUint::from_str_radix(stripped, 16).ok())
372        .map(|n| n.to_str_radix(10))
