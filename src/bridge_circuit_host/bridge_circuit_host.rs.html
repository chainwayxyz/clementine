68/// Generates a Groth16 proof for the Bridge Circuit after performing sanity checks.
69///
70/// This function first validates various conditions such as header chain output,
71/// light client proof, and SPV verification. It then constructs a succinct proof
72/// for the Bridge Circuit. Finally, it converts the succinct proof
73/// into a Groth16 proof using a Circom circuit.
74///
75/// # Arguments
76///
77/// * `bridge_circuit_host_params` - The host parameters containing circuit-related inputs.
78/// * `bridge_circuit_elf` - The compiled ELF binary representing the Bridge Circuit.
79///
80/// # Returns
81///
82/// Returns a Result containing a tuple consisting of:
83/// - `ark_groth16::Proof<Bn254>`: The final Groth16 proof.
84/// - `[u8; 31]`: The Groth16 output.
85/// - `BridgeCircuitBitvmInputs`: The structured inputs for the Bridge Circuit BitVM.
86///
87/// # Errors
88///
89/// This function will return an error if:
90/// - The number of watchtowers does not match expectations.
91/// - The header chain proof output differs from the expected value.
92/// - Light client proof verification fails.
93/// - SPV verification fails.
94/// - The journal hash does not match the expected hash.
95/// - Any serialization/deserialization operation fails.
96/// - The network is unsupported.
97/// - The execution environment cannot be built.
98/// - Proof generation fails.
99/// - Receipt journal conversion fails.
100/// - Computing the image ID fails.
101/// - Converting succinct receipt fails.
102/// - Converting groth16 seal to array fails.
103///
104pub fn prove_bridge_circuit(
105    bridge_circuit_host_params: BridgeCircuitHostParams,
106    bridge_circuit_elf: &[u8],
107) -> Result<(
108    ark_groth16::Proof<Bn254>,
109    [u8; 31],
110    BridgeCircuitBitvmInputs,
111)> {
112    tracing::info!("Starting bridge circuit proof generation");
113    let bridge_circuit_input = bridge_circuit_host_params
114        .clone()
115        .into_bridge_circuit_input();
116
117    let header_chain_proof_output_serialized = borsh::to_vec(&bridge_circuit_input.hcp)
118        .wrap_err("Could not serialize header chain output")?;
119
120    if bridge_circuit_input.lcp.lc_journal != bridge_circuit_host_params.lcp_receipt.journal.bytes {
121        return Err(eyre!("Light client proof output mismatch"));
122    }
123
124    tracing::debug!(target: "ci", "Watchtower challenges: {:?}",
125        bridge_circuit_input.watchtower_inputs);
126
127    let lc_image_id = match bridge_circuit_host_params.network.0 {
128        bitcoin::Network::Bitcoin => MAINNET_LC_IMAGE_ID,
129        bitcoin::Network::Testnet4 => TESTNET4_LC_IMAGE_ID,
130        bitcoin::Network::Signet => DEVNET_LC_IMAGE_ID,
131        bitcoin::Network::Regtest => REGTEST_LC_IMAGE_ID,
132        _ => return Err(eyre!("Unsupported network")),
133    };
134
135    let is_regtest = bridge_circuit_host_params.network.0 == bitcoin::Network::Regtest;
136
137    // Verify light client proof
138    if !is_regtest {
139        bridge_circuit_host_params
140            .lcp_receipt
141            .verify(lc_image_id)
142            .map_err(|_| eyre!("Light client proof verification failed"))?;
143    }
144
145    // Header chain verification
146    if header_chain_proof_output_serialized
147        != bridge_circuit_host_params.headerchain_receipt.journal.bytes
148    {
149        return Err(eyre!("Header chain proof output mismatch"));
150    }
151
152    let header_chain_method_id = match bridge_circuit_host_params.network.0 {
153        bitcoin::Network::Bitcoin => MAINNET_HEADER_CHAIN_METHOD_ID,
154        bitcoin::Network::Testnet4 => TESTNET4_HEADER_CHAIN_METHOD_ID,
155        bitcoin::Network::Signet => SIGNET_HEADER_CHAIN_METHOD_ID,
156        bitcoin::Network::Regtest => REGTEST_HEADER_CHAIN_METHOD_ID,
157        _ => return Err(eyre!("Unsupported network")),
158    };
159
160    // Check for headerchain receipt
161    if bridge_circuit_host_params
162        .headerchain_receipt
163        .verify(header_chain_method_id)
164        .is_err()
165    {
166        return Err(eyre!("Header chain receipt verification failed"));
167    }
168
169    // SPV verification
170    if !bridge_circuit_input.payout_spv.verify(
171        bridge_circuit_input
172            .hcp
173            .chain_state
174            .block_hashes_mmr
175            .clone(),
176    ) {
177        return Err(eyre!("SPV verification failed"));
178    }
179
180    // Make sure the L1 block hash of the LightClientCircuitOutput matches the payout tx block hash
181    let lc_output: LightClientCircuitOutput = borsh::from_slice(
182        bridge_circuit_host_params
183            .lcp_receipt
184            .journal
185            .bytes
186            .as_slice(),
187    )
188    .wrap_err("Failed to deserialize light client circuit output")?;
189
190    let lc_l1_block_hash = lc_output.latest_da_state.block_hash;
191
192    let spv_l1_block_hash = bridge_circuit_input
193        .payout_spv
194        .block_header
195        .compute_block_hash();
196
197    if lc_l1_block_hash != spv_l1_block_hash {
198        return Err(eyre!(
199            "L1 block hash mismatch: expected {:?}, got {:?}",
200            lc_l1_block_hash,
201            spv_l1_block_hash
202        ));
203    }
204
205    let public_inputs: SuccinctBridgeCircuitPublicInputs =
206        SuccinctBridgeCircuitPublicInputs::new(bridge_circuit_input.clone())?;
207
208    let journal_hash = public_inputs.host_journal_hash();
209
210    let mut binding = ExecutorEnv::builder();
211    let env = binding
212        .write_slice(
213            &borsh::to_vec(&bridge_circuit_input)
214                .wrap_err("Failed to serialize bridge circuit input")?,
215        )
216        .add_assumption(bridge_circuit_host_params.headerchain_receipt)
217        .add_assumption(bridge_circuit_host_params.lcp_receipt)
218        .build()
219        .map_err(|e| eyre!("Failed to build execution environment: {}", e))?;
220
221    let prover = default_prover();
222
223    tracing::info!("Checks complete, proving bridge circuit to generate STARK proof");
224
225    let succinct_receipt = prover
226        .prove_with_opts(env, bridge_circuit_elf, &ProverOpts::succinct())
227        .map_err(|e| eyre!("Failed to generate bridge circuit proof: {}", e))?
228        .receipt;
229
230    tracing::info!("Bridge circuit proof (STARK) generated");
231
232    let succinct_receipt_journal: [u8; 32] = succinct_receipt
233        .clone()
234        .journal
235        .bytes
236        .try_into()
237        .map_err(|_| eyre!("Failed to convert journal bytes to array"))?;
238
239    if *journal_hash.as_bytes() != succinct_receipt_journal {
240        return Err(eyre!("Journal hash mismatch"));
241    }
242
243    let bridge_circuit_method_id = compute_image_id(bridge_circuit_elf)
244        .map_err(|e| eyre!("Failed to compute bridge circuit image ID: {}", e))?;
245
246    let combined_method_id_constant =
247        calculate_succinct_output_prefix(bridge_circuit_method_id.as_bytes());
248
249    let (g16_proof, g16_output) = if is_dev_mode() {
250        stark_to_bitvm2_g16_dev_mode(succinct_receipt, &succinct_receipt_journal)?
251    } else {
252        stark_to_bitvm2_g16(
253            succinct_receipt
254                .inner
255                .succinct()
256                .wrap_err("Failed to get succinct receipt")?
257                .clone(),
258            &succinct_receipt_journal,
259        )?
260    };
261
262    tracing::info!("Bridge circuit proof (Groth16) generated");
263
264    let risc0_g16_seal_vec = g16_proof.to_vec();
265    let risc0_g16_256 = risc0_g16_seal_vec[0..256]
266        .try_into()
267        .wrap_err("Failed to convert groth16 seal to array")?;
268    let circuit_g16_proof = CircuitGroth16Proof::from_seal(risc0_g16_256);
269    let ark_groth16_proof: ark_groth16::Proof<Bn254> = circuit_g16_proof.into();
270
271    tracing::debug!(
272        target: "ci",
273        "Circuit debug info:\n\
274        - Combined method ID constant: {:?}\n\
275        - Payout tx block hash: {:?}\n\
276        - Latest block hash: {:?}\n\
277        - Challenge sending watchtowers: {:?}\n\
278        - Deposit constant: {:?}",
279        combined_method_id_constant,
280        public_inputs.payout_tx_block_hash.0,
281        public_inputs.latest_block_hash.0,
282        public_inputs.challenge_sending_watchtowers.0,
283        public_inputs.deposit_constant.0
284    );
285
286    Ok((
287        ark_groth16_proof,
288        g16_output,
289        BridgeCircuitBitvmInputs {
290            payout_tx_block_hash: public_inputs.payout_tx_block_hash.0,
291            latest_block_hash: public_inputs.latest_block_hash.0,
292            challenge_sending_watchtowers: public_inputs.challenge_sending_watchtowers.0,
293            deposit_constant: public_inputs.deposit_constant.0,
294            combined_method_id: combined_method_id_constant,
295        },
296    ))
297}

299/// Constructs an SPV (Simplified Payment Verification) proof.
300///
301/// This function processes block headers, constructs an MMR (Merkle Mountain Range)
302/// for block header commitment, and generates a Merkle proof for the payout transaction's
303/// inclusion in the block.
304///
305/// # Arguments
306///
307/// * `payout_tx` - The payout transaction to prove inclusion for.
308/// * `block_hash_bytes` - A slice of block hashes, each 32 bytes long.
309/// * `payment_block` - The block containing the payout transaction.
310/// * `payment_block_height` - The height of the payment block in the blockchain.
311/// * `genesis_block_height` - The height of the genesis block.
312/// * `payment_tx_index` - The index of the payout transaction in the block's transaction list.
313///
314/// # Returns
315///
316/// Returns a `Result<SPV>` containing:
317/// - The payout transaction wrapped in a `CircuitTransaction`.
318/// - A Merkle proof of the transaction's inclusion in the block.
319/// - The block header.
320/// - An MMR proof of the block header's inclusion in the MMR.
321///
322/// # Errors
323///
324/// This function will return an error if:
325/// - Input parameters are invalid or out of bounds.
326/// - MMR proof generation fails.
327/// - Merkle tree construction fails.
328/// - Payment block height is less than genesis block height.
329/// - Payment transaction index is out of bounds.
330///
331pub fn create_spv(
332    payout_tx: Transaction,
333    block_hash_bytes: &[[u8; 32]],
334    payment_block: bitcoin::Block,
335    payment_block_height: u32,
336    genesis_block_height: u32,
337    payment_tx_index: u32,
338) -> Result<SPV> {
339    // Input validation
340    if payment_block_height < genesis_block_height {
341        return Err(eyre!(
342            "Payment block height ({}) cannot be less than genesis block height ({})",
343            payment_block_height,
344            genesis_block_height
345        ));
346    }
347
348    if payment_tx_index as usize >= payment_block.txdata.len() {
349        return Err(eyre!(
350            "Payment transaction index ({}) out of bounds (block has {} transactions)",
351            payment_tx_index,
352            payment_block.txdata.len()
353        ));
354    }
355
356    let mut mmr_native = MMRNative::new();
357    for block_hash in block_hash_bytes {
358        mmr_native.append(*block_hash);
359    }
360
361    let block_txids: Vec<CircuitTransaction> = payment_block
362        .txdata
363        .iter()
364        .map(|tx| CircuitTransaction(tx.clone()))
365        .collect();
366
367    let mmr_inclusion_proof = mmr_native
368        .generate_proof(payment_block_height - genesis_block_height)
369        .wrap_err("Failed to generate MMR inclusion proof")?;
370
371    let block_mt = BitcoinMerkleTree::new_mid_state(&block_txids);
372
373    let payout_tx_proof = block_mt.generate_proof(payment_tx_index);
374
375    let spv = SPV::new(
376        CircuitTransaction(payout_tx.clone()),
377        payout_tx_proof.clone(),
378        payment_block.header.into(),
379        mmr_inclusion_proof.1.clone(),
380    );
381
382    let mmr_guest = mmr_native.to_guest_mmr();
383
384    // Verify the SPV proof to ensure correctness
385    if !spv.verify(mmr_guest) {
386        return Err(eyre!(
387            "SPV verification failed during creation, please check inputs"
388        ));
389    }
390
391    Ok(spv)
392}
