28/// Generates a Groth16 proof for the Bridge Circuit after performing sanity checks.
29///
30/// This function first validates various conditions such as header chain output,
31/// light client proof, and SPV verification. It then constructs a succinct proof
32/// for the Bridge Circuit. Finally, it converts the succinct proof
33/// into a Groth16 proof using a Circom circuit.
34///
35/// # Arguments
36///
37/// * `bridge_circuit_host_params` - The host parameters containing circuit-related inputs.
38/// * `bridge_circuit_elf` - The compiled ELF binary representing the Bridge Circuit.
39///
40/// # Returns
41///
42/// Returns a tuple consisting of:
43/// - `ark_groth16::Proof<Bn254>`: The final Groth16 proof.
44/// - `[u8; 31]`: The Groth16 output.
45/// - `BridgeCircuitBitvmInputs`: The structured inputs for the Bridge Circuit BitVM.
46///
47/// # Panics
48///
49/// This function will panic if:
50/// - The number of watchtowers does not match expectations.
51/// - The header chain proof output differs from the expected value.
52/// - Light client proof verification fails.
53/// - SPV verification fails.
54/// - The journal hash does not match the expected hash.
55///
56pub fn prove_bridge_circuit(
57    bridge_circuit_host_params: BridgeCircuitHostParams,
58    bridge_circuit_elf: &[u8],
59) -> (
60    ark_groth16::Proof<Bn254>,
61    [u8; 31],
62    BridgeCircuitBitvmInputs,
63) {
64    let bridge_circuit_input = bridge_circuit_host_params
65        .clone()
66        .into_bridge_circuit_input();
67
68    let header_chain_proof_output_serialized =
69        borsh::to_vec(&bridge_circuit_input.hcp).expect("Could not serialize header chain output");
70
71    if bridge_circuit_input.lcp.lc_journal != bridge_circuit_host_params.lcp_receipt.journal.bytes {
72        panic!("Light client proof output mismatch");
73    }
74
75    // if bridge_circuit_host_params.lcp_receipt.verify(LC_IMAGE_ID).is_err()
76    // {
77    //     panic!("Light client proof receipt verification failed");
78    // }
79
80    // Header chain verification
81    if header_chain_proof_output_serialized
82        != bridge_circuit_host_params.headerchain_receipt.journal.bytes
83    {
84        panic!("Header chain proof output mismatch");
85    }
86
87    let header_chain_method_id = match bridge_circuit_host_params.network {
88        bitcoin::Network::Bitcoin => MAINNET_HEADER_CHAIN_METHOD_ID,
89        bitcoin::Network::Testnet4 => TESTNET4_HEADER_CHAIN_METHOD_ID,
90        bitcoin::Network::Signet => SIGNET_HEADER_CHAIN_METHOD_ID,
91        bitcoin::Network::Regtest => REGTEST_HEADER_CHAIN_METHOD_ID,
92        _ => panic!("Unsupported network"),
93    };
94
95    // Check for headerchain receipt
96    if bridge_circuit_host_params
97        .headerchain_receipt
98        .verify(header_chain_method_id)
99        .is_err()
100    {
101        panic!("Header chain receipt verification failed");
102    }
103
104    // SPV verification
105    if !bridge_circuit_input.payout_spv.verify(
106        bridge_circuit_input
107            .hcp
108            .chain_state
109            .block_hashes_mmr
110            .clone(),
111    ) {
112        panic!("SPV verification failed");
113    }
114
115    let public_inputs: SuccinctBridgeCircuitPublicInputs =
116        SuccinctBridgeCircuitPublicInputs::new(bridge_circuit_input.clone());
117
118    let journal_hash = public_inputs.host_journal_hash();
119
120    let mut binding = ExecutorEnv::builder();
121    let env = binding.write_slice(&borsh::to_vec(&bridge_circuit_input).unwrap());
122    // let env = env.add_assumption(bridge_circuit_host_params.lcp_receipt);
123    let env = env.add_assumption(bridge_circuit_host_params.headerchain_receipt);
124    let env = env.build().unwrap();
125    let prover = default_prover();
126
127    tracing::info!("PROVING Bridge CIRCUIT");
128
129    let succinct_receipt = prover
130        .prove_with_opts(env, bridge_circuit_elf, &ProverOpts::succinct())
131        .unwrap()
132        .receipt;
133
134    let succinct_receipt_journal: [u8; 32] = succinct_receipt.journal.bytes.try_into().unwrap();
135
136    if *journal_hash.as_bytes() != succinct_receipt_journal {
137        panic!("Journal hash mismatch");
138    }
139
140    let bridge_circuit_method_id = compute_image_id(bridge_circuit_elf).unwrap();
141    let combined_method_id_constant =
142        calculate_succinct_output_prefix(bridge_circuit_method_id.as_bytes());
143    let (g16_proof, g16_output) = stark_to_snark(
144        succinct_receipt.inner.succinct().unwrap().clone(),
145        &succinct_receipt_journal,
146    );
147    let risc0_g16_seal_vec = g16_proof.to_vec();
148    let risc0_g16_256 = risc0_g16_seal_vec[0..256].try_into().unwrap();
149    let circuit_g16_proof = CircuitGroth16Proof::from_seal(risc0_g16_256);
150    let ark_groth16_proof: ark_groth16::Proof<Bn254> = circuit_g16_proof.into();
151
152    (
153        ark_groth16_proof,
154        g16_output,
155        BridgeCircuitBitvmInputs {
156            payout_tx_block_hash: public_inputs.payout_tx_block_hash.0,
157            latest_block_hash: public_inputs.latest_block_hash.0,
158            challenge_sending_watchtowers: public_inputs.challenge_sending_watchtowers.0,
159            deposit_constant: public_inputs.deposit_constant.0,
160            combined_method_id: combined_method_id_constant,
161        },
162    )
163}

165/// Constructs an SPV (Simplified Payment Verification) proof.
166///
167/// This function decodes a Bitcoin transaction, processes block headers,
168/// constructs an MMR (Merkle Mountain Range) for block header commitment,
169/// and generates a Merkle proof for the payout transaction's inclusion in
170/// the block.
171///
172/// # Arguments
173///
174/// * `payout_tx` - A mutable reference to a byte slice representing the payout transaction.
175/// * `headers` - A byte slice containing block headers, each 80 bytes long.
176/// * `payment_block` - A byte slice representing the full payment block.
177/// * `payment_block_height` - The height of the payment block in the blockchain.
178/// * `payment_tx_index` - The index of the payout transaction in the block's transaction list.
179///
180/// # Returns
181///
182/// Returns an `SPV` struct containing:
183/// - The decoded payout transaction.
184/// - A Merkle proof of the transaction's inclusion in the block.
185/// - The block header.
186/// - An MMR proof of the block header's inclusion in the MMR.
187///
188/// # Panics
189///
190/// This function will panic if:
191/// - Decoding `payout_tx` or `payment_block` fails.
192/// - Any block header chunk is not 80 bytes long.
193/// - Generating the Merkle or MMR proof fails.
194///
195pub fn create_spv(
196    payout_tx: Transaction,
197    block_hash_bytes: &[[u8; 32]],
198    payment_block: bitcoin::Block,
199    payment_block_height: u32,
200    payment_tx_index: u32,
201) -> SPV {
202    let mut mmr_native = MMRNative::new();
203    for block_hash in block_hash_bytes {
204        mmr_native.append(*block_hash);
205    }
206
207    let block_txids: Vec<[u8; 32]> = payment_block
208        .txdata
209        .iter()
210        .map(|tx| tx.compute_txid().as_raw_hash().to_byte_array())
211        .collect();
212
213    let mmr_inclusion_proof = mmr_native.generate_proof(payment_block_height);
214
215    let block_mt = BitcoinMerkleTree::new(block_txids);
216
217    let payout_tx_proof = block_mt.generate_proof(payment_tx_index);
218
219    SPV {
220        transaction: CircuitTransaction(payout_tx),
221        block_inclusion_proof: payout_tx_proof,
222        block_header: payment_block.header.into(),
223        mmr_inclusion_proof: mmr_inclusion_proof.1,
224    }
225}

227/// Generates a Groth16 proof of a bitcoin header chain proof where it only outputs the total work.
228///
229/// This function constructs an execution environment, serializes the provided
230/// input using Borsh, and utilizes a default prover to generate a proof using
231/// the Groth16 proving system.
232///
233/// # Arguments
234///
235/// * `receipt` - A header-chain `Receipt` that serves as an assumption for the proof.
236/// * `input` - A reference to `WorkOnlyCircuitInput` containing the necessary
237///   header chain output data.
238///
239/// # Returns
240///
241/// Returns a new `Receipt` containing the Groth16 proof result.
242///
243pub fn prove_work_only_header_chain_proof(
244    receipt: Receipt,
245    input: &WorkOnlyCircuitInput,
246) -> Receipt {
247    let env = ExecutorEnv::builder()
248        .add_assumption(receipt)
249        .write_slice(&borsh::to_vec(&input).unwrap())
250        .build()
251        .unwrap();
252    let prover = default_prover();
253    prover
254        .prove_with_opts(env, TESTNET4_WORK_ONLY_ELF, &ProverOpts::groth16())
255        .unwrap()
256        .receipt
257}
