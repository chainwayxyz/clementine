47/// Mutex to ensure only one docker operation runs at a time
48/// This prevents conflicts when RISC0_WORK_DIR is set and multiple functions run concurrently
49/// Also in tests where multiple verifiers/operators are running concurrently, this mutex protects the skopeo/udocker operations from conflicting.
50static DOCKER_MUTEX: std::sync::OnceLock<Mutex<()>> = std::sync::OnceLock::new();
51
52fn get_docker_mutex() -> &'static Mutex<()> {
53    DOCKER_MUTEX.get_or_init(|| Mutex::new(()))
54}

56/// Acquires the docker mutex.  
57///  
58/// This mutex guards a unit type (`Mutex<()>`), so it is used purely for mutual exclusion  
59/// and does not protect any shared in-memory data. For that reason, it is safe to ignore  
60/// poisoning and continue using `into_inner()` to recover the guard.  
61///  
62/// However, if a previous thread panicked while holding this mutex, that panic may have  
63/// occurred in the middle of docker / skopeo / udocker operations, potentially leaving  
64/// partially completed external operations (e.g. images, containers, or files on disk) in  
65/// an inconsistent state. We log such poison events for observability but still allow  
66/// subsequent callers to proceed.  
67fn acquire_docker_mutex() -> std::sync::MutexGuard<'static, ()> {
68    match get_docker_mutex().lock() {
69        Ok(guard) => guard,
70        Err(e) => {
71            tracing::warn!("Docker mutex was poisoned; proceeding by recovering the inner lock. Previous docker operation may have been interrupted by a panic.");
72            e.into_inner()
73        }
74    }
75}

77/// Ensures that both `udocker` and `skopeo` binaries are available on the system `PATH`.
78/// Returns an error with a descriptive message if either is missing or cannot be executed.
79fn ensure_udocker_and_skopeo_installed() -> Result<()> {
80    fn check_binary(binary: &str) -> Result<()> {
81        let output = std::process::Command::new(binary).arg("--version").output();
82
83        match output {
84            Ok(output) => {
85                if output.status.success() {
86                    Ok(())
87                } else {
88                    let stderr = String::from_utf8_lossy(&output.stderr);
89                    Err(eyre!(
90                        "`{}` appears to be installed, but `{} --version` failed: {}",
91                        binary,
92                        binary,
93                        stderr
94                    ))
95                }
96            }
97            Err(e) if e.kind() == std::io::ErrorKind::NotFound => Err(eyre!(
98                "`{}` is not installed or not found in PATH. Please install it before running the prover.",
99                binary
100            )),
101            Err(e) => Err(eyre!("Failed to execute `{}`: {}", binary, e)),
102        }
103    }
104
105    check_binary("udocker")?;
106    check_binary("skopeo")?;
107
108    Ok(())
109}

284    Ok((
285        proof_json
286            .try_into()
287            .map_err(|e| eyre!("Failed to convert proof JSON to Seal: {:?}", e))?,
288        output_bytes,
289    ))
290}

309/// Repackages a Docker/OCI image tarball by removing symlinks and copying the files directly to the locations where symlinks would be.  
310///  
311/// # Problem Solved  
312/// This function addresses a known issue with udocker (`udocker load`), where Docker images containing identical layers represented as symlinks can cause failures or incorrect behavior.  
313/// See: https://github.com/indigo-dc/udocker/issues/361  
314///  
315/// # Caching Strategy  
316/// - The function creates a modified tar file (with symlinks removed) in the same directory as the original, appending `_no_symlinks.tar` to the filename.  
317/// - If a cached version exists, its modification time is checked to ensure it is newer than the original.  
318/// - If the modification time is valid, the cached tar's digest is verified against the expected config digest.  
319/// - If digest verification fails, the cached file is deleted and recomputed.  
320/// - The original tar file is never modified; only the cached file is updated or recreated as needed.  
321///  
322/// # Security Considerations  
323/// - Symlink resolution is performed to ensure that files are copied directly, preventing potential symlink attacks or unintended file access.  
324/// - Digest verification is used to detect tampering or corruption of the cached tar file.  
325/// - If digest verification fails, the cache is deleted to prevent use of a potentially compromised file.  
326///  
327/// # Error Conditions  
328/// - Returns an error if the tar file name or parent directory is invalid.  
329/// - Returns an error if file metadata cannot be read or if file operations (copy, delete) fail.  
330/// - Returns an error if digest verification fails or if the cache cannot be recreated.  
331///  
332/// # Returns  
333/// Returns the path to the modified (no-symlinks) tar file.  
334fn remove_symlinks_from_image_tar(path: &Path, expected_config_digest: &str) -> Result<PathBuf> {
335    // Determine cache file path in the same directory as the original tar
336    let cache_file_name = path
337        .file_stem()
338        .and_then(|s| s.to_str())
339        .ok_or_else(|| eyre!("Invalid tar file name: {path:?}"))?
340        .to_string()
341        + "_no_symlinks.tar";
342    let cached_tar_path = path
343        .parent()
344        .ok_or_else(|| eyre!("Tar file has no parent directory: {path:?}"))?
345        .join(&cache_file_name);
346
347    // Check if cached version exists, check modification time first (cheap), then verify digest if needed
348    let use_cache = if cached_tar_path.exists() {
349        // First check modification time (cheap operation)
350        let cached_metadata = fs::metadata(&cached_tar_path).wrap_err(format!(
351            "Failed to get cached tar metadata: {cached_tar_path:?}"
352        ))?;
353        let original_metadata = fs::metadata(path)
354            .wrap_err(format!("Failed to get original tar metadata: {path:?}"))?;
355
356        // Compare modification times - only proceed with digest verification if cache is newer
357        let mtime_valid = match (cached_metadata.modified(), original_metadata.modified()) {
358            (Ok(cached_mtime), Ok(original_mtime)) => cached_mtime >= original_mtime,
359            _ => false,
360        };
361
362        if !mtime_valid {
363            tracing::debug!(
364                "Cached tar is older than original or modification time unavailable, will reprocess: {cached_tar_path:?}"
365            );
366            false
367        } else {
368            // Modification time is valid, now verify digest (expensive operation)
369            match verify_tar_image_digest(&cached_tar_path, expected_config_digest) {
370                Ok(()) => {
371                    tracing::debug!("Cached tar digest verified: {cached_tar_path:?}");
372                    tracing::debug!("Using cached no-symlinks tar file: {cached_tar_path:?}");
373                    true
374                }
375                Err(e) => {
376                    tracing::warn!(
377                        "Cached tar digest verification failed (file may be corrupted or tampered), will delete and recompute: {cached_tar_path:?}, error: {}",
378                        e
379                    );
380                    // Delete the invalid cached file
381                    fs::remove_file(&cached_tar_path).wrap_err(format!(
382                        "Failed to delete invalid cached tar: {cached_tar_path:?}"
383                    ))?;
384                    false
385                }
386            }
387        }
388    } else {
389        tracing::debug!("No cached tar file found, will process: {cached_tar_path:?}");
390        false
391    };
392
393    // If cache is valid, return it directly
394    if use_cache {
395        return Ok(cached_tar_path);
396    }
397
398    // Create a temporary directory for extracting the tar file
399    let tmp_dir = tempdir().wrap_err(format!(
400        "Failed to create temporary directory for processing tar file: {path:?}"
401    ))?;
402    let tmp_path = tmp_dir.path();
403    let extracted_path = tmp_path.join("extracted");
404
405    // Extract tarball to a subfolder within temp directory
406    fs::create_dir(&extracted_path).wrap_err(format!(
407        "Failed to create extracted subdirectory: {extracted_path:?}"
408    ))?;
409    // 1. Extract the original tar
410    let file = fs::File::open(path)?;
411    let mut archive = Archive::new(file);
412    archive.unpack(&extracted_path)?;
413
414    // 2. Repack with symlink dereferencing
415    let out_file = fs::File::create(&cached_tar_path).wrap_err(format!(
416        "Failed to create output tar file: {cached_tar_path:?}"
417    ))?;
418    let mut builder = Builder::new(out_file);
419    builder.follow_symlinks(true);
420    builder
421        .append_dir_all(".", &extracted_path)
422        .wrap_err(format!(
423            "Failed to repack extracted directory into tar: {extracted_path:?}"
424        ))?;
425    builder
426        .finish()
427        .wrap_err(format!("Failed to finalize tar file: {cached_tar_path:?}"))?;
428
429    tracing::debug!("Successfully created flattened tar: {:?}", cached_tar_path);
430    Ok(cached_tar_path)
431}

440/// Reads the config digest from a docker-archive tar using `skopeo inspect --raw`.
441fn inspect_tar_image_config_digest(tar_path: &Path) -> Result<String> {
442    let inspect_output = Command::new("skopeo")
443        .arg("inspect")
444        .arg("--raw")
445        .arg("--no-creds")
446        .arg(format!("docker-archive:{}", tar_path.to_string_lossy()))
447        .output()
448        .wrap_err("skopeo inspect could not be executed")?;
449
450    if !inspect_output.status.success() {
451        let stderr = String::from_utf8_lossy(&inspect_output.stderr);
452        return Err(eyre!("Failed to inspect docker-archive image: {}", stderr));
453    }
454
455    let manifest_json =
456        String::from_utf8(inspect_output.stdout).wrap_err("Failed to parse manifest as UTF-8")?;
457    let manifest: serde_json::Value =
458        serde_json::from_str(&manifest_json).wrap_err("Failed to parse manifest JSON")?;
459
460    // For docker-archive format, the config digest is in manifest.config.digest
461    let config_digest = manifest
462        .get("config")
463        .and_then(|c| c.get("digest"))
464        .and_then(|d| d.as_str())
465        .ok_or_else(|| eyre!("Failed to get config digest from docker-archive manifest"))?;
466
467    Ok(config_digest.to_string())
468}

475) -> Result<()> {

477
478    if computed_config_digest != expected_config_digest {

481            computed_config_digest,
482            expected_config_digest
483        ));
484    }
485
486    tracing::debug!(
487        "Tar file digest verification (inspect only) successful: {:?}",
488        tar_path
489    );
490    Ok(())
491}

493/// Verifies that a docker-archive tar file has the expected config digest by
494/// forcing skopeo to reconstruct the archive first.
495///
496/// # Arguments
497/// * `tar_path` - Path to the docker-archive tar file to verify
498/// * `expected_config_digest` - The expected config digest (e.g., "sha256:...")
499///
500/// # Returns
501/// * `Ok(())` if the computed digest matches the expected digest
502/// * `Err` if verification fails or digests don't match
503pub fn verify_tar_image_digest(tar_path: &Path, expected_config_digest: &str) -> Result<()> {
504    let tmp_dir = tempdir().wrap_err("Failed to create temporary directory for verification")?;
505    let tmp_tar_path = tmp_dir.path().join("verified.tar");
506
507    // Copy from docker-archive to docker-archive - this forces skopeo to recompute all digests
508    // because it reconstructs the image structure. If the tar or manifest is tampered, this will fail.
509    tracing::debug!(
510        "Verifying tar file digest by copying to temporary docker-archive: {:?}",
511        tar_path
512    );
513    let copy_output = Command::new("skopeo")
514        .arg("copy")
515        .arg(format!("docker-archive:{}", tar_path.to_string_lossy()))
516        .arg(format!("docker-archive:{}", tmp_tar_path.to_string_lossy()))
517        .output()
518        .wrap_err("skopeo copy could not be executed for verification")?;
519
520    if !copy_output.status.success() {
521        let stderr = String::from_utf8_lossy(&copy_output.stderr);
522        return Err(eyre!(
523            "skopeo copy failed during tar verification (tar may be corrupted or tampered): {}",
524            stderr
525        ));
526    }
527
528    verify_tar_image_digest_inspect_only(&tmp_tar_path, expected_config_digest)?;
529
530    tracing::debug!("Tar file digest verification successful: {:?}", tar_path);
531    Ok(())
532}

534/// Pulls the image or loads the image from the cache if it exists. It ensures the config digest matches the given digest, or repulls the image if it doesn't. Returns the path to the modified tar file which deleted the symlinks.
535fn pull_or_load_image(
536    image_digest: &str,
537    container_name: &str,
538    image_config_digest: &str,
539) -> Result<PathBuf> {
540    let home_dir = std::env::home_dir().ok_or_else(|| eyre!("Failed to get HOME directory"))?;
541    let images_dir = home_dir.join(".clementine").join(IMAGES_SUBDIR);
542    fs::create_dir_all(&images_dir).wrap_err(format!(
543        "Failed to create images directory for container tarballs: {images_dir:?}"
544    ))?;
545    let tar_file_path = images_dir.join(format!("{container_name}.tar"));
546
547    // 2. Verify local tar digest if it exists
548    let mut need_pull = true;
549    if tar_file_path.exists() {
550        match verify_tar_image_digest(&tar_file_path, image_config_digest) {
551            Ok(()) => {
552                tracing::info!("Local tar config digest verified, skipping pull");
553                need_pull = false;
554            }
555            Err(e) => {
556                tracing::warn!(
557                    "Local tar config digest verification failed for {container_name} in path {tar_file_path:?} (file may be corrupted or tampered), will delete and re-pull: {e}",
558                );
559                fs::remove_file(&tar_file_path).wrap_err("Failed to delete local tar")?;
560            }
561        }
562    }
563
564    // 3. Pull with skopeo if needed
565    if need_pull {
566        tracing::info!("Pulling image via skopeo...");
567        let pull_output = Command::new("skopeo")
568            .arg("copy")
569            .arg(format!("docker://{image_digest}"))
570            .arg("--src-no-creds")
571            .arg(format!(
572                "docker-archive:{}",
573                tar_file_path.to_string_lossy()
574            ))
575            .output()
576            .wrap_err("skopeo copy could not be executed")?;
577        if !pull_output.status.success() {
578            return Err(command_output_error(&pull_output, "skopeo copy"));
579        }
580        // Check if config digest matches the expected digest.
581        verify_tar_image_digest_inspect_only(&tar_file_path, image_config_digest)
582            .wrap_err("Newly pulled image failed digest verification")?;
583    }
584
585    let modified_tar_path = remove_symlinks_from_image_tar(&tar_file_path, image_config_digest)?;
586
587    // Note: Verification is already done inside remove_symlinks_from_image_tar for cached files,
588    // and the newly created file should match since it's derived from the verified original.
589    // However, we verify again here as a final safety check.
590    verify_tar_image_digest(&modified_tar_path, image_config_digest)
591        .wrap_err("Symlink-removed tar failed digest verification")?;
592
593    Ok(modified_tar_path)
594}

623/// Runs the prover container.
624/// skopeo is used to pull images instead of udocker pull because udocker pull doesn't support pulling from a sha digest.
625/// udocker is used instead of docker itself because it requires docker-in-docker to be set up if entities are run with docker.
626/// udocker commands are ran with --allow-root to avoid issues if the container is run as root. If it isn't run as root, --allow-root does nothing.
627fn run_prover_container(
628    image_digest: &str,
629    container_name: &str,
630    image_config_digest: &str,
631    work_dir: &Path,
632) -> Result<()> {
633    let modified_tar_path = pull_or_load_image(image_digest, container_name, image_config_digest)?;
634    // Load tar into udocker
635    let load_output = Command::new("udocker")
636        .arg("--allow-root")
637        .arg("load")
638        .arg("-i")
639        .arg(&modified_tar_path)
640        .output()
641        .wrap_err("udocker load could not be executed")?;
642    if !load_output.status.success() {
643        return Err(command_output_error(
644            &load_output,
645            format!("udocker load -i {modified_tar_path:?}").as_str(),
646        ));
647    }
648
649    // udocker load returns the image id of the loaded image on the last line of the stdout.
650    // also this output is different every time udocker load is run (even if the same .tar is used), so we need to parse the output to get the image id.
651    let output_str = String::from_utf8(load_output.stdout)
652        .wrap_err("Failed to parse udocker load stdout as UTF-8")?;
653    tracing::debug!("udocker load stdout: {:?}", output_str);
654    let udocker_image_id = output_str
655        .lines()
656        .last()
657        .ok_or_else(|| eyre!("No output lines from udocker load"))?
658        .trim_matches(&['[', ']', '\'', ' '][..])
659        .to_string();
660    tracing::debug!("Loaded udocker image id for {container_name}: {udocker_image_id}");
661
662    struct UdockerCleanupGuard {
663        container_name: String,
664        image_id: String,
665    }
666    impl Drop for UdockerCleanupGuard {
667        fn drop(&mut self) {
668            // Note: If udocker run --rm succeeds, the container may already be removed,
669            // but udocker rm will fail harmlessly (errors are ignored here).
670            // Also, container may not exist if udocker create fails for some reason, but this Drop impl also cleans up the image.
671            let _ = Command::new("udocker")
672                .arg("--allow-root")
673                .arg("rm")
674                .arg(&self.container_name)
675                .output();
676
677            // every udocker load creates a new container image (even if the same .tar is used), if we don't clean up the old container image, it will accumulate and consume disk space.
678            let _ = Command::new("udocker")
679                .arg("--allow-root")
680                .arg("rmi")
681                .arg(&self.image_id)
682                .output();
683
684            tracing::debug!(
685                "Cleaned up udocker container {} and image {}",
686                self.container_name,
687                self.image_id
688            );
689        }
690    }
691    let _udocker_guard = UdockerCleanupGuard {
692        container_name: container_name.to_string(),
693        image_id: udocker_image_id.clone(),
694    };
695
696    // Remove any stale container with the same name before creating a new one.
697    // This is to prevent stale containers being used by previous versions. There might be better ways so that we don't do this if it's not necessary.
698    let rm_output = Command::new("udocker")
699        .arg("--allow-root")
700        .arg("rm")
701        .arg(container_name)
702        .output()
703        .wrap_err("udocker rm could not be executed")?;
704
705    if !rm_output.status.success() {
706        let stderr = String::from_utf8_lossy(&rm_output.stderr);
707        // Ignore error if container doesn't exist (we're just cleaning up stale containers)
708        if !stderr.contains("invalid container id") && !stderr.contains("container not found") {
709            return Err(eyre!("udocker rm {container_name} failed: {stderr}"));
710        }
711    }
712
713    // Create the container using udocker create
714    let create_output = Command::new("udocker")
715        .arg("--allow-root")
716        .arg("create")
717        .arg(format!("--name={container_name}"))
718        .arg(&udocker_image_id)
719        .output()
720        .wrap_err("udocker create could not be executed")?;
721    if !create_output.status.success() {
722        let stderr = String::from_utf8_lossy(&create_output.stderr);
723        if stderr.contains("container name already exists") {
724            tracing::warn!(
725                "udocker create: container name '{}' already exists despite pre-cleanup. Error: {}",
726                container_name,
727                stderr
728            );
729        } else {
730            return Err(eyre!(
731                "udocker create failed for {container_name}: {stderr}"
732            ));
733        }
734    }
735
736    // Setup container with P2 exec mode (doesn't require user namespaces or root privileges)
737    let setup_output = Command::new("udocker")
738        .arg("--allow-root")
739        .arg("setup")
740        .arg("--execmode=P2")
741        .arg(container_name)
742        .output()
743        .wrap_err("udocker setup could not be executed")?;
744    if !setup_output.status.success() {
745        return Err(command_output_error(
746            &setup_output,
747            format!("udocker setup {container_name}").as_str(),
748        ));
749    }
750
751    // Run container with volume mount
752    let run_output = Command::new("udocker")
753        .arg("--allow-root")
754        .arg("run")
755        .arg("--rm")
756        .arg("-v")
757        .arg(format!("{}:/mnt", work_dir.to_string_lossy()))
758        .arg(container_name)
759        .output()
760        .wrap_err("udocker run could not be executed")?;
761    if !run_output.status.success() {
762        return Err(command_output_error(
763            &run_output,
764            format!("udocker run {container_name}").as_str(),
765        ));
766    }
767
768    Ok(())
769}

964        .try_into()
965        .wrap_err("Failed to convert succinct control root to 32 bytes")?;
966    succinct_control_root_bytes.reverse();
967    let succinct_control_root_bytes: String = succinct_control_root_bytes.encode_hex();
968    let a1_str = succinct_control_root_bytes[0..32].to_string();
969    let a0_str = succinct_control_root_bytes[32..64].to_string();
970    tracing::debug!("Succinct control root a0: {:?}", a0_str);
971    tracing::debug!("Succinct control root a1: {:?}", a1_str);
972    let a0_dec = to_decimal(&a0_str)
973        .ok_or_else(|| eyre!("Failed to convert succinct control root a0 to decimal"))?;
974    let a1_dec = to_decimal(&a1_str)
975        .ok_or_else(|| eyre!("Failed to convert succinct control root a1 to decimal"))?;
976    tracing::debug!("Succinct control root a0 dec: {:?}", a0_dec);
977    tracing::debug!("Succinct control root a1 dec: {:?}", a1_dec);
978
979    let id_bn254_fr_bits: Vec<String> = ID_BN254_FR_BITS_DEV_BRIDGE
980        .iter()
981        .map(|&bit| bit.to_string())
982        .collect();
983
984    let mut seal_json: Value = serde_json::json!({});
985
986    seal_json["journal_digest_bits"] = journal_bits.into();
987    seal_json["pre_state_digest_bits"] = pre_state_digest_bits.into();
988    seal_json["post_state_digest_bits"] = post_state_digest_bits.into();
989    seal_json["id_bn254_fr_bits"] = id_bn254_fr_bits.into();

992        seal_path,
993        serde_json::to_string_pretty(&seal_json)
994            .wrap_err("Failed to convert seal JSON to string")?,
995    )
996    .wrap_err("Failed to write seal file")?;
997
998    run_prover_container(
999        DEV_STARK_TO_BITVM2_IMAGE_DIGEST,
1000        DEV_STARK_TO_BITVM2_CONTAINER_NAME,

1012        serde_json::from_str(&proof_content).wrap_err("Failed to parse proof JSON")?;

1015    let parsed_json: Value =
1016        serde_json::from_str(&output_content_dec).wrap_err("Failed to parse output JSON")?;
1017    let output_str = parsed_json[0]
1018        .as_str()
1019        .wrap_err("Failed to get output string from JSON")?; // Extracts the string from the JSON array
1020
1021    // Step 2: Convert the decimal string to BigUint and then to hexadecimal
1022    let output_content_hex = BigUint::from_str_radix(output_str, 10)
1023        .wrap_err("Failed to parse decimal string")?
1024        .to_str_radix(16);
1025
1026    // If the length of the hexadecimal string is odd, add a leading zero
1027    let output_content_hex = if output_content_hex.len() % 2 == 0 {
1028        output_content_hex
1029    } else {
1030        format!("0{output_content_hex}")
1031    };
1032
1033    // Step 3: Decode the hexadecimal string to a byte vector
1034    let output_byte_vec =
1035        hex::decode(&output_content_hex).wrap_err("Failed to decode hex string")?;
1036    // Create our target 31-byte array, initialized to all zeros.
1037    let mut output_bytes = [0u8; 31];
1038
1039    // Calculate the starting position in the destination array.
1040    // This ensures the bytes are right-aligned, effectively padding with leading zeros.
1041    let start_index = 31 - output_byte_vec.len();
1042
1043    // Copy the decoded bytes from the vector into the correct slice of the array.
1044    output_bytes[start_index..].copy_from_slice(&output_byte_vec);
1045    Ok((
1046        proof_json
1047            .try_into()
1048            .map_err(|e| eyre!("Failed to convert proof JSON to Seal: {:?}", e))?,
1049        output_bytes,
1050    ))
1051}

1199    /// Collect all paths under `root`, returned as sorted relative paths.
1200    fn collect_relative_paths(
1201        root: &Path,
1202    ) -> std::result::Result<Vec<PathBuf>, Box<dyn std::error::Error>> {
1203        fn walk(
1204            root: &Path,
1205            current: &Path,
1206            out: &mut Vec<PathBuf>,
1207        ) -> std::result::Result<(), Box<dyn std::error::Error>> {
1208            for entry in fs::read_dir(current)? {
1209                let entry = entry?;
1210                let path = entry.path();
1211                let relative = path.strip_prefix(root)?.to_path_buf();
1212                out.push(relative);
1213
1214                if path.is_dir() {
1215                    walk(root, &path, out)?;
1216                }
1217            }
1218            Ok(())
1219        }
1220
1221        let mut paths = Vec::new();
1222        walk(root, root, &mut paths)?;
1223        paths.sort();
1224        Ok(paths)
1225    }
