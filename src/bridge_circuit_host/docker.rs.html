24/// Convert a STARK proof to a SNARK proof. Taken from risc0-groth16 and modified slightly.
25pub fn stark_to_bitvm2_g16(
26    succinct_receipt: SuccinctReceipt<ReceiptClaim>,
27    journal: &[u8],
28) -> Result<(Seal, [u8; 31])> {
29    let ident_receipt = risc0_zkvm::recursion::identity_p254(&succinct_receipt)
30        .map_err(|e| eyre!("Failed to create identity receipt: {:?}", e))?;
31    let identity_p254_seal_bytes = ident_receipt.get_seal_bytes();
32    let receipt_claim = succinct_receipt
33        .claim
34        .value()
35        .wrap_err("Failed to get receipt claim value")?;
36    tracing::debug!("Journal for stark_to_bitvm2_g16: {:?}", journal);
37
38    // This part is from risc0-groth16
39    if !is_x86_architecture() {
40        return Err(eyre!(
41            "stark_to_snark is only supported on x86 architecture"
42        ));
43    }
44    if !is_docker_installed() {
45        return Err(eyre!("Please install docker first")); // Maybe check this at startup...
46    }
47
48    let tmp_dir = tempdir().wrap_err("Failed to create temporary directory")?;
49    let work_var = std::env::var("RISC0_WORK_DIR").ok();
50    let work_dir = work_var.as_ref().map(Path::new).unwrap_or(tmp_dir.path());
51    tracing::debug!("work_dir: {:?}", work_dir);
52
53    std::fs::write(work_dir.join("seal.r0"), identity_p254_seal_bytes.clone())
54        .wrap_err("Failed to write seal file")?;
55    let seal_path = work_dir.join("input.json");
56    let proof_path = work_dir.join("proof.json");
57    let output_path = work_dir.join("public.json");
58    let mut seal_json = Vec::new();
59    to_json(&*identity_p254_seal_bytes, &mut seal_json)
60        .map_err(|e| eyre!("Failed to convert seal to JSON: {:?}", e))?;
61    std::fs::write(seal_path.clone(), seal_json).wrap_err("Failed to write seal JSON")?;
62
63    let pre_state: risc0_zkvm::MaybePruned<SystemState> = receipt_claim.clone().pre;
64    tracing::debug!("pre_state: {:?}", pre_state);
65    let pre_state_digest: Digest = pre_state.clone().digest();
66    tracing::debug!("pre_state_digest: {:?}", pre_state_digest);
67    let pre_state_digest_bits: Vec<String> = pre_state_digest
68        .as_bytes()
69        .iter()
70        .flat_map(|&byte| (0..8).rev().map(move |i| ((byte >> i) & 1).to_string()))
71        .collect();
72    tracing::debug!("pre_state_digest_bits: {:?}", pre_state_digest_bits);
73    let post_state: risc0_zkvm::MaybePruned<SystemState> = receipt_claim.clone().post;
74    tracing::debug!("post_state: {:?}", post_state);
75    let post_state_digest: Digest = post_state.clone().digest();
76    let post_state_digest_bits: Vec<String> = post_state_digest
77        .as_bytes()
78        .iter()
79        .flat_map(|&byte| (0..8).rev().map(move |i| ((byte >> i) & 1).to_string()))
80        .collect();
81    tracing::debug!("post_state_digest_bits: {:?}", post_state_digest_bits);
82
83    let mut journal_bits = Vec::new();
84    for byte in journal {
85        for i in 0..8 {
86            journal_bits.push((byte >> (7 - i)) & 1);
87        }
88    }
89    tracing::debug!("journal_bits len: {:?}", journal_bits.len());
90
91    let succinct_verifier_params = SuccinctReceiptVerifierParameters::default();
92    tracing::debug!("Succinct verifier params: {:?}", succinct_verifier_params);
93    let succinct_control_root = succinct_verifier_params.control_root;
94    tracing::debug!("Succinct control root: {:?}", succinct_control_root);
95    let mut succinct_control_root_bytes: [u8; 32] = succinct_control_root
96        .as_bytes()
97        .try_into()
98        .wrap_err("Failed to convert succinct control root to 32 bytes")?;
99    succinct_control_root_bytes.reverse();
100    let succinct_control_root_bytes: String = succinct_control_root_bytes.encode_hex();
101    let a1_str = succinct_control_root_bytes[0..32].to_string();
102    let a0_str = succinct_control_root_bytes[32..64].to_string();
103    tracing::debug!("Succinct control root a0: {:?}", a0_str);
104    tracing::debug!("Succinct control root a1: {:?}", a1_str);
105    let a0_dec = to_decimal(&a0_str)
106        .ok_or_else(|| eyre!("Failed to convert succinct control root a0 to decimal"))?;
107    let a1_dec = to_decimal(&a1_str)
108        .ok_or_else(|| eyre!("Failed to convert succinct control root a1 to decimal"))?;
109    tracing::debug!("Succinct control root a0 dec: {:?}", a0_dec);
110    tracing::debug!("Succinct control root a1 dec: {:?}", a1_dec);
111    tracing::debug!("CONTROL_ID: {:?}", ident_receipt.control_id);
112    let mut id_bn254_fr_bits: Vec<String> = ident_receipt
113        .control_id
114        .as_bytes()
115        .iter()
116        .flat_map(|&byte| (0..8).rev().map(move |i| ((byte >> i) & 1).to_string()))
117        .collect();
118    tracing::debug!("id_bn254_fr_bits: {:?}", id_bn254_fr_bits);
119
120    // remove 248th and 249th bits
121    id_bn254_fr_bits.remove(248);
122    id_bn254_fr_bits.remove(248);
123
124    tracing::debug!(
125        "id_bn254_fr_bits after removing 2 extra bits: {:?}",
126        id_bn254_fr_bits
127    );
128
129    let mut seal_json: Value = {
130        let file_content = fs::read_to_string(&seal_path).wrap_err("Failed to read seal file")?;
131        serde_json::from_str(&file_content).wrap_err("Failed to parse seal JSON")?
132    };
133
134    seal_json["journal_digest_bits"] = journal_bits.into();
135    seal_json["pre_state_digest_bits"] = pre_state_digest_bits.into();
136    seal_json["post_state_digest_bits"] = post_state_digest_bits.into();
137    seal_json["id_bn254_fr_bits"] = id_bn254_fr_bits.into();
138    seal_json["control_root"] = vec![a0_dec, a1_dec].into();
139    std::fs::write(
140        seal_path,
141        serde_json::to_string_pretty(&seal_json).wrap_err("Failed to write updated seal JSON")?,
142    )
143    .wrap_err("Failed to write seal file")?;
144
145    let output = Command::new("docker")
146        .arg("run")
147        .arg("--pull=always")
148        .arg("--rm")
149        .arg("--platform=linux/amd64") // Force linux/amd64 platform
150        .arg("-v")
151        .arg(format!("{}:/mnt", work_dir.to_string_lossy()))
152        .arg("ozancw/risc0-to-bitvm2-groth16-prover@sha256:840b833190a86c87eb71473cd72bcaa3e3b90cefa9bf4ae785ab2de61bc9d262")
153        .stdout(Stdio::piped())
154        .stderr(Stdio::piped())
155        .output()
156        .wrap_err("Failed to execute docker command")?;
157
158    if !output.status.success() {
159        return Err(eyre!(
160            "STARK to SNARK prover docker image returned failure: {:?}",
161            output
162        ));
163    }
164
165    tracing::debug!("proof_path: {:?}", proof_path);
166    let proof_content =
167        std::fs::read_to_string(proof_path).wrap_err("Failed to read proof file")?;
168    let output_content_dec =
169        std::fs::read_to_string(output_path).wrap_err("Failed to read output file")?;
170    let proof_json: ProofJson =
171        serde_json::from_str(&proof_content).wrap_err("Failed to parse proof JSON")?;
172
173    let parsed_json: Value =
174        serde_json::from_str(&output_content_dec).wrap_err("Failed to parse output JSON")?;
175    let output_str = parsed_json[0]
176        .as_str()
177        .ok_or_else(|| eyre!("Failed to get output string from JSON"))?;
178
179    // Step 2: Convert the decimal string to BigUint and then to hexadecimal
180    let output_content_hex = BigUint::from_str_radix(output_str, 10)
181        .wrap_err("Failed to parse decimal string")?
182        .to_str_radix(16);
183
184    // If the length of the hexadecimal string is odd, add a leading zero
185    let output_content_hex = if output_content_hex.len() % 2 == 0 {
186        output_content_hex
187    } else {
188        format!("0{output_content_hex}")
189    };
190
191    // Step 3: Decode the hexadecimal string to a byte vector
192    let output_byte_vec =
193        hex::decode(&output_content_hex).wrap_err("Failed to decode hex string")?;
194    let output_bytes: [u8; 31] = output_byte_vec
195        .as_slice()
196        .try_into()
197        .wrap_err("Failed to convert output bytes to array")?;
198
199    Ok((
200        proof_json
201            .try_into()
202            .map_err(|e| eyre!("Failed to convert proof JSON to Seal: {:?}", e))?,
203        output_bytes,
204    ))
205}

354    );
355    let inner_receipt = InnerReceipt::Groth16(g16_receipt);
356    Ok(Receipt::new(inner_receipt, journal.to_vec()))
357}

419        .iter()
420        .flat_map(|&byte| (0..8).rev().map(move |i| ((byte >> i) & 1).to_string()))
421        .collect();
422    tracing::debug!("post_state_digest_bits: {:?}", post_state_digest_bits);
423
424    let mut journal_bits = Vec::new();
425    for byte in journal {
426        for i in 0..8 {
427            journal_bits.push((byte >> (7 - i)) & 1);
428        }
429    }
430    tracing::debug!("journal_bits len: {:?}", journal_bits.len());
431
432    let succinct_verifier_params = SuccinctReceiptVerifierParameters::default();
433    tracing::debug!("Succinct verifier params: {:?}", succinct_verifier_params);
434    let succinct_control_root = succinct_verifier_params.control_root;
435    tracing::debug!("Succinct control root: {:?}", succinct_control_root);
436    let mut succinct_control_root_bytes: [u8; 32] = succinct_control_root
437        .as_bytes()
438        .try_into()
439        .wrap_err("Failed to convert succinct control root to 32 bytes")?;
440    succinct_control_root_bytes.reverse();
441    let succinct_control_root_bytes: String = succinct_control_root_bytes.encode_hex();
442    let a1_str = succinct_control_root_bytes[0..32].to_string();
443    let a0_str = succinct_control_root_bytes[32..64].to_string();
444    tracing::debug!("Succinct control root a0: {:?}", a0_str);
445    tracing::debug!("Succinct control root a1: {:?}", a1_str);
446    let a0_dec = to_decimal(&a0_str)
447        .ok_or_else(|| eyre!("Failed to convert succinct control root a0 to decimal"))?;
448    let a1_dec = to_decimal(&a1_str)
449        .ok_or_else(|| eyre!("Failed to convert succinct control root a1 to decimal"))?;
450    tracing::debug!("Succinct control root a0 dec: {:?}", a0_dec);
451    tracing::debug!("Succinct control root a1 dec: {:?}", a1_dec);
452
453    let id_bn254_fr_bits: Vec<String> = ID_BN254_FR_BITS_DEV_BRIDGE
454        .iter()
455        .map(|&bit| bit.to_string())
456        .collect();
457
458    let mut seal_json: Value = serde_json::json!({});
459
460    seal_json["journal_digest_bits"] = journal_bits.into();
461    seal_json["pre_state_digest_bits"] = pre_state_digest_bits.into();
462    seal_json["post_state_digest_bits"] = post_state_digest_bits.into();
463    seal_json["id_bn254_fr_bits"] = id_bn254_fr_bits.into();
464    seal_json["control_root"] = vec![a0_dec, a1_dec].into();
465    std::fs::write(
466        seal_path,
467        serde_json::to_string_pretty(&seal_json)
468            .wrap_err("Failed to convert seal JSON to string")?,
469    )
470    .wrap_err("Failed to write seal file")?;
471
472    let output = Command::new("docker")
473        .arg("run")
474        .arg("--pull=always")
475        .arg("--rm")
476        .arg("--platform=linux/amd64") // Force linux/amd64 platform
477        .arg("-v")
478        .arg(format!("{}:/mnt", work_dir.to_string_lossy()))
479        .arg("ozancw/dev-risc0-to-bitvm2-groth16-prover@sha256:9f1d8515b9c44a1280979bbcab327ec36041fae6dd0c4923997f084605f9f9e7")
480        .stdout(Stdio::piped())
481        .stderr(Stdio::piped())
482        .output()
483        .wrap_err("Failed to execute docker command")?;
484
485    if !output.status.success() {
486        return Err(eyre!(
487            "STARK to SNARK prover docker image returned failure: {:?}",
488            output
489        ));
490    }
491
492    tracing::debug!("proof_path: {:?}", proof_path);
493    let proof_content =
494        std::fs::read_to_string(proof_path).wrap_err("Failed to read proof file")?;
495    let output_content_dec =
496        std::fs::read_to_string(output_path).wrap_err("Failed to read output file")?;
497    tracing::debug!("output content: {:?}", output_content_dec);
498    let proof_json: ProofJson =
499        serde_json::from_str(&proof_content).wrap_err("Failed to parse proof JSON")?;
500
501    // Convert output_content_dec from decimal to hex
502    let parsed_json: Value =
503        serde_json::from_str(&output_content_dec).wrap_err("Failed to parse output JSON")?;
504    let output_str = parsed_json[0]
505        .as_str()
506        .wrap_err("Failed to get output string from JSON")?; // Extracts the string from the JSON array
507
508    // Step 2: Convert the decimal string to BigUint and then to hexadecimal
509    let output_content_hex = BigUint::from_str_radix(output_str, 10)
510        .wrap_err("Failed to parse decimal string")?
511        .to_str_radix(16);
512
513    // If the length of the hexadecimal string is odd, add a leading zero
514    let output_content_hex = if output_content_hex.len() % 2 == 0 {
515        output_content_hex
516    } else {
517        format!("0{output_content_hex}")
518    };
519
520    // Step 3: Decode the hexadecimal string to a byte vector
521    let output_byte_vec =
522        hex::decode(&output_content_hex).wrap_err("Failed to decode hex string")?;
523    // Create our target 31-byte array, initialized to all zeros.
524    let mut output_bytes = [0u8; 31];
525
526    // Calculate the starting position in the destination array.
527    // This ensures the bytes are right-aligned, effectively padding with leading zeros.
528    let start_index = 31 - output_byte_vec.len();
529
530    // Copy the decoded bytes from the vector into the correct slice of the array.
531    output_bytes[start_index..].copy_from_slice(&output_byte_vec);
532    Ok((
533        proof_json
534            .try_into()
535            .map_err(|e| eyre!("Failed to convert proof JSON to Seal: {:?}", e))?,
536        output_bytes,
537    ))
538}
