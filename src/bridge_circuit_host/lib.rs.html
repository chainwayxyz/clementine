26/// Fetches the light client proof for a given L1 block height.
27///
28/// This function queries an Citrea light client prover RPC endpoint to obtain a light client proof
29/// for a specified L1 height. The proof is then decoded and deserialized into an `InnerReceipt`,
30/// which is further processed to extract a valid `Receipt`.
31///
32/// # Arguments
33/// * `l1_height` - A `u32` representing the L1 block height for which the proof is requested.
34/// * `client` - An instance of `RpcClient` used to send the JSON-RPC request.
35///
36/// # Returns
37/// * `Ok((LightClientProof, Receipt))` - A tuple containing:
38///   - `LightClientProof`: The extracted proof with journal bytes and L2 height.
39///   - `Receipt`: The transaction receipt parsed from the proof.
40///
41/// # Panics
42/// This function will panic if:
43/// * The RPC response does not contain a valid `"proof"` string.
44/// * The proof is not a valid hex string.
45/// * The proof cannot be deserialized into `InnerReceipt`.
46/// * The receipt cannot be extracted from the `InnerReceipt`.
47/// * The RPC response does not contain a valid `"lastL2Height"` string.
48///
49/// Example:
50/// ```rust, ignore
51/// use alloy_rpc_client::RpcClient;
52/// use alloy_rpc_client::ClientBuilder;
53/// use bridge_circuit_host::fetch_light_client_proof;
54///
55/// const LIGHT_CLIENT_PROVER_URL: &str = "https://light-client-prover.testnet.citrea.xyz/";
56///
57/// #[tokio::main]
58/// async fn main() {
59///     let light_client_rpc_client =
60///         ClientBuilder::default().http(LIGHT_CLIENT_PROVER_URL.parse().unwrap());
61///
62///     let (light_client_proof, lcp_receipt) =
63///         fetch_light_client_proof(72471, light_client_rpc_client)
64///             .await
65///             .unwrap();
66/// }
67/// ```
68pub async fn fetch_light_client_proof(
69    l1_height: u32,
70    client: RpcClient,
71) -> Result<(LightClientProof, Receipt), ()> {
72    let request = json!({
73        "l1_height": l1_height
74    });
75
76    let response: serde_json::Value = client
77        .request("lightClientProver_getLightClientProofByL1Height", request)
78        .await
79        .unwrap();
80
81    let raw_proof = response["proof"].as_str().expect("Proof is missing");
82    let proof_str = raw_proof
83        .strip_prefix("0x")
84        .expect("Invalid proof format")
85        .to_string();
86
87    let bytes = decode(proof_str).expect("Invalid hex");
88    let decoded: InnerReceipt = bincode::deserialize(&bytes).expect("Failed to deserialize");
89    let receipt = receipt_from_inner(decoded).expect("Failed to create receipt");
90
91    let l2_height = response["lightClientProofOutput"]["lastL2Height"]
92        .as_str()
93        .expect("l2 height is not a string");
94
95    Ok((
96        LightClientProof {
97            lc_journal: receipt.journal.bytes.clone(),
98            l2_height: l2_height.to_string(),
99        },
100        receipt,
101    ))
102}

104/// Fetches the storage proof for a given deposit index and transaction ID.
105///
106/// This function interacts with an Citrea RPC endpoint to retrieve a storage proof,
107/// which includes proof details for both the UTXO and the deposit index.
108///
109/// # Arguments
110/// * `l2_height` - A reference to `String` representing the L2 block height. (e.g. "0x123a")
111/// * `deposit_index` - A `u32` representing the deposit index.
112/// * `client` - An instance of `RpcClient` used to make the JSON-RPC request.
113///
114/// # Returns
115/// Returns a `StorageProof` struct containing serialized storage proofs for the UTXO and deposit index.
116///
117/// # Errors
118/// * This function will panic if:
119///   * `keccak256(UTXOS_STORAGE_INDEX)` does not return a valid 32-byte slice.
120///   * The RPC request to `eth_getProof` fails.
121///   * The response from the RPC call cannot be deserialized into an `EIP1186AccountProofResponse`.
122///
123/// Example:
124/// ```rust,ignore
125/// use alloy_rpc_client::ClientBuilder;
126/// use bridge_circuit_host::fetch_storage_proof;
127/// use hex_literal::hex;
128///
129/// const CITREA_TESTNET_RPC: &str = "https://rpc.testnet.citrea.xyz/";
130///
131/// #[tokio::main]
132/// async fn main() {
133///     let citrea_rpc_client = ClientBuilder::default().http(CITREA_TESTNET_RPC.parse().unwrap());
134///
135///     let storage_proof = fetch_storage_proof(
136///         &"0xabc".to_string(),
137///         37,
138///         hex!("BB25103468A467382ED9F585129AD40331B54425155D6F0FAE8C799391EE2E7F"),
139///         citrea_rpc_client,
140///     )
141///     .await;
142/// }
143/// ```
144pub async fn fetch_storage_proof(
145    l2_height: &String,
146    deposit_index: u32,
147    client: RpcClient,
148) -> StorageProof {
149    let ind = deposit_index;
150    let tx_index: u32 = ind * 2;
151
152    let storage_address_wd_utxo_bytes = keccak256(UTXOS_STORAGE_INDEX);
153    let storage_address_wd_utxo: U256 = U256::from_be_bytes(
154        <[u8; 32]>::try_from(&storage_address_wd_utxo_bytes[..])
155            .expect("Slice with incorrect length"),
156    );
157
158    // Storage key address calculation UTXO
159    let storage_key_wd_utxo: alloy_primitives::Uint<256, 4> =
160        storage_address_wd_utxo + U256::from(tx_index);
161    let storage_key_wd_utxo_hex =
162        format!("0x{}", hex::encode(storage_key_wd_utxo.to_be_bytes::<32>()));
163
164    // Storage key address calculation Vout
165    let storage_key_vout: alloy_primitives::Uint<256, 4> =
166        storage_address_wd_utxo + U256::from(tx_index + 1);
167    let storage_key_vout_hex = format!("0x{}", hex::encode(storage_key_vout.to_be_bytes::<32>()));
168
169    // Storage key address calculation Deposit
170    let storage_address_deposit_bytes = keccak256(DEPOSIT_STORAGE_INDEX);
171    let storage_address_deposit: U256 = U256::from_be_bytes(
172        <[u8; 32]>::try_from(&storage_address_deposit_bytes[..])
173            .expect("Slice with incorrect length"),
174    );
175
176    let storage_key_deposit: alloy_primitives::Uint<256, 4> =
177        storage_address_deposit + U256::from(deposit_index);
178    let storage_key_deposit_hex = hex::encode(storage_key_deposit.to_be_bytes::<32>());
179    let storage_key_deposit_hex = format!("0x{}", storage_key_deposit_hex);
180
181    let request = json!([
182        CONTRACT_ADDRESS,
183        [
184            storage_key_wd_utxo_hex,
185            storage_key_vout_hex,
186            storage_key_deposit_hex
187        ],
188        l2_height
189    ]);
190
191    let response: serde_json::Value = client.request("eth_getProof", request).await.unwrap();
192
193    let response: EIP1186AccountProofResponse = serde_json::from_value(response).unwrap();
194
195    let serialized_utxo = serde_json::to_string(&response.storage_proof[0]).unwrap();
196
197    let serialized_vout = serde_json::to_string(&response.storage_proof[1]).unwrap();
198
199    let serialized_deposit = serde_json::to_string(&response.storage_proof[2]).unwrap();
200
201    StorageProof {
202        storage_proof_utxo: serialized_utxo,
203        storage_proof_vout: serialized_vout,
204        storage_proof_deposit_txid: serialized_deposit,
205        index: ind,
206    }
207}

209/// Converts an `InnerReceipt` into a `Receipt`, ensuring all required fields are present.
210///
211/// # Arguments
212/// * `inner` - The `InnerReceipt` to extract data from.
213///
214/// # Returns
215/// Returns a `Receipt` if all required fields are found, otherwise returns an error.
216///
217/// # Errors
218/// This function can return an error in the following cases:
219/// * If `inner.claim()` is empty.
220/// * If `claim.value()` is empty.
221/// * If `claim.output.value()` is empty.
222/// * If `output` is `None`.
223/// * If `output.journal.value()` is empty.
224fn receipt_from_inner(inner: InnerReceipt) -> eyre::Result<Receipt> {
225    let mb_claim = inner.claim().or_else(|_| bail!("Claim is empty"))?;
226    let claim = mb_claim
227        .value()
228        .or_else(|_| bail!("Claim content is empty"))?;
229    let output = claim
230        .output
231        .value()
232        .or_else(|_| bail!("Output content is empty"))?;
233    let Some(output) = output else {
234        bail!("Output body is empty");
235    };
236    let journal = output
237        .journal
238        .value()
239        .or_else(|_| bail!("Journal content is empty"))?;
240    Ok(Receipt::new(inner, journal))
241}
