1//! Signing information for taproot spends.
2
3use bitcoin::TapNodeHash;
4use serde::{Deserialize, Serialize};
5
6/// Information to re-sign an RBF transaction.
7/// Specifically the merkle root of the taproot to keyspend with and the output index of the utxo to be
8/// re-signed.
9///
10/// - Not needed for SinglePlusAnyoneCanPay RBF txs.
11/// - Not needed for CPFP.
12/// - Only signs for a keypath spend
13// Consider adding `#[cfg(any(test, feature = "test-fields"))]` back to
14// `annex` and `additional_taproot_output_count` fields to reduce struct size in production.
15// These fields are only used for testing large transaction scenarios.
16#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
17pub struct RbfSigningInfo {
18    /// The output index of the UTXO to be re-signed.
19    pub vout: u32,
20    /// The tweak merkle root for the taproot keyspend.
21    pub tweak_merkle_root: Option<TapNodeHash>,
22    /// Annex data (used for testing large transaction scenarios).
23    pub annex: Option<Vec<u8>>,
24    /// Additional taproot output count (used for testing large transaction scenarios).
25    pub additional_taproot_output_count: Option<u32>,
26}
27
28/// Contains information about the spend path that is needed to sign the utxo.
29#[derive(Copy, Clone, Debug, Eq, PartialEq, Serialize, Deserialize)]
30pub enum TapTweakData {
31    /// Key path spend with an optional merkle root for tweaking.
32    KeyPath(Option<TapNodeHash>),
33    /// Script path spend.
34    ScriptPath,
35    /// Unknown spend path.
36    Unknown,
37}
