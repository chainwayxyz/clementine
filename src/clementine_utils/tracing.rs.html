8/// Initializes a [`tracing`] subscriber depending on the environment.
9/// [`EnvFilter`] is used with an optional default level. Sets up the
10/// [`color_eyre`] handler.
11///
12/// # Log Formats
13///
14/// - `json` **JSON** is used when `LOG_FORMAT=json`
15/// - `human` **Human-readable** direct logs are used when `LOG_FORMAT` is not
16///   set to `json`.
17///
18/// ## CI
19///
20/// In CI, logging is always in the human-readable format with output to the
21/// console. The `INFO_LOG_FILE` env var can be used to set an optional log file
22/// output. If not set, only console logging is used.
23///
24/// # Backtraces
25///
26/// Backtraces are enabled by default for tests. Error backtraces otherwise
27/// depend on the `RUST_LIB_BACKTRACE` env var. Please read [`color_eyre`]
28/// documentation for more details.
29///
30/// # Parameters
31///
32/// - `default_level`: Default level ranges from 0 to 5. This is overwritten through the
33///   `RUST_LOG` env var.
34///
35/// # Returns
36///
37/// Returns `Err` in CI if the file logging cannot be initialized.  Already
38/// initialized errors are ignored, so this function can be called multiple
39/// times safely.
40pub fn initialize_logger(default_level: Option<LevelFilter>) -> Result<(), BridgeError> {
41    let is_ci = std::env::var("CI")
42        .map(|v| v == "true" || v == "1")
43        .unwrap_or(false);
44
45    // UNCOMMENT TO DEBUG TOKIO TASKS
46    // console_subscriber::init();
47
48    if cfg!(test) {
49        // Enable full backtraces for tests
50        std::env::set_var("RUST_LIB_BACKTRACE", "full");
51        std::env::set_var("RUST_BACKTRACE", "full");
52    }
53
54    // Initialize color-eyre for better error handling and backtraces
55    let mut hook_builder =
56        color_eyre::config::HookBuilder::default().add_frame_filter(Box::new(|frames| {
57            // Frames with names starting with any of the str's below will be filtered out
58            let filters = &[
59                "std::",
60                "test::",
61                "tokio::",
62                "core::",
63                "<core::",
64                "<alloc::",
65                "start_thread",
66                "<tonic::",
67                "<futures::",
68                "<tower::",
69                "<hyper",
70                "hyper",
71                "__rust_try",
72                "<axum::",
73                "<F as ",
74                "clone",
75            ];
76
77            frames.retain(|frame| {
78                !filters.iter().any(|f| {
79                    let name = if let Some(name) = frame.name.as_ref() {
80                        name.as_str()
81                    } else {
82                        return true;
83                    };
84
85                    name.starts_with(f)
86                })
87            });
88        }));
89
90    // When JSON logs are enabled, use a theme without color codes.
91    if is_json_logs() {
92        hook_builder = hook_builder.theme(color_eyre::config::Theme::new());
93    }
94
95    let _ = hook_builder.install();
96
97    if is_ci {
98        let info_log_file = std::env::var("INFO_LOG_FILE").ok();
99        if let Some(file_path) = info_log_file {
100            try_set_global_subscriber(env_subscriber_with_file(&file_path)?);
101            tracing::trace!("Using file logging in CI, outputting to {}", file_path);
102        } else {
103            try_set_global_subscriber(env_subscriber_to_human(default_level));
104            tracing::trace!("Using console logging in CI");
105            tracing::warn!(
106                "CI is set but INFO_LOG_FILE is missing, only console logs will be used."
107            );
108        }
109    } else if is_json_logs() {
110        try_set_global_subscriber(env_subscriber_to_json(default_level));
111        tracing::trace!("Using JSON logging");
112    } else {
113        try_set_global_subscriber(env_subscriber_to_human(default_level));
114        tracing::trace!("Using human-readable logging");
115    }
116
117    tracing::info!("Tracing initialized successfully.");
118    Ok(())
119}

121fn try_set_global_subscriber<S>(subscriber: S)
122where
123    S: Subscriber + Send + Sync + 'static,
124{
125    match tracing::subscriber::set_global_default(subscriber) {
126        Ok(_) => {}
127        // Statically, the only error possible is "already initialized"
128        Err(_) => {
129            #[cfg(test)]
130            tracing::trace!("Tracing is already initialized, skipping without errors...");
131            #[cfg(not(test))]
132            tracing::info!(
133                "Unexpected double initialization of tracing, skipping without errors..."
134            );
135        }
136    }
137}

139fn env_subscriber_with_file(path: &str) -> Result<Box<dyn Subscriber + Send + Sync>, BridgeError> {
140    if let Some(parent_dir) = std::path::Path::new(path).parent() {
141        std::fs::create_dir_all(parent_dir).map_err(|e| {
142            BridgeError::ConfigError(format!(
143                "Failed to create log directory '{}': {}",
144                parent_dir.display(),
145                e
146            ))
147        })?;
148    }
149
150    let file = File::create(path).map_err(|e| BridgeError::ConfigError(e.to_string()))?;
151
152    let file_filter = EnvFilter::from_default_env()
153        .add_directive("info".parse().expect("It should parse info level"))
154        .add_directive("ci=debug".parse().expect("It should parse ci debug level"));
155
156    let console_filter = EnvFilter::builder()
157        .with_default_directive(LevelFilter::WARN.into())
158        .from_env_lossy();
159
160    let file_layer = fmt::layer()
161        .with_writer(file)
162        .with_ansi(false)
163        .with_file(true)
164        .with_line_number(true)
165        .with_target(true)
166        .with_thread_ids(true)
167        .with_thread_names(true)
168        .with_filter(file_filter)
169        .boxed();
170
171    let console_layer = fmt::layer()
172        .with_test_writer()
173        .with_file(true)
174        .with_line_number(true)
175        .with_target(true)
176        .with_filter(console_filter)
177        .boxed();
178
179    Ok(Box::new(
180        Registry::default().with(file_layer).with(console_layer),
181    ))
182}

184fn env_subscriber_to_json(level: Option<LevelFilter>) -> Box<dyn Subscriber + Send + Sync> {
185    let filter = match level {
186        Some(lvl) => EnvFilter::builder()
187            .with_default_directive(lvl.into())
188            .from_env_lossy(),
189        None => EnvFilter::from_default_env(),
190    };
191
192    let json_layer = fmt::layer::<Registry>()
193        .with_test_writer()
194        // .with_timer(time::UtcTime::rfc_3339())
195        .with_file(true)
196        .with_line_number(true)
197        .with_thread_ids(true)
198        .with_thread_names(true)
199        .with_target(true)
200        .json();
201    // .with_current_span(true)z
202    // .with_span_list(true)
203    // To see how long each span takes, uncomment this.
204    // .with_span_events(FmtSpan::CLOSE)
205
206    Box::new(tracing_subscriber::registry().with(json_layer).with(filter))
207}

209fn env_subscriber_to_human(level: Option<LevelFilter>) -> Box<dyn Subscriber + Send + Sync> {
210    let filter = match level {
211        Some(lvl) => EnvFilter::builder()
212            .with_default_directive(lvl.into())
213            .from_env_lossy(),
214        None => EnvFilter::from_default_env(),
215    };
216
217    let standard_layer = fmt::layer()
218        .with_test_writer()
219        // .with_timer(time::UtcTime::rfc_3339())
220        .with_file(true)
221        .with_line_number(true)
222        // To see how long each span takes, uncomment this.
223        // .with_span_events(FmtSpan::CLOSE)
224        .with_target(true);
225
226    Box::new(
227        tracing_subscriber::registry()
228            .with(standard_layer)
229            .with(filter),
230    )
231}

233fn is_json_logs() -> bool {
234    std::env::var("LOG_FORMAT")
235        .map(|v| v.eq_ignore_ascii_case("json"))
236        .unwrap_or(false)
