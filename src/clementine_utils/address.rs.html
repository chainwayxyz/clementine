9/// A helper to construct a `TaprootBuilder` from a slice of script buffers, forming the script tree.
10/// Finds the needed depth the script tree needs to be to fit all the scripts and inserts the scripts.
11pub fn taproot_builder_with_scripts(scripts: impl Into<Vec<ScriptBuf>>) -> TaprootBuilder {
12    // doesn't clone if its already an owned Vec
13    let mut scripts: Vec<ScriptBuf> = scripts.into();
14    let builder = TaprootBuilder::new();
15    let num_scripts = scripts.len();
16
17    // Special return cases for n = 0 or n = 1
18    match num_scripts {
19        0 => return builder,
20        1 => {
21            return builder
22                .add_leaf(0, scripts.remove(0))
23                .expect("one root leaf added on empty builder")
24        }
25        _ => {}
26    }
27
28    let deepest_layer_depth: u8 = ((num_scripts - 1).ilog2() + 1) as u8;
29
30    let num_empty_nodes_in_final_depth = 2_usize.pow(deepest_layer_depth.into()) - num_scripts;
31    let num_nodes_in_final_depth = num_scripts - num_empty_nodes_in_final_depth;
32
33    scripts
34        .into_iter()
35        .enumerate()
36        .fold(builder, |acc, (i, script)| {
37            let is_node_in_last_minus_one_depth = (i >= num_nodes_in_final_depth) as u8;
38
39            acc.add_leaf(
40                deepest_layer_depth - is_node_in_last_minus_one_depth,
41                script,
42            )
43            .expect("algorithm tested to be correct")
44        })
45}

47/// Calculates the depth of each leaf in a balanced Taproot tree structure.
48/// The returned Vec contains the depth for each script at the corresponding index.
49pub fn calculate_taproot_leaf_depths(num_scripts: usize) -> Vec<u8> {
50    match num_scripts {
51        0 => return vec![],
52        1 => return vec![0],
53        _ => {}
54    }
55
56    let deepest_layer_depth: u8 = ((num_scripts - 1).ilog2() + 1) as u8;
57
58    let num_empty_nodes_in_final_depth = 2_usize.pow(deepest_layer_depth.into()) - num_scripts;
59    let num_nodes_in_final_depth = num_scripts - num_empty_nodes_in_final_depth;
60
61    (0..num_scripts)
62        .map(|i| {
63            let is_node_in_last_minus_one_depth = (i >= num_nodes_in_final_depth) as u8;
64            deepest_layer_depth - is_node_in_last_minus_one_depth
65        })
66        .collect()
67}

69/// Creates a taproot address with given scripts and internal key.
70pub fn create_taproot_address(
71    scripts: &[ScriptBuf],
72    internal_key: Option<XOnlyPublicKey>,
73    network: bitcoin::Network,
74) -> (Address, TaprootSpendInfo) {
75    // Build script tree
76    let taproot_builder = taproot_builder_with_scripts(scripts);
77    // Finalize the tree
78    let tree_info = match internal_key {
79        Some(xonly_pk) => taproot_builder
80            .finalize(&SECP, xonly_pk)
81            .expect("builder return is finalizable"),
82        None => taproot_builder
83            .finalize(&SECP, *UNSPENDABLE_XONLY_PUBKEY)
84            .expect("builder return is finalizable"),
85    };
86
87    // Create the address
88    let taproot_address: Address = Address::p2tr_tweaked(tree_info.output_key(), network);
89
90    (taproot_address, tree_info)
91}

93/// Creates an OP_RETURN output with the given data slice.
94///
95/// # Arguments
96///
97/// * `slice` - The data to embed in the OP_RETURN output.
98///
99/// # Returns
100///
101/// A [`TxOut`] with an OP_RETURN script containing the provided data.
102pub fn op_return_txout<S: AsRef<[u8]>>(slice: S) -> TxOut {
103    let buf = slice.as_ref().to_vec();
104    let push_bytes = PushBytesBuf::try_from(buf).expect("data too large for op_return");
105    TxOut {
106        value: Amount::ZERO,
107        script_pubkey: ScriptBuf::new_op_return(push_bytes),
108    }
109}
