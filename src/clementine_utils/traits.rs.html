7/// A trait for entities that have a name, operator, verifier, etc.

9/// and to provide a human-readable name for the entity for task names.
10pub trait NamedEntity: Sync + Send + 'static {

12    /// state machines with different owners in the database.
13    ///
14    /// ## Example
15    /// "operator", "verifier", "user"
16    const ENTITY_NAME: &'static str;
17
18    /// Consumer ID for the tx sender task.
19    const TX_SENDER_CONSUMER_ID: &'static str;
20
21    /// Consumer ID for the finalized block task with no automation.
22    const FINALIZED_BLOCK_CONSUMER_ID_NO_AUTOMATION: &'static str;
23
24    /// Consumer ID for the finalized block task with automation.

28/// Trait to extract last 20 bytes (for address derivation).
29pub trait Last20Bytes {
30    /// Extract the last 20 bytes.
31    fn last_20_bytes(&self) -> [u8; 20];
32}
33
34/// Fallible version of [`Last20Bytes`].
35pub trait TryLast20Bytes {
36    /// Extract the last 20 bytes, or return an error.
37    fn try_last_20_bytes(self) -> Result<[u8; 20], BridgeError>;
38}
39
40impl Last20Bytes for [u8; 32] {
41    fn last_20_bytes(&self) -> [u8; 20] {
42        self.as_slice()
43            .try_last_20_bytes()
44            .expect("will not happen")
45    }
46}
47
48impl TryLast20Bytes for &[u8] {
49    fn try_last_20_bytes(self) -> Result<[u8; 20], BridgeError> {
50        if self.len() < 20 {
51            return Err(eyre::eyre!("Input is too short to contain 20 bytes").into());
52        }
53        let mut result = [0u8; 20];
54        result.copy_from_slice(&self[self.len() - 20..]);
55        Ok(result)
56    }
57}

59/// Extension trait for [`ScriptBuf`].
60pub trait ScriptBufExt {
61    /// Try to extract the taproot public key from a P2TR script.
62    fn try_get_taproot_pk(&self) -> Result<XOnlyPublicKey, BridgeError>;
63}
64
65impl ScriptBufExt for ScriptBuf {
66    fn try_get_taproot_pk(&self) -> Result<XOnlyPublicKey, BridgeError> {
67        if !self.is_p2tr() {
68            return Err(eyre::eyre!("Script is not a valid P2TR script (not 34 bytes)").into());
69        }
70
71        Ok(XOnlyPublicKey::from_slice(&self.as_bytes()[2..34])
72            .wrap_err("Failed to parse XOnlyPublicKey from script")?)
73    }
