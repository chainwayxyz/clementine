179#[derive(Debug, Clone, Serialize, Deserialize)]
180pub struct BitcoinMerkleTree {
181    depth: u32,
182    nodes: Vec<Vec<[u8; 32]>>,
183}
184
185impl BitcoinMerkleTree {
186    pub fn new(transactions: Vec<[u8; 32]>) -> Self {
187        // assert!(depth > 0, "Depth must be greater than 0");
188        // assert!(depth <= 254, "Depth must be less than or equal to 254");
189        // assert!(
190        //     u32::pow(2, (depth) as u32) >= transactions.len() as u32,
191        //     "Too many transactions for this depth"
192        // );
193        let depth = (transactions.len() - 1).ilog(2) + 1;
194        let mut tree = BitcoinMerkleTree {
195            depth,
196            nodes: vec![],
197        };
198
199        // Populate leaf nodes
200        tree.nodes.push(vec![]);
201        for tx in transactions.iter() {
202            tree.nodes[0].push(*tx);
203        }
204
205        // Construct the tree
206        let mut curr_level_offset: usize = 1;
207        let mut prev_level_size = transactions.len();
208        let mut prev_level_index_offset = 0;
209        let mut preimage: [u8; 64] = [0; 64];
210        while prev_level_size > 1 {
211            // println!("curr_level_offset: {}", curr_level_offset);
212            // println!("prev_level_size: {}", prev_level_size);
213            // println!("prev_level_index_offset: {}", prev_level_index_offset);
214            tree.nodes.push(vec![]);
215            for i in 0..(prev_level_size / 2) {
216                preimage[..32].copy_from_slice(
217                    &tree.nodes[curr_level_offset - 1_usize][prev_level_index_offset + i * 2],
218                );
219                preimage[32..].copy_from_slice(
220                    &tree.nodes[curr_level_offset - 1][prev_level_index_offset + i * 2 + 1],
221                );
222                let combined_hash = calculate_double_sha256(&preimage);
223                tree.nodes[curr_level_offset].push(combined_hash);
224            }
225            if prev_level_size % 2 == 1 {
226                let mut preimage: [u8; 64] = [0; 64];
227                preimage[..32].copy_from_slice(
228                    &tree.nodes[curr_level_offset - 1]
229                        [prev_level_index_offset + prev_level_size - 1],
230                );
231                preimage[32..].copy_from_slice(
232                    &tree.nodes[curr_level_offset - 1]
233                        [prev_level_index_offset + prev_level_size - 1],
234                );
235                let combined_hash = calculate_double_sha256(&preimage);
236                tree.nodes[curr_level_offset].push(combined_hash);
237            }
238            curr_level_offset += 1;
239            prev_level_size = prev_level_size.div_ceil(2);
240            prev_level_index_offset = 0;
241        }
242        tree
243    }
244
245    // Returns the Merkle root
246    pub fn root(&self) -> [u8; 32] {
247        self.nodes[self.nodes.len() - 1][0]
248    }
249
250    pub fn get_idx_path(&self, index: u32) -> Vec<[u8; 32]> {
251        assert!(index < self.nodes[0].len() as u32, "Index out of bounds");
252        let mut path = vec![];
253        let mut level = 0;
254        let mut i = index;
255        while level < self.nodes.len() as u32 - 1 {
256            if i % 2 == 1 {
257                path.push(self.nodes[level as usize][i as usize - 1]);
258            } else if (self.nodes[level as usize].len() - 1) as u32 == i {
259                path.push(self.nodes[level as usize][i as usize]);
260            } else {
261                path.push(self.nodes[level as usize][(i + 1) as usize]);
262            }
263
264            level += 1;
265            i /= 2;
266        }
267
268        path
269    }
270
271    pub fn calculate_root_with_merkle_proof(
272        &self,
273        txid: [u8; 32],
274        idx: u32,
275        merkle_proof: Vec<[u8; 32]>,
276    ) -> [u8; 32] {
277        let mut preimage: [u8; 64] = [0; 64];
278        let mut combined_hash: [u8; 32] = txid;
279        let mut index = idx;
280        let mut level: u32 = 0;
281        while level < self.depth {
282            if index % 2 == 0 {
283                preimage[..32].copy_from_slice(&combined_hash);
284                preimage[32..].copy_from_slice(&merkle_proof[level as usize]);
285                combined_hash = calculate_double_sha256(&preimage);
286            } else {
287                preimage[..32].copy_from_slice(&merkle_proof[level as usize]);
288                preimage[32..].copy_from_slice(&combined_hash);
289                combined_hash = calculate_double_sha256(&preimage);
290            }
291            level += 1;
292            index /= 2;
293        }
294        combined_hash
295    }
296}

298pub fn calculate_double_sha256(input: &[u8]) -> [u8; 32] {
299    let mut hasher = Sha256::default();
300    hasher.update(input);
301    let result = hasher.finalize_reset();
302    hasher.update(result);
303    hasher.finalize().into()
304}

306fn _get_block_merkle_proof(
307    block: Block,
308    target_txid: Txid,
309) -> Result<(usize, Vec<u8>), BridgeError> {
310    let mut txid_index = 0;
311    let txids = block
312        .txdata
313        .iter()
314        .enumerate()
315        .map(|(i, tx)| {
316            if tx.compute_txid() == target_txid {
317                txid_index = i;
318            }
319
320            if i == 0 {
321                [0; 32]
322            } else {
323                let wtxid = tx.compute_wtxid();
324                wtxid.as_byte_array().to_owned()
325            }
326        })
327        .collect::<Vec<_>>();
328
329    let merkle_tree = BitcoinMerkleTree::new(txids.clone());
330    let _witness_root = block.witness_root().expect("Failed to get witness root");
331    let witness_idx_path =
332        merkle_tree.get_idx_path(txid_index.try_into().expect("Failed to convert index"));
333
334    let _root = merkle_tree.calculate_root_with_merkle_proof(
335        txids[txid_index],
336        txid_index.try_into().expect("Failed to convert index"),
337        witness_idx_path.clone(),
338    );
339
340    Ok((txid_index, witness_idx_path.into_iter().flatten().collect()))
341}

356async fn handle_operator_call(url: String, command: OperatorCommands) {
357    let config = create_minimal_config();
358    let mut operator =
359        clementine_core::rpc::get_clients(vec![url], ClementineOperatorClient::new, &config, true)
360            .await
361            .expect("Exists")[0]
362            .clone();
363
364    match command {
365        OperatorCommands::GetDepositKeys {
366            deposit_outpoint_txid,
367            deposit_outpoint_vout,
368        } => {
369            println!(
370                "Getting deposit keys for outpoint {}:{}",
371                deposit_outpoint_txid, deposit_outpoint_vout
372            );
373            let params = clementine_core::rpc::clementine::DepositParams {
374                security_council: Some(clementine::SecurityCouncil {
375                    pks: vec![],
376                    threshold: 0,
377                }),
378                deposit: Some(Deposit {
379                    deposit_outpoint: Some(Outpoint {
380                        txid: deposit_outpoint_txid.into(),
381                        vout: deposit_outpoint_vout,
382                    }),
383                    deposit_data: Some(DepositData::BaseDeposit(BaseDeposit {
384                        evm_address: vec![1; 20],
385                        recovery_taproot_address: String::new(),
386                    })),
387                }),
388                actors: Some(Actors {
389                    verifiers: Some(VerifierPublicKeys {
390                        verifier_public_keys: vec![],
391                    }),
392                    watchtowers: Some(XOnlyPublicKeys {
393                        xonly_public_keys: vec![],
394                    }),
395                    operators: Some(XOnlyPublicKeys {
396                        xonly_public_keys: vec![],
397                    }),
398                }),
399            };
400            let response = operator
401                .get_deposit_keys(Request::new(params))
402                .await
403                .expect("Failed to make a request");
404            println!("Get deposit keys response: {:?}", response);
405        }
406        OperatorCommands::GetParams => {
407            let params = operator
408                .get_params(Empty {})
409                .await
410                .expect("Failed to make a request");
411            println!("Operator params: {:?}", params);
412        }
413        OperatorCommands::Withdraw {
414            withdrawal_id,
415            input_signature,
416            input_outpoint_txid,
417            input_outpoint_vout,
418            output_script_pubkey,
419            output_amount,
420        } => {
421            println!("Processing withdrawal with id {}", withdrawal_id);
422
423            let params = clementine_core::rpc::clementine::WithdrawParams {
424                withdrawal_id,
425                input_signature: input_signature.as_bytes().to_vec(),
426                input_outpoint: Some(Outpoint {
427                    txid: input_outpoint_txid.as_bytes().to_vec(),
428                    vout: input_outpoint_vout,
429                }),
430                output_script_pubkey: output_script_pubkey.as_bytes().to_vec(),
431                output_amount,
432            };
433            operator
434                .withdraw(Request::new(params))
435                .await
436                .expect("Failed to make a request");
437        }
438        OperatorCommands::Vergen => {
439            let params = Empty {};
440            let response = operator
441                .vergen(Request::new(params))
442                .await
443                .expect("Failed to make a request");
444            println!("Vergen response:\n{}", response.into_inner().response);
445        }
446    }
447}

857            for (i, result) in withdraw_responses.iter().enumerate() {

860                        success,
861                    )) => {
862                        let txid = bitcoin::Txid::from_byte_array(
863                            success
864                                .txid
865                                .clone()
866                                .try_into()
867                                .expect("Failed to convert txid to array"),
868                        );
869                        println!("Operator {}: Withdrawal successful, txid: {}", i, txid);
870                    }
871                    Some(clementine_core::rpc::clementine::withdraw_result::Result::Error(
872                        error,
873                    )) => {
874                        println!("Operator {}: Withdrawal failed: {}", i, error.error);
875                    }
876                    None => {
877                        println!("Operator {}: Unknown result", i);
878                    }
879                }
880            }
881        }
